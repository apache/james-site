217     public void testSubject() throws Exception {
218         assertEquals("Subject: ", decode(Fields.subject("").getRaw()));
219         assertEquals("Subject: test", decode(Fields.subject("test").getRaw()));
220         assertEquals("Subject: =?ISO-8859-1?Q?Sm=F8rebr=F8d?=", decode(Fields
221                 .subject("Sm\370rebr\370d").getRaw()));
222 
223         String seventyEight = "12345678901234567890123456789012345678901234567890123456789012345678";
224         assertEquals("Subject:\r\n " + seventyEight, decode(Fields.subject(
225                 seventyEight).getRaw()));
226 
227         String seventyNine = seventyEight + "9";
228         String expected = "Subject: =?US-ASCII?Q?1234567890123456789012345678901234?="
229                 + "\r\n =?US-ASCII?Q?56789012345678901234567890123456789?=";
230         assertEquals(expected, decode(Fields.subject(seventyNine).getRaw()));
231     }
