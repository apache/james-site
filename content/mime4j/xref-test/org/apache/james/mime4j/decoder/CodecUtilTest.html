49      public void testEncodeQuotedPrintableLargeInput() throws Exception {
50          StringBuffer sb = new StringBuffer();
51          for (int i = 0; i < 1024 * 5; i++) {
52              sb.append((char) ('0' + (i % 10)));
53          }
54          String expected = sb.toString().replaceAll("(\\d{75})", "$1=\r\n");
55          
56          InputStream in = new ByteArrayInputStream(sb.toString().getBytes("US-ASCII"));
57          ByteArrayOutputStream out = new ByteArrayOutputStream();
58          CodecUtil.encodeQuotedPrintableBinary(in, out);
59          String actual = new String(out.toByteArray(), "US-ASCII");
60          assertEquals(expected, actual);
61      }

72      public void testBase64OutputStream() throws Exception {

82      private String roundtripUsingOutputStream(String input) throws IOException {
83          ByteArrayOutputStream out2 = new ByteArrayOutputStream();
84          Base64OutputStream outb64 = new Base64OutputStream(out2, 76);
85          CodecUtil.copy(new ByteArrayInputStream(input.getBytes()), outb64);
86          outb64.flush();
87          outb64.close();
88          
89          InputStream is = new Base64InputStream(new ByteArrayInputStream(out2.toByteArray()));
90          ByteArrayOutputStream outRoundtrip = new ByteArrayOutputStream();
91          CodecUtil.copy(is, outRoundtrip);
92          String output = new String(outRoundtrip.toByteArray());
93          return output;
94      }
