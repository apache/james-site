66      public void testGetMimeType() {
67          Message parent = null;
68          Message child = null;
69          
70          parent = new Message();
71          child = new Message();
72          child.setParent(parent);
73          parent.setHeader(headerMultipartDigest);
74          child.setHeader(headerEmpty);
75          assertEquals("multipart/digest, empty", "message/rfc822", 
76                  child.getMimeType());
77          child.setHeader(headerTextPlain);
78          assertEquals("multipart/digest, text/plain", "text/plain", 
79                  child.getMimeType());
80          child.setHeader(headerMessageRFC822);
81          assertEquals("multipart/digest, message/rfc822", "message/rfc822", 
82                  child.getMimeType());
83          
84          parent = new Message();
85          child = new Message();
86          child.setParent(parent);
87          parent.setHeader(headerMultipartMixed);
88          child.setHeader(headerEmpty);
89          assertEquals("multipart/mixed, empty", "text/plain", 
90                  child.getMimeType());
91          child.setHeader(headerTextPlain);
92          assertEquals("multipart/mixed, text/plain", "text/plain", 
93                  child.getMimeType());
94          child.setHeader(headerMessageRFC822);
95          assertEquals("multipart/mixed, message/rfc822", "message/rfc822", 
96                  child.getMimeType());
97          
98          child = new Message();
99          child.setHeader(headerEmpty);
100         assertEquals("null, empty", "text/plain", child.getMimeType());
101         child.setHeader(headerTextPlain);
102         assertEquals("null, text/plain", "text/plain", child.getMimeType());
103         child.setHeader(headerMessageRFC822);
104         assertEquals("null, message/rfc822", "message/rfc822", 
105                 child.getMimeType());
106     }

108     public void testIsMultipart() {
109         Message m = new Message();
110         
111         m.setHeader(headerEmpty);
112         assertTrue("empty", !m.isMultipart());
113         
114         m.setHeader(headerTextPlain);
115         assertTrue("text/plain", !m.isMultipart());
116         
117         m.setHeader(headerMultipartDigest);
118         assertTrue("multipart/digest", m.isMultipart());
119         
120         m.setHeader(headerMultipartMixed);
121         assertTrue("multipart/mixed", m.isMultipart());
122     }

441     public void testDisposeGetsPropagatedToBody() throws Exception {
442         DummyBody body1 = new DummyBody();
443         BodyPart part1 = new BodyPart();
444         part1.setHeader(headerEmpty);
445         part1.setBody(body1);
446 
447         DummyBody body2 = new DummyBody();
448         BodyPart part2 = new BodyPart();
449         part2.setHeader(headerEmpty);
450         part2.setBody(body2);
451 
452         Multipart mp = new Multipart("mixed");
453         mp.addBodyPart(part1);
454         mp.addBodyPart(part2);
455 
456         Message m = new Message();
457         m.setHeader(headerMultipartMixed);
458         m.setBody(mp);
459 
460         assertFalse(body1.disposed);
461         assertFalse(body2.disposed);
462 
463         m.dispose();
464 
465         assertTrue(body1.disposed);
466         assertTrue(body2.disposed);
467     }
