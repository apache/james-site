49      public void sendMail(MailAddress sender, Collection recipients, MimeMessage msg) throws MessagingException {
50          Object[] mailObjects = new Object[]{sender, recipients, new MimeMessageCopyOnWriteProxy(msg)};
51          mails.add(mailObjects);
52      }

54      public void sendMail(MailAddress sender, Collection recipients, InputStream msg) throws MessagingException {
55          Object[] mailObjects = new Object[]{sender, recipients, msg};
56          mails.add(mailObjects);
57      }

59      public void sendMail(Mail mail) throws MessagingException {
60          int bodySize = mail.getMessage().getSize();
61          try {
62              if (m_maxMessageSizeBytes != 0 && m_maxMessageSizeBytes < bodySize) throw new MessageSizeException();
63          } catch (MessageSizeException e) {
64              throw new MessagingException("message size exception is nested", e);
65          }
66          sendMail(mail.getSender(), mail.getRecipients(), mail.getMessage());
67      }
68  
69      public void sendMail(MimeMessage message) throws MessagingException {
70          // taken from class org.apache.james.James 
71          MailAddress sender = new MailAddress((InternetAddress)message.getFrom()[0]);
72          Collection recipients = new HashSet();
73          Address addresses[] = message.getAllRecipients();
74          if (addresses != null) {
75              for (int i = 0; i < addresses.length; i++) {
76                  // Javamail treats the "newsgroups:" header field as a
77                  // recipient, so we want to filter those out.
78                  if ( addresses[i] instanceof InternetAddress ) {
79                      recipients.add(new MailAddress((InternetAddress)addresses[i]));
80                  }
81              }
82          }
83          sendMail(sender, recipients, message);
84      }
