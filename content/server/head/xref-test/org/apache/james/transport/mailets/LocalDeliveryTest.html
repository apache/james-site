318     /**
319      * @param mail
320      * @param expectedMails
321      * @throws MessagingException
322      */
323     private void assertDeliveryWorked(Mail mail, HashMap expectedMails) throws MessagingException {
324         assertEquals(Mail.GHOST, mail.getState());
325         Iterator mboxes = mailboxes.keySet().iterator();
326         while (mboxes.hasNext()) {
327             String mboxName = (String) mboxes.next();
328             MailRepository inMemorySpoolRepository = (MailRepository) mailboxes.get(mboxName);
329             assertExpectedMailsInRepository(mail, (String[]) expectedMails.get(mboxName), inMemorySpoolRepository);
330             
331         }
332         
333         MailRepository sentMailsRepository = mockMailServer.getSentMailsRepository();
334         MailRepository errorsMailRepository = new InMemorySpoolRepository();
335         
336         Iterator keys = sentMailsRepository.list();
337         while (keys.hasNext()) {
338             String nextKey = (String) keys.next();
339             Mail m = sentMailsRepository.retrieve(nextKey);
340             if (Mail.ERROR.equals(m.getState())) {
341                 errorsMailRepository.store(m);
342                 sentMailsRepository.remove(nextKey);
343             } else {
344                 assertEquals("Found a mail in outgoing having a state different from ERROR or ROOT", Mail.DEFAULT, m.getState());
345             }
346             
347             ContainerUtil.dispose(m);
348         }
349         
350         assertExpectedMailsInRepository(mail, (String[]) expectedMails.get("resent"), sentMailsRepository);
351         assertExpectedMailsInRepository(mail, (String[]) expectedMails.get("errors"), errorsMailRepository);
352         
353         ContainerUtil.dispose(errorsMailRepository);
354     }

357     /**
358      * @param mail
359      * @param expectedMails
360      * @param mboxName
361      * @param inMemorySpoolRepository
362      * @throws MessagingException
363      */
364     private void assertExpectedMailsInRepository(Mail mail, String[] expectedDeliveries, MailRepository inMemorySpoolRepository) throws MessagingException {
365         List c = expectedDeliveries != null ? new ArrayList(Arrays.asList(expectedDeliveries)) : new ArrayList();
366         Iterator i = inMemorySpoolRepository.list();
367         System.out.println("check: "+c.size()+"|"+inMemorySpoolRepository);
368         for (int j = 0; j < c.size(); j++) {
369             assertTrue("No mails have been found in the repository", i.hasNext());
370             String next = (String) i.next();
371             assertNotNull("Mail has not been stored", next);
372             Mail storedMail = inMemorySpoolRepository.retrieve(next);
373             assertNotNull("Mail cannot be retrieved", storedMail);
374             String recipient = storedMail.getRecipients().iterator().next().toString();
375             assertTrue("Message has been delivered to the wrong user", c.contains(recipient));
376             c.remove(recipient);
377         }
378         assertEquals("Message has not been delivered to all expected recipients", c.size(), 0);
379         if (i.hasNext()) {
380             String tooMuch = (String) i.next();
381             Mail m = inMemorySpoolRepository.retrieve(tooMuch);
382             fail("Found unexpected mail for recipient "+m.getRecipients().iterator().next().toString()+"!");
383         }
384     }
