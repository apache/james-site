159     public void verifyLastMail(String sender, String recipient, MimeMessage msg) throws IOException, MessagingException {
160         Mail mailData = m_mailServer.getLastMail();
161         assertNotNull("mail received by mail server", mailData);
162 
163         if (sender == null && recipient == null && msg == null) fail("no verification can be done with all arguments null");
164 
165         if (sender != null) assertEquals("sender verfication", sender, mailData.getSender().toString());
166         if (recipient != null) assertTrue("recipient verfication", mailData.getRecipients().contains(new MailAddress(recipient)));
167         if (msg != null) {
168             ByteArrayOutputStream bo1 = new ByteArrayOutputStream();
169             msg.writeTo(bo1);
170             ByteArrayOutputStream bo2 = new ByteArrayOutputStream();
171             mailData.getMessage().writeTo(bo2);
172             assertEquals(bo1.toString(),bo2.toString());
173             assertEquals("message verification", msg, mailData.getMessage());
174         }
175     }
