55          if (mappings != null) {
56              Iterator mappingsIt = virtualUserTable.getAllMappings().keySet().iterator();
57      
58      
59              while(mappingsIt.hasNext()) {
60                  String key = mappingsIt.next().toString();
61                  String args[] = key.split("@");
62          
63                  Collection map = (Collection) mappings.get(key);
64          
65                  Iterator mapIt = map.iterator();
66          
67                  while (mapIt.hasNext()) {
68                      try {
69                          removeMapping(args[0], args[1], mapIt.next().toString());
70                      } catch (InvalidMappingException e) {
71                          e.printStackTrace();
72                      }
73                  }
74              }
75          }
76          ContainerUtil.dispose(virtualUserTable);
77      }

79      private void removeMapping(String user, String domain, String rawMapping) throws InvalidMappingException {
80          if (rawMapping.startsWith(VirtualUserTable.ERROR_PREFIX)) {
81              removeMapping(user, domain, rawMapping.substring(VirtualUserTable.ERROR_PREFIX.length()), ERROR_TYPE);
82          } else if (rawMapping.startsWith(VirtualUserTable.REGEX_PREFIX)) {
83              removeMapping(user, domain, rawMapping.substring(VirtualUserTable.REGEX_PREFIX.length()), REGEX_TYPE);
84          } else if (rawMapping.startsWith(VirtualUserTable.ALIASDOMAIN_PREFIX)) {
85              removeMapping(user, domain, rawMapping.substring(VirtualUserTable.ALIASDOMAIN_PREFIX.length()), ALIASDOMAIN_TYPE);
86          } else {
87              removeMapping(user, domain, rawMapping, ADDRESS_TYPE);
88          }
89      }

279     public void testRecursiveMapping() throws ErrorMappingException {
280         String user1 = "user1";
281         String user2 = "user2";
282         String user3 = "user3";
283         String domain1 = "domain1";
284         String domain2 = "domain2";
285         String domain3 = "domain3";
286         boolean exception1 = false;
287     
288         virtualUserTable.setRecursiveMapping(true);
289        
290         try {
291             assertNull("No mappings",virtualUserTable.getAllMappings());
292      
293             assertTrue("Add mapping", addMapping(user1, domain1, user2 + "@" + domain2, ADDRESS_TYPE));
294             assertTrue("Add mapping", addMapping(user2, domain2, user3 + "@" + domain3, ADDRESS_TYPE));
295             assertEquals("Recursive mapped", virtualUserTable.getMappings(user1, domain1).iterator().next(),user3 + "@" + domain3);
296 
297             assertTrue("Add mapping", addMapping(user3, domain3, user1 + "@" + domain1, ADDRESS_TYPE));
298             try {
299                 virtualUserTable.getMappings(user1, domain1);
300             } catch (ErrorMappingException e) {
301                 exception1 = true;
302             }
303             assertTrue("Exception thrown on to many mappings", exception1);
304             
305             // disable recursive mapping
306             virtualUserTable.setRecursiveMapping(false);
307             assertEquals("Not recursive mapped", virtualUserTable.getMappings(user1, domain1).iterator().next(),user2 + "@" + domain2);
308             
309         } catch (InvalidMappingException e) {
310             fail("Storing failed");
311         }
312     }
