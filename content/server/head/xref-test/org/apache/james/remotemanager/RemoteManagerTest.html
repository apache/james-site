123     protected String getLastLine(List list) {
124         if (list == null || list.isEmpty()) return null;
125         return (String)list.get(list.size()-1);
126     }

128     protected List readAnswer() {
129         return readAnswer(1);
130     }
131     
132     protected List readAnswer(int numLines) {
133         List allAnswerLines = new ArrayList();
134         try {
135             if (numLines > 0) {
136                 for (int i = 0; i < numLines; i++) {
137                     allAnswerLines.add(m_reader.readLine());
138                 }
139             } else {
140                 String line = m_reader.readLine();
141                 allAnswerLines.add(line);
142                 
143                 while (m_reader.ready()) {
144                     allAnswerLines.add(m_reader.readLine());
145                 }
146             }
147             return allAnswerLines;
148         } catch (IOException e) {
149             return null;
150         }
151     }
152 
153     protected void sendCommand(String command) throws IOException {
154         m_writer.println(command);
155         m_writer.flush();
156     }

163         m_writer = new InternetPrintWriter(new BufferedOutputStream(m_telnetClient.getOutputStream(), 1024), true);
164         
165         readAnswer(3);
166     }

374     public void testListUsers() throws IOException {
375         finishSetUp(m_testConfiguration);
376         connect();
377         login();
378 
379         String[] users = new String[] {"ccc", "aaa", "dddd", "bbbbb"};
380 
381         for (int i = 0; i < users.length; i++) {
382             String user = users[i];
383             sendCommand("adduser " + user + " test");
384             readAnswer(1);
385         }
386 
387         delay();
388 
389         sendCommand("listusers");
390         List list = readAnswer(5);
391 
392         assertEquals("user count line", "Existing accounts " + users.length, list.get(0));
393         
394         List readUserNames = new ArrayList();
395         for (Iterator iterator = list.iterator(); iterator.hasNext();) {
396             String answerLine = (String) iterator.next();
397             if (!answerLine.startsWith("user: ")) continue;
398             readUserNames.add(answerLine.substring(6));
399         }
400         assertEquals("user count", users.length, readUserNames.size());
401 
402         for (int i = 0; i < users.length; i++) {
403             String user = users[i];
404             assertTrue("name found", readUserNames.contains(user));
405         }
406     }

488 
489         //TODO: is this correct? setting self as alias!
