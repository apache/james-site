31  public interface ImapResponseComposer {
32  
33      /**
34       * Writes a standard tagged OK response on completion of a command. Response
35       * is writen as:
36       * 
37       * <pre>
38       *        a01 OK COMMAND_NAME completed.
39       * </pre>
40       * 
41       * @param command
42       *            The ImapCommand which was completed.
43       */
44      public abstract void commandComplete(final ImapCommand command,
45              final String tag) throws IOException;
46  
47      /**
48       * Writes a standard tagged OK response on completion of a command, with a
49       * response code (eg READ-WRITE) Response is writen as:
50       * 
51       * <pre>
52       *        a01 OK [responseCode] COMMAND_NAME completed.
53       * </pre>
54       * 
55       * @param command
56       *            The ImapCommand which was completed.
57       * @param responseCode
58       *            A string response code to send to the client.
59       */
60      public abstract void commandComplete(final ImapCommand command,
61              final String responseCode, final String tag) throws IOException;
62  
63      /**
64       * Writes a standard NO response on command failure, together with a
65       * descriptive message. Response is writen as:
66       * 
67       * <pre>
68       *        a01 NO COMMAND_NAME failed. &lt;reason&gt;
69       * </pre>
70       * 
71       * @param command
72       *            The ImapCommand which failed.
73       * @param reason
74       *            A message describing why the command failed.
75       */
76      public abstract void commandFailed(final ImapCommand command,
77              final String reason, final String tag) throws IOException;
78  
79      /**
80       * Writes a standard NO response on command failure, together with a
81       * descriptive message. Response is writen as:
82       * 
83       * <pre>
84       *        a01 NO [responseCode] COMMAND_NAME failed. &lt;reason&gt;
85       * </pre>
86       * 
87       * @param command
88       *            The ImapCommand which failed.
89       * @param responseCode
90       *            The Imap response code to send.
91       * @param reason
92       *            A message describing why the command failed.
93       */
94      public abstract void commandFailed(ImapCommand command,
95              String responseCode, String reason, final String tag)
96              throws IOException;

98      /**
99       * Writes a standard BAD response on command error, together with a

101      * 
102      * <pre>
103      *        a01 BAD &lt;message&gt;
104      * </pre>
105      * 
106      * @param message
107      *            The descriptive error message.
108      */
109     public abstract void commandError(final String message, final String tag)

123             throws IOException;

126      * Writes an untagged OK response, with the supplied response code, and an

128      * 
129      * @param responseCode
