22  import org.apache.james.api.imap.message.response.imap4rev1.StatusResponseFactory;
23  import org.apache.james.api.imap.process.ImapProcessor;
24  import org.apache.james.imap.processor.imap4rev1.fetch.FetchProcessor;

30  public class Imap4Rev1ProcessorFactory {
31  
32      public static final ImapProcessor createDefaultChain(
33              final ImapProcessor chainEndProcessor,
34              final MailboxManagerProvider mailboxManagerProvider,
35              final StatusResponseFactory statusResponseFactory) {
36  
37          final LogoutProcessor logoutProcessor = new LogoutProcessor(
38                  chainEndProcessor, statusResponseFactory);
39          final CapabilityProcessor capabilityProcessor = new CapabilityProcessor(
40                  logoutProcessor, statusResponseFactory);
41          final CheckProcessor checkProcessor = new CheckProcessor(
42                  capabilityProcessor, statusResponseFactory);
43          final LoginProcessor loginProcessor = new LoginProcessor(
44                  checkProcessor, mailboxManagerProvider, statusResponseFactory);
45          final RenameProcessor renameProcessor = new RenameProcessor(
46                  loginProcessor, mailboxManagerProvider, statusResponseFactory);
47          final DeleteProcessor deleteProcessor = new DeleteProcessor(
48                  renameProcessor, mailboxManagerProvider, statusResponseFactory);
49          final CreateProcessor createProcessor = new CreateProcessor(
50                  deleteProcessor, mailboxManagerProvider, statusResponseFactory);
51          final CloseProcessor closeProcessor = new CloseProcessor(
52                  createProcessor, statusResponseFactory);
53          final UnsubscribeProcessor unsubscribeProcessor = new UnsubscribeProcessor(
54                  closeProcessor, mailboxManagerProvider, statusResponseFactory);
55          final SubscribeProcessor subscribeProcessor = new SubscribeProcessor(
56                  unsubscribeProcessor, mailboxManagerProvider,
57                  statusResponseFactory);
58          final CopyProcessor copyProcessor = new CopyProcessor(
59                  subscribeProcessor, mailboxManagerProvider,
60                  statusResponseFactory);
61          final AuthenticateProcessor authenticateProcessor = new AuthenticateProcessor(
62                  copyProcessor, statusResponseFactory);
63          final ExpungeProcessor expungeProcessor = new ExpungeProcessor(
64                  authenticateProcessor, mailboxManagerProvider,
65                  statusResponseFactory);
66          final ExamineProcessor examineProcessor = new ExamineProcessor(
67                  expungeProcessor, mailboxManagerProvider, statusResponseFactory);
68          final AppendProcessor appendProcessor = new AppendProcessor(
69                  examineProcessor, mailboxManagerProvider, statusResponseFactory);
70          final StoreProcessor storeProcessor = new StoreProcessor(
71                  appendProcessor, statusResponseFactory);
72          final NoopProcessor noopProcessor = new NoopProcessor(storeProcessor,
73                  statusResponseFactory);
74          final StatusProcessor statusProcessor = new StatusProcessor(
75                  noopProcessor, mailboxManagerProvider, statusResponseFactory);
76          final LSubProcessor lsubProcessor = new LSubProcessor(statusProcessor,
77                  mailboxManagerProvider, statusResponseFactory);
78          final ListProcessor listProcessor = new ListProcessor(lsubProcessor,
79                  mailboxManagerProvider, statusResponseFactory);
80          final SearchProcessor searchProcessor = new SearchProcessor(
81                  listProcessor, statusResponseFactory);
82          final SelectProcessor selectProcessor = new SelectProcessor(
83                  searchProcessor, mailboxManagerProvider, statusResponseFactory);
84          final ImapProcessor result = new FetchProcessor(selectProcessor,
85                  statusResponseFactory);
86          return result;
87      }
