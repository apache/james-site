110      * @return <code>ImapCommandMessage</code>, not null

113         ImapMessage result;
114         try {
115 
116             ImapMessage message = decode(command, request, tag);
117             setupLogger(message);
118             result = message;

121             getLog().debug("Cannot parse protocol ", e);
122             result = messageFactory.taggedBad(tag, command,
123                     HumanReadableTextKey.ILLEGAL_ARGUMENTS);
124         }
125         return result;
126     }

221         } else {
222             return mailbox;

226     /**
227      * Reads one <code>date</code> argument from the request.
228      * 
229      * @param request
230      *            <code>ImapRequestLineReader</code>, not null
231      * @return <code>DayMonthYear</code>, not null
232      * @throws ProtocolException
233      */
234     public DayMonthYear date(ImapRequestLineReader request)
235             throws ProtocolException {
236 
237         final char one = request.consume();
238         final char two = request.consume();
239         final int day;
240         if (two == '-') {
241             day = DecoderUtils.decodeFixedDay(' ', one);
242         } else {
243             day = DecoderUtils.decodeFixedDay(one, two);
244             nextIsDash(request);
245         }
246 
247         final char monthFirstChar = request.consume();
248         final char monthSecondChar = request.consume();
249         final char monthThirdChar = request.consume();
250         final int month = DecoderUtils.decodeMonth(monthFirstChar,
251                 monthSecondChar, monthThirdChar) + 1;
252         nextIsDash(request);
253         final char milleniumChar = request.consume();
254         final char centuryChar = request.consume();
255         final char decadeChar = request.consume();
256         final char yearChar = request.consume();
257         final int year = DecoderUtils.decodeYear(milleniumChar, centuryChar,
258                 decadeChar, yearChar);
259         final DayMonthYear result = new DayMonthYear(day, month, year);
260         return result;
261     }

266         if (next != '-') {
267             throw new ProtocolException("Expected dash but was " + next);
268         }
269     }

287     /**
288      * Reads the next "word from the request, comprising all characters up to
289      * the next SPACE. Characters are tested by the supplied CharacterValidator,
290      * and an exception is thrown if invalid characters are encountered.
291      */
292     protected static String consumeWord(ImapRequestLineReader request,
293             CharacterValidator validator) throws ProtocolException {
294         StringBuffer atom = new StringBuffer();
295 
296         char next = request.nextWordChar();
297         while (!isWhitespace(next)) {
298             if (validator.isValid(next)) {
299                 atom.append(next);
300                 request.consume();
301             } else {
302                 throw new ProtocolException("Invalid character: '" + next + "'");
303             }
304             next = request.nextChar();
305         }
306         return atom.toString();
307     }
308 
309     private static boolean isWhitespace(char next) {
310         return (next == ' ' || next == '\n' || next == '\r' || next == '\t');
311     }

313     /**
314      * Reads an argument of type "literal" from the request, in the format: "{"
315      * charCount "}" CRLF *CHAR8 Note before calling, the request should be
316      * positioned so that nextChar is '{'. Leading whitespace is not skipped in
317      * this method.
318      * 
319      * @param charset ,
320      *            or null for <code>US-ASCII</code>
321      */
322     protected String consumeLiteral(final ImapRequestLineReader request,
323             final Charset charset) throws ProtocolException {
324         if (charset == null) {
325             return consumeLiteral(request, US_ASCII);
326         } else {
327             // The 1st character must be '{'
328             consumeChar(request, '{');
329 
330             StringBuffer digits = new StringBuffer();
331             char next = request.nextChar();
332             while (next != '}' && next != '+') {
333                 digits.append(next);
334                 request.consume();
335                 next = request.nextChar();
336             }
337 
338             // If the number is *not* suffixed with a '+', we *are* using a
339             // synchronized literal,
340             // and we need to send command continuation request before reading
341             // data.
342             boolean synchronizedLiteral = true;
343             // '+' indicates a non-synchronized literal (no command continuation
344             // request)
345             if (next == '+') {
346                 synchronizedLiteral = false;
347                 consumeChar(request, '+');
348             }
349 
350             // Consume the '}' and the newline
351             consumeChar(request, '}');
352             consumeCRLF(request);
353 
354             if (synchronizedLiteral) {
355                 request.commandContinuationRequest();
356             }
357 
358             final int size = Integer.parseInt(digits.toString());
359             final byte[] bytes = new byte[size];
360             request.read(bytes);
361             final ByteBuffer buffer = ByteBuffer.wrap(bytes);
362             return decode(charset, buffer);
363         }
364     }
365 
366     private String decode(final Charset charset, final ByteBuffer buffer)
367             throws ProtocolException {
368         try {
369 
370             final String result = charset.newDecoder().onMalformedInput(
371                     CodingErrorAction.REPORT).onUnmappableCharacter(
372                     CodingErrorAction.REPORT).decode(buffer).toString();
373             return result;
374 
375         } catch (IllegalStateException e) {
376             throw new ProtocolException("Bad character encoding", e);
377         } catch (MalformedInputException e) {
378             throw new ProtocolException("Bad character encoding", e);
379         } catch (UnmappableCharacterException e) {
380             throw new ProtocolException("Bad character encoding", e);
381         } catch (CharacterCodingException e) {
382             throw new ProtocolException("Bad character encoding", e);
383         }
384     }

423     /**
424      * Reads a quoted string value from the request.
425      */
426     protected String consumeQuoted(ImapRequestLineReader request,
427             Charset charset) throws ProtocolException {
428         if (charset == null) {
429             return consumeQuoted(request, US_ASCII);
430         } else {
431             // The 1st character must be '"'
432             consumeChar(request, '"');
433             final QuotedStringDecoder decoder = new QuotedStringDecoder(charset);
434             final String result = decoder.decode(request);
435             consumeChar(request, '"');
436             return result;
437         }
438     }

450     /**
451      * Reads a "flags" argument from the request.
452      */
453     public Flags flagList(ImapRequestLineReader request)
454             throws ProtocolException {
455         Flags flags = new Flags();
456         request.nextWordChar();
457         consumeChar(request, '(');
458         CharacterValidator validator = new NoopCharValidator();
459         String nextWord = consumeWord(request, validator);
460         while (!nextWord.endsWith(")")) {
461             DecoderUtils.setFlag(nextWord, flags);
462             nextWord = consumeWord(request, validator);
463         }
464         // Got the closing ")", may be attached to a word.
465         if (nextWord.length() > 1) {
466             int parenIndex = nextWord.indexOf(')');
467             if (parenIndex > 0) {
468                 final String nextFlag = nextWord.substring(0, parenIndex);
469                 DecoderUtils.setFlag(nextFlag, flags);
470             }
471         }
472 
473         return flags;
474     }

483     private long readDigits(final ImapRequestLineReader request, int add,
484             final long total, final boolean first) throws ProtocolException {
485         final char next;
486         if (first) {
487             next = request.nextWordChar();
488         } else {
489             request.consume();
490             next = request.nextChar();
491         }
492         final long currentTotal = (10 * total) + add;
493         switch (next) {
494             case '0':
495                 return readDigits(request, 0, currentTotal, false);
496             case '1':
497                 return readDigits(request, 1, currentTotal, false);
498             case '2':
499                 return readDigits(request, 2, currentTotal, false);
500             case '3':
501                 return readDigits(request, 3, currentTotal, false);
502             case '4':
503                 return readDigits(request, 4, currentTotal, false);
504             case '5':
505                 return readDigits(request, 5, currentTotal, false);
506             case '6':
507                 return readDigits(request, 6, currentTotal, false);
508             case '7':
509                 return readDigits(request, 7, currentTotal, false);
510             case '8':
511                 return readDigits(request, 8, currentTotal, false);
512             case '9':
513                 return readDigits(request, 9, currentTotal, false);
514             case '.':
515             case ' ':
516             case '>':
517             case '\r':
518             case '\n':
519             case '\t':
520                 return currentTotal;
521             default:
522                 throw new ProtocolException("Expected a digit but was " + next);
523         }
524     }

569     /**
570      * Reads a "message set" argument, and parses into an IdSet. Currently only
571      * supports a single range of values.
572      */
573     public IdRange[] parseIdRange(ImapRequestLineReader request)
574             throws ProtocolException {
575         CharacterValidator validator = new MessageSetCharValidator();
576         String nextWord = consumeWord(request, validator);
577 
578         int commaPos = nextWord.indexOf(',');
579         if (commaPos == -1) {
580             return new IdRange[] { parseRange(nextWord) };
581         }
582 
583         ArrayList rangeList = new ArrayList();
584         int pos = 0;
585         while (commaPos != -1) {
586             String range = nextWord.substring(pos, commaPos);
587             IdRange set = parseRange(range);
588             rangeList.add(set);
589 
590             pos = commaPos + 1;
591             commaPos = nextWord.indexOf(',', pos);
592         }
593         String range = nextWord.substring(pos);
594         rangeList.add(parseRange(range));
595         return (IdRange[]) rangeList.toArray(new IdRange[rangeList.size()]);
596     }
597 
598     private IdRange parseRange(String range) throws ProtocolException {
599         int pos = range.indexOf(':');
600         try {
601             if (pos == -1) {
602                 long value = parseLong(range);
603                 return new IdRange(value);
604             } else {
605                 long lowVal = parseLong(range.substring(0, pos));
606                 long highVal = parseLong(range.substring(pos + 1));
607                 return new IdRange(lowVal, highVal);
608             }
609         } catch (NumberFormatException e) {
610             throw new ProtocolException("Invalid message set.", e);
611         }
612     }
613 
614     private long parseLong(String value) {
615         if (value.length() == 1 && value.charAt(0) == '*') {
616             return Long.MAX_VALUE;
617         }
618         return Long.parseLong(value);
619     }

642     public static class ATOM_CHARValidator implements CharacterValidator {

645         }

649         }

670     /**
671      * Decodes contents of a quoted string. Charset aware. One shot, not thread
672      * safe.
673      */
674     private static class QuotedStringDecoder {
675         /** Decoder suitable for charset */
676         private final CharsetDecoder decoder;
677 
678         /** byte buffer will be filled then flushed to character buffer */
679         private final ByteBuffer buffer;
680 
681         /** character buffer may be dynamically resized */
682         CharBuffer charBuffer;
683 
684         public QuotedStringDecoder(Charset charset) {
685             decoder = charset.newDecoder();
686             buffer = ByteBuffer.allocate(QUOTED_BUFFER_INITIAL_CAPACITY);
687             charBuffer = CharBuffer.allocate(QUOTED_BUFFER_INITIAL_CAPACITY);
688         }
689 
690         public String decode(ImapRequestLineReader request)
691                 throws ProtocolException {
692             try {
693                 decoder.reset();
694                 char next = request.nextChar();
695                 while (next != '"') {
696                     // fill up byte buffer before decoding
697                     if (!buffer.hasRemaining()) {
698                         decodeByteBufferToCharacterBuffer(false);
699                     }
700                     if (next == '\\') {
701                         request.consume();
702                         next = request.nextChar();
703                         if (!isQuotedSpecial(next)) {
704                             throw new ProtocolException(
705                                     "Invalid escaped character in quote: '"
706                                             + next + "'");
707                         }
708                     }
709                     // TODO: nextChar does not report accurate chars so safe to
710                     // cast to byte
711                     buffer.put((byte) next);
712                     request.consume();
713                     next = request.nextChar();
714                 }
715                 completeDecoding();
716                 final String result = charBuffer.toString();
717                 return result;
718 
719             } catch (IllegalStateException e) {
720                 throw new ProtocolException("Bad character encoding", e);
721             }
722         }
723 
724         private void completeDecoding() throws ProtocolException {
725             decodeByteBufferToCharacterBuffer(true);
726             flush();
727             charBuffer.flip();
728         }
729 
730         private void flush() throws ProtocolException {
731             final CoderResult coderResult = decoder.flush(charBuffer);
732             if (coderResult.isOverflow()) {
733                 upsizeCharBuffer();
734                 flush();
735             } else if (coderResult.isError()) {
736                 throw new ProtocolException("Bad character encoding");
737             }
738         }

740         /**
741          * Decodes contents of the byte buffer to the character buffer. The
742          * character buffer will be replaced by a larger one if required.
743          * 
744          * @param endOfInput
745          *            is the input ended
746          */

753         private CoderResult decodeMoreBytesToCharacterBuffer(
754                 final boolean endOfInput) throws ProtocolException {
755             final CoderResult coderResult = decoder.decode(buffer, charBuffer,
756                     endOfInput);
757             if (coderResult.isOverflow()) {
758                 upsizeCharBuffer();
759                 return decodeMoreBytesToCharacterBuffer(endOfInput);
760             } else if (coderResult.isError()) {
761                 throw new ProtocolException("Bad character encoding");
762             } else if (coderResult.isUnderflow()) {
763                 buffer.clear();
764             }
765             return coderResult;
766         }

770          */
771         private void upsizeCharBuffer() {
772             final int oldCapacity = charBuffer.capacity();
773             CharBuffer oldBuffer = charBuffer;
774             charBuffer = CharBuffer.allocate(oldCapacity
775                     + QUOTED_BUFFER_INITIAL_CAPACITY);

777             charBuffer.put(oldBuffer);
