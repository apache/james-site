62      public int decode(final CharSequence sectionSpecification)
63              throws ProtocolException {
64          init();
65          sectionType = decode(0, sectionSpecification);
66          prunePath();
67          return sectionType;
68      }

70      private void prunePath() {
71          if (path != null) {
72              final int length = path.length;
73              if (used < length) {
74                  final int[] newPath = new int[used];
75                  System.arraycopy(path, 0, newPath, 0, used);
76                  path = newPath;
77              }
78          }
79      }

81      private int decode(final int at, final CharSequence sectionSpecification)
82              throws ProtocolException {
83          final int result;
84          final int length = sectionSpecification.length();
85          if (at < length) {
86              final char next = sectionSpecification.charAt(at);
87              switch (next) {
88                  case '.':
89                      separator();
90                      result = decode(at + 1, sectionSpecification);
91                      break;
92  
93                  case '0':
94                      result = digit(at, sectionSpecification, 0);
95                      break;
96  
97                  case '1':
98                      result = digit(at, sectionSpecification, 1);
99                      break;
100 
101                 case '2':
102                     result = digit(at, sectionSpecification, 2);
103                     break;
104 
105                 case '3':
106                     result = digit(at, sectionSpecification, 3);
107                     break;
108 
109                 case '4':
110                     result = digit(at, sectionSpecification, 4);
111                     break;
112 
113                 case '5':
114                     result = digit(at, sectionSpecification, 5);
115                     break;
116 
117                 case '6':
118                     result = digit(at, sectionSpecification, 6);
119                     break;
120 
121                 case '7':
122                     result = digit(at, sectionSpecification, 7);
123                     break;
124 
125                 case '8':
126                     result = digit(at, sectionSpecification, 8);
127                     break;
128 
129                 case '9':
130                     result = digit(at, sectionSpecification, 9);
131                     break;
132 
133                 case 't':
134                 case 'T':
135                     result = text(at, sectionSpecification);
136                     break;
137 
138                 case 'h':
139                 case 'H':
140                     result = header(at, sectionSpecification);
141                     break;
142 
143                 case 'm':
144                 case 'M':
145                     result = mime(at, sectionSpecification);
146                     break;
147 
148                 default:
149                     throw new ProtocolException("Did not expect '" + next
150                             + "' here in body specification.");
151             }
152         } else {
153             storePartial();
154             result = CONTENT;
155         }
156         return result;
157     }

159     private int mime(int at, CharSequence sectionSpecification)
160             throws ProtocolException {
161         if (sectionSpecification.length() == at + 4) {
162             mustBeI(sectionSpecification, at + 1);
163             mustBeM(sectionSpecification, at + 2);
164             mustBeE(sectionSpecification, at + 3);
165             storePartial();
166         } else {
167             throw new ProtocolException("Unknown body specification");
168         }
169         return MIME;
170     }

292     private int header(int at, CharSequence sectionSpecification)
293             throws ProtocolException {
294         final int result;
295         final int length = sectionSpecification.length();
296         if (length > at + 5) {
297             mustBeE(sectionSpecification, at + 1);
298             mustBeA(sectionSpecification, at + 2);
299             mustBeD(sectionSpecification, at + 3);
300             mustBeE(sectionSpecification, at + 4);
301             mustBeR(sectionSpecification, at + 5);
302             storePartial();
303             if (length == at + 6) {
304                 result = HEADER;
305             } else {
306                 result = headerFields(at + 6, sectionSpecification);
307             }
308         } else {
309             throw new ProtocolException("Unknown body specification");
310         }
311         return result;
312     }

314     private int headerFields(int at, CharSequence sectionSpecification)
315             throws ProtocolException {
316         final int result;
317         final int length = sectionSpecification.length();
318         if (length > at + 7) {
319             mustBeDot(sectionSpecification, at);
320             mustBeF(sectionSpecification, at + 1);
321             mustBeI(sectionSpecification, at + 2);
322             mustBeE(sectionSpecification, at + 3);
323             mustBeL(sectionSpecification, at + 4);
324             mustBeD(sectionSpecification, at + 5);
325             mustBeS(sectionSpecification, at + 6);
326             final char next = sectionSpecification.charAt(at + 7);
327             final int namesStartAt;
328             switch (next) {
329                 case ' ':
330                     result = HEADER_FIELDS;
331                     namesStartAt = skipSpaces(at + 7, sectionSpecification);
332                     break;
333                 case '.':
334                     if (length > at + 10) {
335                         mustBeN(sectionSpecification, at + 8);
336                         mustBeO(sectionSpecification, at + 9);
337                         mustBeT(sectionSpecification, at + 10);
338                         result = HEADER_NOT_FIELDS;
339                         namesStartAt = skipSpaces(at + 11, sectionSpecification);
340                     } else {
341                         throw new ProtocolException(
342                                 "Unknown body specification");
343                     }
344                     break;
345                 default:
346                     throw new ProtocolException("Unknown body specification");
347             }
348             mustBeOpenParen(sectionSpecification, namesStartAt);
349             readHeaderNames(namesStartAt + 1, sectionSpecification);
350 
351         } else {
352             throw new ProtocolException("Unknown body specification");
353         }
354         return result;
355     }

357     private void readHeaderNames(final int at,
358             final CharSequence sectionSpecification) throws ProtocolException {
359         names = new ArrayList();
360         final int firstWordStart = skipSpaces(at, sectionSpecification);
361         readHeaderNames(firstWordStart, firstWordStart, sectionSpecification);
362     }
363 
364     private void readHeaderNames(final int at, final int lastWordStart,
365             final CharSequence sectionSpecification) throws ProtocolException {
366         if (at < sectionSpecification.length()) {
367             final char next = sectionSpecification.charAt(at);
368             switch (next) {
369                 case ' ':
370                     readName(lastWordStart, at, sectionSpecification);
371                     final int nextWord = skipSpaces(at, sectionSpecification);
372                     readHeaderNames(nextWord, nextWord, sectionSpecification);
373                     break;
374                 case ')':
375                     readName(lastWordStart, at, sectionSpecification);
376                     break;
377                 default:
378                     readHeaderNames(at + 1, lastWordStart, sectionSpecification);
379             }
380         } else {
381             throw new ProtocolException("Closing parenthesis missing.");
382         }
383     }

385     private void readName(final int wordStart, final int wordFinish,
386             final CharSequence sectionSpecification) {
387         if (wordStart <= wordFinish) {
388             final CharSequence word = sectionSpecification.subSequence(
389                     wordStart, wordFinish);
390             final String name = word.toString();
391             names.add(name);
392         }
393     }

395     private int skipSpaces(final int at, final CharSequence sectionSpecification) {
396         final int result;
397         if (at < sectionSpecification.length()) {
398             final char next = sectionSpecification.charAt(at);
399             if (next == ' ') {
400                 result = skipSpaces(at + 1, sectionSpecification);
401             } else {
402                 result = at;
403             }
404         } else {
405             result = at;
406         }
407         return result;
408     }

410     private int text(int at, CharSequence sectionSpecification)
411             throws ProtocolException {
412         if (sectionSpecification.length() == at + 4) {
413             mustBeE(sectionSpecification, at + 1);
414             mustBeX(sectionSpecification, at + 2);
415             mustBeT(sectionSpecification, at + 3);
416             storePartial();
417         } else {
418             throw new ProtocolException("Unknown body specification");
419         }
420         return TEXT;
421     }

423     private int digit(final int at, final CharSequence sectionSpecification,
424             int digit) throws ProtocolException {

431     private void init() {
432         sectionType = CONTENT;
433         resetPartial();
434         path = null;
435         used = 0;
436         names = null;
437     }

439     private void resetPartial() {
440         partial = 0;
441     }
442 
443     private void separator() {
444         storePartial();
445     }
446 
447     private void storePartial() {
448         if (partial > 0) {
449             ensureSpaceForOneInPath();
450             path[used++] = partial;
451             resetPartial();
452         }
453     }
454 
455     private void ensureSpaceForOneInPath() {
456         if (path == null) {
457             path = new int[ARRAY_INITIAL_SIZE];
458         } else {
459             final int length = path.length;
460             if (used >= length) {
461                 int[] newPath = new int[length + ARRAY_INCREMENT];
462                 System.arraycopy(path, 0, newPath, 0, length);
463                 path = newPath;
464             }
465         }
466     }
467 
468     private void digit(int digit) {
469         partial = (partial * 10) + digit;
470     }
