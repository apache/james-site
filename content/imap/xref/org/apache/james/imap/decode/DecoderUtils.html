67      public static void setFlag(final String flagString, final Flags flags) {
68          if (flagString.equalsIgnoreCase(MessageFlags.ANSWERED_ALL_CAPS)) {
69              flags.add(Flags.Flag.ANSWERED);
70          } else if (flagString.equalsIgnoreCase(MessageFlags.DELETED_ALL_CAPS)) {
71              flags.add(Flags.Flag.DELETED);
72          } else if (flagString.equalsIgnoreCase(MessageFlags.DRAFT_ALL_CAPS)) {
73              flags.add(Flags.Flag.DRAFT);
74          } else if (flagString.equalsIgnoreCase(MessageFlags.FLAGGED_ALL_CAPS)) {
75              flags.add(Flags.Flag.FLAGGED);
76          } else if (flagString.equalsIgnoreCase(MessageFlags.SEEN_ALL_CAPS)) {
77              flags.add(Flags.Flag.SEEN);
78          } else {
79              if (flagString.equalsIgnoreCase(MessageFlags.RECENT_ALL_CAPS)) {
80                  // RFC3501 specifically excludes /Recent
81                  // The /Recent flag should be set automatically by the server
82              } else {
83                  // RFC3501 allows novel flags
84                  flags.add(flagString);
85              }
86          }
87      }

89      /**
90       * Decodes the given string as a standard IMAP date-time.
91       * 
92       * @param dateString
93       *            standard IMAP date-time
94       * @return <code>Date</code> with time component, not null
95       * @throws ProtocolException
96       *             when this conversion fails
97       */
98      public static final Date decodeDateTime(CharSequence chars)
99              throws ProtocolException {
100         if (isDateTime(chars)) {
101             final char dayHigh = chars.charAt(0);
102             final char dayLow = chars.charAt(1);
103             final int day = decodeFixedDay(dayHigh, dayLow);
104 
105             final char monthFirstChar = chars.charAt(3);
106             final char monthSecondChar = chars.charAt(4);
107             final char monthThirdChar = chars.charAt(5);
108             final int month = decodeMonth(monthFirstChar, monthSecondChar,
109                     monthThirdChar);
110 
111             final char milleniumChar = chars.charAt(7);
112             final char centuryChar = chars.charAt(8);
113             final char decadeChar = chars.charAt(9);
114             final char yearChar = chars.charAt(10);
115             final int year = decodeYear(milleniumChar, centuryChar, decadeChar,
116                     yearChar);
117 
118             final char zoneDeterminent = chars.charAt(21);
119             final char zoneDigitOne = chars.charAt(22);
120             final char zoneDigitTwo = chars.charAt(23);
121             final char zoneDigitThree = chars.charAt(24);
122             final char zoneDigitFour = chars.charAt(25);
123             final int offset = decodeZone(zoneDeterminent, zoneDigitOne,
124                     zoneDigitTwo, zoneDigitThree, zoneDigitFour);
125 
126             final char hourHigh = chars.charAt(12);
127             final char hourLow = chars.charAt(13);
128             final int hour = applyHourOffset(offset, decodeNumber(hourHigh,
129                     hourLow));
130 
131             final char minuteHigh = chars.charAt(15);
132             final char minuteLow = chars.charAt(16);
133             final int minute = applyMinuteOffset(offset, decodeNumber(
134                     minuteHigh, minuteLow));
135 
136             final char secondHigh = chars.charAt(18);
137             final char secondLow = chars.charAt(19);
138             final int second = decodeNumber(secondHigh, secondLow);
139 
140             final GregorianCalendar calendar = new GregorianCalendar(TimeZone
141                     .getTimeZone("GMT"), Locale.US);
142             calendar.clear();
143             calendar.set(year, month, day, hour, minute, second);
144             final Date result = calendar.getTime();
145             return result;
146         } else {
147             final String message;
148             if (chars == null) {
149                 message = "Expected a date-time but was nothing.";
150             } else {
151                 message = new StringBuffer("Expected a date-time but was ")
152                         .append(chars.toString()).toString();
153             }
154 
155             throw new ProtocolException(message);
156         }
157 
158     }

160     private static boolean isDateTime(CharSequence chars) {
161         final boolean result;
162         if (chars == null) {
163             result = false;
164         } else if (chars.length() < 20) {
165             // Be liberal in what you accept
166             result = false;
167         } else {
168             result = true;
169         }
170         return result;
171     }

173     private static int applyMinuteOffset(final int offset, final int minutes) {
174         final int result = minutes
175                 - ((Math.abs(offset) % 100) * (offset == 0 ? 0 : offset > 0 ? 1
176                         : -1));
177         return result;
178     }

180     private static int applyHourOffset(final int offset, final int hours) {
181         final int result = hours - (offset / 100);
182         return result;
183     }

185     public static int decodeNumber(final char high, final char low)
186             throws ProtocolException {
187         return (10 * decodeDigit(high)) + decodeDigit(low);
188     }

190     public static int decodeZone(char zoneDeterminent, char zoneDigitOne,
191             char zoneDigitTwo, char zoneDigitThree, char zoneDigitFour)
192             throws ProtocolException {
193         if (isInvalidZone(zoneDeterminent, zoneDigitOne, zoneDigitTwo,
194                 zoneDigitThree, zoneDigitFour)) {
195             throw createTimeZoneException(zoneDeterminent, zoneDigitOne,
196                     zoneDigitTwo, zoneDigitThree, zoneDigitFour);
197         }
198         final int sign;
199         if (zoneDeterminent == '+') {
200             sign = 1;
201         } else if (zoneDeterminent == '-') {
202             sign = -1;
203         } else {
204             throw createTimeZoneException(zoneDeterminent, zoneDigitOne,
205                     zoneDigitTwo, zoneDigitThree, zoneDigitFour);
206         }
207         final int result = sign
208                 * ((1000 * decodeDigit(zoneDigitOne))
209                         + (100 * decodeDigit(zoneDigitTwo))
210                         + (10 * decodeDigit(zoneDigitThree)) + decodeDigit(zoneDigitFour));
211         return result;
212     }

214     private static ProtocolException createTimeZoneException(
215             char zoneDeterminent, char zoneDigitOne, char zoneDigitTwo,
216             char zoneDigitThree, char zoneDigitFour) {
217         return new ProtocolException("Expected time-zone but was "
218                 + zoneDeterminent + zoneDigitOne + zoneDigitTwo
219                 + zoneDigitThree + zoneDigitFour);
220     }

222     private static boolean isInvalidZone(char zoneDeterminent,
223             char zoneDigitOne, char zoneDigitTwo, char zoneDigitThree,
224             char zoneDigitFour) {
225         final boolean result;
226         if (zoneDeterminent == '+' || zoneDeterminent == '-') {
227             result = !(isSimpleDigit(zoneDigitOne)
228                     && isSimpleDigit(zoneDigitTwo)
229                     && isSimpleDigit(zoneDigitThree) && isSimpleDigit(zoneDigitFour));
230         } else {
231             result = true;
232         }
233         return result;
234     }

249      * Decodes a year.
250      * 
251      * @param milleniumChar
252      *            first digit
253      * @param centuryChar
254      *            second digit
255      * @param decadeChar
256      *            third digit
257      * @param yearChar
258      *            forth digit
259      * @return {@link Calendar} year
260      * @throws ProtocolException
261      */
262     public static int decodeYear(final char milleniumChar,
263             final char centuryChar, final char decadeChar, final char yearChar)
264             throws ProtocolException {
265         return (decodeDigit(milleniumChar) * 1000)
266                 + (decodeDigit(centuryChar) * 100)
267                 + (decodeDigit(decadeChar) * 10) + decodeDigit(yearChar);
268     }

270     /**
271      * Decodes asn IMAP <code>date-month</code> to a {@link Calendar} month.
272      * 
273      * @param monthFirstChar
274      *            first character in a month triple
275      * @param monthSecondChar
276      *            second character in a month triple
277      * @param monthThirdChar
278      *            third character in a month triple
279      * @return {@link Calendar} month (<code>JAN</code>=0)
280      * @throws ProtocolException
281      */
282     public static int decodeMonth(final char monthFirstChar,
283             final char monthSecondChar, final char monthThirdChar)
284             throws ProtocolException {
285         final int result;
286         // Bitwise magic! Eliminate possibility by three switches
287         int possibleMonths = ALL_MONTH_BITS;
288         switch (monthFirstChar) {
289             case 'J':
290             case 'j':
291                 possibleMonths &= (JAN_BIT | JUN_BIT | JUL_BIT);
292                 break;
293             case 'F':
294             case 'f':
295                 possibleMonths &= FEB_BIT;
296                 break;
297             case 'M':
298             case 'm':
299                 possibleMonths &= (MAR_BIT | MAY_BIT);
300                 break;
301             case 'A':
302             case 'a':
303                 possibleMonths &= (APR_BIT | AUG_BIT);
304                 break;
305             case 'S':
306             case 's':
307                 possibleMonths &= SEP_BIT;
308                 break;
309             case 'O':
310             case 'o':
311                 possibleMonths &= OCT_BIT;
312                 break;
313             case 'N':
314             case 'n':
315                 possibleMonths &= NOV_BIT;
316                 break;
317             case 'D':
318             case 'd':
319                 possibleMonths &= DEC_BIT;
320                 break;
321             default:
322                 possibleMonths = 0;
323                 break;
324         }
325         switch (monthSecondChar) {
326             case 'A':
327             case 'a':
328                 possibleMonths &= (JAN_BIT | MAR_BIT | MAY_BIT);
329                 break;
330             case 'E':
331             case 'e':
332                 possibleMonths &= (FEB_BIT | SEP_BIT | DEC_BIT);
333                 break;
334             case 'P':
335             case 'p':
336                 possibleMonths &= (APR_BIT);
337                 break;
338             case 'U':
339             case 'u':
340                 possibleMonths &= (JUN_BIT | JUL_BIT | AUG_BIT);
341                 break;
342             case 'C':
343             case 'c':
344                 possibleMonths &= OCT_BIT;
345                 break;
346             case 'O':
347             case 'o':
348                 possibleMonths &= NOV_BIT;
349                 break;
350             default:
351                 possibleMonths = 0;
352                 break;
353         }
354         switch (monthThirdChar) {
355             case 'N':
356             case 'n':
357                 possibleMonths &= (JAN_BIT | JUN_BIT);
358                 break;
359             case 'B':
360             case 'b':
361                 possibleMonths &= FEB_BIT;
362                 break;
363             case 'R':
364             case 'r':
365                 possibleMonths &= (MAR_BIT | APR_BIT);
366                 break;
367             case 'Y':
368             case 'y':
369                 possibleMonths &= MAY_BIT;
370                 break;
371             case 'L':
372             case 'l':
373                 possibleMonths &= JUL_BIT;
374                 break;
375             case 'G':
376             case 'g':
377                 possibleMonths &= AUG_BIT;
378                 break;
379             case 'P':
380             case 'p':
381                 possibleMonths &= SEP_BIT;
382                 break;
383             case 'T':
384             case 't':
385                 possibleMonths &= OCT_BIT;
386                 break;
387             case 'V':
388             case 'v':
389                 possibleMonths &= NOV_BIT;
390                 break;
391             case 'C':
392             case 'c':
393                 possibleMonths &= DEC_BIT;
394                 break;
395             default:
396                 possibleMonths = 0;
397                 break;
398         }
399         switch (possibleMonths) {
400             case JAN_BIT:
401                 result = Calendar.JANUARY;
402                 break;
403             case FEB_BIT:
404                 result = Calendar.FEBRUARY;
405                 break;
406             case MAR_BIT:
407                 result = Calendar.MARCH;
408                 break;
409             case APR_BIT:
410                 result = Calendar.APRIL;
411                 break;
412             case MAY_BIT:
413                 result = Calendar.MAY;
414                 break;
415             case JUN_BIT:
416                 result = Calendar.JUNE;
417                 break;
418             case JUL_BIT:
419                 result = Calendar.JULY;
420                 break;
421             case AUG_BIT:
422                 result = Calendar.AUGUST;
423                 break;
424             case SEP_BIT:
425                 result = Calendar.SEPTEMBER;
426                 break;
427             case OCT_BIT:
428                 result = Calendar.OCTOBER;
429                 break;
430             case NOV_BIT:
431                 result = Calendar.NOVEMBER;
432                 break;
433             case DEC_BIT:
434                 result = Calendar.DECEMBER;
435                 break;
436             default:
437                 throw new ProtocolException("Expected month name but was "
438                         + monthFirstChar + monthSecondChar + monthThirdChar);
439         }
440         return result;
441     }

443     public static int decodeFixedDay(final char dayHigh, final char dayLow)
444             throws ProtocolException {
445         int result = decodeDigit(dayLow);
446         switch (dayHigh) {
447             case '0':
448                 return result;
449             case '1':
450                 return result += 10;
451             case '2':
452                 return result += 20;
453             case '3':
454                 return result += 30;
455             case ' ':
456                 return result;
457         }
458         throw new ProtocolException("Expected SP, 0, 1, 2, or 3 but was "
459                 + dayHigh);
460     }

462     /**
463      * Decodes a number character into a <code>0-9</code> digit.
464      * 
465      * @param character
466      * @return a digit
467      * @throws ProtocolException
468      *             if the char is not a digit
469      */
470     public static final int decodeDigit(char character)
471             throws ProtocolException {
472         final int result = character - ASCII_ZERO;
473         if (result < 0 || result > 9) {
474             throw new ProtocolException("Expected a digit but was '"
475                     + character + "'");
476         }
477         return result;
478     }
