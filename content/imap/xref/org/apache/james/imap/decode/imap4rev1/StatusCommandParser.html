21  import org.apache.james.api.imap.ImapCommand;
22  import org.apache.james.api.imap.ImapConstants;

24  import org.apache.james.api.imap.imap4rev1.Imap4Rev1CommandFactory;

27  import org.apache.james.imap.decode.ImapRequestLineReader;
28  import org.apache.james.imap.decode.InitialisableCommandFactory;
29  import org.apache.james.imap.decode.ProtocolException;
30  import org.apache.james.imap.decode.base.AbstractImapCommandParser;

32  class StatusCommandParser extends AbstractImapCommandParser implements
33          InitialisableCommandFactory {
34      public StatusCommandParser() {
35      }
36  
37      /**
38       * @see org.apache.james.imap.decode.InitialisableCommandFactory#init(org.apache.james.api.imap.imap4rev1.Imap4Rev1CommandFactory)
39       */
40      public void init(Imap4Rev1CommandFactory factory) {
41          final ImapCommand command = factory.getStatus();
42          setCommand(command);
43      }
44  
45      StatusDataItems statusDataItems(ImapRequestLineReader request)
46              throws ProtocolException {
47          StatusDataItems items = new StatusDataItems();
48  
49          request.nextWordChar();
50          consumeChar(request, '(');
51          CharacterValidator validator = new NoopCharValidator();
52          String nextWord = consumeWord(request, validator);
53          while (!nextWord.endsWith(")")) {
54              addItem(nextWord, items);
55              nextWord = consumeWord(request, validator);
56          }
57          // Got the closing ")", may be attached to a word.
58          if (nextWord.length() > 1) {
59              addItem(nextWord.substring(0, nextWord.length() - 1), items);
60          }
61  
62          return items;
63      }
64  
65      private void addItem(String nextWord, StatusDataItems items)
66              throws ProtocolException {
67          if (nextWord.equals(ImapConstants.STATUS_MESSAGES)) {
68              items.setMessages(true);
69          } else if (nextWord.equals(ImapConstants.STATUS_RECENT)) {
70              items.setRecent(true);
71          } else if (nextWord.equals(ImapConstants.STATUS_UIDNEXT)) {
72              items.setUidNext(true);
73          } else if (nextWord.equals(ImapConstants.STATUS_UIDVALIDITY)) {
74              items.setUidValidity(true);
75          } else if (nextWord.equals(ImapConstants.STATUS_UNSEEN)) {
76              items.setUnseen(true);
77          } else {
78              throw new ProtocolException("Unknown status item: '" + nextWord
79                      + "'");
80          }
81      }
82  
83      protected ImapMessage decode(ImapCommand command,
84              ImapRequestLineReader request, String tag) throws ProtocolException {
85          final String mailboxName = mailbox(request);
86          final StatusDataItems statusDataItems = statusDataItems(request);
87          endLine(request);
88          final Imap4Rev1MessageFactory factory = getMessageFactory();
89          final ImapMessage result = factory.createStatusMessage(command,
90                  mailboxName, statusDataItems, tag);
91          return result;
92      }
93  }
