20  package org.apache.james.test.functional;
21  
22  import java.io.BufferedReader;
23  import java.io.InputStream;
24  import java.io.InputStreamReader;
25  import java.util.ArrayList;
26  import java.util.List;
27  
28  
29  /**
30   * A builder which generates a ProtocolSession from a test file.
31   * 
32   * @author Darrell DeBoer <darrell@apache.org>
33   * 
34   * @version $Revision: 709874 $
35   */
36  public class FileProtocolSessionBuilder {
37      private static final String SERVER_CONTINUATION_TAG = "S: \\+";
38  
39      private static final String CLIENT_TAG = "C:";
40  
41      private static final String SERVER_TAG = "S:";
42  
43      private static final String OPEN_UNORDERED_BLOCK_TAG = "SUB {";
44  
45      private static final String CLOSE_UNORDERED_BLOCK_TAG = "}";
46  
47      private static final String COMMENT_TAG = "#";
48  
49      private static final String SESSION_TAG = "SESSION:";
50  
51      /**
52       * Builds a ProtocolSession by reading lines from the test file with the
53       * supplied name.
54       * 
55       * @param fileName
56       *            The name of the protocol session file.
57       * @return The ProtocolSession
58       */
59      public ProtocolSession buildProtocolSession(String fileName)
60              throws Exception {
61          ProtocolSession session = new ProtocolSession();
62          addTestFile(fileName, session);
63          return session;
64      }
65  
66      /**
67       * Adds all protocol elements from a test file to the ProtocolSession
68       * supplied.
69       * 
70       * @param fileName
71       *            The name of the protocol session file.
72       * @param session
73       *            The ProtocolSession to add the elements to.
74       */
75      public void addTestFile(String fileName, ProtocolSession session)
76              throws Exception {
77          // Need to find local resource.
78          InputStream is = this.getClass().getResourceAsStream(fileName);
79          if (is == null) {
80              throw new Exception("Test Resource '" + fileName + "' not found.");
81          }
82  
83          addProtocolLinesFromStream(is, session, fileName);
84      }
85  
86      /**
87       * Reads ProtocolElements from the supplied InputStream and adds them to the
88       * ProtocolSession.
89       * 
90       * @param is
91       *            The input stream containing the protocol definition.
92       * @param session
93       *            The ProtocolSession to add elements to.
94       * @param fileName
95       *            The name of the source file, for error messages.
96       */
97      public void addProtocolLinesFromStream(InputStream is,
98              ProtocolSession session, String fileName) throws Exception {
99          int sessionNumber = -1;
100         BufferedReader reader = new BufferedReader(new InputStreamReader(is));
101         String next;
102         int lineNumber = -1;
103         String lastClientMsg = "";
104         while ((next = reader.readLine()) != null) {
105             String location = fileName + ":" + lineNumber;
106             if (SERVER_CONTINUATION_TAG.equals(next)) {
107                 session.CONT(sessionNumber);
108             } else if (next.startsWith(CLIENT_TAG)) {
109                 String clientMsg = "";
110                 if (next.length() > 3) {
111                     clientMsg = next.substring(3);
112                 }
113                 session.CL(sessionNumber, clientMsg);
114                 lastClientMsg = clientMsg;
115             } else if (next.startsWith(SERVER_TAG)) {
116                 String serverMsg = "";
117                 if (next.length() > 3) {
118                     serverMsg = next.substring(3);
119                 }
120                 session.SL(sessionNumber, serverMsg, location, lastClientMsg);
121             } else if (next.startsWith(OPEN_UNORDERED_BLOCK_TAG)) {
122                 List unorderedLines = new ArrayList(5);
123                 next = reader.readLine();
124 
125                 while (!next.startsWith(CLOSE_UNORDERED_BLOCK_TAG)) {
126                     if (!next.startsWith(SERVER_TAG)) {
127                         throw new Exception(
128                                 "Only 'S: ' lines are permitted inside a 'SUB {' block.");
129                     }
130                     String serverMsg = next.substring(3);
131                     unorderedLines.add(serverMsg);
132                     next = reader.readLine();
133                     lineNumber++;
134                 }
135 
136                 session.SUB(sessionNumber, unorderedLines, location,
137                         lastClientMsg);
138             } else if (next.startsWith(COMMENT_TAG)
139                     || next.trim().length() == 0) {
140                 // ignore these lines.
141             } else if (next.startsWith(SESSION_TAG)) {
142                 String number = next.substring(SESSION_TAG.length()).trim();
143                 if (number.length() == 0) {
144                     throw new Exception("No session number specified");
145                 }
146                 sessionNumber = Integer.parseInt(number);
147             } else {
148                 String prefix = next;
149                 if (next.length() > 3) {
150                     prefix = next.substring(0, 3);
151                 }
152                 throw new Exception("Invalid line prefix: " + prefix);
153             }
154             lineNumber++;
155         }
156     }
