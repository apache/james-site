32  /**
33   * <p>
34   * Represents an <code>RFC2060</code> status response. The five specified
35   * status server responses (<code>OK<code>.
36   * <code>NO</code>, <code>BAD</code>, <code>PREAUTH</code>
37   * and <code>BYE</code>) are modeled by this single interface.
38   * They are differentiated by {@link #getServerResponseType()}
39   * </p>
40   */

53       * @return if tagged response, the tag. Otherwise null.
54       */
55      public String getTag();

262         /**
263          * Creates an extension response code. Names that do not begin with 'X'

265          * 
266          * @param name
267          *            extension code, not null
268          * @return <code>ResponseCode</code>, not null
269          */
270         public static ResponseCode createExtension(String name) {
271             StringBuffer buffer = new StringBuffer(name.length() + 2);
272             if (!name.startsWith("X")) {
273                 buffer.append('X');
274             }
275             buffer.append(name);
276             final ResponseCode result = new ResponseCode(buffer.toString());
277             return result;
278         }

329         public int hashCode() {
