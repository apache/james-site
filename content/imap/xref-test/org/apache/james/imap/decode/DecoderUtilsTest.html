243     private void dateDecode(String in, TimeZone zone) throws Exception {
244         Date date = DecoderUtils.decodeDateTime(in);
245         String out = formatAsImap(date, zone);
246         assertEquals("Round trip", in, out);
247     }
248 
249     private String formatAsImap(Date date, TimeZone zone) {
250         assertNotNull(date);
251         FastDateFormat format = FastDateFormat.getInstance(
252                 "dd-MMM-yyyy hh:mm:ss Z", zone, Locale.US);
253         String out = format.format(date);
254         if (out.charAt(0) == '0') {
255             out = ' ' + out.substring(1, out.length());
256         }
257         return out;
258     }
