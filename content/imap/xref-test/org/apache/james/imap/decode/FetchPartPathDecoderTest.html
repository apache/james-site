170     private void checkReadNames(String base, String[] names) throws Exception {
171         base = base + " (";
172         for (int i = 0; i < names.length; i++) {
173             base = base + ' ' + names[i];
174         }
175         base = base + ')';
176         decoder.decode(base);
177         Collection results = decoder.getNames();
178         assertNotNull(results);
179         Iterator it = results.iterator();
180         for (int i = 0; i < names.length; i++) {
181             assertEquals(names[i], it.next());
182         }
183     }
184 
185     private void checkEndingPermutations(int[] values) throws Exception {
186         String base = "";
187         boolean first = true;
188         for (int i = 0; i < values.length; i++) {
189             if (first) {
190                 first = false;
191             } else {
192                 base = base + ".";
193             }
194             base = base + values[i];
195         }
196         checkPath(values, base);
197         checkPath(values, base + ".TEXT");
198         checkPath(values, base + ".HEADER");
199         checkPath(values, base + ".MIME");
200         checkPath(values, base + ".HEADER.FIELDS.NOT ()");
201         checkPath(values, base + ".HEADER.FIELDS ()");
202     }
203 
204     private void checkPath(int[] expected, String encoded) throws Exception {
205         decoder.decode(encoded);
206         final int[] path = decoder.getPath();
207         assertNotNull(path);
208         assertEquals(expected.length, path.length);
209         for (int i = 0; i < expected.length; i++) {
210             assertEquals(expected[i], path[i]);
211         }
212     }
