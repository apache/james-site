376     private void checkValid(String input, final SearchKey key) throws Exception {
377         ImapRequestLineReader reader = new ImapRequestLineReader(
378                 new ByteArrayInputStream(input.getBytes("US-ASCII")),
379                 new ByteArrayOutputStream());
380 
381         assertEquals(key, parser.searchKey(reader, null, false));
382     }
