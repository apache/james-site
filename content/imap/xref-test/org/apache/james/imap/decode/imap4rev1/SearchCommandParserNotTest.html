72      public void testShouldParseNotSequence() throws Exception {
73          IdRange[] range = { new IdRange(Long.MAX_VALUE, 100), new IdRange(110),
74                  new IdRange(200, 201), new IdRange(400, Long.MAX_VALUE) };
75          SearchKey notdKey = SearchKey.buildSequenceSet(range);
76          SearchKey key = SearchKey.buildNot(notdKey);
77          checkValid("NOT *:100,110,200:201,400:*\r\n", key);
78      }
79  
80      public void testShouldParseNotUid() throws Exception {
81          IdRange[] range = { new IdRange(Long.MAX_VALUE, 100), new IdRange(110),
82                  new IdRange(200, 201), new IdRange(400, Long.MAX_VALUE) };
83          SearchKey notdKey = SearchKey.buildUidSet(range);
84          SearchKey key = SearchKey.buildNot(notdKey);
85          checkValid("NOT UID *:100,110,200:201,400:*\r\n", key);
86      }

88      public void testShouldParseNotHeaderKey() throws Exception {
89          SearchKey notdKey = SearchKey.buildHeader("FROM", "Smith");
90          SearchKey key = SearchKey.buildNot(notdKey);
91          checkValid("NOT HEADER FROM Smith\r\n", key);
92          checkValid("NOT header FROM Smith\r\n", key);
93      }
94  
95      public void testShouldParseNotDateParameterKey() throws Exception {
96          SearchKey notdKey = SearchKey.buildSince(new DayMonthYear(11, 1, 2001));
97          SearchKey key = SearchKey.buildNot(notdKey);
98          checkValid("NOT since 11-Jan-2001\r\n", key);
99          checkValid("NOT SINCE 11-Jan-2001\r\n", key);
100     }

102     public void testShouldParseNotStringParameterKey() throws Exception {
103         SearchKey notdKey = SearchKey.buildFrom("Smith");
104         SearchKey key = SearchKey.buildNot(notdKey);
105         checkValid("NOT FROM Smith\r\n", key);
106         checkValid("NOT FROM \"Smith\"\r\n", key);
107     }

109     public void testShouldParseNotStringQuotedParameterKey() throws Exception {

115     public void testShouldParseNotNoParameterKey() throws Exception {
116         SearchKey notdKey = SearchKey.buildNew();
117         SearchKey key = SearchKey.buildNot(notdKey);
118         checkValid("NOT NEW\r\n", key);
119         checkValid("Not NEW\r\n", key);
120         checkValid("not new\r\n", key);
121     }

123     private void checkValid(String input, final SearchKey key) throws Exception {
124         ImapRequestLineReader reader = new ImapRequestLineReader(
125                 new ByteArrayInputStream(input.getBytes("US-ASCII")),
126                 new ByteArrayOutputStream());
127 
128         assertEquals(key, parser.searchKey(reader, null, false));
129     }
