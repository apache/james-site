386     private void check(SearchKey key, SearchQuery.Criterion criterion)
387             throws Exception {
388         SearchQuery query = new SearchQuery();
389         query.andCriteria(criterion);
390         check(key, query);
391     }
392 
393     private void check(SearchKey key, SearchQuery query) throws Exception {
394         MailboxSession mailboxSession = (MailboxSession) mock(
395                 MailboxSession.class).proxy();
396         session.expects(once()).method("getAttribute").with(
397                 eq(ImapSessionUtils.MAILBOX_SESSION_ATTRIBUTE_SESSION_KEY))
398                 .will(returnValue((MailboxSession) mailboxSession));
399         session.expects(once()).method("getAttribute").with(
400                 eq(ImapSessionUtils.SELECTED_MAILBOX_ATTRIBUTE_SESSION_KEY))
401                 .will(returnValue((Mailbox) mailbox.proxy()));
402         session.expects(once()).method("unsolicitedResponses").with(eq(true),
403                 eq(false)).will(returnValue(new ArrayList()));
404         mailbox.expects(once()).method("search").with(eq(query),
405                 eq(FetchGroupImpl.MINIMAL), eq(mailboxSession)).will(
406                 returnValue(new ArrayList().iterator()));
407         responder.expects(once()).method("respond").with(
408                 eq(new SearchResponse(EMPTY)));
409         SearchRequest message = new SearchRequest(imapCommand, key, false, TAG);
410         processor.doProcess(message, (ImapSession) session.proxy(), TAG,
411                 (ImapCommand) command.proxy(), (Responder) responder.proxy());
412     }

414     private void expectOk() {
415         StatusResponse response = (StatusResponse) statusResponse.proxy();
416         serverResponseFactory.expects(once()).method("taggedOk").with(eq(TAG),
417                 same(imapCommand), eq(HumanReadableTextKey.COMPLETED)).will(
418                 returnValue(response));
419         responder.expects(once()).method("respond").with(same(response));
420     }
