130     private Address mockAddress(final String name, final String domainList,
131             final String mailbox, final String host) {
132         Mock address = mock(Address.class);
133         address.expects(once()).method("getPersonalName").will(
134                 returnValue(name));
135         address.expects(once()).method("getAtDomainList").will(
136                 returnValue(domainList));
137         address.expects(once()).method("getMailboxName").will(
138                 returnValue(mailbox));
139         address.expects(once()).method("getHostName").will(returnValue(host));
140         return (Address) address.proxy();
141     }
