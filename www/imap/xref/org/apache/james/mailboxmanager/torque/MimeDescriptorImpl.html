39      public static MimeDescriptorImpl build(final InputStream stream)
40              throws IOException {
41          final MimeTokenStream parser = MimeTokenStream
42                  .createMaximalDescriptorStream();
43          parser.parse(stream);
44          parser.setRecursionMode(RecursionMode.M_NO_RECURSE);
45          return createDescriptor(parser);
46      }

48      private static MimeDescriptorImpl createDescriptor(
49              final MimeTokenStream parser) throws IOException, MimeException {
50          int next = parser.next();
51          final Collection headers = new ArrayList();
52          while (next != MimeTokenStream.T_BODY
53                  && next != MimeTokenStream.T_END_OF_STREAM
54                  && next != MimeTokenStream.T_START_MULTIPART) {
55              if (next == MimeTokenStream.T_FIELD) {
56                  headers.add(new Header(parser.getFieldName(), parser
57                          .getFieldValue().trim()));
58              }
59              next = parser.next();
60          }
61  
62          final MimeDescriptorImpl mimeDescriptorImpl;
63          switch (next) {
64              case MimeTokenStream.T_BODY:
65                  mimeDescriptorImpl = simplePartDescriptor(parser, headers);
66                  break;
67              case MimeTokenStream.T_START_MULTIPART:
68                  mimeDescriptorImpl = compositePartDescriptor(parser, headers);
69                  break;
70              case MimeTokenStream.T_END_OF_STREAM:
71                  throw new MimeException("Premature end of stream");
72              default:
73                  throw new MimeException("Unexpected parse state");
74          }
75          return mimeDescriptorImpl;
76      }

78      private static MimeDescriptorImpl compositePartDescriptor(
79              final MimeTokenStream parser, final Collection headers)
80              throws IOException {
81          MaximalBodyDescriptor descriptor = (MaximalBodyDescriptor) parser
82                  .getBodyDescriptor();
83          MimeDescriptorImpl mimeDescriptor = createDescriptor(0, 0, descriptor,
84                  null, headers);
85          int next = parser.next();
86          while (next != MimeTokenStream.T_END_MULTIPART
87                  && next != MimeTokenStream.T_END_OF_STREAM) {
88              if (next == MimeTokenStream.T_START_BODYPART) {
89                  mimeDescriptor.addPart(createDescriptor(parser));
90              }
91              next = parser.next();
92          }
93          return mimeDescriptor;
94      }

96      private static MimeDescriptorImpl simplePartDescriptor(
97              final MimeTokenStream parser, final Collection headers)
98              throws IOException {
99          MaximalBodyDescriptor descriptor = (MaximalBodyDescriptor) parser
100                 .getBodyDescriptor();
101         final MimeDescriptorImpl mimeDescriptorImpl;
102         if ("message".equalsIgnoreCase(descriptor.getMediaType())
103                 && "rfc822".equalsIgnoreCase(descriptor.getSubType())) {
104             final CountingInputStream messageStream = new CountingInputStream(
105                     parser.getDecodedInputStream());
106             MimeDescriptorImpl embeddedMessageDescriptor = build(messageStream);
107             final int octetCount = messageStream.getOctetCount();
108             final int lineCount = messageStream.getLineCount();
109 
110             mimeDescriptorImpl = createDescriptor(octetCount, lineCount,
111                     descriptor, embeddedMessageDescriptor, headers);
112         } else {
113             final InputStream body = parser.getInputStream();
114             long bodyOctets = 0;
115             long lines = 0;
116             for (int n = body.read(); n >= 0; n = body.read()) {
117                 if (n == '\r') {
118                     lines++;
119                 }
120                 bodyOctets++;
121             }
122 
123             mimeDescriptorImpl = createDescriptor(bodyOctets, lines,
124                     descriptor, null, headers);
125         }
126         return mimeDescriptorImpl;
127     }

129     private static MimeDescriptorImpl createDescriptor(long bodyOctets,
130             long lines, MaximalBodyDescriptor descriptor,
131             MimeDescriptor embeddedMessage, final Collection headers) {
132         final String contentDescription = descriptor.getContentDescription();
133         final String contentId = descriptor.getContentId();
134 
135         final String subType = descriptor.getSubType();
136         final String type = descriptor.getMediaType();
137         final String transferEncoding = descriptor.getTransferEncoding();
138         final Collection contentTypeParameters = new ArrayList();
139         final Map valuesByName = descriptor.getContentTypeParameters();
140         for (final Iterator it = valuesByName.keySet().iterator(); it.hasNext();) {
141             final String name = (String) it.next();
142             final String value = (String) valuesByName.get(name);
143             contentTypeParameters.add(new Header(name, value));
144         }
145         final String codeset = descriptor.getCharset();
146         Header header;
147         if (codeset == null) {
148             if ("TEXT".equals(type)) {
149                 header = new Header("charset", "us-ascii");
150             } else {
151                 header = null;
152             }
153         } else {
154             header = new Header("charset", codeset);
155         }
156         if (header != null) {
157             contentTypeParameters.add(header);
158         }
159         final String boundary = descriptor.getBoundary();
160         if (boundary != null) {
161             contentTypeParameters.add(new Header("boundary", boundary));
162         }
163         final List languages = descriptor.getContentLanguage();
164         final String disposition = descriptor.getContentDispositionType();
165         final Map dispositionParams = descriptor
166                 .getContentDispositionParameters();
167         final Collection parts = new ArrayList();
168         final String location = descriptor.getContentLocation();
169         final String md5 = descriptor.getContentMD5Raw();
170         final MimeDescriptorImpl mimeDescriptorImpl = new MimeDescriptorImpl(
171                 bodyOctets, contentDescription, contentId, lines, subType,
172                 type, transferEncoding, headers, contentTypeParameters,
173                 languages, disposition, dispositionParams, embeddedMessage,
174                 parts, location, md5);
175         return mimeDescriptorImpl;
176     }

305     private static final class CountingInputStream extends InputStream {
306 
307         private final InputStream in;
308 
309         private int lineCount;
310 
311         private int octetCount;
312 
313         private CountingInputStream(InputStream in) {
314             super();
315             this.in = in;
316         }
317 
318         public int read() throws IOException {
319             int next = in.read();
320             if (next > 0) {
321                 octetCount++;
322                 if (next == '\r') {
323                     lineCount++;
324                 }
325             }
326             return next;
327         }
328 
329         public final int getLineCount() {
330             return lineCount;
331         }
332 
333         public final int getOctetCount() {
334             return octetCount;
335         }
336     }
