118     public MessageResult appendMessage(MimeMessage message, Date internalDate,
119             FetchGroup fetchGroup, MailboxSession mailboxSession)
120             throws MailboxManagerException {
121 
122         try {
123             checkAccess();
124 
125             final MailboxRow myMailboxRow = reserveNextUid();
126 
127             if (myMailboxRow != null) {
128                 try {
129                     // To be thread safe, we first get our own copy and the
130                     // exclusive
131                     // Uid
132                     // TODO create own message_id and assign uid later
133                     // at the moment it could lead to the situation that uid 5
134                     // is
135                     // inserted long before 4, when
136                     // mail 4 is big and comes over a slow connection.
137                     long uid = myMailboxRow.getLastUid();
138                     this.mailboxRow = myMailboxRow;
139 
140                     MessageRow messageRow = new MessageRow();
141                     messageRow.setMailboxId(getMailboxRow().getMailboxId());
142                     messageRow.setUid(uid);
143                     messageRow.setInternalDate(internalDate);
144 
145                     final int size = size(message);
146                     messageRow.setSize(size);
147                     populateFlags(message, messageRow);
148 
149                     addHeaders(message, messageRow);
150 
151                     MessageBody mb = populateBody(message);
152                     messageRow.addMessageBody(mb);
153 
154                     save(messageRow);
155                     MessageResult messageResult = fillMessageResult(messageRow,
156                             fetchGroup);
157                     getUidChangeTracker().found(messageResult);
158                     return messageResult;
159                 } catch (Exception e) {
160                     throw new MailboxManagerException(e);
161                 }
162             } else {
163                 // mailboxRow==null
164                 throw new MailboxManagerException("Mailbox has been deleted");
165             }
166         } catch (InterruptedException e) {
167             throw new MailboxManagerException(e);
168         }
169     }

195     private void addHeaders(MimeMessage message, MessageRow messageRow)
196             throws MessagingException, TorqueException {
197         int line_number = 0;
198 
199         for (Enumeration lines = message.getAllHeaderLines(); lines
200                 .hasMoreElements();) {
201             String line = (String) lines.nextElement();
202             int colon = line.indexOf(": ");
203             if (colon > 0) {
204                 line_number++;
205                 MessageHeader mh = new MessageHeader();
206                 mh.setLineNumber(line_number);
207                 mh.setField(line.substring(0, colon));
208                 // TODO avoid unlikely IOOB Exception
209                 mh.setValue(line.substring(colon + 2));
210                 messageRow.addMessageHeader(mh);
211             }
212         }
213     }

235             MailboxManagerException {
236         final MailboxRow myMailboxRow;
237         try {
238             lock.writeLock().acquire();

250     private Criteria criteriaForMessageSet(MessageRange set)
251             throws MailboxManagerException {
252         Criteria criteria = new Criteria();
253         criteria.addAscendingOrderByColumn(MessageRowPeer.UID);
254         if (set.getType() == MessageRange.TYPE_ALL) {
255             // empty Criteria = everything
256         } else if (set.getType() == MessageRange.TYPE_UID) {
257 
258             if (set.getUidFrom() == set.getUidTo()) {
259                 criteria.add(MessageRowPeer.UID, set.getUidFrom());
260             } else {
261                 Criteria.Criterion criterion1 = criteria.getNewCriterion(
262                         MessageRowPeer.UID, new Long(set.getUidFrom()),
263                         Criteria.GREATER_EQUAL);
264                 if (set.getUidTo() > 0) {
265                     Criteria.Criterion criterion2 = criteria.getNewCriterion(
266                             MessageRowPeer.UID, new Long(set.getUidTo()),
267                             Criteria.LESS_EQUAL);
268                     criterion1.and(criterion2);
269                 }
270                 criteria.add(criterion1);
271             }
272         } else {
273             throw new MailboxManagerException("Unsupported MessageSet: "
274                     + set.getType());
275         }
276         return criteria;
277     }

279     public Iterator getMessages(final MessageRange set, FetchGroup fetchGroup,
280             MailboxSession mailboxSession) throws MailboxManagerException {
281         try {
282             lock.readLock().acquire();
283             try {
284                 checkAccess();
285                 UidRange range = uidRangeForMessageSet(set);
286                 try {
287                     Criteria c = criteriaForMessageSet(set);
288                     c.add(MessageFlagsPeer.MAILBOX_ID, getMailboxRow()
289                             .getMailboxId());
290                     return getMessages(fetchGroup, range, c);
291                 } catch (TorqueException e) {
292                     throw new MailboxManagerException(e);
293                 } catch (MessagingException e) {
294                     throw new MailboxManagerException(e);
295                 }
296             } finally {
297                 lock.readLock().release();
298             }
299         } catch (InterruptedException e) {
300             throw new MailboxManagerException(e);
301         }
302     }

306             MailboxManagerException {

322             throws MailboxManagerException {
323         if (set.getType() == MessageRange.TYPE_UID) {
324             return new UidRange(set.getUidFrom(), set.getUidTo());
325         } else if (set.getType() == MessageRange.TYPE_ALL) {

329                     + set.getType());

354                 checkAccess();
355                 final Criteria criterion = queryRecentFlagSet();
356                 final List messageRows = getMailboxRow().getMessageRows(

362                     results[count++] = row.getUid();
363                 }

368                 return results;

380     private Criteria queryRecentFlagSet() {
381         final Criteria criterion = new Criteria();
382         criterion.addJoin(MessageFlagsPeer.MAILBOX_ID,
383                 MessageRowPeer.MAILBOX_ID);
384         criterion.addJoin(MessageRowPeer.UID, MessageFlagsPeer.UID);
385 
386         MessageFlagsPeer.addFlagsToCriteria(new Flags(Flags.Flag.RECENT), true,
387                 criterion);

420                         return null;
421                     }
422                 } catch (TorqueException e) {
423                     throw new MailboxManagerException(e);

425                     throw new MailboxManagerException(e);
426                 }

473     private Iterator doExpunge(final MessageRange set, FetchGroup fetchGroup)
474             throws MailboxManagerException {
475         checkAccess();
476         try {
477             // TODO put this into a serializable transaction
478             final Criteria c = criteriaForMessageSet(set);
479             c.addJoin(MessageRowPeer.MAILBOX_ID, MessageFlagsPeer.MAILBOX_ID);
480             c.addJoin(MessageRowPeer.UID, MessageFlagsPeer.UID);
481             c.add(MessageRowPeer.MAILBOX_ID, getMailboxRow().getMailboxId());
482             c.add(MessageFlagsPeer.DELETED, true);
483 
484             final List messageRows = getMailboxRow().getMessageRows(c);
485             final long[] uids = uids(messageRows);
486             final OrFetchGroup orFetchGroup = new OrFetchGroup(fetchGroup,
487                     FetchGroup.FLAGS);
488             final TorqueResultIterator resultIterator = new TorqueResultIterator(
489                     messageRows, orFetchGroup);
490             // ensure all results are loaded before deletion
491             Collection messageResults = IteratorUtils.toList(resultIterator);
492 
493             for (Iterator iter = messageRows.iterator(); iter.hasNext();) {
494                 MessageRow messageRow = (MessageRow) iter.next();
495                 Criteria todelc = new Criteria();
496                 todelc
497                         .add(MessageRowPeer.MAILBOX_ID, messageRow
498                                 .getMailboxId());
499                 todelc.add(MessageRowPeer.UID, messageRow.getUid());
500                 MessageRowPeer.doDelete(todelc);
501             }
502             getUidChangeTracker().expunged(uids);
503             return messageResults.iterator();
504         } catch (Exception e) {
505             throw new MailboxManagerException(e);
506         }
507     }

511         long[] results = new long[size];
512         for (int i = 0; i < size; i++) {
513             final MessageRow messageRow = (MessageRow) messageRows.get(i);
514             results[i] = (messageRow).getUid();
515         }

536     private Iterator doSetFlags(Flags flags, boolean value, boolean replace,
537             final MessageRange set, FetchGroup fetchGroup,
538             MailboxSession mailboxSession) throws MailboxManagerException {
539         checkAccess();
540         try {
541             // TODO put this into a serializeable transaction
542             final List messageRows = getMailboxRow().getMessageRows(
543                     criteriaForMessageSet(set));
544             UidRange uidRange = uidRangeForMessageSet(set);
545             getUidChangeTracker().found(uidRange,
546                     MessageRowUtils.toMessageFlags(messageRows));
547             for (Iterator iter = messageRows.iterator(); iter.hasNext();) {
548                 final MessageRow messageRow = (MessageRow) iter.next();
549                 final MessageFlags messageFlags = messageRow.getMessageFlags();
550                 if (messageFlags != null) {
551                     if (replace) {
552                         messageFlags.setFlags(flags);
553                     } else {
554                         Flags current = messageFlags.getFlagsObject();
555                         if (value) {
556                             current.add(flags);
557                         } else {
558                             current.remove(flags);
559                         }
560                         messageFlags.setFlags(current);
561                     }
562                     messageFlags.save();
563                 }
564             }
565             final OrFetchGroup orFetchGroup = new OrFetchGroup(fetchGroup,
566                     FetchGroup.FLAGS);
567             final TorqueResultIterator resultIterator = new TorqueResultIterator(
568                     messageRows, orFetchGroup);
569             final org.apache.james.mailboxmanager.impl.MessageFlags[] messageFlags = resultIterator
570                     .getMessageFlags();
571             tracker.flagsUpdated(messageFlags, mailboxSession.getSessionId());
572             tracker.found(uidRange, messageFlags);
573             return resultIterator;
574         } catch (Exception e) {
575             throw new MailboxManagerException(e);
576         }
577     }

662     public Iterator search(SearchQuery query, FetchGroup fetchGroup,
663             MailboxSession mailboxSession) throws MailboxManagerException {
664         try {
665             lock.readLock().acquire();
666             try {
667                 checkAccess();
668 
669                 final Criteria criterion = preSelect(query);
670                 final List rows = MessageRowPeer
671                         .doSelectJoinMessageFlags(criterion);
672                 final List filteredMessages = new ArrayList();
673                 for (Iterator it = rows.iterator(); it.hasNext();) {
674                     final MessageRow row = (MessageRow) it.next();
675                     try {
676                         if (searches.isMatch(query, row)) {
677                             filteredMessages.add(row);
678                         }
679                     } catch (TorqueException e) {
680                         getLog()
681                                 .info(
682                                         "Cannot test message against search criteria. Will continue to test other messages.",
683                                         e);
684                         if (getLog().isDebugEnabled())
685                             getLog().debug("UID: " + row.getUid());
686                     }
687                 }
688 
689                 return getResults(fetchGroup, filteredMessages);
690             } catch (TorqueException e) {
691                 throw new MailboxManagerException(e);
692             } finally {
693                 lock.readLock().release();
694             }
695         } catch (InterruptedException e) {
696             throw new MailboxManagerException(e);
697         }
698     }

700     private Criteria preSelect(SearchQuery query) {
701         final Criteria results = new Criteria();
702         final List criteria = query.getCriterias();
703         if (criteria.size() == 1) {
704             final Criterion criterion = (Criterion) criteria.get(0);
705             if (criterion instanceof SearchQuery.UidCriterion) {
706                 final SearchQuery.UidCriterion uidCriterion = (SearchQuery.UidCriterion) criterion;
707                 preSelectUid(results, uidCriterion);
708             }
709         }
710         return results;
711     }
712 
713     private void preSelectUid(final Criteria results,
714             final SearchQuery.UidCriterion uidCriterion) {
715         final NumericRange[] ranges = uidCriterion.getOperator().getRange();
716         for (int i = 0; i < ranges.length; i++) {
717             final long low = ranges[i].getLowValue();
718             final long high = ranges[i].getHighValue();
719             if (low == Long.MAX_VALUE) {
720                 results.add(MessageRowPeer.UID, high, Criteria.LESS_EQUAL);
721             } else if (low == high) {
722                 results.add(MessageRowPeer.UID, low);
723             } else {
724                 final Criteria.Criterion fromCriterion = results
725                         .getNewCriterion(MessageRowPeer.UID, new Long(low),
726                                 Criteria.GREATER_EQUAL);
727                 if (high > 0 && high < Long.MAX_VALUE) {
728                     final Criteria.Criterion toCriterion = results
729                             .getNewCriterion(MessageRowPeer.UID,
730                                     new Long(high), Criteria.LESS_EQUAL);
731                     fromCriterion.and(toCriterion);
732                 }
733                 results.add(fromCriterion);
734             }
735         }

778                 final MailboxRow mailbox = reserveNextUid();

786                     // is

789                     long uid = mailbox.getLastUid();

794                     newRow.setUid(uid);

796                     newRow.setSize(fromRow.getSize());

801                     for (Iterator iterator = headers.iterator(); iterator
802                             .hasNext();) {

804                                 .next();

806                                 fromHeader.getField(), fromHeader.getValue(),
