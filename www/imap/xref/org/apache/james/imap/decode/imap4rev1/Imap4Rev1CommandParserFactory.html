134     /**
135      * @see org.apache.james.imap.decode.imap4rev1.ImapCommandParserFactory#getParser(java.lang.String)
136      */
137     public ImapCommandParser getParser(String commandName) {
138         Class cmdClass = (Class) _imapCommands.get(commandName.toUpperCase());
139 
140         if (cmdClass == null) {
141             return null;
142         } else {
143             return createCommand(cmdClass);
144         }
145     }
146 
147     private ImapCommandParser createCommand(Class commandClass) {
148         final Log logger = getLog();
149         try {
150             ImapCommandParser cmd = (ImapCommandParser) commandClass
151                     .newInstance();
152             initialiseParser(commandClass, logger, cmd);
153             return cmd;
154         } catch (Exception e) {
155             if (logger.isWarnEnabled()) {
156                 logger.warn("Create command instance failed: ", e);
157             }
158             // TODO: would probably be better to manage this in protocol
159             // TODO: this runtime will produce a nasty disconnect for the client
160             throw new RuntimeException("Could not create command instance: "
161                     + commandClass.getName(), e);
162         }
163     }
164 
165     protected void initialiseParser(Class commandClass, final Log logger,
166             ImapCommandParser cmd) {
167         setupLogger(cmd);
168         if (logger.isDebugEnabled()) {
169             logger.debug("Created command " + commandClass);
170         }
171 
172         if (cmd instanceof DelegatingImapCommandParser) {
173             ((DelegatingImapCommandParser) cmd).setParserFactory(this);
174         }
175 
176         if (cmd instanceof MessagingImapCommandParser) {
177             final MessagingImapCommandParser messagingImapCommandParser = (MessagingImapCommandParser) cmd;
178             messagingImapCommandParser.setMessageFactory(messageFactory);
179             messagingImapCommandParser
180                     .setStatusResponseFactory(statusResponseFactory);
181         }
182 
183         if (cmd instanceof InitialisableCommandFactory) {
184             ((InitialisableCommandFactory) cmd).init(commandFactory);
185         }
186     }
