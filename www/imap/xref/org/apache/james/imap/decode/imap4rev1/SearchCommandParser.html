61      /**
62       * Parses the request argument into a valid search term.
63       * 
64       * @param request
65       *            <code>ImapRequestLineReader</code>, not null
66       * @param charset
67       *            <code>Charset</code> or null if there is no charset
68       * @param isFirstToken
69       *            true when this is the first token read, false otherwise
70       */
71      public SearchKey searchKey(ImapRequestLineReader request, Charset charset,
72              boolean isFirstToken) throws ProtocolException,
73              IllegalCharsetNameException, UnsupportedCharsetException {
74          final char next = request.nextChar();
75          if (next >= '0' && next <= '9' || next == '*') {
76              return sequenceSet(request);
77          } else if (next == '(') {
78              return paren(request, charset);
79          } else {
80              final int cap = consumeAndCap(request);
81              switch (cap) {
82                  case 'A':
83                      return a(request);
84                  case 'B':
85                      return b(request, charset);
86                  case 'C':
87                      return c(request, isFirstToken, charset);
88                  case 'D':
89                      return d(request);
90                  case 'E':
91                      throw new ProtocolException("Unknown search key");
92                  case 'F':
93                      return f(request, charset);
94                  case 'G':
95                      throw new ProtocolException("Unknown search key");
96                  case 'H':
97                      return header(request, charset);
98                  case 'I':
99                      throw new ProtocolException("Unknown search key");
100                 case 'J':
101                     throw new ProtocolException("Unknown search key");
102                 case 'K':
103                     return keyword(request);
104                 case 'L':
105                     return larger(request);
106                 case 'M':
107                     throw new ProtocolException("Unknown search key");
108                 case 'N':
109                     return n(request, charset);
110                 case 'O':
111                     return o(request, charset);
112                 case 'P':
113                     throw new ProtocolException("Unknown search key");
114                 case 'Q':
115                     throw new ProtocolException("Unknown search key");
116                 case 'R':
117                     return recent(request);
118                 case 'S':
119                     return s(request, charset);
120                 case 'T':
121                     return t(request, charset);
122                 case 'U':
123                     return u(request);
124                 default:
125                     throw new ProtocolException("Unknown search key");
126             }
127         }
128     }

130     private SearchKey paren(ImapRequestLineReader request, Charset charset)
131             throws ProtocolException {
132         request.consume();
133         List keys = new ArrayList();
134         addUntilParen(request, keys, charset);
135         return SearchKey.buildAnd(keys);
136     }
137 
138     private void addUntilParen(ImapRequestLineReader request, List keys,
139             Charset charset) throws ProtocolException {
140         final char next = request.nextWordChar();
141         if (next == ')') {
142             request.consume();
143         } else {
144             final SearchKey key = searchKey(request, null, false);
145             keys.add(key);
146             addUntilParen(request, keys, charset);
147         }
148     }

150     private int consumeAndCap(ImapRequestLineReader request)
151             throws ProtocolException {
152         final char next = request.consume();
153         final int cap = next > 'Z' ? next ^ 32 : next;
154         return cap;
155     }

158             throws ProtocolException {
159         final SearchKey result;

162         result = SearchKey.buildCc(value);
163         return result;

166     private SearchKey c(ImapRequestLineReader request,
167             final boolean isFirstToken, final Charset charset)
168             throws ProtocolException, IllegalCharsetNameException,
169             UnsupportedCharsetException {
170         final int next = consumeAndCap(request);
171         switch (next) {
172             case 'C':
173                 return cc(request, charset);
174             case 'H':
175                 return charset(request, isFirstToken);
176             default:
177                 throw new ProtocolException("Unknown search key");
178         }
179     }

181     private SearchKey charset(ImapRequestLineReader request,
182             final boolean isFirstToken) throws ProtocolException,
183             IllegalCharsetNameException, UnsupportedCharsetException {
184         final SearchKey result;
185         nextIsA(request);
186         nextIsR(request);
187         nextIsS(request);
188         nextIsE(request);
189         nextIsT(request);
190         nextIsSpace(request);
191         if (!isFirstToken) {
192             throw new ProtocolException("Unknown search key");
193         }
194         final String value = astring(request);
195         final Charset charset = Charset.forName(value);
196         request.nextWordChar();
197         result = searchKey(request, charset, false);
198         return result;
199     }

904     public SearchKey decode(ImapRequestLineReader request)
905             throws ProtocolException, IllegalCharsetNameException,
906             UnsupportedCharsetException {
907         request.nextWordChar();
908         final SearchKey firstKey = searchKey(request, null, true);
909         final SearchKey result;
910         if (request.nextChar() == ' ') {
911             List keys = new ArrayList();
912             keys.add(firstKey);
913             while (request.nextChar() == ' ') {
914                 request.nextWordChar();
915                 final SearchKey key = searchKey(request, null, false);
916                 keys.add(key);
917             }
918             result = SearchKey.buildAnd(keys);
919         } else {
920             result = firstKey;
921         }
922         endLine(request);
923         return result;
924     }

937     private synchronized void loadCharsetNames() {
938         if (charsetNames == null) {
939             charsetNames = new HashSet();
940             for (final Iterator it = Charset.availableCharsets().values()
941                     .iterator(); it.hasNext();) {
942                 final Charset charset = (Charset) it.next();
943                 final Set aliases = charset.aliases();
944                 charsetNames.addAll(aliases);
945             }
946         }
