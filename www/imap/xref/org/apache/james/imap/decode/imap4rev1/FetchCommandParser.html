48      public FetchData fetchRequest(ImapRequestLineReader request)
49              throws ProtocolException {
50          FetchData fetch = new FetchData();
51  
52          char next = nextNonSpaceChar(request);
53          if (request.nextChar() == '(') {
54              consumeChar(request, '(');
55  
56              next = nextNonSpaceChar(request);
57              while (next != ')') {
58                  addNextElement(request, fetch);
59                  next = nextNonSpaceChar(request);
60              }
61              consumeChar(request, ')');
62          } else {
63              addNextElement(request, fetch);
64  
65          }
66  
67          return fetch;
68      }

70      private void addNextElement(ImapRequestLineReader reader, FetchData fetch)
71              throws ProtocolException {
72          // String name = element.toString();
73          String name = readWord(reader, " [)\r\n");
74          char next = reader.nextChar();
75          // Simple elements with no '[]' parameters.
76          if (next != '[') {
77              if ("FAST".equalsIgnoreCase(name)) {
78                  fetch.setFlags(true);
79                  fetch.setInternalDate(true);
80                  fetch.setSize(true);
81              } else if ("FULL".equalsIgnoreCase(name)) {
82                  fetch.setFlags(true);
83                  fetch.setInternalDate(true);
84                  fetch.setSize(true);
85                  fetch.setEnvelope(true);
86                  fetch.setBody(true);
87              } else if ("ALL".equalsIgnoreCase(name)) {
88                  fetch.setFlags(true);
89                  fetch.setInternalDate(true);
90                  fetch.setSize(true);
91                  fetch.setEnvelope(true);
92              } else if ("FLAGS".equalsIgnoreCase(name)) {
93                  fetch.setFlags(true);
94              } else if ("RFC822.SIZE".equalsIgnoreCase(name)) {
95                  fetch.setSize(true);
96              } else if ("ENVELOPE".equalsIgnoreCase(name)) {
97                  fetch.setEnvelope(true);
98              } else if ("INTERNALDATE".equalsIgnoreCase(name)) {
99                  fetch.setInternalDate(true);
100             } else if ("BODY".equalsIgnoreCase(name)) {
101                 fetch.setBody(true);
102             } else if ("BODYSTRUCTURE".equalsIgnoreCase(name)) {
103                 fetch.setBodyStructure(true);
104             } else if ("UID".equalsIgnoreCase(name)) {
105                 fetch.setUid(true);
106             } else if ("RFC822".equalsIgnoreCase(name)) {
107                 fetch.add(BodyFetchElement.createRFC822(), false);
108             } else if ("RFC822.HEADER".equalsIgnoreCase(name)) {
109                 fetch.add(BodyFetchElement.createRFC822Header(), true);
110             } else if ("RFC822.TEXT".equalsIgnoreCase(name)) {
111                 fetch.add(BodyFetchElement.createRFC822Text(), false);
112             } else {
113                 throw new ProtocolException("Invalid fetch attribute: " + name);
114             }
115         } else {
116             consumeChar(reader, '[');
117 
118             String parameter = readWord(reader, "]");
119 
120             consumeChar(reader, ']');
121 
122             final Long firstOctet;
123             final Long numberOfOctets;
124             if (reader.nextChar() == '<') {
125                 consumeChar(reader, '<');
126                 firstOctet = new Long(number(reader));
127                 if (reader.nextChar() == '.') {
128                     consumeChar(reader, '.');
129                     numberOfOctets = new Long(nzNumber(reader));
130                 } else {
131                     numberOfOctets = null;
132                 }
133                 consumeChar(reader, '>');
134             } else {
135                 firstOctet = null;
136                 numberOfOctets = null;
137             }
138 
139             final BodyFetchElement bodyFetchElement = createBodyElement(
140                     parameter, firstOctet, numberOfOctets);
141             final boolean isPeek = isPeek(name);
142             fetch.add(bodyFetchElement, isPeek);
143         }
144     }
145 
146     private boolean isPeek(String name) throws ProtocolException {
147         final boolean isPeek;
148         if ("BODY".equalsIgnoreCase(name)) {
149             isPeek = false;
150         } else if ("BODY.PEEK".equalsIgnoreCase(name)) {
151             isPeek = true;
152         } else {
153             throw new ProtocolException("Invalid fetch attibute: " + name
154                     + "[]");
155         }
156         return isPeek;
157     }

159     private BodyFetchElement createBodyElement(String parameter,
160             Long firstOctet, Long numberOfOctets) throws ProtocolException {
161         final String responseName = "BODY[" + parameter + "]";
162         FetchPartPathDecoder decoder = new FetchPartPathDecoder();
163         decoder.decode(parameter);
164         final int sectionType = getSectionType(decoder);
165 
166         final List names = decoder.getNames();
167         final int[] path = decoder.getPath();
168         final BodyFetchElement bodyFetchElement = new BodyFetchElement(
169                 responseName, sectionType, path, names, firstOctet,
170                 numberOfOctets);
171         return bodyFetchElement;
172     }
173 
174     private int getSectionType(FetchPartPathDecoder decoder)
175             throws ProtocolException {
176         final int specifier = decoder.getSpecifier();
177         final int sectionType;
178         switch (specifier) {
179             case FetchPartPathDecoder.CONTENT:
180                 sectionType = BodyFetchElement.CONTENT;
181                 break;
182             case FetchPartPathDecoder.HEADER:
183                 sectionType = BodyFetchElement.HEADER;
184                 break;
185             case FetchPartPathDecoder.HEADER_FIELDS:
186                 sectionType = BodyFetchElement.HEADER_FIELDS;
187                 break;
188             case FetchPartPathDecoder.HEADER_NOT_FIELDS:
189                 sectionType = BodyFetchElement.HEADER_NOT_FIELDS;
190                 break;
191             case FetchPartPathDecoder.MIME:
192                 sectionType = BodyFetchElement.MIME;
193                 break;
194             case FetchPartPathDecoder.TEXT:
195                 sectionType = BodyFetchElement.TEXT;
196                 break;
197             default:
198                 throw new ProtocolException("Section type is unsupported.");
199         }
200         return sectionType;
201     }

203     private String readWord(ImapRequestLineReader request, String terminator)
204             throws ProtocolException {
205         StringBuffer buf = new StringBuffer();
206         char next = request.nextChar();
207         while (terminator.indexOf(next) == -1) {
208             buf.append(next);
209             request.consume();
210             next = request.nextChar();
211         }
212         return buf.toString();
213     }
214 
215     private char nextNonSpaceChar(ImapRequestLineReader request)
216             throws ProtocolException {
217         char next = request.nextChar();
218         while (next == ' ') {
219             request.consume();
220             next = request.nextChar();
221         }
222         return next;
223     }

225     protected ImapMessage decode(ImapCommand command,
226             ImapRequestLineReader request, String tag, boolean useUids)
227             throws ProtocolException {
228         IdRange[] idSet = parseIdRange(request);
229         FetchData fetch = fetchRequest(request);
230         endLine(request);
231 
232         final Imap4Rev1MessageFactory factory = getMessageFactory();
233         final ImapMessage result = factory.createFetchMessage(command, useUids,
234                 idSet, fetch, tag);
235         return result;
236     }
