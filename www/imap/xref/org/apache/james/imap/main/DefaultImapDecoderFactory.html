33  /**
34   * TODO: this is temporary: should let the container do the coupling. TODO:
35   * convert to POJO
36   */
37  public class DefaultImapDecoderFactory implements ImapDecoderFactory {
38  
39      public static final ImapDecoder createDecoder() {
40          final UnpooledStatusResponseFactory unpooledStatusResponseFactory = new UnpooledStatusResponseFactory();
41          final Imap4Rev1MessageFactory messageFactory = new BaseImap4Rev1MessageFactory(
42                  unpooledStatusResponseFactory);
43          final Imap4Rev1CommandFactory commandFactory = new StandardImap4Rev1CommandFactory();
44          final ImapCommandParserFactory imapCommands = new Imap4Rev1CommandParserFactory(
45                  messageFactory, commandFactory, unpooledStatusResponseFactory);
46          final ImapDecoder result = new DefaultImapDecoder(messageFactory,
47                  imapCommands);
48          return result;
49      }
50  
51      public ImapDecoder buildImapDecoder() {
52          return createDecoder();
53      }
54  
55  }
