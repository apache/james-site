55      public ChannelImapResponseWriter(final WritableByteChannel out,
56              final int bufferSize) {
57          this.out = out;
58          skipNextSpace = false;

60          usAscii = Charset.forName("US-ASCII");

63      /**
64       * Writes the message provided to the client, prepended with the untagged

81      private void write(final ByteBuffer buffer) throws IOException {
82          while (out.write(buffer) > 0) {
83              // Write all
84          }
85      }

87      private void writeASCII(final String string) throws IOException {
88          final ByteBuffer buffer = usAscii.encode(string);
89          write(buffer);
90      }

92      private void write(byte[] bytes) throws IOException {
93          final ByteBuffer wrap = ByteBuffer.wrap(bytes);
94          write(wrap);
95      }

135     public void quote(String message) throws IOException {
136         space();
137         final int length = message.length();
138         buffer.clear();
139         buffer.put(BYTE_DQUOTE);
140         for (int i = 0; i < length; i++) {
141             writeIfFull();
142             char character = message.charAt(i);
143             if (character == ImapConstants.BACK_SLASH || character == DQUOTE) {
144                 buffer.put(BYTE_BACK_SLASH);
145             }
146             writeIfFull();
147             // 7-bit ASCII only
148             if (character > 128) {
149                 buffer.put(BYTE_QUESTION);
150             } else {
151                 buffer.put((byte) character);
152             }
153         }
154         writeIfFull();
155         buffer.put(BYTE_DQUOTE);
156         buffer.flip();
157         write(buffer);
158     }
159 
160     private void writeIfFull() throws IOException {
161         if (!buffer.hasRemaining()) {
162             buffer.flip();
163             write(buffer);
164             buffer.clear();
165         }
166     }

206     public void literal(Literal literal) throws IOException {
207         space();
208         write(BYTES_OPEN_BRACE);
209         final long size = literal.size();
210         writeASCII(Long.toString(size));
211         write(BYTES_CLOSE_BRACE);
212         write(BYTES_LINE_END);
213         if (size > 0) {

230     private void upperCaseAscii(String message, boolean quote)
231             throws IOException {
232         space();
233         final int length = message.length();
234         buffer.clear();
235         if (quote) {
236             buffer.put(BYTE_DQUOTE);
237         }
238         for (int i = 0; i < length; i++) {
239             writeIfFull();
240             final char next = message.charAt(i);
241             if (next >= 'a' && next <= 'z') {
242                 buffer.put((byte) (next - LOWER_CASE_OFFSET));
243             } else {
244                 buffer.put((byte) (next));
245             }
246         }
247         writeIfFull();
248         if (quote) {
249             buffer.put(BYTE_DQUOTE);
250         }
251         buffer.flip();
252         write(buffer);
253     }

255     public void quoteUpperCaseAscii(String message) throws IOException {
