40  public class StatusProcessor extends AbstractMailboxAwareProcessor {
41  
42      public StatusProcessor(final ImapProcessor next,
43              final MailboxManagerProvider mailboxManagerProvider,
44              final StatusResponseFactory factory) {
45          super(next, mailboxManagerProvider, factory);
46      }
47  
48      protected boolean isAcceptable(ImapMessage message) {
49          return (message instanceof StatusRequest);
50      }
51  
52      protected void doProcess(ImapRequest message, ImapSession session,
53              String tag, ImapCommand command, Responder responder) {
54          final StatusRequest request = (StatusRequest) message;
55          final String mailboxName = request.getMailboxName();
56          final StatusDataItems statusDataItems = request.getStatusDataItems();
57          final Log logger = getLog();
58          final MailboxSession mailboxSession = ImapSessionUtils
59                  .getMailboxSession(session);
60  
61          try {
62              String fullMailboxName = buildFullName(session, mailboxName);
63  
64              if (logger != null && logger.isDebugEnabled()) {
65                  logger.debug("Status called on mailbox named " + mailboxName
66                          + " (" + fullMailboxName + ")");
67              }
68  
69              final MailboxManager mailboxManager = getMailboxManager(session);
70              final Mailbox mailbox = mailboxManager.getMailbox(fullMailboxName,
71                      false);
72  
73              final Long messages = messages(statusDataItems, mailboxSession,
74                      mailbox);
75              final Long recent = recent(statusDataItems, mailboxSession, mailbox);
76              final Long uidNext = uidNext(statusDataItems, mailboxSession,
77                      mailbox);
78              final Long uidValidity = uidValidity(statusDataItems,
79                      mailboxSession, mailbox);
80              final Long unseen = unseen(statusDataItems, mailboxSession, mailbox);
81  
82              final STATUSResponse response = new STATUSResponse(messages,
83                      recent, uidNext, uidValidity, unseen, mailboxName);
84              responder.respond(response);
85              unsolicitedResponses(session, responder, false);
86              okComplete(command, tag, responder);
87  
88          } catch (MailboxManagerException e) {
89              no(command, tag, responder, e);
90          }
91      }
92  
93      private Long unseen(final StatusDataItems statusDataItems,
94              final MailboxSession mailboxSession, final Mailbox mailbox)
95              throws MailboxManagerException {
96          final Long unseen;
97          if (statusDataItems.isUnseen()) {
98              final int unseenCountValue = mailbox.getUnseenCount(mailboxSession);
99              unseen = new Long(unseenCountValue);
100         } else {
101             unseen = null;
102         }
103         return unseen;
104     }
105 
106     private Long uidValidity(final StatusDataItems statusDataItems,
107             final MailboxSession mailboxSession, final Mailbox mailbox)
108             throws MailboxManagerException {
109         final Long uidValidity;
110         if (statusDataItems.isUidValidity()) {
111             final long uidValidityValue = mailbox
112                     .getUidValidity(mailboxSession);
113             uidValidity = new Long(uidValidityValue);
114         } else {
115             uidValidity = null;
116         }
117         return uidValidity;
118     }
119 
120     private Long uidNext(final StatusDataItems statusDataItems,
121             final MailboxSession mailboxSession, final Mailbox mailbox)
122             throws MailboxManagerException {
123         final Long uidNext;
124         if (statusDataItems.isUidNext()) {
125             final long uidNextValue = mailbox.getUidNext(mailboxSession);
126             uidNext = new Long(uidNextValue);
127         } else {
128             uidNext = null;
129         }
130         return uidNext;
131     }
132 
133     private Long recent(final StatusDataItems statusDataItems,
134             final MailboxSession mailboxSession, final Mailbox mailbox)
135             throws MailboxManagerException {
136         final Long recent;
137         if (statusDataItems.isRecent()) {
138             final int recentCount = mailbox.recent(false, mailboxSession).length;
139             recent = new Long(recentCount);
140         } else {
141             recent = null;
142         }
143         return recent;
144     }
145 
146     private Long messages(final StatusDataItems statusDataItems,
147             final MailboxSession mailboxSession, final Mailbox mailbox)
148             throws MailboxManagerException {
149         final Long messages;
150         if (statusDataItems.isMessages()) {
151             final int messageCount = mailbox.getMessageCount(mailboxSession);
152             messages = new Long(messageCount);
153         } else {
154             messages = null;
155         }
156         return messages;
157     }
