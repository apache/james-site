350     /**
351      * @throws IOException
352      * @see org.apache.james.imap.encode.ImapResponseComposer#statusResponse(java.lang.String,
353      *      org.apache.james.api.imap.ImapCommand, java.lang.String,
354      *      java.lang.String, Collection, long, java.lang.String)
355      */
356     public void statusResponse(String tag, ImapCommand command, String type,
357             String responseCode, Collection parameters, long number, String text)
358             throws IOException {
359         if (tag == null) {
360             untagged();
361         } else {
362             tag(tag);
363         }
364         message(type);
365         if (responseCode != null) {
366             openSquareBracket();
367             message(responseCode);
368             if (parameters != null && !parameters.isEmpty()) {
369                 openParen();
370                 for (Iterator it = parameters.iterator(); it.hasNext();) {
371                     final String parameter = (String) it.next();
372                     message(parameter);
373                 }
374                 closeParen();
375             }
376             if (number > 0) {
377                 message(number);
378             }
379             closeSquareBracket();
380         }
381         if (command != null) {
382             commandName(command);
383         }
384         if (text != null && !"".equals(text)) {
385             message(text);
386         }
387         end();
388     }

395     public void statusResponse(Long messages, Long recent, Long uidNext,
396             Long uidValidity, Long unseen, String mailboxName)
397             throws IOException {
398         untagged();
399         message(STATUS_COMMAND_NAME);
400         quote(mailboxName);
401         openParen();
402 
403         if (messages != null) {
404             message(STATUS_MESSAGES);
405             final long messagesValue = messages.longValue();
406             message(messagesValue);
407         }
408 
409         if (recent != null) {

412             message(recentValue);
413         }
414 
415         if (uidNext != null) {
416             message(STATUS_UIDNEXT);
417             final long uidNextValue = uidNext.longValue();
418             message(uidNextValue);
419         }
420 
421         if (uidValidity != null) {

437     public void listResponse(String typeName, List attributes,
438             String hierarchyDelimiter, String name) throws IOException {
439         untagged();
440         message(typeName);
441         openParen();
442         if (attributes != null) {
443             for (Iterator it = attributes.iterator(); it.hasNext();) {
444                 final String attribute = (String) it.next();
445                 message(attribute);
446             }
447         }
448         closeParen();
449 
450         if (hierarchyDelimiter == null) {
451             message(NIL);
452         } else {
453             quote(hierarchyDelimiter);
454         }
455 
456         quote(name);
457 
458         end();

492     public void flags(Flags flags) throws IOException {
493         message(FLAGS);
494         openParen();
495         if (flags.contains(Flags.Flag.ANSWERED)) {
496             message("\\Answered");

505             message("\\Flagged");
506         }
507         if (flags.contains(Flags.Flag.RECENT)) {
508             message("\\Recent");
509         }
510         if (flags.contains(Flags.Flag.SEEN)) {
511             message("\\Seen");
512         }
