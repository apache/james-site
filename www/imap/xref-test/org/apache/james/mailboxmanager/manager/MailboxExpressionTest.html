46      public void testIsWild() throws Exception {
47          assertTrue(create("*").isWild());
48          assertTrue(create("%").isWild());
49          assertTrue(create("One*").isWild());
50          assertTrue(create("*One").isWild());
51          assertTrue(create("A*A").isWild());
52          assertTrue(create("One%").isWild());
53          assertTrue(create("%One").isWild());
54          assertTrue(create("A%A").isWild());
55          assertFalse(create("").isWild());
56          assertFalse(create(null).isWild());
57          assertFalse(create("ONE").isWild());
58      }

243     public void testMultipleFreeWildcards() throws Exception {
244         MailboxExpression expression = create(SECOND_PART + '*' + PART + '*'
245                 + SECOND_PART + "**");
246         assertTrue(expression.isExpressionMatch(SECOND_PART + PART
247                 + SECOND_PART, '.'));
248         assertTrue(expression.isExpressionMatch(SECOND_PART + '.' + PART + '.'
249                 + SECOND_PART, '.'));
250         assertTrue(expression.isExpressionMatch(SECOND_PART + "tosh.bosh"
251                 + PART + "tosh.bosh" + SECOND_PART + "boshtosh", '.'));
252         assertFalse(expression.isExpressionMatch(SECOND_PART + '.'
253                 + PART.substring(1) + '.' + SECOND_PART, '.'));
254         assertTrue(expression.isExpressionMatch(SECOND_PART + '.'
255                 + PART.substring(1) + '.' + SECOND_PART + PART + '.'
256                 + SECOND_PART + "toshbosh", '.'));
257         assertFalse(expression.isExpressionMatch(SECOND_PART + '.'
258                 + PART.substring(1) + '.' + SECOND_PART + PART + '.'
259                 + SECOND_PART.substring(1), '.'));
260         assertTrue(expression.isExpressionMatch(SECOND_PART + "tosh.bosh"
261                 + PART + "tosh.bosh" + PART + SECOND_PART + "boshtosh" + PART
262                 + SECOND_PART, '.'));
263         assertFalse(expression.isExpressionMatch(SECOND_PART.substring(1)
264                 + "tosh.bosh" + PART + "tosh.bosh" + SECOND_PART
265                 + PART.substring(1) + SECOND_PART + "boshtosh" + PART
266                 + SECOND_PART.substring(1), '.'));
267     }
