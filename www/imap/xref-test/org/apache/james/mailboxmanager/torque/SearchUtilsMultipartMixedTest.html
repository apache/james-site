114     protected void setUp() throws Exception {
115         super.setUp();
116         row = new MessageRow();
117         row.addMessageHeader(new MessageHeader(ImapConstants.RFC822_FROM,
118                 "Alex <alex@example.org"));
119         row.addMessageHeader(new MessageHeader(ImapConstants.RFC822_TO,
120                 "Harry <harry@example.org"));
121         row.addMessageHeader(new MessageHeader(ImapConstants.RFC822_SUBJECT,
122                 "A Mixed Multipart Mail"));
123         row.addMessageHeader(new MessageHeader(ImapConstants.RFC822_DATE,
124                 "Thu, 14 Feb 2008 12:00:00 +0000 (GMT)"));
125         row.addMessageHeader(new MessageHeader("Content-Type",
126                 "multipart/mixed;boundary=1729"));
127         row.addMessageBody(new MessageBody(Charset.forName("us-ascii").encode(
128                 BODY).array()));
129         searches = new MessageSearches();
130         recent = new ArrayList();
131     }
