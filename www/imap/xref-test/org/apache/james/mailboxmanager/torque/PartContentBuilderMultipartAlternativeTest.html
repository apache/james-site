29  public class PartContentBuilderMultipartAlternativeTest extends TestCase {

31      private static final String CONTENT_TYPE_PLAIN = "text/plain;charset=us-ascii";
32  
33      private static final String CONTENT_TYPE_HTML = "text/html;charset=us-ascii";
34  
35      private static final String CONTENT_TYPE_XHTML = "application/xhtml;charset=us-ascii";
36  
37      private static final String CONTENT_TYPE = "Content-Type";
38  
39      private static final String ALT_PLAIN_BODY = "Rhubarb!Rhubard!Rhubard!\r\n";
40  
41      private static final String ALT_XHTML_BODY = "<!DOCTYPE html PUBLIC '-//W3C//DTD XHTML 1.0 Strict//EN' 'http://www.w3.org/TR/xhtml1/DTD/xhtml1-strict.dtd'>"
42              + "<html><head><title>Rhubard!</title></head><body><p>Rhubarb!Rhubard!Rhubard!</p></body></html>\r\n";
43  
44      private static final String ALT_HTML_BODY = "<html><head><title>Rhubard!</title></head><body><p>Rhubarb!Rhubard!Rhubard!</p></body></html>\r\n";
45  
46      private static final String ALT_PART_XHTML = CONTENT_TYPE + ": "
47              + CONTENT_TYPE_XHTML + "\r\n" + "\r\n" + ALT_XHTML_BODY;
48  
49      private static final String ALT_PART_HTML = CONTENT_TYPE + ": "
50              + CONTENT_TYPE_HTML + "\r\n" + "\r\n" + ALT_HTML_BODY;
51  
52      private static final String ALT_PART_PLAIN = CONTENT_TYPE + ": "
53              + CONTENT_TYPE_PLAIN + "\r\n" + "\r\n" + ALT_PLAIN_BODY;
54  
55      private static final String MULTIPART_ALTERNATIVE = "From: Samual Smith <samual@example.org>\r\n"
56              + "To: John Smith <john@example.org>\r\n"
57              + "Date: Sun, 10 Feb 2008 08:00:00 -0800 (PST)\r\n"
58              + "Subject: Rhubarb!\r\n"
59              + "Content-Type: multipart/alternative;boundary=4242\r\n"
60              + "\r\n"
61              + "--4242\r\n"
62              + ALT_PART_PLAIN
63              + "\r\n--4242\r\n"
64              + ALT_PART_HTML
65              + "\r\n--4242\r\n" + ALT_PART_XHTML + "\r\n--4242\r\n";

67      PartContentBuilder builder;
68  
69      protected void setUp() throws Exception {
70          super.setUp();
71          builder = new PartContentBuilder();
72      }
73  
74      protected void tearDown() throws Exception {
75          super.tearDown();
76      }
77  
78      public void testShouldLocatePartsOfMultipartAlterative() throws Exception {
79          assertEquals(ALT_PLAIN_BODY, bodyContent(MULTIPART_ALTERNATIVE, 1));
80          assertEquals(ALT_HTML_BODY, bodyContent(MULTIPART_ALTERNATIVE, 2));
81          assertEquals(ALT_XHTML_BODY, bodyContent(MULTIPART_ALTERNATIVE, 3));
82      }
83  
84      public void testShouldLocateHeadersOfMultipartAlterative() throws Exception {
85          checkContentType(CONTENT_TYPE_PLAIN, MULTIPART_ALTERNATIVE, 1);
86          checkContentType(CONTENT_TYPE_HTML, MULTIPART_ALTERNATIVE, 2);
87          checkContentType(CONTENT_TYPE_XHTML, MULTIPART_ALTERNATIVE, 3);
88      }
89  
90      public void testShouldLocateFullContentOfMultipartAlterative()
91              throws Exception {
92          assertEquals(ALT_PART_PLAIN, fullContent(MULTIPART_ALTERNATIVE, 1));
93          assertEquals(ALT_PART_HTML, fullContent(MULTIPART_ALTERNATIVE, 2));
94          assertEquals(ALT_PART_XHTML, fullContent(MULTIPART_ALTERNATIVE, 3));
95      }

97      private String fullContent(String mail, int position) throws Exception {
98          InputStream in = new ByteArrayInputStream(Charset.forName("us-ascii")
99                  .encode(mail).array());
100         builder.parse(in);
101         builder.to(position);
102         StringBuffer buffer = new StringBuffer();
103         builder.getFullContent().writeTo(buffer);
104         return buffer.toString();
105     }
106 
107     private String bodyContent(String mail, int position) throws Exception {
108         InputStream in = new ByteArrayInputStream(Charset.forName("us-ascii")
109                 .encode(mail).array());
110         builder.parse(in);
111         builder.to(position);
112         StringBuffer buffer = new StringBuffer();
113         builder.getMimeBodyContent().writeTo(buffer);
114         return buffer.toString();
115     }
116 
117     private void checkContentType(String contentType, String mail, int position)
118             throws Exception {
119         List headers = headers(mail, position);
120         assertEquals(1, headers.size());
121         Header header = (Header) headers.get(0);
122         assertEquals(CONTENT_TYPE, header.getName());
123         assertEquals(contentType, header.getValue());
124     }
125 
126     private List headers(String mail, int position) throws Exception {
127         InputStream in = new ByteArrayInputStream(Charset.forName("us-ascii")
128                 .encode(mail).array());
129         builder.parse(in);
130         builder.to(position);
131         return builder.getMimeHeaders();
132     }
