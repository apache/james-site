56              builder.append();

58              builder.append();
59              builder.setFile("multipart-mixed-complex.mail");
60              builder.append();

62              builder.append();
63              builder.setFile("wild-mixed-alt.mail");
64              builder.append();
65              builder.setFile("wild-mixed.mail");
66              builder.append();
67              builder.setFile("rfc822-resent.mail");
68              builder.append();
69              builder.setFile("rfc822-trace.mail");
70              builder.append();
71              builder.setFile("wild-alt-reply4.mail");
72              builder.append();
73              builder.resetFetch().setRfc822Size(true);
74              builder.fetchAllMessages();
75              builder.resetFetch().setRfc(true);
76              builder.fetchAllMessages();
77              builder.resetFetch().setRfcHeaders(true);
78              builder.fetchAllMessages();
79              builder.resetFetch().setRfcText(true);
80              builder.fetchAllMessages();
81          } finally {
82              builder.quit();
83          }
84      }

291     public static void rename(ScriptBuilder builder) throws Exception {
292         setupSearch(builder, true);
293         builder.select();
294         String originalMailbox = builder.getMailbox();
295         builder.getFetch().setFlagsFetch(true).bodyPeekHeaders(
296                 ScriptBuilder.Fetch.SELECT_HEADERS).setUid(true);
297         builder.fetchAllMessages();
298         builder.setMailbox("other").create().select().append();
299         builder.setMailbox("base").create().select();
300         builder.rename(originalMailbox, "moved").setMailbox("moved").select()
301                 .fetchAllMessages();
302         builder.setMailbox(originalMailbox).select();
303         builder.rename("other", "base");
304         builder.setMailbox(originalMailbox).select();
305         builder.setMailbox("moved").select();
306         builder.setMailbox("other").select();
307         builder.setMailbox("base").select();
308         builder.setMailbox("BOGUS").select();
309         builder.setMailbox("WHATEVER").select();
310         builder.rename("other", originalMailbox);
311         builder.setMailbox(originalMailbox).select();
312         builder.setMailbox("moved").select();
313         builder.setMailbox("other").select();
314         builder.setMailbox("base").select();
315         builder.setMailbox("BOGUS").select();
316         builder.setMailbox("WHATEVER").select();
317         builder.rename("BOGUS", "WHATEVER");
318         builder.rename(originalMailbox, "INBOX");
319         builder.rename(originalMailbox, "inbox");
320         builder.rename(originalMailbox, "Inbox");
321         builder.setMailbox(originalMailbox).select();
322         builder.setMailbox("moved").select();
323         builder.setMailbox("other").select();
324         builder.setMailbox("base").select();
325         builder.setMailbox("BOGUS").select();
326         builder.setMailbox("WHATEVER").select();
327         ;
328         builder.setMailbox("BOGUS").delete();
329         builder.setMailbox("WHATEVER").delete();
330         builder.setMailbox(originalMailbox).delete();
331         builder.setMailbox("base").delete();
332         builder.setMailbox("other").delete();
333         builder.setMailbox("moved").delete();
334         builder.quit();
335     }

459         builder.body(COMMON_LETTER).undraft().unflagged().answered().search();
460         builder.to(COMMON_LETTER).draft().flagged().answered().search();
461         builder.to(COMMON_LETTER).smaller(10000).all().draft().search();
462         builder.bcc(COMMON_LETTER).larger(1000).search();
463         builder.from(COMMON_LETTER).larger(1000).flagged().search();
464         builder.from(COMMON_LETTER).to(COMMON_LETTER).answered().flagged()
465                 .all().body(COMMON_LETTER).sentbefore(2009, 1, 1).search();
466         builder.or().openParen().from(COMMON_LETTER).to(COMMON_LETTER)
467                 .answered().flagged().all().body(COMMON_LETTER).sentbefore(
468                         2009, 1, 1).closeParen().openParen().header(HEADER,
469                         "\"\"").draft().closeParen().search();
470         builder.or().openParen().cc(COMMON_LETTER).text(COMMON_LETTER).unseen()
471                 .larger(1000).all().body(COMMON_LETTER).senton(2008, 4, 8)
472                 .closeParen().openParen().header(HEADER, "\"\"").draft()
473                 .closeParen().search();
474         builder.or().openParen().cc(COMMON_LETTER).to(COMMON_LETTER).draft()
475                 .unseen().all().text(COMMON_LETTER).sentsince(2000, 1, 1)
476                 .closeParen().openParen().header(HEADER, "\"\"").draft()
477                 .closeParen().search();
478         builder.or().openParen().or().openParen().or().openParen().not().text(
479                 COMMON_LETTER).cc(COMMON_LETTER).unseen().flagged().all().body(
480                 COMMON_LETTER).not().senton(2008, 3, 1).closeParen()
