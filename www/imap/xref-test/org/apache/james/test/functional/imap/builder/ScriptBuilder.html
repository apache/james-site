432     public ScriptBuilder fetchSection(String section) throws Exception {

512     public static final class Flags {
513         private StringBuffer flags;
514 
515         private StringBuffer msn;
516 
517         private boolean first;
518 
519         private boolean silent;
520 
521         private boolean add;
522 
523         private boolean subtract;
524 
525         public Flags() {
526             add = false;
527             subtract = false;
528             silent = false;
529             first = true;
530             flags = new StringBuffer("(");
531             msn = new StringBuffer();
532         }
533 
534         public Flags msn(long number) {
535             msn.append(number);
536             msn.append(' ');
537             return this;
538         }
539 
540         public Flags range(long low, long high) {
541             msn.append(low);
542             msn.append(':');
543             msn.append(high);
544             msn.append(' ');
545             return this;
546         }
547 
548         public Flags rangeTill(long number) {
549             msn.append("*:");
550             msn.append(number);
551             msn.append(' ');
552             return this;
553         }
554 
555         public Flags rangeFrom(long number) {
556             msn.append(number);
557             msn.append(":* ");
558             return this;
559         }
560 
561         public Flags add() {
562             add = true;
563             subtract = false;
564             return this;
565         }
566 
567         public Flags subtract() {
568             add = false;
569             subtract = true;
570             return this;
571         }
572 
573         public Flags silent() {
574             silent = true;
575             return this;
576         }
577 
578         public Flags deleted() {
579             return append("\\DELETED");
580         }
581 
582         public Flags flagged() {
583             return append("\\FLAGGED");
584         }
585 
586         public Flags answered() {
587             return append("\\ANSWERED");
588         }
589 
590         public Flags seen() {
591             return append("\\SEEN");
592         }
593 
594         public Flags draft() {
595             return append("\\DRAFT");
596         }
597 
598         public String command() {
599             String flags;
600             if (add) {
601                 flags = " +FLAGS ";
602             } else if (subtract) {
603                 flags = " -FLAGS ";
604             } else {
605                 flags = " FLAGS ";
606             }
607             if (silent) {
608                 flags = flags + ".SILENT";
609             }
610             return "STORE " + msn + flags + this.flags + ")";
611         }
612 
613         private Flags append(String term) {
614             if (first) {
615                 first = false;
616             } else {
617                 flags.append(' ');
618             }
619             flags.append(term);
620             return this;
621         }
622     }

655         }

657         private Search append(long term) {

660 
661         private Search append(String term) {
662             if (first) {
663                 first = false;
664             } else {
665                 buffer.append(' ');
666             }
667             buffer.append(term);
668             return this;
669         }
670 
671         private Search date(int year, int month, int day) {
672             append(day);
673             switch (month) {
674                 case 1:
675                     buffer.append("-Jan-");
676                     break;
677                 case 2:
678                     buffer.append("-Feb-");
679                     break;
680                 case 3:
681                     buffer.append("-Mar-");
682                     break;
683                 case 4:
684                     buffer.append("-Apr-");
685                     break;
686                 case 5:
687                     buffer.append("-May-");
688                     break;
689                 case 6:
690                     buffer.append("-Jun-");
691                     break;
692                 case 7:
693                     buffer.append("-Jul-");
694                     break;
695                 case 8:
696                     buffer.append("-Aug-");
697                     break;
698                 case 9:
699                     buffer.append("-Sep-");
700                     break;
701                 case 10:
702                     buffer.append("-Oct-");
703                     break;
704                 case 11:
705                     buffer.append("-Nov-");
706                     break;
707                 case 12:
708                     buffer.append("-Dec-");
709                     break;
710             }
711             buffer.append(year);
712             return this;
713         }
