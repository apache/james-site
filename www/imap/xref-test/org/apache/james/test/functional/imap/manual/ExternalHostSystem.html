29  public class ExternalHostSystem implements HostSystem {
30  
31      public static final HostSystem createLocalImap() {
32          final ExternalHostSystem result = new ExternalHostSystem("localhost",
33                  143, new SystemLoggingMonitor());
34          return result;
35      }

37      private final InetSocketAddress address;
38  
39      private final Monitor monitor;
40  
41      public ExternalHostSystem(final String host, final int port,
42              final Monitor monitor) {
43          super();
44          this.address = new InetSocketAddress(host, port);
45          this.monitor = monitor;
46      }
47  
48      public boolean addUser(String user, String password) throws Exception {
49          monitor.note("Please ensure user '" + user + "' with password '"
50                  + password + "' exists.");
51          return true;
52      }

54      public Session newSession(Continuation continuation) throws Exception {
55          final SocketChannel channel = SocketChannel.open(address);
56          channel.configureBlocking(false);
57          final SessionImpl result = new SessionImpl(channel, monitor);
58          return result;
59      }

61      public void reset() throws Exception {
62          monitor.note("Please reset system.");
63      }

65      public interface Monitor {
66          void note(String message);
67      }
68  
69      public static final class SystemLoggingMonitor implements Monitor {
70  
71          public void note(String message) {
72              System.out.println(message);
73          }
74  
75      }
76  
77      private final static class SessionImpl implements Session {
78  
79          private static final byte[] CRLF = { '\r', '\n' };
80  
81          private final SocketChannel socket;
82  
83          private final Monitor monitor;
84  
85          private final ByteBuffer readBuffer;
86  
87          private final Charset ascii;
88  
89          private final ByteBuffer lineEndBuffer;
90  
91          private boolean first = true;
92  
93          public SessionImpl(final SocketChannel socket, final Monitor monitor) {
94              super();
95              this.socket = socket;
96              this.monitor = monitor;
97              readBuffer = ByteBuffer.allocateDirect(2048);
98              ascii = Charset.forName("US-ASCII");
99              lineEndBuffer = ByteBuffer.wrap(CRLF);
100         }
101 
102         public String readLine() throws Exception {
103             StringBuffer buffer = new StringBuffer();
104             readlineInto(buffer);
105             final String result;
106             if (first) {
107                 // fake shabang
108                 monitor.note("<-" + buffer.toString());
109                 result = "* OK IMAP4rev1 Server ready";
110                 first = false;
111             } else {
112                 result = buffer.toString();
113                 monitor.note("<-" + result);
114             }
115             return result;
116         }
117 
118         private void readlineInto(StringBuffer buffer) throws Exception {
119             while (socket.read(readBuffer) == 0)
120                 ;
121             readBuffer.flip();
122             while (readOneMore(buffer))
123                 ;
124             readBuffer.compact();
125         }
126 
127         private boolean readOneMore(StringBuffer buffer) throws Exception {
128             final boolean result;
129             if (readBuffer.hasRemaining()) {
130                 char next = (char) readBuffer.get();
131                 if (next == '\n') {
132                     result = false;
133                 } else if (next == '\r') {
134                     result = true;
135                 } else {
136                     buffer.append(next);
137                     result = true;
138                 }
139             } else {
140                 readBuffer.clear();
141                 readlineInto(buffer);
142                 result = true;
143             }
144             return result;
145         }
146 
147         public void start() throws Exception {
148             while (!socket.finishConnect()) {
149                 monitor.note("connecting...");
150                 Thread.sleep(10);
151             }
152         }
153 
154         public void stop() throws Exception {
155             monitor.note("closing");
156             socket.close();
157         }
158 
159         public void writeLine(String line) throws Exception {
160             monitor.note("-> " + line);
161             ByteBuffer writeBuffer = ascii.encode(line);
162             while (writeBuffer.hasRemaining()) {
163                 socket.write(writeBuffer);
164             }
165             lineEndBuffer.rewind();
166             while (lineEndBuffer.hasRemaining()) {
167                 socket.write(lineEndBuffer);
168             }
169         }
170     }
