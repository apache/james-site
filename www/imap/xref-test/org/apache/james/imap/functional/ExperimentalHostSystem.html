154     static class ByteBufferInputStream extends InputStream {
155 
156         ByteBuffer buffer = ByteBuffer.allocate(16384);
157 
158         CharsetEncoder encoder = Charset.forName("ASCII").newEncoder();
159 
160         boolean readLast = true;
161 
162         public int read() throws IOException {
163             if (!readLast) {
164                 readLast = true;
165                 buffer.flip();
166             }
167             int result = -1;
168             if (buffer.hasRemaining()) {
169                 result = buffer.get();
170             }
171             return result;
172         }
173 
174         public void nextLine(String line) {
175             if (buffer.position() > 0 && readLast) {
176                 buffer.compact();
177             }
178             encoder.encode(CharBuffer.wrap(line), buffer, true);
179             buffer.put((byte) '\r');
180             buffer.put((byte) '\n');
181             readLast = false;
182         }
183     }

185     static class ByteBufferOutputStream extends OutputStream {
186         ByteBuffer buffer = ByteBuffer.allocate(65536);
187 
188         Charset ascii = Charset.forName("ASCII");
189 
190         Continuation continuation;
191 
192         boolean matchPlus = false;
193 
194         boolean matchCR = false;
195 
196         boolean matchLF = false;
197 
198         public ByteBufferOutputStream(Continuation continuation) {
199             this.continuation = continuation;
200         }
201 
202         public void write(String message) throws IOException {
203             ascii.newEncoder().encode(CharBuffer.wrap(message), buffer, true);
204         }
205 
206         public void write(int b) throws IOException {
207             buffer.put((byte) b);
208             if (b == '\n' && matchPlus && matchCR && matchLF) {
209                 matchPlus = false;
210                 matchCR = false;
211                 matchLF = false;
212                 continuation.doContinue();
213             } else if (b == '\n') {
214                 matchLF = true;
215                 matchPlus = false;
216                 matchCR = false;
217             } else if (b == '+' && matchLF) {
218                 matchPlus = true;
219                 matchCR = false;
220             } else if (b == '\r' && matchPlus && matchLF) {
221                 matchCR = true;
222             } else {
223                 matchPlus = false;
224                 matchCR = false;
225                 matchLF = false;
226             }
227         }
228 
229         public String nextLine() throws Exception {
230             buffer.flip();
231             byte last = 0;
232             while (buffer.hasRemaining()) {
233                 byte next = buffer.get();
234                 if (last == '\r' && next == '\n') {
235                     break;
236                 }
237                 last = next;
238             }
239             final ByteBuffer readOnlyBuffer = buffer.asReadOnlyBuffer();
240             readOnlyBuffer.flip();
241             int limit = readOnlyBuffer.limit() - 2;
242             if (limit < 0) {
243                 limit = 0;
244             }
245             readOnlyBuffer.limit(limit);
246             String result = ascii.decode(readOnlyBuffer).toString();
247             buffer.compact();
248             return result;
249         }
250     }
