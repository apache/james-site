66                  .proxy());

73      public void testShouldParseSilentDraftFlagged() throws Exception {
74          IdRange[] ranges = { new IdRange(1) };
75          Flags flags = new Flags();
76          flags.add(Flags.Flag.DRAFT);
77          flags.add(Flags.Flag.FLAGGED);
78          check("1 FLAGS.SILENT (\\Draft \\Flagged)\r\n", ranges, true, null,
79                  flags, false, "A01");
80      }
81  
82      private void check(String input, final IdRange[] idSet, boolean silent,
83              Boolean sign, final Flags flags, final boolean useUids, String tag)
84              throws Exception {
85          ImapRequestLineReader reader = new ImapRequestLineReader(
86                  new ByteArrayInputStream(input.getBytes("US-ASCII")),
87                  new ByteArrayOutputStream());
88  
89          Constraint[] constraints = { eq(command), eq(idSet), eq(silent),
90                  eq(sign), eq(flags), eq(useUids), same(tag) };
91          mockMessageFactory.expects(once()).method("createStoreMessage").with(
92                  constraints).will(returnValue(message));
93          parser.decode(command, reader, tag, useUids);
94      }
