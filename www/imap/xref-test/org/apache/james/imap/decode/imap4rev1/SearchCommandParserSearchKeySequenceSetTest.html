72      public void testAllNumbers() throws Exception {
73  
74          IdRange[] range = { new IdRange(2), new IdRange(4), new IdRange(9),
75                  new IdRange(16), new IdRange(25), new IdRange(36),
76                  new IdRange(49), new IdRange(64), new IdRange(81),
77                  new IdRange(100) };
78          check("2,4,9,16,25,36,49,64,81,100", range);
79      }
80  
81      public void testEndStar() throws Exception {
82          IdRange[] range = { new IdRange(8), new IdRange(9, 10),
83                  new IdRange(17), new IdRange(100, Long.MAX_VALUE) };
84          check("8,9:10,17,100:*", range);
85      }
86  
87      public void testStartStar() throws Exception {
88          IdRange[] range = { new IdRange(Long.MAX_VALUE, 9), new IdRange(15),
89                  new IdRange(799, 820) };
90          check("*:9,15,799:820", range);
91      }
92  
93      private void check(String sequence, IdRange[] range) throws Exception {
94          checkUid(sequence, range);
95          checkSequence(sequence, range);
96      }
97  
98      private void checkUid(String sequence, IdRange[] range) throws Exception {
99          SearchKey key = SearchKey.buildUidSet(range);
100         checkValid("UID " + sequence, key);
101         checkValid("uid " + sequence, key);
102         checkValid("Uid " + sequence, key);
103     }
104 
105     private void checkSequence(String sequence, IdRange[] range)
106             throws Exception {
107         SearchKey key = SearchKey.buildSequenceSet(range);
108         checkValid(sequence, key);
109         checkValid(sequence, key);
110         checkValid(sequence, key);
111     }
112 
113     private void checkValid(String input, final SearchKey key) throws Exception {
114         input = input + "\r\n";
115         ImapRequestLineReader reader = new ImapRequestLineReader(
116                 new ByteArrayInputStream(input.getBytes("US-ASCII")),
117                 new ByteArrayOutputStream());
118 
119         final SearchKey searchKey = parser.searchKey(reader, null, false);
120         assertEquals(key, searchKey);
121     }
