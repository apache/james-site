225     private void expectOk() {
226         StatusResponse response = (StatusResponse) statusResponse.proxy();
227         serverResponseFactory.expects(once()).method("taggedOk").with(eq(TAG),
228                 same(imapCommand), eq(HumanReadableTextKey.COMPLETED)).will(
229                 returnValue(response));
230         responder.expects(once()).method("respond").with(same(response));
231     }
232 
233     private void expectSubscriptions() {
234         session.expects(once()).method("getAttribute").with(
235                 eq(ImapSessionUtils.MAILBOX_MANAGER_ATTRIBUTE_SESSION_KEY))
236                 .will(returnValue(manager.proxy()));
237         session.expects(once()).method("getAttribute").with(
238                 eq(ImapSessionUtils.MAILBOX_USER_ATTRIBUTE_SESSION_KEY)).will(
239                 returnValue(USER));

241         manager.expects(once()).method("subscriptions").with(eq(USER)).will(
242                 returnValue(subscriptions));
