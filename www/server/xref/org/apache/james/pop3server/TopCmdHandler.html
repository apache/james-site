33  /***
34    * Handles TOP command
35    */
36  public class TopCmdHandler implements CommandHandler {
37  
38      /***
39       * @see org.apache.james.pop3server.CommandHandler#onCommand(POP3Session)
40       */
41      public void onCommand(POP3Session session) {
42          doTOP(session,session.getCommandArgument());
43      }

45      /***
46       * Handler method called upon receipt of a TOP command.
47       * This command retrieves the top N lines of a specified
48       * message in the mailbox.
49       *
50       * The expected command format is
51       *  TOP [mail message number] [number of lines to return]
52       *
53       * @param command the command parsed by the parseCommand method
54       * @param argument the first argument parsed by the parseCommand method
55       */
56      private void doTOP(POP3Session session,String arguments) {
57          String argument = "";
58          String argument1 = "";
59          int pos = arguments.indexOf(" ");
60          if (pos > 0) {
61              argument = arguments.substring(0,pos);
62              argument1 = arguments.substring(pos+1);
63          }
64          String responseString = null;
65          if (session.getHandlerState() == POP3Handler.TRANSACTION) {
66              int num = 0;
67              int lines = 0;
68              try {
69                  num = Integer.parseInt(argument);
70                  lines = Integer.parseInt(argument1);
71              } catch (NumberFormatException nfe) {
72                  responseString = POP3Handler.ERR_RESPONSE + " Usage: TOP [mail number] [Line number]";
73                  session.writeResponse(responseString);
74                  return;
75              }
76              try {
77                  Mail mc = (Mail) session.getUserMailbox().get(num);
78                  if (mc != POP3Handler.DELETED) {
79                      responseString = POP3Handler.OK_RESPONSE + " Message follows";
80                      session.writeResponse(responseString);
81                      try {
82                          for (Enumeration e = mc.getMessage().getAllHeaderLines(); e.hasMoreElements(); ) {
83                              session.writeResponse(e.nextElement().toString());
84                          }
85                          session.writeResponse("");
86                          ExtraDotOutputStream edouts =
87                                  new ExtraDotOutputStream(session.getOutputStream());
88                          OutputStream nouts = new BytesWrittenResetOutputStream(edouts,
89                                                                    session.getWatchdog(),
90                                                                    session.getConfigurationData().getResetLength());
91                          writeMessageContentTo(mc.getMessage(),nouts,lines);
92                          nouts.flush();
93                          edouts.checkCRLFTerminator();
94                          edouts.flush();
95                      } finally {
96                          session.writeResponse(".");
97                      }
98                  } else {
99                      StringBuffer responseBuffer =
100                         new StringBuffer(64)
101                                 .append(POP3Handler.ERR_RESPONSE)
102                                 .append(" Message (")
103                                 .append(num)
104                                 .append(") already deleted.");
105                     responseString = responseBuffer.toString();
106                     session.writeResponse(responseString);
107                 }
108             } catch (IOException ioe) {
109                 responseString = POP3Handler.ERR_RESPONSE + " Error while retrieving message.";
110                 session.writeResponse(responseString);
111             } catch (MessagingException me) {
112                 responseString = POP3Handler.ERR_RESPONSE + " Error while retrieving message.";
113                 session.writeResponse(responseString);
114             } catch (IndexOutOfBoundsException iob) {
115                 StringBuffer exceptionBuffer =
116                     new StringBuffer(64)
117                             .append(POP3Handler.ERR_RESPONSE)
118                             .append(" Message (")
119                             .append(num)
120                             .append(") does not exist.");
121                 responseString = exceptionBuffer.toString();
122                 session.writeResponse(responseString);
123             }
124         } else {
125             responseString = POP3Handler.ERR_RESPONSE;
126             session.writeResponse(responseString);
127         }
128     }

131     /***
132      * Writes the content of the message, up to a total number of lines, out to 
133      * an OutputStream.
134      *
135      * @param out the OutputStream to which to write the content
136      * @param lines the number of lines to write to the stream
137      *
138      * @throws MessagingException if the MimeMessage is not set for this MailImpl
139      * @throws IOException if an error occurs while reading or writing from the stream
140      */
141     public void writeMessageContentTo(MimeMessage message, OutputStream out, int lines)
142         throws IOException, MessagingException {
143         String line;
144         BufferedReader br;
145         if (message != null) {
146             br = new BufferedReader(new InputStreamReader(message.getRawInputStream()));
147             try {
148                 while (lines-- > 0) {
149                     if ((line = br.readLine()) == null) {
150                         break;
151                     }
152                     line += "\r\n";
153                     out.write(line.getBytes());
154                 }
155             } finally {
156                 br.close();
157             }
158         } else {
159             throw new MessagingException("No message set for this MailImpl.");
160         }
161     }
