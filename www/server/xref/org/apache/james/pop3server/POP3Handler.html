167     /***
168      * @see org.apache.james.core.AbstractJamesHandler#handleProtocol()
169      */
170     protected void handleProtocol() throws IOException {
171         handlerState = AUTHENTICATION_READY;
172         authenticatedUser = "unknown";
173 
174         sessionEnded = false;
175         resetState();
176 
177         // Initially greet the connector
178         // Format is:  Sat, 24 Jan 1998 13:16:09 -0500
179         responseBuffer.append(OK_RESPONSE)
180                     .append(" ")
181                     .append(theConfigData.getHelloName())
182                     .append(" POP3 server (")
183                     .append(POP3Handler.softwaretype)
184                     .append(") ready ");
185         String responseString = clearResponseBuffer();
186         writeLoggedFlushedResponse(responseString);
187 
188         //Session started - RUN all connect handlers
189         List connectHandlers = handlerChain.getConnectHandlers();
190         if(connectHandlers != null) {
191             int count = connectHandlers.size();
192             for(int i = 0; i < count; i++) {
193                 ((ConnectHandler)connectHandlers.get(i)).onConnect(this);
194                 if(sessionEnded) {
195                     break;
196                 }
197             }
198         }
199 
200         
201         theWatchdog.start();
202         while(!sessionEnded) {
203           //Reset the current command values
204           curCommandName = null;
205           curCommandArgument = null;
206           mode = COMMAND_MODE;
207 
208           //parse the command
209           String cmdString =  readCommandLine();
210           if (cmdString == null) {
211               break;
212           }
213           int spaceIndex = cmdString.indexOf(" ");
214           if (spaceIndex > 0) {
215               curCommandName = cmdString.substring(0, spaceIndex);
216               curCommandArgument = cmdString.substring(spaceIndex + 1);
217           } else {
218               curCommandName = cmdString;
219           }
220           curCommandName = curCommandName.toUpperCase(Locale.US);
221 
222           if (getLogger().isDebugEnabled()) {
223               // Don't display password in logger
224               if (!curCommandName.equals("PASS")) {
225                   getLogger().debug("Command received: " + cmdString);
226               } else {
227                   getLogger().debug("Command received: PASS <password omitted>");
228               }
229           }
230 
231           //fetch the command handlers registered to the command
232           List commandHandlers = handlerChain.getCommandHandlers(curCommandName);
233           if(commandHandlers == null) {
234               //end the session
235               break;
236           } else {
237               int count = commandHandlers.size();
238               for(int i = 0; i < count; i++) {
239                   ((CommandHandler)commandHandlers.get(i)).onCommand(this);
240                   theWatchdog.reset();
241                   //if the response is received, stop processing of command handlers
242                   if(mode != COMMAND_MODE) {
243                       break;
244                   }
245               }
246 
247           }
248         }
249         theWatchdog.stop();
250         if (getLogger().isInfoEnabled()) {
251             StringBuffer logBuffer =
252                 new StringBuffer(128)
253                     .append("Connection for ")
254                     .append(getUser())
255                     .append(" from ")
256                     .append(remoteHost)
257                     .append(" (")
258                     .append(remoteIP)
259                     .append(") closed.");
260             getLogger().info(logBuffer.toString());
261         }
262     }

277     /***
278      * Resets the handler data to a basic state.
279      */
280     protected void resetHandler() {
281         // Clear user data
282         authenticatedUser = null;
283         userInbox = null;
284         if (userMailbox != null) {
285             Iterator i = userMailbox.iterator();
286             while (i.hasNext()) {
287                 ContainerUtil.dispose(i.next());
288             }
289             userMailbox.clear();
290             userMailbox = null;
291         }
292 
293         if (backupUserMailbox != null) {
294             backupUserMailbox.clear();
295             backupUserMailbox = null;
296         }
297 
298         // Clear config data
299         theConfigData = null;
300     }

302     /***
303      * Implements a "stat".  If the handler is currently in
304      * a transaction state, this amounts to a rollback of the
305      * mailbox contents to the beginning of the transaction.
306      * This method is also called when first entering the
307      * transaction state to initialize the handler copies of the
308      * user inbox.
309      *
310      */
311     public void stat() {
312         userMailbox = new ArrayList();
313         userMailbox.add(DELETED);
314         try {
315             for (Iterator it = userInbox.list(); it.hasNext(); ) {
316                 String key = (String) it.next();
317                 Mail mc = userInbox.retrieve(key);
318                 // Retrieve can return null if the mail is no longer in the store.
319                 // In this case we simply continue to the next key
320                 if (mc == null) {
321                     continue;
322                 }
323                 userMailbox.add(mc);
324             }
325         } catch(MessagingException e) {
326             // In the event of an exception being thrown there may or may not be anything in userMailbox
327             getLogger().error("Unable to STAT mail box ", e);
328         }
329         finally {
330             backupUserMailbox = (ArrayList) userMailbox.clone();
331         }
332     }

334     /***
335      * Reads a line of characters off the command line.
336      *
337      * @return the trimmed input line
338      * @throws IOException if an exception is generated reading in the input characters
339      */
340     public final String readCommandLine() throws IOException {
341         for (;;) try {
342             String commandLine = inReader.readLine();
343             if (commandLine != null) {
344                 commandLine = commandLine.trim();
345             }
346             return commandLine;
347         } catch (CRLFTerminatedReader.TerminationException te) {
348             writeLoggedFlushedResponse("-ERR Syntax error at character position " + te.position() + ". CR and LF must be CRLF paired.  See RFC 1939 #3.");
349         }
350     }
