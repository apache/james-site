29  /***
30   * <p>Accepts POP3 connections on a server socket and dispatches them to POP3Handlers.</p>
31   *
32   * <p>Also responsible for loading and parsing POP3 specific configuration.</p>
33   *
34   * @version 1.0.0, 24/04/1999
35   */
36  public class POP3Server extends AbstractJamesService implements POP3ServerMBean {
37  
38      /***
39       * The handler chain - POP3handlers can lookup handlerchain to obtain
40       * Command handlers , Message handlers and connection handlers
41       */
42      POP3HandlerChain handlerChain = new POP3HandlerChain();
43  
44      /***
45       * The internal mail server service
46       */
47      private MailServer mailServer;
48  
49      /***
50       * The user repository for this server - used to authenticate users.
51       */
52      private UsersRepository users;
53  
54      /***
55       * The number of bytes to read before resetting
56       * the connection timeout timer.  Defaults to
57       * 20 KB.
58       */
59      private int lengthReset = 20 * 1024;
60  
61      /***
62       * The configuration data to be passed to the handler
63       */
64      private POP3HandlerConfigurationData theConfigData
65          = new POP3HandlerConfigurationDataImpl();
66  
67      private ServiceManager serviceManager;
68  
69      public void setMailServer(MailServer mailServer) {
70          this.mailServer = mailServer;
71      }
72  
73      public void setUsers(UsersRepository users) {
74          this.users = users;
75      }
76  
77      /***
78       * @see org.apache.avalon.framework.service.Serviceable#service(ServiceManager) 
79       */
80      public void service( final ServiceManager componentManager )
81          throws ServiceException {
82          super.service(componentManager);
83          serviceManager = componentManager;
84          MailServer mailServer = (MailServer)componentManager.lookup( MailServer.ROLE );
85          setMailServer(mailServer);
86          UsersRepository users = (UsersRepository)componentManager.lookup( UsersRepository.ROLE );
87          setUsers(users);
88      }
89  
90      /***
91       * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
92       */
93      public void configure(final Configuration configuration) throws ConfigurationException {
94          super.configure(configuration);
95          if (isEnabled()) {
96              Configuration handlerConfiguration = configuration.getChild("handler");
97              lengthReset = handlerConfiguration.getChild("lengthReset").getValueAsInteger(lengthReset);
98              if (getLogger().isInfoEnabled()) {
99                  getLogger().info("The idle timeout will be reset every " + lengthReset + " bytes.");
100             }
101             //set the logger
102             ContainerUtil.enableLogging(handlerChain,getLogger());
103 
104             try {
105                 ContainerUtil.service(handlerChain,serviceManager);
106             } catch (ServiceException e) {
107                 if (getLogger().isErrorEnabled()) {
108                     getLogger().error("Failed to service handlerChain",e);
109                 }
110                 throw new ConfigurationException("Failed to service handlerChain");
111             }
112             
113             //read from the XML configuration and create and configure each of the handlers
114             ContainerUtil.configure(handlerChain,handlerConfiguration.getChild("handlerchain"));
115 
116         }
117     }
