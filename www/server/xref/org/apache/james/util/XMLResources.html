35  /*
36   * This is derived from the SQLResources.java class that we have been
37   * using to provide SQL strings based upon the particular drive being
38   * used.
39   *
40   * The format is indentical to the format used by SQLResources.  The
41   * only difference are the names of the elements and attributes.  A
42   * mapping is as follows:
43   *
44   *      sqlResources            xmlResources
45   *      ------------            ------------
46   *      sqlResources            resources
47   *      dbMatchers              matchers
48   *      dbMatcher               matcher
49   *      db                      for
50   *      databaseProductName     match
51   *      sqlDefs                 group
52   *      sql                     resource
53   *
54   * This class provides String resources defined in XML.  Resources are
55   * organized into groups, and identified by name.  For each resource
56   * there can be a standard value, and custom values matched by regular
57   * expression.
58   *
59   * The structure of the XML file is:
60   *
61   *  &lt;resources&gt;
62   *    &lt;matchers&gt;
63   *      &lt;matcher for="<i>label</i>" match="<i>regular expression</i>"/&gt;
64   *      ...
65   *    &lt;/matchers&gt;
66   *    &lt;group name="<i>group name</i>"&gt;
67   *      &lt;resource name="<i>resouce name</i>" [for="<i>match label</i>"]&gt;<i>text, including ${placeholders}, which will be replaced at runtime.</i>&lt;/resource&gt;
68   *      ...
69   *    &lt;/group&gt;
70   *    <i>... more &lt;group&gt; elements ...</i>
71   *  &lt;/resources&gt;
72   * 
73   */
74  public class XMLResources
75  {

77       * A map of statement types to resource strings
78       */
79      private Map m_resource = new HashMap();

89      private Perl5Util m_perl5Util = new Perl5Util();

91      /***
92       * Configures an XMLResources object to provide string statements from a file.
93       * 
94       * Parameters encoded as $(parameter} in the input file are
95       * replace by values from the parameters Map, if the named parameter exists.
96       * Parameter values may also be specified in the resourceSection element.
97       * 
98       * @param xmlFile    the input file containing the string definitions
99       * @param group      xml element containing the strings to be used
100      * @param select     if customized elements exist for this value, use them instead of the default
101      * @param configParameters a map of parameters (name-value string pairs) which are
102      *                   replaced where found in the input strings
103      */
104     public void init(File xmlFile, String group,
105                      String select, Map configParameters)
106         throws Exception
107     {
108         // Parse the xmlFile as an XML document.
109         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
110         DocumentBuilder builder = factory.newDocumentBuilder();
111         Document doc = builder.parse(xmlFile);
112 
113         // First process the matchers, to select the statements to use.
114         Element matcherElement = 
115             (Element)(doc.getElementsByTagName("matchers").item(0));
116         String selectTag = null;
117         if ( matcherElement != null ) {
118             selectTag = match(select, matcherElement);
119             m_perl5Util = null;     // release the PERL matcher!
120         }
121 
122         // Now get the section defining strings for the group required.
123         NodeList sections = doc.getElementsByTagName("group");
124         int sectionsCount = sections.getLength();
125         Element sectionElement = null;
126         for (int i = 0; i < sectionsCount; i++ ) {
127             sectionElement = (Element)(sections.item(i));
128             String sectionName = sectionElement.getAttribute("name");
129             if ( sectionName != null && sectionName.equals(group) ) {
130                 break;
131             }
132 
133         }
134         if ( sectionElement == null ) {
135             StringBuffer exceptionBuffer =
136                 new StringBuffer(64)
137                         .append("Error loading string definition file. ")
138                         .append("The element named \'")
139                         .append(group)
140                         .append("\' does not exist.");
141             throw new RuntimeException(exceptionBuffer.toString());
142         }
143 
144         // Get parameters defined within the file as defaults,
145         // and use supplied parameters as overrides.
146         Map parameters = new HashMap();
147         // First read from the <params> element, if it exists.
148         Element parametersElement = 
149             (Element)(sectionElement.getElementsByTagName("parameters").item(0));
150         if ( parametersElement != null ) {
151             NamedNodeMap params = parametersElement.getAttributes();
152             int paramCount = params.getLength();
153             for (int i = 0; i < paramCount; i++ ) {
154                 Attr param = (Attr)params.item(i);
155                 String paramName = param.getName();
156                 String paramValue = param.getValue();
157                 parameters.put(paramName, paramValue);
158             }
159         }
160         // Then copy in the parameters supplied with the call.
161         parameters.putAll(configParameters);
162 
163         // 2 maps - one for storing default statements,
164         // the other for statements with a "for" attribute matching this 
165         // connection.
166         Map defaultStrings = new HashMap();
167         Map selectTagStrings = new HashMap();
168 
169         // Process each string resource, replacing string parameters,
170         // and adding to the appropriate map..
171         NodeList resDefs = sectionElement.getElementsByTagName("resource");
172         int resCount = resDefs.getLength();
173         for ( int i = 0; i < resCount; i++ ) {
174             // See if this needs to be processed (is default or product specific)
175             Element resElement = (Element)(resDefs.item(i));
176             String resSelect = resElement.getAttribute("for");
177             Map resMap;
178             if ( resSelect.equals("")) {
179                 // default
180                 resMap = defaultStrings;
181             }
182             else if (resSelect.equals(selectTag) ) {
183                 // Specific to this product
184                 resMap = selectTagStrings;
185             }
186             else {
187                 // for a different product
188                 continue;
189             }
190 
191             // Get the key and value for this string resource.
192             String resKey = resElement.getAttribute("name");
193             if ( resKey == null ) {
194                 // ignore elements without a "name" attribute.
195                 continue;
196             }
197             String resString = resElement.getFirstChild().getNodeValue();
198 
199             // Do parameter replacements for this string resource.
200             Iterator paramNames = parameters.keySet().iterator();
201             while ( paramNames.hasNext() ) {
202                 String paramName = (String)paramNames.next();
203                 String paramValue = (String)parameters.get(paramName);
204 
205                 StringBuffer replaceBuffer =
206                     new StringBuffer(64)
207                             .append("${")
208                             .append(paramName)
209                             .append("}");
210                 resString = substituteSubString(resString, replaceBuffer.toString(), paramValue);
211             }
212 
213             // See if we already have registered a string of this value
214             String shared = (String) stringTable.get(resString);
215             // If not, register it -- we will use it next time
216             if (shared == null) {
217                 stringTable.put(resString, resString);
218             } else {
219                 resString = shared;
220             }
221 
222             // Add to the resMap - either the "default" or the "product" map
223             resMap.put(resKey, resString);
224         }
225 
226         // Copy in default strings, then overwrite product-specific ones.
227         m_resource.putAll(defaultStrings);
228         m_resource.putAll(selectTagStrings);
229     }

231     /***
232      * Compares the "select" value against a set of regular expressions
233      * defined in XML.  The first successful match defines the name of a
234      * selector tag. This value is then used to choose the specific
235      * expressions to use.
236      *
237      * @param select the String to be checked
238      * @param matchersElement the XML element containing selector patterns
239      *
240      * @return the selector tag that will be used to select custom resources
241      *
242      */
243     private String match(String select, Element matchersElement)
244         throws MalformedPerl5PatternException
245     {
246         String selectTagName = select;
247     
248         NodeList matchers = matchersElement.getElementsByTagName("matcher");
249         for ( int i = 0; i < matchers.getLength(); i++ ) {
250             // Get the values for this matcher element.
251             Element matcher = (Element)matchers.item(i);
252             String matchName = matcher.getAttribute("for");
253             StringBuffer selectTagPatternBuffer =
254                 new StringBuffer(64)
255                         .append("/")
256                         .append(matcher.getAttribute("match"))
257                         .append("/i");
258 
259             // If the select string matches the pattern, use the match
260             // name from this matcher.
261             if ( m_perl5Util.match(selectTagPatternBuffer.toString(), selectTagName) ) {
262                 return matchName;
263             }
264         }
265         return null;
266     }

268     /***
269      * Replace substrings of one string with another string and return altered string.
270      * @param input input string
271      * @param find the string to replace
272      * @param replace the string to replace with
273      * @return the substituted string
274      */
275     static private String substituteSubString( String input, 
276                                                String find,
277                                                String replace )
278     {
279         int find_length = find.length();
280         int replace_length = replace.length();
281 
282         StringBuffer output = new StringBuffer(input);
283         int index = input.indexOf(find);
284         int outputOffset = 0;
285 
286         while ( index > -1 ) {
287             output.replace(index + outputOffset, index + outputOffset + find_length, replace);
288             outputOffset = outputOffset + (replace_length - find_length);
289 
290             index = input.indexOf(find, index + find_length);
291         }
292 
293         String result = output.toString();
294         return result;
295     }

309     /***
310      * Returns a named string for the specified connection, replacing
311      * parameters with the values set.
312      * 
313      * @param name     the name of the String resource required.
314      * @param required true if the resource is required
315      * @return the requested resource
316      * @throws ConfigurationException
317      *         if a required resource cannot be found.
318      */
319     public String getString(String name, boolean required)
320     {
321         String str = getString(name);
322 
323         if (str == null && required) {
324             StringBuffer exceptionBuffer =
325                 new StringBuffer(64)
326                         .append("Required String resource: '")
327                         .append(name)
328                         .append("' was not found.");
329             throw new RuntimeException(exceptionBuffer.toString());
330         }
331         return str;
332     }

347     /***
348      * Returns a named string, replacing parameters with the values set.
349      * 
350      * @param name          the name of the String resource required.
351      * @param parameters    a map of parameters (name-value string pairs) which are
352      *                      replaced where found in the input strings
353      * @return the requested resource
354      */
355     static public String replaceParameters(String str, Map parameters)
356     {
357         // Do parameter replacements for this string resource.
358         Iterator paramNames = parameters.keySet().iterator();
359         while ( paramNames.hasNext() ) {
360             String paramName = (String)paramNames.next();
361             String paramValue = (String)parameters.get(paramName);
362 
363             StringBuffer replaceBuffer = new StringBuffer(64).append("${").append(paramName).append("}");
364             str = substituteSubString(str, replaceBuffer.toString(), paramValue);
365         }
366 
367         return str;
368     }
