41  /***
42   * Represents a single server socket managed by a connection manager.
43   * The connection manager will spawn a single ServerConnection for each
44   * server socket that the connection manager is managing.
45   *
46   */
47  public class ServerConnection extends AbstractLogEnabled
48      implements Initializable, Runnable {
49  
50      /***
51       * This is a hack to deal with the fact that there appears to be
52       * no platform-independent way to break out of a ServerSocket
53       * accept() call.  On some platforms closing either the ServerSocket
54       * itself, or its associated InputStream, causes the accept
55       * method to exit.  Unfortunately, this behavior is not consistent
56       * across platforms.  The deal with this, we introduce a polling
57       * loop of 20 seconds for the server socket.  This introduces a
58       * cost across platforms, but is necessary to maintain cross-platform
59       * functionality.
60       */
61      private static int POLLING_INTERVAL = 20*1000;

135     public ServerConnection(ServerSocket serverSocket,
136                             ConnectionHandlerFactory handlerFactory,
137                             ThreadPool threadPool,
138                             int timeout,
139                             int maxOpenConn, int maxOpenConnPerIP) {
140         this.serverSocket = serverSocket;
141         this.handlerFactory = handlerFactory;
142         connThreadPool = threadPool;
143         socketTimeout = timeout;
144         this.maxOpenConn = maxOpenConn;
145         this.maxOpenConnPerIP = maxOpenConnPerIP;
146     }

157     /***
158      * The dispose operation is called by the owning ConnectionManager
159      * at the end of its lifecycle.  Cleans up the server connection, forcing
160      * everything to finish.
161      */
162     public void dispose() {
163         if (getLogger().isDebugEnabled()) {
164             getLogger().debug("Disposing server connection..." + this.toString());
165         }
166         synchronized( this ) {
167             if( null != serverConnectionThread ) {
168                 // Execution of this block means that the run() method
169                 // hasn't finished yet.  So we interrupt the thread
170                 // to terminate run() and wait for the run() method
171                 // to finish.  The notifyAll() at the end of run() will
172                 // wake this thread and allow dispose() to end.
173                 Thread thread = serverConnectionThread;
174                 serverConnectionThread = null;
175                 thread.interrupt();
176                 try {
177                     
178                     serverSocket.close();
179                     
180                 } catch (IOException ie) {
181                     // Ignored - we're doing this to break out of the
182                     // accept.  This minimizes the time required to
183                     // shutdown the server.  Unfortunately, this is
184                     // not guaranteed to work on all platforms.  See
185                     // the comments for POLLING_INTERVAL
186                 }
187                 try {
188                     if (POLLING_INTERVAL > 0) {
189                         wait(2L*POLLING_INTERVAL);
190                     } else {
191                         wait();
192                     }
193                 } catch (InterruptedException ie) {
194                     // Expected - just complete dispose()
195                 }
196             }
197             ContainerUtil.dispose(runnerPool);
198             runnerPool = null;
199         }
200 
201         getLogger().debug("Closed server connection - cleaning up clients - " + this.toString());
202 
203         synchronized (clientConnectionRunners) {
204             Iterator runnerIterator = clientConnectionRunners.iterator();
205             while( runnerIterator.hasNext() ) {
206                 ClientConnectionRunner runner = (ClientConnectionRunner)runnerIterator.next();
207                 runner.dispose();
208                 runner = null;
209             }
210             clientConnectionRunners.clear();
211             clearConnectionPerIP();
212             
213         }
214 
215         getLogger().debug("Cleaned up clients - " + this.toString());
216 
217     }

219     /***
220      * Returns a ClientConnectionRunner in the set managed by this ServerConnection object.
221      *
222      * @param clientConnectionRunner the ClientConnectionRunner to be added
223      */
224     private ClientConnectionRunner addClientConnectionRunner()
225             throws Exception {
226         synchronized (clientConnectionRunners) {
227             ClientConnectionRunner clientConnectionRunner = (ClientConnectionRunner)runnerPool.get();
228             clientConnectionRunners.add(clientConnectionRunner);
229             if (getLogger().isDebugEnabled()) {
230                 getLogger().debug("Adding one connection for a total of " + clientConnectionRunners.size());
231             }
232             return clientConnectionRunner;
233         }
234     }

236     /***
237      * Removes a ClientConnectionRunner from the set managed by this ServerConnection object.
238      *
239      * @param clientConnectionRunner the ClientConnectionRunner to be removed
240      */
241     private void removeClientConnectionRunner(ClientConnectionRunner clientConnectionRunner) {
242 
243        /*
244         * checking runnerPool avoids 'dead-lock' when service is disposing :
245         * (dispose() calls dispose all runners)
246         * but runner is 'running' and cleans up on exit
247         * this situation will result in a dead-lock on 'clientConnectionRunners'
248         */
249         if( runnerPool == null ) {
250             getLogger().info("ServerConnection.removeClientConnectionRunner - dispose has been called - so just return : " + clientConnectionRunner );
251             return;
252         }
253         
254         synchronized (clientConnectionRunners) {
255             if (clientConnectionRunners.remove(clientConnectionRunner)) {
256                 if (getLogger().isDebugEnabled()) {
257                     getLogger().debug("Releasing one connection, leaving a total of " + clientConnectionRunners.size());
258                 }
259                 runnerPool.put(clientConnectionRunner);
260             }
261         }
262     }

264     /***
265      * Raise the connectionCount for the given ipAdrress
266      * @param ip raise the connectionCount for the given ipAddress
267      */
268     private synchronized void addConnectionPerIP (String ip) {
269         connectionPerIP.put(ip,Integer.toString(getConnectionPerIP(ip) +1));
270     }

272     /***
273      * Get the count of connections for the given ip
274      * @param ip the ipAddress to get the connections for. 
275      * @return count
276      */
277     private synchronized int getConnectionPerIP(String ip) {
278         int count = 0;
279         String curCount = null;
280         Object c = connectionPerIP.get(ip);
281         
282         if (c != null) {
283             curCount = c.toString();
284             
285             if (curCount != null) {
286                 return Integer.parseInt(curCount);
287             }
288         }
289         return count;
290     }
291     
292     /***
293      * Set the connection count for the given ipAddress
294      * @param ip ipAddres for which we want to set the count
295      */
296     private synchronized void removeConnectionPerIP (String ip) {
297 
298         int count = getConnectionPerIP(ip);
299         if (count > 1) {
300             connectionPerIP.put(ip,Integer.toString(count -1));
301         } else {
302             // not need this entry any more
303             connectionPerIP.remove(ip);
304         }
305 
306     }

308     /***
309      * Clear the connection count map
310      */
311     private synchronized void clearConnectionPerIP () {
312         connectionPerIP.clear();
313     }

315     /***
316      * Provides the body for the thread of execution for a ServerConnection.
317      * Connections made to the server socket are passed to an appropriate,
318      * newly created, ClientConnectionRunner
319      */
320     public void run() {
321         serverConnectionThread = Thread.currentThread();
322 
323         int ioExceptionCount = 0;
324         try {
325             serverSocket.setSoTimeout(POLLING_INTERVAL);
326         } catch (SocketException se) {
327             // Ignored - for the moment
328         }
329 
330         if ((getLogger().isDebugEnabled()) && (serverConnectionThread != null)) {
331             StringBuffer debugBuffer =
332                 new StringBuffer(128)
333                     .append(serverConnectionThread.getName())
334                     .append(" is listening on ")
335                     .append(serverSocket.toString());
336             getLogger().debug(debugBuffer.toString());
337         }
338         while( !Thread.currentThread().interrupted() && null != serverConnectionThread ) {
339             try {
340                 Socket clientSocket = null;
341                 try {
342                     clientSocket = serverSocket.accept();
343                 } catch( InterruptedIOException iioe ) {
344                     // This exception is expected upon ServerConnection shutdown.
345                     // See the POLLING_INTERVAL comment
346                     continue;
347                 } catch( IOException se ) {
348                     if (ioExceptionCount > 0) {
349                         getLogger().error( "Fatal exception while listening on server socket.  Terminating connection.", se );
350                         break;
351                     } else {
352                         continue;
353                     }
354                 } catch( SecurityException se ) {
355                     getLogger().error( "Fatal exception while listening on server socket.  Terminating connection.", se );
356                     break;
357                 }
358                 ClientConnectionRunner runner = null;
359                 synchronized (clientConnectionRunners) {
360                     if ((maxOpenConn > 0) && (clientConnectionRunners.size() >= maxOpenConn)) {
361                         if (getLogger().isWarnEnabled()) {
362                            getLogger().warn("Maximum number of open connections exceeded - refusing connection.  Current number of connections is " + clientConnectionRunners.size());
363                            if (getLogger().isWarnEnabled()) {
364                                Iterator runnerIterator = clientConnectionRunners.iterator();
365                                getLogger().info("Connections: ");
366                                while( runnerIterator.hasNext() ) {
367                                    getLogger().info("    " + ((ClientConnectionRunner)runnerIterator.next()).toString());
368                                }
369                            }
370                         }
371                         try {
372                             clientSocket.close();
373                         } catch (IOException ignored) {
374                             // We ignore this exception, as we already have an error condition.
375                         }
376                         continue;
377                     } else if ((maxOpenConnPerIP > 0) && (getConnectionPerIP(clientSocket.getInetAddress().getHostAddress()) >= maxOpenConnPerIP)) {
378                         getLogger().info("Maximum number of open connections per IP exceeded - refusing connection.  Current number of connections is " + getConnectionPerIP(clientSocket.getInetAddress().getHostAddress()));
379                             
380                         if (getLogger().isWarnEnabled()) {
381                                 
382                             getLogger().warn("Maximum number of open connections per IP exceeded - refusing connection.  Current number of connections is " + getConnectionPerIP(clientSocket.getInetAddress().getHostAddress()));
383                         }
384                         try {
385                             clientSocket.close();
386                         } catch (IOException ignored) {
387                             // We ignore this exception, as we already have an error condition.
388                         }
389                         continue;
390                         
391                     } else {
392                         addConnectionPerIP(clientSocket.getInetAddress().getHostAddress());
393                         clientSocket.setSoTimeout(socketTimeout);
394                         runner = addClientConnectionRunner();
395                         runner.setSocket(clientSocket);
396                         
397                     }
398                 }
399                 setupLogger( runner );
400                 try {
401                     connThreadPool.execute( runner );
402                 } catch (Exception e) {
403                     // This error indicates that the underlying thread pool
404                     // is out of threads.  For robustness, we catch this and
405                     // cleanup
406                     getLogger().error("Internal error - insufficient threads available to service request.  " +
407                                       Thread.activeCount() + " threads in service request pool.", e);
408                     try {
409                         removeConnectionPerIP(clientSocket.getInetAddress().getHostAddress());
410                         clientSocket.close();
411                     } catch (IOException ignored) {
412                         // We ignore this exception, as we already have an error condition.
413                     }
414                     // In this case, the thread will not remove the client connection runner,
415                     // so we must.
416                     removeClientConnectionRunner(runner);
417                 }
418             } catch( IOException ioe ) {
419                 getLogger().error( "Exception accepting connection", ioe );
420             } catch( Throwable e ) {
421                 getLogger().error( "Exception executing client connection runner: " + e.getMessage(), e );
422             }
423         }
424         synchronized( this ) {
425             serverConnectionThread = null;
426             Thread.currentThread().interrupted();
427             notifyAll();
428         }
429     }

431     /***
432      * An inner class to provide the actual body of the thread of execution
433      * that occurs upon a client connection.
434      *
435      */
436     class ClientConnectionRunner extends AbstractLogEnabled
437         implements Poolable, Runnable  {
438 
439         /***
440          * The Socket that this client connection is using for transport.
441          */
442         private Socket clientSocket;
443 
444         /***
445          * The thread of execution associated with this client connection.
446          */
447         private Thread clientSocketThread;
448 
449         /***
450          * Returns string for diagnostic logging
451          */
452         public String toString() {
453             return getClass().getName() + " for " + clientSocket + " on " + clientSocketThread;
454         }
455 
456         public ClientConnectionRunner() {
457         }
458         
459         /***
460          * The dispose operation that terminates the runner.  Should only be
461          * called by the ServerConnection that owns the ClientConnectionRunner
462          */
463         public void dispose() {
464             synchronized( this ) {
465                 if (null != clientSocketThread) {
466                     // Execution of this block means that the run() method
467                     // hasn't finished yet.  So we interrupt the thread
468                     // to terminate run() and wait for the run() method
469                     // to finish.  The notifyAll() at the end of run() will
470                     // wake this thread and allow dispose() to end.
471                     clientSocketThread.interrupt();
472                     clientSocketThread = null;
473                     try {
474                         wait();
475                     } catch (InterruptedException ie) {
476                         // Expected - return from the method
477                     }
478                 }
479             }
480         }
481 
482         /***
483          * Sets the socket for a ClientConnectionRunner.
484          *
485          * @param socket the client socket associated with this ClientConnectionRunner
486          */
487         public void setSocket(Socket socket) {
488             clientSocket = socket;
489         }
490 
491         /***
492          * Provides the body for the thread of execution dealing with a particular client
493          * connection.  An appropriate ConnectionHandler is created, applied, executed,
494          * and released.
495          */
496         public void run() {
497             ConnectionHandler handler = null;
498             try {
499                 clientSocketThread = Thread.currentThread();
500 
501                 handler = ServerConnection.this.handlerFactory.createConnectionHandler();
502                 String connectionString = null;
503                 if( getLogger().isDebugEnabled() ) {
504                     connectionString = getConnectionString();
505                     String message = "Starting " + connectionString;
506                     getLogger().debug( message );
507                 }
508 
509                 handler.handleConnection(clientSocket);
510 
511                 if( getLogger().isDebugEnabled() ) {
512                     String message = "Ending " + connectionString;
513                     getLogger().debug( message );
514                 }
515 
516             } catch( Throwable e ) {
517                 getLogger().error( "Error handling connection", e );
518             } finally {
519 
520                 // remove this connection from map!
521                 removeConnectionPerIP(clientSocket.getInetAddress().getHostAddress());
522                 
523                 // Close the underlying socket
524                 try {
525                     if (clientSocket != null) {
526                         clientSocket.close();
527                     }
528                 } catch( IOException ioe ) {
529                     getLogger().warn( "Error shutting down connection", ioe );
530                 }
531 
532                 clientSocket = null;
533 
534                 // Null out the thread, notify other threads to encourage
535                 // a context switch
536                 synchronized( this ) {
537                     clientSocketThread = null;
538 
539                     Thread.currentThread().interrupted();
540 
541                     // Release the handler and kill the reference to the handler factory
542                     //
543                     // This needs to be done after the clientSocketThread is nulled out,
544                     // otherwise we could trash a reused ClientConnectionRunner
545                     if (handler != null) {
546                         ServerConnection.this.handlerFactory.releaseConnectionHandler( handler );
547                         handler = null;
548                     }
549 
550                     // Remove this runner from the list of active connections.
551                     ServerConnection.this.removeClientConnectionRunner(this);
552 
553                     notifyAll();
554                 }
555             }
556         }
557 
558         /***
559          * Helper method to return a formatted string with connection transport information.
560          *
561          * @return a formatted string
562          */
563         private String getConnectionString() {
564             if (clientSocket == null) {
565                 return "invalid socket";
566             }
567             StringBuffer connectionBuffer
568                 = new StringBuffer(256)
569                     .append("connection on ")
570                     .append(clientSocket.getLocalAddress().getHostAddress().toString())
571                     .append(":")
572                     .append(clientSocket.getLocalPort())
573                     .append(" from ")
574                     .append(clientSocket.getInetAddress().getHostAddress().toString())
575                     .append(":")
576                     .append(clientSocket.getPort());
577             return connectionBuffer.toString();
578         }
579     }
