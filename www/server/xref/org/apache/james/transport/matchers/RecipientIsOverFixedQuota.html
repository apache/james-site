18  package org.apache.james.transport.matchers;
19  
20  import java.util.Locale;
21  
22  import javax.mail.MessagingException;
23  
24  import org.apache.mailet.Mail;
25  import org.apache.mailet.MailAddress;
26  
27  /***
28   * <P>Checks whether a recipient has exceeded a maximum allowed quota for messages
29   * standing in his inbox. Such quota is <I>the same</I> for all users.</P>
30   * <P>Will check if the total size of all his messages in the inbox are greater
31   * than a certain number of bytes.  You can use 'k' and 'm' as optional postfixes.
32   * In other words, "1m" is the same as writing "1024k", which is the same as
33   * "1048576".</P>
34   * <P>Here follows an example of a config.xml definition:</P>
35   * <PRE><CODE>
36   * &lt;processor name="transport"&gt;
37   * .
38   * .
39   * .
40   *    &lt;mailet match=match="RecipientIsOverFixedQuota=40M" class="ToProcessor"&gt;
41   *       &lt;processor&gt; error &lt;/processor&gt;
42   *       &lt;notice&gt;The recipient has exceeded maximum allowed size quota&lt;/notice&gt;
43   *    &lt;/mailet&gt;
44   * .
45   * .
46   * .
47   * &lt;/processor&gt;
48   * </CODE></PRE>
49   *
50   * @version 1.0.0, 2003-05-11
51   */
52  
53  public class RecipientIsOverFixedQuota extends AbstractStorageQuota {
54      private long quota = 0;
55  
56      /***
57       * Standard matcher initialization.
58       * Does a <CODE>super.init()</CODE> and parses the common storage quota amount from
59       * <I>config.xml</I> for later use.
60       */
61      public void init() throws MessagingException {
62          super.init();
63          quota = parseQuota(getCondition().trim().toLowerCase(Locale.US));
64      }
65  
66      protected long getQuota(MailAddress recipient, Mail _) throws MessagingException {
67          return quota;
68      }
69  }
