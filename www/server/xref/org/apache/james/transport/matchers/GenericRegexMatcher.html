18  package org.apache.james.transport.matchers;
19  
20  import org.apache.mailet.GenericMatcher;
21  import org.apache.mailet.Mail;
22  import org.apache.oro.text.regex.MalformedPatternException;
23  import org.apache.oro.text.regex.Pattern;
24  import org.apache.oro.text.regex.Perl5Compiler;
25  import org.apache.oro.text.regex.Perl5Matcher;
26  
27  import javax.mail.MessagingException;
28  import javax.mail.internet.MimeMessage;
29  import java.util.Collection;
30  
31  /***
32   * This is a generic matcher that uses regular expressions.  If any of
33   * the regular expressions match, the matcher is considered to have
34   * matched.  This is an abstract class that must be subclassed to feed
35   * patterns.  Patterns are provided by calling the compile method.  A
36   * subclass will generally call compile() once during init(), but it
37   * could subclass match(), and call it as necessary during message
38   * processing (e.g., if a file of expressions changed). 
39   *
40   * @
41   */
42  
43  abstract public class GenericRegexMatcher extends GenericMatcher {
44      protected Object[][] patterns;
45  
46      public void compile(Object[][] patterns) throws MalformedPatternException {
47          // compile a bunch of regular expressions
48          this.patterns = patterns;
49          for (int i = 0; i < patterns.length; i++) {
50              String pattern = (String)patterns[i][1];
51              patterns[i][1] = new Perl5Compiler().compile(pattern, Perl5Compiler.READ_ONLY_MASK | Perl5Compiler.SINGLELINE_MASK);
52          }
53      }
54  
55      abstract public void init() throws MessagingException;
56  
57      public Collection match(Mail mail) throws MessagingException {
58          MimeMessage message = mail.getMessage();
59          Perl5Matcher matcher = new Perl5Matcher();
60  
61          //Loop through all the patterns
62          if (patterns != null) for (int i = 0; i < patterns.length; i++) {
63              //Get the header name
64              String headerName = (String)patterns[i][0];
65              //Get the patterns for that header
66              Pattern pattern = (Pattern)patterns[i][1];
67              //Get the array of header values that match that
68              String headers[] = message.getHeader(headerName);
69              //Loop through the header values
70              if (headers != null) for (int j = 0; j < headers.length; j++) {
71                  if (matcher.matches(headers[j], pattern)) {
72                      // log("Match: " + headerName + "[" + j + "]: " + headers[j]);
73                      return mail.getRecipients();
74                  }
75                  //log("       " + headerName + "[" + j + "]: " + headers[j]);
76              }
77          }
78          return null;
79      }
80  }
