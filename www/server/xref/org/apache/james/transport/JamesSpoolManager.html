37  /***
38   * Manages the mail spool.  This class is responsible for retrieving
39   * messages from the spool, directing messages to the appropriate
40   * processor, and removing them from the spool when processing is
41   * complete.
42   *
43   * @version CVS $Revision: 420948 $ $Date: 2006-07-11 20:47:39 +0200 (mar, 11 lug 2006) $
44   */

113     /***
114      * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
115      */
116     public void configure(Configuration conf) throws ConfigurationException {
117         numThreads = conf.getChild("threads").getValueAsInteger(1);
118         
119         String processorClass = conf.getChild("processorClass").getValue("org.apache.james.transport.StateAwareProcessorList");
120         try {
121             processorList = (MailProcessor) Thread.currentThread().getContextClassLoader().loadClass(processorClass).newInstance();
122         } catch (Exception e1) {
123             getLogger().error("Unable to instantiate spoolmanager processor: "+processorClass, e1);
124             throw new ConfigurationException("Instantiation exception: "+processorClass, e1);
125         }
126         
127         try {
128             ContainerUtil.enableLogging(processorList, getLogger());
129             ContainerUtil.service(processorList, compMgr);
130         } catch (ServiceException e) {
131             getLogger().error(e.getMessage(), e);
132             throw new ConfigurationException("Servicing failed with error: "+e.getMessage(),e);
133         }
134         
135         ContainerUtil.configure(processorList, conf);
136     }

138     /***
139      * @see org.apache.avalon.framework.activity.Initializable#initialize()
140      */
141     public void initialize() throws Exception {
142 
143         getLogger().info("JamesSpoolManager init...");
144         
145         ContainerUtil.initialize(processorList);
146         
147         if (getLogger().isInfoEnabled()) {
148             StringBuffer infoBuffer =
149                 new StringBuffer(64)
150                     .append("Spooler Manager uses ")
151                     .append(numThreads)
152                     .append(" Thread(s)");
153             getLogger().info(infoBuffer.toString());
154         }
155 
156         active = true;
157         numActive = 0;
158         spoolThreads = new java.util.ArrayList(numThreads);
159         for ( int i = 0 ; i < numThreads ; i++ ) {
160             Thread reader = new Thread(this, "Spool Thread #" + i);
161             spoolThreads.add(reader);
162             reader.start();
163         }
164     }

166     /***
167      * This routinely checks the message spool for messages, and processes
168      * them as necessary
169      */
170     public void run() {
171 
172         if (getLogger().isInfoEnabled())
173         {
174             getLogger().info("Run JamesSpoolManager: "
175                              + Thread.currentThread().getName());
176             getLogger().info("Spool=" + spool.getClass().getName());
177         }
178 
179         numActive++;
180         while(active) {
181             String key = null;
182             try {
183                 Mail mail = (Mail)spool.accept();
184                 key = mail.getName();
185                 if (getLogger().isDebugEnabled()) {
186                     StringBuffer debugBuffer =
187                         new StringBuffer(64)
188                                 .append("==== Begin processing mail ")
189                                 .append(mail.getName())
190                                 .append("====");
191                     getLogger().debug(debugBuffer.toString());
192                 }
193 
194                 processorList.service(mail);
195 
196                 // Only remove an email from the spool is processing is
197                 // complete, or if it has no recipients
198                 if ((Mail.GHOST.equals(mail.getState())) ||
199                     (mail.getRecipients() == null) ||
200                     (mail.getRecipients().size() == 0)) {
201                     spool.remove(key);
202                     if (getLogger().isDebugEnabled()) {
203                         StringBuffer debugBuffer =
204                             new StringBuffer(64)
205                                     .append("==== Removed from spool mail ")
206                                     .append(mail.getName())
207                                     .append("====");
208                         getLogger().debug(debugBuffer.toString());
209                     }
210                 }
211                 else {
212                     // spool.remove() has a side-effect!  It unlocks the
213                     // message so that other threads can work on it!  If
214                     // we don't remove it, we must unlock it!
215                     spool.store(mail);
216                     spool.unlock(key);
217                     // Do not notify: we simply updated the current mail
218                     // and we are able to reprocess it now.
219                 }
220                 ContainerUtil.dispose(mail);
221                 mail = null;
222             } catch (InterruptedException ie) {
223                 getLogger().info("Interrupted JamesSpoolManager: " + Thread.currentThread().getName());
224             } catch (Throwable e) {
225                 if (getLogger().isErrorEnabled()) {
226                     getLogger().error("Exception processing " + key + " in JamesSpoolManager.run "
227                                       + e.getMessage(), e);
228                 }
229                 /* Move the mail to ERROR state?  If we do, it could be
230                  * deleted if an error occurs in the ERROR processor.
231                  * Perhaps the answer is to resolve that issue by
232                  * having a special state for messages that are not to
233                  * be processed, but aren't to be deleted?  The message
234                  * would already be in the spool, but would not be
235                  * touched again.
236                 if (mail != null) {
237                     try {
238                         mail.setState(Mail.ERROR);
239                         spool.store(mail);
240                     }
241                 }
242                 */
243             }
244         }
245         if (getLogger().isInfoEnabled())
246         {
247             getLogger().info("Stop JamesSpoolManager: " + Thread.currentThread().getName());
248         }
249         numActive--;
250     }

252     /***
253      * The dispose operation is called at the end of a components lifecycle.
254      * Instances of this class use this method to release and destroy any
255      * resources that they own.
256      *
257      * This implementation shuts down the LinearProcessors managed by this
258      * JamesSpoolManager
259      *
260      */
261     public void dispose() {
262         getLogger().info("JamesSpoolManager dispose...");
263         active = false; // shutdown the threads
264         for (Iterator it = spoolThreads.iterator(); it.hasNext(); ) {
265             ((Thread) it.next()).interrupt(); // interrupt any waiting accept() calls.
266         }
267 
268         long stop = System.currentTimeMillis() + 60000;
269         // give the spooler threads one minute to terminate gracefully
270         while (numActive != 0 && stop > System.currentTimeMillis()) {
271             try {
272                 Thread.sleep(1000);
273             } catch (Exception ignored) {}
274         }
275         getLogger().info("JamesSpoolManager thread shutdown completed.");
276         
277         ContainerUtil.dispose(processorList);
278     }
