44  /***
45   * CommandListservManager is the default implementation of {@link ICommandListservManager}.
46   * It loads all the configured {@link IListServCommand}s and delegates to them at runtime.
47   * <br />
48   *
49   * It isn't responsible for procesing messages sent to the main mailing list, but is responsible for
50   * individual commands sent by users, such as: info, subscribe, etc...
51   * <br />
52   *
53   * Requests sent to the CommandListservManager take the form of:
54   * <pre>
55   * &lt;listName&gt;-&lt;commandName&gt;@domain
56   * </pre>
57   *
58   * If the command isn't recognized an error will be sent using {@link #onError}.
59   * <br />
60   * <br />
61   *
62   * The configuration for this mailet sould be in the 'root' processor block.
63   * <pre>
64   * &lt;mailet match="CommandListservMatcher=announce@localhost" class="CommandListservManager"&gt;
65   *  &lt;listName&gt;announce&lt;/listName&gt;
66   *  &lt;displayName&gt;Announce mailing list&lt;/displayName&gt;
67   *  &lt;listOwner&gt;owner@localhost&lt;/listOwner&gt;
68   *  &lt;repositoryName&gt;list-announce&lt;/repositoryName&gt;
69   *  &lt;listDomain&gt;localhost&lt;/listDomain&gt;
70   *
71   *  &lt;commandpackages&gt;
72   *     &lt;commandpackage&gt;org.apache.james.transport.mailets.listservcommands&lt;/commandpackage&gt;
73   *  &lt;/commandpackages&gt;
74   *
75   *  &lt;commands&gt;
76   *     &lt;command name="subscribe" class="Subscribe"/&gt;
77   *     &lt;command name="subscribe-confirm" class="SubscribeConfirm"/&gt;
78   *     &lt;command name="unsubscribe" class="UnSubscribe"/&gt;
79   *     &lt;command name="unsubscribe-confirm" class="UnSubscribeConfirm"/&gt;
80   *     &lt;command name="error" class="ErrorCommand"/&gt;
81   *     &lt;command name="owner" class="Owner"/&gt;
82   *     &lt;command name="info" class="Info"/&gt;
83   *  &lt;/commands&gt;
84   * &lt;/mailet&gt;
85   * </pre>
86   *
87   * <br />
88   * <br />
89   * Todo: refine the command matching so we can have more sophistciated commands such as:
90   * <pre>
91   * &lt;listName&gt;-&lt;commandName&gt;-&lt;optCommandParam&gt;@domain
92   * </pre>
93   *
94   * @version CVS $Revision: 423745 $ $Date: 2006-07-20 05:47:46 +0200 (gio, 20 lug 2006) $
95   * @since 2.2.0
96   */

277     /***
278      * <p>Called by the mailet container to allow the mailet to process a
279      * message.</p>
280      *
281      * <p>This method is declared abstract so subclasses must override it.</p>
282      *
283      * @param mail - the Mail object that contains the MimeMessage and
284      *          routing information
285      * @throws MessagingException - if an exception occurs that interferes with the mailet's normal operation
286      *          occurred
287      */
288     public void service(Mail mail) throws MessagingException {
289         if (mail.getRecipients().size() != 1) {
290             getMailetContext().bounce(mail, "You can only send one command at a time to this listserv manager.");
291             return;
292         }
293         MailAddress mailAddress = (MailAddress) mail.getRecipients().iterator().next();
294         IListServCommand command = getCommandTarget(mailAddress);
295 
296         if (command == null) {
297             //don't recognize the command
298             onError(mail,
299                     "unknown command",
300                     xmlResources.getString("command.not.understood", getStandardProperties()));
301         }
302         command.onCommand(mail);
303 
304         mail.setState(Mail.GHOST);
305     }

312     protected String getCommandName(MailAddress mailAddress) {
313         String user = mailAddress.getUser();
314         int index = user.indexOf('-', listName.length());
315         String commandName = user.substring(++index);
316         return commandName;

342     /***
343      * Load an initialize all of the available commands
344      * @param configuration
345      * @throws ConfigurationException
346      */
347     protected void loadCommands(Configuration configuration) throws Exception {
348         final Configuration commandConfigurations = configuration.getChild("commands");
349         final Configuration[] commandConfs = commandConfigurations.getChildren("command");
350         for (int index = 0; index < commandConfs.length; index++) {
351             Configuration commandConf = commandConfs[index];
352             String commandName = commandConf.getAttribute("name").toLowerCase();
353             String className = commandConf.getAttribute("class");
354             loadCommand(commandName, className, commandConf);
355         }
356     }

358     /***
359      * Loads and initializes a single command
360      *
361      * @param commandName
362      * @param className
363      * @param configuration
364      * @throws ConfigurationException
365      */
366     protected void loadCommand(String commandName,
367                                String className,
368                                Configuration configuration)
369             throws ConfigurationException, ClassNotFoundException, IllegalAccessException, InstantiationException {
370         ClassLoader theClassLoader = getClass().getClassLoader();
371         for (Iterator it = commandPackages.iterator(); it.hasNext();) {
372             String packageName = (String) it.next();
373 
374             IListServCommand listServCommand = null;
375             try {
376                 listServCommand = (IListServCommand) theClassLoader.loadClass(packageName + className).newInstance();
377             } catch (Exception e) {
378                 //ignore
379                 continue;
380             }
381             listServCommand.init(this, configuration);
382             commandMap.put(commandName, listServCommand);
383             return;
384         }
385 
386         throw new ConfigurationException("Unable to load listservcommand: " + commandName);
387     }

389     /***
390      * loads all of the packages for the commands
391      *
392      * @param configuration
393      * @throws ConfigurationException
394      */
395     protected void loadCommandPackages(Configuration configuration) throws ConfigurationException {
396         commandPackages.add("");
397         final Configuration packageConfiguration = configuration.getChild("commandpackages");
398         final Configuration[] pkgConfs = packageConfiguration.getChildren("commandpackage");
399         for (int index = 0; index < pkgConfs.length; index++) {
400             Configuration conf = pkgConfs[index];
401             String packageName = conf.getValue().trim();
402             if (!packageName.endsWith(".")) {
403                 packageName += ".";
404             }
405             commandPackages.add(packageName);
406         }
407     }

409     /***
410      * Retrieves a data field, potentially defined by a super class.
411      * @return null if not found, the object otherwise
412      */
413     protected static Object getField(Object instance, String name) throws IllegalAccessException {
414         Class clazz = instance.getClass();
415         Field[] fields;
416         while (clazz != null) {
417             fields = clazz.getDeclaredFields();
418             for (int index = 0; index < fields.length; index++) {
419                 Field field = fields[index];
420                 if (field.getName().equals(name)) {
421                     field.setAccessible(true);
422                     return field.get(instance);
423                 }
424             }
425             clazz = clazz.getSuperclass();
426         }
427 
428         return null;
429     }
