40  /*** <P>Manages for each local user a "white list" of remote addresses whose messages
41   * should never be blocked as spam.</P>
42   * <P>The normal behaviour is to check, for a local sender, if a remote recipient
43   * is already in the list: if not, it will be automatically inserted.
44   * This is under the interpretation that if a local sender <I>X</I> sends a message to a
45   * remote recipient <I>Y</I>, then later on if a message is sent by <I>Y</I> to <I>X</I> it should be
46   * considered always valid and never blocked; hence <I>Y</I> should be in the white list
47   * of <I>X</I>.</P>
48   * <P>Another mode of operations is when a local sender sends a message to <I>whitelistManagerAddress</I>
49   * with one of three specific values in the subject, to
50   * (i) send back a message displaying a list of the addresses in his own list;
51   * (ii) insert some new addresses in his own list;
52   * (iii) remove some addresses from his own list.
53   * In all this cases the message will be ghosted and the postmaster will reply
54   * to the sender.</P>
55   * <P> The sender name is always converted to its primary name (handling aliases).</P>
56   * <P>Sample configuration:</P>
57   * <PRE><CODE>
58   * &lt;mailet match="SMTPAuthSuccessful" class="WhiteListManager"&gt;
59   *   &lt;repositoryPath&gt; db://maildb &lt;/repositoryPath&gt;
60   *   &lt;!--
61   *     If true automatically inserts the local sender to remote recipients entries in the whitelist (default is false).
62   *   --&gt;
63   *   &lt;automaticInsert&gt;true&lt;/automaticInsert&gt;
64   *   &lt;!--
65   *     Set this to an email address of the "whitelist manager" to send commands to (default is null).
66   *   --&gt;
67   *   &lt;whitelistManagerAddress&gt;whitelist.manager@xxx.yyy&lt;/whitelistManagerAddress&gt;
68   *   &lt;!--
69   *     Set this to a unique text that you can use (by sending a message to the "whitelist manager" above)
70   *     to tell the mailet to send back the contents of the white list (default is null).
71   *   --&gt;
72   *   &lt;displayFlag&gt;display whitelist&lt;/displayFlag&gt;
73   *   &lt;!--
74   *     Set this to a unique text that you can use (by sending a message to the "whitelist manager" above)
75   *     to tell the mailet to insert some new remote recipients to the white list (default is null).
76   *   --&gt;
77   *   &lt;insertFlag&gt;insert whitelist&lt;/insertFlag&gt;
78   *   &lt;!--
79   *     Set this to a unique text that you can use (by sending a message to the "whitelist manager" above)
80   *     to tell the mailet to remove some remote recipients from the white list (default is null).
81   *   --&gt;
82   *   &lt;removeFlag&gt;remove whitelist&lt;/removeFlag&gt;
83   * &lt;/mailet&gt;
84   * </CODE></PRE>
85   *
86   * @see org.apache.james.transport.matchers.IsInWhiteList
87   * @version SVN $Revision: $ $Date: $
88   * @since 2.3.0
89   */

160     public void init() throws MessagingException {
161         automaticInsert = new Boolean(getInitParameter("automaticInsert")).booleanValue();
162         log("automaticInsert: " + automaticInsert);
163 
164         displayFlag = getInitParameter("displayFlag");
165         insertFlag = getInitParameter("insertFlag");
166         removeFlag = getInitParameter("removeFlag");
167         
168         String whitelistManagerAddressString = getInitParameter("whitelistManagerAddress");
169         if (whitelistManagerAddressString != null) {
170             whitelistManagerAddressString = whitelistManagerAddressString.trim();
171             log("whitelistManagerAddress: " + whitelistManagerAddressString);
172             try {
173                 whitelistManagerAddress = new MailAddress(whitelistManagerAddressString);
174             }
175             catch (javax.mail.internet.ParseException pe) {
176                 throw new MessagingException("Bad whitelistManagerAddress", pe);
177             }
178             
179             if (displayFlag != null) {
180                 displayFlag = displayFlag.trim();
181                 log("displayFlag: " + displayFlag);
182             }
183             else {
184                 log("displayFlag is null");
185             }
186             if (insertFlag != null) {
187                 insertFlag = insertFlag.trim();
188                 log("insertFlag: " + insertFlag);
189             }
190             else {
191                 log("insertFlag is null");
192             }
193             if (removeFlag != null) {
194                 removeFlag = removeFlag.trim();
195                 log("removeFlag: " + removeFlag);
196             }
197             else {
198                 log("removeFlag is null");
199             }
200         }
201         else {
202             log("whitelistManagerAddress is null; will ignore commands");
203         }
204         
205         String repositoryPath = getInitParameter("repositoryPath");
206         if (repositoryPath != null) {
207             log("repositoryPath: " + repositoryPath);
208         }
209         else {
210             throw new MessagingException("repositoryPath is null");
211         }
212 
213         ServiceManager serviceManager = (ServiceManager) getMailetContext().getAttribute(Constants.AVALON_COMPONENT_MANAGER);
214 
215         try {
216             // Get the DataSourceSelector block
217             DataSourceSelector datasources = (DataSourceSelector) serviceManager.lookup(DataSourceSelector.ROLE);
218             // Get the data-source required.
219             int stindex =   repositoryPath.indexOf("://") + 3;
220             String datasourceName = repositoryPath.substring(stindex);
221             datasource = (DataSourceComponent) datasources.select(datasourceName);
222         } catch (Exception e) {
223             throw new MessagingException("Can't get datasource", e);
224         }
225 
226          try {
227             // Get the UsersRepository
228             usersStore = (UsersStore)serviceManager.lookup(UsersStore.ROLE);
229             localusers = (UsersRepository)usersStore.getRepository("LocalUsers");
230         } catch (Exception e) {
231             throw new MessagingException("Can't get the local users repository", e);
232         }
233 
234         try {
235             initSqlQueries(datasource.getConnection(), getMailetContext());
236         } catch (Exception e) {
237             throw new MessagingException("Exception initializing queries", e);
238         }        
239         
240         selectByPK = sqlQueries.getSqlString("selectByPK", true);
241         selectBySender = sqlQueries.getSqlString("selectBySender", true);
242         insert = sqlQueries.getSqlString("insert", true);
243         deleteByPK = sqlQueries.getSqlString("deleteByPK", true);
244     }

246     /*** Services the mailet.
247      */    
248     public void service(Mail mail) throws MessagingException {
249         
250         // check if it's a local sender
251         MailAddress senderMailAddress = mail.getSender();
252         if (senderMailAddress == null) {
253             return;
254         }
255         if (!getMailetContext().isLocalEmail(senderMailAddress)) {
256             // not a local sender, so return
257             return;
258         }
259         
260         Collection recipients = mail.getRecipients();
261         
262         if (recipients.size() == 1
263         && whitelistManagerAddress != null
264         && whitelistManagerAddress.equals(recipients.toArray()[0])) {
265             
266             mail.setState(Mail.GHOST);
267         
268             String subject = mail.getMessage().getSubject();
269             if (displayFlag != null && displayFlag.equals(subject)) {
270                 manageDisplayRequest(mail);
271             }
272             else if (insertFlag != null && insertFlag.equals(subject)) {
273                 manageInsertRequest(mail);
274             }
275             else if (removeFlag != null && removeFlag.equals(subject)) {
276                 manageRemoveRequest(mail);
277             }
278             else {
279                 StringWriter sout = new StringWriter();
280                 PrintWriter out = new PrintWriter(sout, true);
281                 out.println("Answering on behalf of: " + whitelistManagerAddress);
282                 out.println("ERROR: Unknown command in the subject line: " + subject);
283                 sendReplyFromPostmaster(mail, sout.toString());
284             }
285             return;
286         }
287         
288         if (automaticInsert) {
289             checkAndInsert(senderMailAddress, recipients);
290         }
291         
292     }

302     /*** Loops through each address in the recipient list, checks if in the senders
303      * list and inserts in it otherwise.
304      */    
305     private void checkAndInsert(MailAddress senderMailAddress, Collection recipients) throws MessagingException {
306         String senderUser = senderMailAddress.getUser().toLowerCase(Locale.US);
307         String senderHost = senderMailAddress.getHost().toLowerCase(Locale.US);
308         
309         senderUser = getPrimaryName(senderUser);
310         
311         Connection conn = null;
312         PreparedStatement selectStmt = null;
313         PreparedStatement insertStmt = null;
314         boolean dbUpdated = false;
315         
316         try {
317             
318             for (Iterator i = recipients.iterator(); i.hasNext(); ) {
319                 ResultSet selectRS = null;
320                 try {
321                     MailAddress recipientMailAddress = (MailAddress)i.next();
322                     String recipientUser = recipientMailAddress.getUser().toLowerCase(Locale.US);
323                     String recipientHost = recipientMailAddress.getHost().toLowerCase(Locale.US);
324                     
325                     if (getMailetContext().isLocalServer(recipientHost)) {
326                         // not a remote recipient, so skip
327                         continue;
328                     }
329                     
330                     if (conn == null) {
331                         conn = datasource.getConnection();
332                     }
333                     
334                     if (selectStmt == null) {
335                         selectStmt = conn.prepareStatement(selectByPK);
336                     }
337                     selectStmt.setString(1, senderUser);
338                     selectStmt.setString(2, senderHost);
339                     selectStmt.setString(3, recipientUser);
340                     selectStmt.setString(4, recipientHost);
341                     selectRS = selectStmt.executeQuery();
342                     if (selectRS.next()) {
343                         //This address was already in the list
344                         continue;
345                     }
346                     
347                     if (insertStmt == null) {
348                         insertStmt = conn.prepareStatement(insert);
349                     }
350                     insertStmt.setString(1, senderUser);
351                     insertStmt.setString(2, senderHost);
352                     insertStmt.setString(3, recipientUser);
353                     insertStmt.setString(4, recipientHost);
354                     insertStmt.executeUpdate();
355                     dbUpdated = true;
356                     
357                 } finally {
358                     theJDBCUtil.closeJDBCResultSet(selectRS);
359                 }
360                 
361                 //Commit our changes if necessary.
362                 if (conn != null && dbUpdated && !conn.getAutoCommit()) {
363                     conn.commit();
364                     dbUpdated = false;
365                 }
366             }
367         } catch (SQLException sqle) {
368             log("Error accessing database", sqle);
369             throw new MessagingException("Exception thrown", sqle);
370         } finally {
371             theJDBCUtil.closeJDBCStatement(selectStmt);
372             theJDBCUtil.closeJDBCStatement(insertStmt);
373             //Rollback our changes if necessary.
374             try {
375                 if (conn != null && dbUpdated && !conn.getAutoCommit()) {
376                     conn.rollback();
377                     dbUpdated = false;
378                 }
379             }
380             catch (Exception e) {}
381             theJDBCUtil.closeJDBCConnection(conn);
382         }
383     }

387     private void manageDisplayRequest(Mail mail)
388     throws MessagingException {
389         MailAddress senderMailAddress = mail.getSender();
390         String senderUser = senderMailAddress.getUser().toLowerCase(Locale.US);
391         String senderHost = senderMailAddress.getHost().toLowerCase(Locale.US);
392         
393         senderUser = getPrimaryName(senderUser);
394         
395         Connection conn = null;
396         PreparedStatement selectStmt = null;
397         ResultSet selectRS = null;
398         
399         StringWriter sout = new StringWriter();
400         PrintWriter out = new PrintWriter(sout, true);
401         
402         try {
403             out.println("Answering on behalf of: " + whitelistManagerAddress);
404             out.println("Displaying white list of " + (new MailAddress(senderUser, senderHost)) + ":");
405             out.println();
406             
407             conn = datasource.getConnection();

409             selectStmt.setString(1, senderUser);
410             selectStmt.setString(2, senderHost);
411             selectRS = selectStmt.executeQuery();

413                 MailAddress mailAddress =
414                     new MailAddress(selectRS.getString(1), selectRS.getString(2));
415                 out.println(mailAddress.toInternetAddress().toString());
416             }
417             
418             out.println();
419             out.println("Finished");
420             
421             sendReplyFromPostmaster(mail, sout.toString());
422                         
423         } catch (SQLException sqle) {
424             out.println("Error accessing the database");
425             sendReplyFromPostmaster(mail, sout.toString());
426             throw new MessagingException("Error accessing database", sqle);
427         } finally {
428             theJDBCUtil.closeJDBCResultSet(selectRS);
429             theJDBCUtil.closeJDBCStatement(selectStmt);
430             theJDBCUtil.closeJDBCConnection(conn);
431         }

453             out.println("Answering on behalf of: " + whitelistManagerAddress);
454             out.println("Inserting in the white list of " + (new MailAddress(senderUser, senderHost)) + " ...");
455             out.println();
456             
457             MimeMessage message = mail.getMessage() ;
458             
459             Object content= message.getContent();
460             
461             if (message.getContentType().startsWith("text/plain")
462             && content instanceof String) {
463                 StringTokenizer st = new StringTokenizer((String) content, " \t\n\r\f,;:<>");
464                 while (st.hasMoreTokens()) {
465                     ResultSet selectRS = null;
466                     try {
467                         MailAddress recipientMailAddress;
468                         try {
469                             recipientMailAddress = new MailAddress(st.nextToken());
470                         }
471                         catch (javax.mail.internet.ParseException pe) {
472                             continue;
473                         }
474                         String recipientUser = recipientMailAddress.getUser().toLowerCase(Locale.US);
475                         String recipientHost = recipientMailAddress.getHost().toLowerCase(Locale.US);
476                         
477                         if (getMailetContext().isLocalServer(recipientHost)) {
478                             // not a remote recipient, so skip
479                             continue;
480                         }
481                         
482                         if (conn == null) {
483                             conn = datasource.getConnection();
484                         }
485                         
486                         if (selectStmt == null) {
487                             selectStmt = conn.prepareStatement(selectByPK);
488                         }
489                         selectStmt.setString(1, senderUser);
490                         selectStmt.setString(2, senderHost);
491                         selectStmt.setString(3, recipientUser);
492                         selectStmt.setString(4, recipientHost);
493                         selectRS = selectStmt.executeQuery();
494                         if (selectRS.next()) {
495                             //This address was already in the list
496                             out.println("Skipped:  " + recipientMailAddress);
497                             continue;
498                         }
499                         
500                         if (insertStmt == null) {
501                             insertStmt = conn.prepareStatement(insert);
502                         }
503                         insertStmt.setString(1, senderUser);
504                         insertStmt.setString(2, senderHost);
505                         insertStmt.setString(3, recipientUser);
506                         insertStmt.setString(4, recipientHost);
507                         insertStmt.executeUpdate();
508                         dbUpdated = true;
509                         out.println("Inserted: " + recipientMailAddress);
510                         
511                     } finally {
512                         theJDBCUtil.closeJDBCResultSet(selectRS);
513                     }
514                 }
515                 
516                 if (dbUpdated) {
517                     log("Insertion request issued by " + senderMailAddress);
518                 }
519                 //Commit our changes if necessary.
520                 if (conn != null && dbUpdated && !conn.getAutoCommit()) {
521                     conn.commit() ;
522                     dbUpdated = false;
523                 }
524             }
525             else {
526                 out.println("The message must be plain - no action");
527             }
528             
529             out.println();
530             out.println("Finished");
531             
532             sendReplyFromPostmaster(mail, sout.toString());
533             
534         } catch (SQLException sqle) {
535             out.println("Error accessing the database");
536             sendReplyFromPostmaster(mail, sout.toString());
537             throw new MessagingException("Error accessing the database", sqle);
538         } catch (IOException ioe) {
539             out.println("Error getting message content");
540             sendReplyFromPostmaster(mail, sout.toString());
541             throw new MessagingException("Error getting message content", ioe);
542         } finally {
543             theJDBCUtil.closeJDBCStatement(selectStmt);
544             theJDBCUtil.closeJDBCStatement(insertStmt);
545             //Rollback our changes if necessary.
546             try {
547                 if (conn != null && dbUpdated && !conn.getAutoCommit()) {
548                     conn.rollback() ;
549                     dbUpdated = false;
550                 }
551             }
552             catch (Exception e) {}
553             theJDBCUtil.closeJDBCConnection(conn);
554         }
555     }

680     private void sendReplyFromPostmaster(Mail mail, String stringContent) throws MessagingException {
681         try {
682             MailAddress notifier = getMailetContext().getPostmaster();
683             
684             MailAddress senderMailAddress = mail.getSender();
685             
686             MimeMessage message = mail.getMessage();
687             //Create the reply message
688             MimeMessage reply = new MimeMessage(Session.getDefaultInstance(System.getProperties(), null));
689             
690             //Create the list of recipients in the Address[] format
691             InternetAddress[] rcptAddr = new InternetAddress[1];
692             rcptAddr[0] = senderMailAddress.toInternetAddress();
693             reply.setRecipients(Message.RecipientType.TO, rcptAddr);
694             
695             //Set the sender...
696             reply.setFrom(notifier.toInternetAddress());
697             
698             //Create the message body
699             MimeMultipart multipart = new MimeMultipart();
700             //Add message as the first mime body part
701             MimeBodyPart part = new MimeBodyPart();
702             part.setContent(stringContent, "text/plain");
703             part.setHeader(RFC2822Headers.CONTENT_TYPE, "text/plain");
704             multipart.addBodyPart(part);
705             
706             reply.setContent(multipart);
707             reply.setHeader(RFC2822Headers.CONTENT_TYPE, multipart.getContentType());
708             
709             //Create the list of recipients in our MailAddress format
710             Set recipients = new HashSet();
711             recipients.add(senderMailAddress);
712             
713             //Set additional headers
714             if (reply.getHeader(RFC2822Headers.DATE)==null){
715                 reply.setHeader(RFC2822Headers.DATE, rfc822DateFormat.format(new java.util.Date()));
716             }
717             String subject = message.getSubject();
718             if (subject == null) {
719                 subject = "";
720             }
721             if (subject.indexOf("Re:") == 0){
722                 reply.setSubject(subject);
723             } else {
724                 reply.setSubject("Re:" + subject);
725             }
726             reply.setHeader(RFC2822Headers.IN_REPLY_TO, message.getMessageID());
727             
728             //Send it off...
729             getMailetContext().sendMail(notifier, recipients, reply);
730         }
731         catch (Exception e) {
732             log("Exception found sending reply", e);
733         }
734     }
