30  /***
31   * <P>Generates a response to the reverse-path address.
32   * Note that this is different than a mail-client's
33   * reply, which would use the Reply-To or From header.</P>
34   * <P>Bounced messages are attached in their entirety (headers and
35   * content) and the resulting MIME part type is "message/rfc822".<BR>
36   * The reverse-path and the Return-Path header of the response is set to "null" ("<>"),
37   * meaning that no reply should be sent.</P>
38   * <P>A sender of the notification message can optionally be specified.
39   * If one is not specified, the postmaster's address will be used.<BR>
40   * A notice text can be specified, and in such case will be inserted into the
41   * notification inline text.<BR>
42   * If the notified message has an "error message" set, it will be inserted into the
43   * notification inline text. If the <CODE>attachStackTrace</CODE> init parameter
44   * is set to true, such error message will be attached to the notification message.<BR>
45   * <P>Supports the <CODE>passThrough</CODE> init parameter (true if missing).</P>
46   *
47   * <P>Sample configuration:</P>
48   * <PRE><CODE>
49   * &lt;mailet match="All" class="Bounce">
50   *   &lt;sender&gt;<I>an address or postmaster or sender or unaltered, default=postmaster</I>&lt;/sender&gt;
51   *   &lt;attachError&gt;<I>true or false, default=false</I>&lt;/attachError&gt;
52   *   &lt;message&gt;<I>notice attached to the original message text (optional)</I>&lt;/message&gt;
53   *   &lt;prefix&gt;<I>optional subject prefix prepended to the original message</I>&lt;/prefix&gt;
54   *   &lt;inline&gt;<I>see {@link Resend}, default=none</I>&lt;/inline&gt;
55   *   &lt;attachment&gt;<I>see {@link Resend}, default=message</I>&lt;/attachment&gt;
56   *   &lt;passThrough&gt;<I>true or false, default=true</I>&lt;/passThrough&gt;
57   *   &lt;fakeDomainCheck&gt;<I>true or false, default=true</I>&lt;/fakeDomainCheck&gt;
58   *   &lt;debug&gt;<I>true or false, default=false</I>&lt;/debug&gt;
59   * &lt;/mailet&gt;
60   * </CODE></PRE>
61   *
62   * <P>The behaviour of this mailet is equivalent to using Resend with the following
63   * configuration:</P>
64   * <PRE><CODE>
65   * &lt;mailet match="All" class="Resend">
66   *   &lt;sender&gt;<I>an address or postmaster or sender or unaltered</I>&lt;/sender&gt;
67   *   &lt;attachError&gt;<I>true or false</I>&lt;/attachError&gt;
68   *   &lt;message&gt;<I><B>dynamically built</B></I>&lt;/message&gt;
69   *   &lt;prefix&gt;<I>a string</I>&lt;/prefix&gt;
70   *   &lt;passThrough&gt;true or false&lt;/passThrough&gt;
71   *   &lt;fakeDomainCheck&gt;<I>true or false</I>&lt;/fakeDomainCheck&gt;
72   *   &lt;recipients&gt;<B>sender</B>&lt;/recipients&gt;
73   *   &lt;reversePath&gt;null&lt;/reversePath&gt;
74   *   &lt;inline&gt;see {@link Resend}&lt;/inline&gt;
75   *   &lt;attachment&gt;see {@link Resend}&lt;/attachment&gt;
76   *   &lt;isReply&gt;true&lt;/isReply&gt;
77   *   &lt;debug&gt;<I>true or false</I>&lt;/debug&gt;
78   * &lt;/mailet&gt;
79   * </CODE></PRE>
80   * <P><I>notice</I> and <I>sendingAddress</I> can be used instead of
81   * <I>message</I> and <I>sender</I>; such names are kept for backward compatibility.</P>
82   *
83   * @version CVS $Revision: 411536 $ $Date: 2006-06-04 10:24:19 +0200 (dom, 04 giu 2006) $
84   * @since 2.2.0
85   */
86  public class Bounce extends AbstractNotify {

139      * @return <CODE>SpecialAddress.NULL</CODE> (the meaning of bounce)

149     /***
150      * Service does the hard work,and redirects the originalMail in the form specified.
151      * Checks that the original return path is not empty,
152      * and then calls super.service(originalMail), otherwise just returns.
153      *
154      * @param originalMail the mail to process and redirect
155      * @throws MessagingException if a problem arises formulating the redirected mail
156      */
157     public void service(Mail originalMail) throws MessagingException {
158         if (originalMail.getSender() == null) {
159             if (isDebug)
160                 log("Processing a bounce request for a message with an empty reverse-path.  No bounce will be sent.");
161             if(!getPassThrough(originalMail)) {
162                 originalMail.setState(Mail.GHOST);
163             }
164             return;
165         }
166 
167         if (isDebug)
168             log("Processing a bounce request for a message with a reverse path.  The bounce will be sent to " + originalMail.getSender().toString());
169 
170         super.service(originalMail);
171     }
