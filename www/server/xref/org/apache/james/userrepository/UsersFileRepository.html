37  /***
38   * Implementation of a Repository to store users on the File System.
39   *
40   * Requires a configuration element in the .conf.xml file of the form:
41   *  &lt;repository destinationURL="file://path-to-root-dir-for-repository"
42   *              type="USERS"
43   *              model="SYNCHRONOUS"/&gt;
44   * Requires a logger called UsersRepository.
45   *
46   *
47   * @version CVS $Revision: 420948 $
48   *
49   */
50  public class UsersFileRepository
51      extends AbstractLogEnabled
52      implements UsersRepository, Configurable, Serviceable, Initializable {
53   
54      /***
55       * Whether 'deep debugging' is turned on.
56       */
57      protected static boolean DEEP_DEBUG = false;
58  
59      private Store store;
60      private ObjectRepository objectRepository;
61  
62      /***
63       * The destination URL used to define the repository.
64       */
65      private String destination;
66  
67      public void setStore(Store store) {
68          this.store = store;
69      }
70  
71      /***
72       * @see org.apache.avalon.framework.service.Serviceable#service(ServiceManager)
73       */
74      public void service( final ServiceManager componentManager )
75          throws ServiceException {
76  
77          try {
78              setStore((Store)componentManager.lookup( Store.ROLE ));
79          } catch (Exception e) {
80              final String message = "Failed to retrieve Store component:" + e.getMessage();
81              getLogger().error( message, e );
82              throw new ServiceException ("", message, e );
83          }
84      }
85  
86      /***
87       * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
88       */
89      public void configure( final Configuration configuration )
90          throws ConfigurationException {
91  
92          destination = configuration.getChild( "destination" ).getAttribute( "URL" );
93  
94          if (!destination.endsWith(File.separator)) {
95              destination += File.separator;
96          }
97      }
98  
99      /***
100      * @see org.apache.avalon.framework.activity.Initializable#initialize()
101      */
102     public void initialize()
103         throws Exception {
104 
105         try {
106             //prepare Configurations for object and stream repositories
107             final DefaultConfiguration objectConfiguration
108                 = new DefaultConfiguration( "repository",
109                                             "generated:UsersFileRepository.compose()" );
110 
111             objectConfiguration.setAttribute( "destinationURL", destination );
112             objectConfiguration.setAttribute( "type", "OBJECT" );
113             objectConfiguration.setAttribute( "model", "SYNCHRONOUS" );
114 
115             objectRepository = (ObjectRepository)store.select( objectConfiguration );
116             if (getLogger().isDebugEnabled()) {
117                 StringBuffer logBuffer =
118                     new StringBuffer(192)
119                             .append(this.getClass().getName())
120                             .append(" created in ")
121                             .append(destination);
122                 getLogger().debug(logBuffer.toString());
123             }
124         } catch (Exception e) {
125             if (getLogger().isErrorEnabled()) {
126                 getLogger().error("Failed to initialize repository:" + e.getMessage(), e );
127             }
128             throw e;
129         }
130     }

132     /***
133      * List users in repository.
134      *
135      * @return Iterator over a collection of Strings, each being one user in the repository.

139     }

141     /***
142      * Update the repository with the specified user object. A user object
143      * with this username must already exist.
144      *
145      * @param user the user to be added.
146      *
147      * @return true if successful.
148      */
149     public synchronized boolean addUser(User user) {
150         String username = user.getUserName();
151         if (contains(username)) {
152             return false;
153         }
154         try {
155             objectRepository.put(username, user);
156         } catch (Exception e) {
157             throw new RuntimeException("Exception caught while storing user: " + e );
158         }
159         return true;
160     }
161 
162     public void addUser(String name, Object attributes) {
163         if (attributes instanceof String) {
164             User newbie = new DefaultUser(name, "SHA");
165             newbie.setPassword( (String) attributes);
166             addUser(newbie);
167         }
168         else {
169             throw new RuntimeException("Improper use of deprecated method"
170                                        + " - use addUser(User user)");
171         }
172     }
173 
174     public boolean addUser(String username, String password) {
175         User newbie = new DefaultJamesUser(username, "SHA");
176         newbie.setPassword(password);
177         return addUser(newbie);
178     }

180     public synchronized User getUserByName(String name) {
181         if (contains(name)) {
182             try {
183                 return (User)objectRepository.get(name);
184             } catch (Exception e) {
185                 throw new RuntimeException("Exception while retrieving user: "
186                                            + e.getMessage());
187             }
188         } else {
189             return null;
190         }
191     }
192 
193     public User getUserByNameCaseInsensitive(String name) {
194         String realName = getRealName(name);
195         if (realName == null ) {
196             return null;
197         }
198         return getUserByName(realName);
199     }

201     public String getRealName(String name) {
202         Iterator it = list();
203         while (it.hasNext()) {
204             String temp = (String) it.next();
205             if (name.equalsIgnoreCase(temp)) {
206                 return temp;
207             }
208         }
209         return null;
210     }

212     public boolean updateUser(User user) {
213         String username = user.getUserName();
214         if (!contains(username)) {
215             return false;
216         }
217         try {
218             objectRepository.put(username, user);

220             throw new RuntimeException("Exception caught while storing user: " + e );
221         }
222         return true;
223     }

225     public synchronized void removeUser(String name) {
226         objectRepository.remove(name);
227     }

229     public boolean contains(String name) {
230         return objectRepository.containsKey(name);
231     }
232 
233     public boolean containsCaseInsensitive(String name) {
234         Iterator it = list();
235         while (it.hasNext()) {
236             if (name.equalsIgnoreCase((String)it.next())) {
237                 return true;
238             }
239         }
240         return false;
241     }

243     public boolean test(String name, String password) {
244         User user;
245         try {
246             if (contains(name)) {
247                 user = (User) objectRepository.get(name);
248             } else {
249                return false;
250             }
251         } catch (Exception e) {
252             throw new RuntimeException("Exception retrieving User" + e);
253         }
254         return user.verifyPassword(password);
255     }

257     public int countUsers() {
258         int count = 0;
259         for (Iterator it = list(); it.hasNext(); it.next()) {
260             count++;
261         }
262         return count;
263     }
