104     /***
105      * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
106      */
107     public void configure( final Configuration configuration )
108         throws ConfigurationException {
109 
110         final boolean autodiscover =
111             configuration.getChild( "autodiscover" ).getValueAsBoolean( true );
112 
113         if (autodiscover) {
114             getLogger().info("Autodiscovery is enabled - trying to discover your system's DNS Servers");
115             String[] serversArray = ResolverConfig.getCurrentConfig().servers();
116             if (serversArray != null) {
117                 for ( int i = 0; i < serversArray.length; i++ ) {
118                     dnsServers.add(serversArray[ i ]);
119                     getLogger().info("Adding autodiscovered server " + serversArray[i]);
120                 }
121             }
122         }
123 
124         singleIPPerMX = configuration.getChild( "singleIPperMX" ).getValueAsBoolean( false ); 
125         
126         // Get the DNS servers that this service will use for lookups
127         final Configuration serversConfiguration = configuration.getChild( "servers" );
128         final Configuration[] serverConfigurations =
129             serversConfiguration.getChildren( "server" );
130 
131         for ( int i = 0; i < serverConfigurations.length; i++ ) {
132             dnsServers.add( serverConfigurations[ i ].getValue() );
133         }
134 
135         if (dnsServers.isEmpty()) {
136             getLogger().info("No DNS servers have been specified or found by autodiscovery - adding 127.0.0.1");
137             dnsServers.add("127.0.0.1");
138         }
139 
140         final boolean authoritative =
141             configuration.getChild( "authoritative" ).getValueAsBoolean( false );
142         // TODO: Check to see if the credibility field is being used correctly.  From the
143         //       docs I don't think so
144         dnsCredibility = authoritative ? Credibility.AUTH_ANSWER : Credibility.NONAUTH_ANSWER;
145 
146         maxCacheSize = (int) configuration.getChild( "maxcachesize" ).getValueAsLong( maxCacheSize );
147     }

149     /***
150      * @see org.apache.avalon.framework.activity.Initializable#initialize()
151      */
152     public void initialize()
153         throws Exception {
154 
155         getLogger().debug("DNSServer init...");
156 
157         // If no DNS servers were configured, default to local host
158         if (dnsServers.isEmpty()) {
159             try {
160                 dnsServers.add( InetAddress.getLocalHost().getHostName() );
161             } catch ( UnknownHostException ue ) {
162                 dnsServers.add( "127.0.0.1" );
163             }
164         }
165 
166         //Create the extended resolver...
167         final String[] serversArray = (String[])dnsServers.toArray(new String[0]);
168 
169         if (getLogger().isInfoEnabled()) {
170             for(int c = 0; c < serversArray.length; c++) {
171                 getLogger().info("DNS Server is: " + serversArray[c]);
172             }
173         }
174 
175         try {
176             resolver = new ExtendedResolver( serversArray );
177             Lookup.setDefaultResolver(resolver);
178         } catch (UnknownHostException uhe) {
179             getLogger().fatalError("DNS service could not be initialized.  The DNS servers specified are not recognized hosts.", uhe);
180             throw uhe;
181         }
182 
183         cache = new Cache (DClass.IN);
184         cache.setMaxEntries(maxCacheSize);
185         Lookup.setDefaultCache(cache, DClass.IN);
186         
187         getLogger().debug("DNSServer ...init end");
188     }

200     /***
201      * <p>Return a prioritized unmodifiable list of MX records
202      * obtained from the server.</p>
203      *
204      * @param hostname domain name to look up
205      *
206      * @return a list of MX records corresponding to this mail domain
207      */
208     public List findMXRecordsRaw(String hostname) {
209         Record answers[] = lookup(hostname, Type.MX);
210         List servers = new ArrayList();
211         if (answers == null) {
212             return servers;
213         }
214 
215         MXRecord mxAnswers[] = new MXRecord[answers.length];
216         for (int i = 0; i < answers.length; i++) {
217             mxAnswers[i] = (MXRecord)answers[i];
218         }
219 
220         Arrays.sort(mxAnswers, mxComparator);
221 
222         for (int i = 0; i < mxAnswers.length; i++) {
223             servers.add(mxAnswers[i].getTarget ().toString ());
224             getLogger().debug(new StringBuffer("Found MX record ").append(mxAnswers[i].getTarget ().toString ()).toString());
225         }
226         return servers;
227     }

229     /***
230      * <p>Return a prioritized unmodifiable list of host handling mail
231      * for the domain.</p>
232      * 
233      * <p>First lookup MX hosts, then MX hosts of the CNAME adress, and
234      * if no server is found return the IP of the hostname</p>
235      *
236      * @param hostname domain name to look up
237      *
238      * @return a unmodifiable list of handling servers corresponding to
239      *         this mail domain name
240      */
241     public Collection findMXRecords(String hostname) {
242         List servers = new ArrayList();
243         try {
244             servers = findMXRecordsRaw(hostname);
245             return Collections.unmodifiableCollection(servers);
246         } finally {
247             //If we found no results, we'll add the original domain name if
248             //it's a valid DNS entry
249             if (servers.size () == 0) {
250                 StringBuffer logBuffer =
251                     new StringBuffer(128)
252                             .append("Couldn't resolve MX records for domain ")
253                             .append(hostname)
254                             .append(".");
255                 getLogger().info(logBuffer.toString());
256                 Record cnames[] = lookup(hostname, Type.CNAME);
257                 Collection cnameMXrecords = null;
258                 if (cnames!=null && cnames.length > 0) {
259                     cnameMXrecords = findMXRecordsRaw(((CNAMERecord) cnames[0]).getTarget().toString());
260                 } else {
261                     logBuffer = new StringBuffer(128)
262                             .append("Couldn't find CNAME records for domain ")
263                             .append(hostname)
264                             .append(".");
265                     getLogger().info(logBuffer.toString());
266                 }
267                 if (cnameMXrecords==null) {
268                     try {
269                         getByName(hostname);
270                         servers.add(hostname);
271                     } catch (UnknownHostException uhe) {
272                         // The original domain name is not a valid host,
273                         // so we can't add it to the server list.  In this
274                         // case we return an empty list of servers
275                         logBuffer = new StringBuffer(128)
276                                   .append("Couldn't resolve IP address for host ")
277                                   .append(hostname)
278                                   .append(".");
279                         getLogger().error(logBuffer.toString());
280                     }
281                 } else {
282                     servers.addAll(cnameMXrecords);
283                 }
284             }
285         }
286     }

301     /***
302      * Looks up DNS records of the specified type for the specified name
303      *
304      * @param namestr the name of the host to be looked up
305      * @param querysent whether the query has already been sent to the DNS servers
306      * @param type the type of record desired
307      */
308     private Record[] rawDNSLookup(String namestr, boolean querysent, int type) {
309         Name name = null;
310         try {
311             name = Name.fromString(namestr, Name.root);
312         } catch (TextParseException tpe) {
313             // TODO: Figure out how to handle this correctly.
314             getLogger().error("Couldn't parse name " + namestr, tpe);
315             return null;
316         }
317         int dclass = DClass.IN;
318 
319         SetResponse cached = cache.lookupRecords(name, type, dnsCredibility);
320         if (cached.isSuccessful()) {
321             getLogger().debug(new StringBuffer(256)
322                              .append("Retrieving MX record for ")
323                              .append(name).append(" from cache")
324                              .toString());
325 
326             return processSetResponse(cached);
327         }
328         else if (cached.isNXDOMAIN() || cached.isNXRRSET()) {
329             return null;
330         }
331         else if (querysent) {
332             return null;
333         }
334         else {
335             getLogger().debug(new StringBuffer(256)
336                              .append("Looking up MX record for ")
337                              .append(name)
338                              .toString());
339             Record question = Record.newRecord(name, type, dclass);
340             Message query = Message.newQuery(question);
341             Message response = null;
342 
343             try {
344                 response = resolver.send(query);
345             }
346             catch (Exception ex) {
347                 getLogger().warn("Query error!", ex);
348                 return null;
349             }
350 
351             int rcode = response.getHeader().getRcode();
352             if (rcode == Rcode.NOERROR || rcode == Rcode.NXDOMAIN) {
353                 cached = cache.addMessage(response);
354                 if (cached != null && cached.isSuccessful()) {
355                     return processSetResponse(cached);
356                 }
357             }
358 
359             if (rcode != Rcode.NOERROR) {
360                 return null;
361             }
362 
363             return rawDNSLookup(namestr, true, type);
364         }
365     }

367     protected Record[] processSetResponse(SetResponse sr) {
368         Record [] answers;
369         int answerCount = 0, n = 0;
370 
371         RRset [] rrsets = sr.answers();
372         answerCount = 0;
373         for (int i = 0; i < rrsets.length; i++) {
374             answerCount += rrsets[i].size();
375         }
376 
377         answers = new Record[answerCount];
378 
379         for (int i = 0; i < rrsets.length; i++) {
380             Iterator iter = rrsets[i].rrs();
381             while (iter.hasNext()) {
382                 Record r = (Record)iter.next();
383                 answers[n++] = r;
384             }
385         }
386         return answers;
387     }

389     /* RFC 2821 section 5 requires that we sort the MX records by their
390      * preference, and introduce a randomization.  This Comparator does
391      * comparisons as normal unless the values are equal, in which case
392      * it "tosses a coin", randomly speaking.
393      *
394      * This way MX record w/preference 0 appears before MX record
395      * w/preference 1, but a bunch of MX records with the same preference
396      * would appear in different orders each time.
397      *
398      * Reminder for maintainers: the return value on a Comparator can
399      * be counter-intuitive for those who aren't used to the old C
400      * strcmp function:
401      *
402      * < 0 ==> a < b
403      * = 0 ==> a = b
404      * > 0 ==> a > b
405      */
406     private static class MXRecordComparator implements Comparator {
407         private final static Random random = new Random();
408         public int compare (Object a, Object b) {
409             int pa = ((MXRecord)a).getPriority();
410             int pb = ((MXRecord)b).getPriority();
411             return (pa == pb) ? (512 - random.nextInt(1024)) : pa - pb;
412         }
413     }

415     /*
416      * Returns an Iterator over org.apache.mailet.HostAddress, a
417      * specialized subclass of javax.mail.URLName, which provides
418      * location information for servers that are specified as mail
419      * handlers for the given hostname.  This is done using MX records,
420      * and the HostAddress instances are returned sorted by MX priority.
421      * If no host is found for domainName, the Iterator returned will be
422      * empty and the first call to hasNext() will return false.  The
423      * Iterator is a nested iterator: the outer iteration is over the
424      * results of the MX record lookup, and the inner iteration is over
425      * potentially multiple A records for each MX record.  DNS lookups
426      * are deferred until actually needed.
427      *
428      * @since v2.2.0a16-unstable
429      * @param domainName - the domain for which to find mail servers
430      * @return an Iterator over HostAddress instances, sorted by priority
431      */
432     public Iterator getSMTPHostAddresses(final String domainName) {
433         return new Iterator() {
434             private Iterator mxHosts = findMXRecords(domainName).iterator();
435             private Iterator addresses = null;
436 
437             public boolean hasNext() {
438                 /* Make sure that when next() is called, that we can
439                  * provide a HostAddress.  This means that we need to
440                  * have an inner iterator, and verify that it has
441                  * addresses.  We could, for example, run into a
442                  * situation where the next mxHost didn't have any valid
443                  * addresses.
444                  */
445                 if ((addresses == null || !addresses.hasNext()) && mxHosts.hasNext()) do {
446                     final String nextHostname = (String)mxHosts.next();
447                     InetAddress[] addrs = null;
448                     try {
449                         if (singleIPPerMX) {
450                             addrs = new InetAddress[] {getByName(nextHostname)};
451                         } else {
452                             addrs = getAllByName(nextHostname);
453                         }
454                     } catch (UnknownHostException uhe) {
455                         // this should never happen, since we just got
456                         // this host from mxHosts, which should have
457                         // already done this check.
458                         StringBuffer logBuffer = new StringBuffer(128)
459                                                  .append("Couldn't resolve IP address for discovered host ")
460                                                  .append(nextHostname)
461                                                  .append(".");
462                         getLogger().error(logBuffer.toString());
463                     }
464                     final InetAddress[] ipAddresses = addrs;
465 
466                     addresses = new Iterator() {
467                         int i = 0;
468 
469                         public boolean hasNext() {
470                             return ipAddresses != null && i < ipAddresses.length;
471                         }
472 
473                         public Object next() {
474                             return new org.apache.mailet.HostAddress(nextHostname, "smtp://" + ipAddresses[i++].getHostAddress());
475                         }
476 
477                         public void remove() {
478                             throw new UnsupportedOperationException ("remove not supported by this iterator");
479                         }
480                     };
481                 } while (!addresses.hasNext() && mxHosts.hasNext());
482 
483                 return addresses != null && addresses.hasNext();
484             }
485 
486             public Object next() {
487                 return addresses != null ? addresses.next() : null;
488             }
489 
490             public void remove() {
491                 throw new UnsupportedOperationException ("remove not supported by this iterator");
492             }
493         };
494     }

496     /* java.net.InetAddress.get[All]ByName(String) allows an IP literal
497      * to be passed, and will recognize it even with a trailing '.'.
498      * However, org.xbill.DNS.Address does not recognize an IP literal
499      * with a trailing '.' character.  The problem is that when we
500      * lookup an MX record for some domains, we may find an IP address,
501      * which will have had the trailing '.' appended by the time we get
502      * it back from dnsjava.  An MX record is not allowed to have an IP
503      * address as the right-hand-side, but there are still plenty of
504      * such records on the Internet.  Since java.net.InetAddress can
505      * handle them, for the time being we've decided to support them.
506      *
507      * These methods are NOT intended for use outside of James, and are
508      * NOT declared by the org.apache.james.services.DNSServer.  This is
509      * currently a stopgap measure to be revisited for the next release.
510      */
511 
512     private static String allowIPLiteral(String host) {
513         if ((host.charAt(host.length() - 1) == '.')) {
514             String possible_ip_literal = host.substring(0, host.length() - 1);
515             if (org.xbill.DNS.Address.isDottedQuad(possible_ip_literal)) {
516                 host = possible_ip_literal;
517             }
518         }
519         return host;
520     }
