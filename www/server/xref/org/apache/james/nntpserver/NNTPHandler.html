261     /***
262      * @see org.apache.james.core.AbstractJamesHandler#handleProtocol()
263      */
264     protected void handleProtocol() throws IOException {
265         // section 7.1
266         if ( theConfigData.getNNTPRepository().isReadOnly() ) {
267             StringBuffer respBuffer =
268                 new StringBuffer(128)
269                     .append("201 ")
270                     .append(theConfigData.getHelloName())
271                     .append(" NNTP Service Ready, posting prohibited");
272             writeLoggedFlushedResponse(respBuffer.toString());
273         } else {
274             StringBuffer respBuffer =
275                 new StringBuffer(128)
276                         .append("200 ")
277                         .append(theConfigData.getHelloName())
278                         .append(" NNTP Service Ready, posting permitted");
279             writeLoggedFlushedResponse(respBuffer.toString());
280         }
281 
282         theWatchdog.start();
283         while (parseCommand(inReader.readLine())) {
284             theWatchdog.reset();
285         }
286         theWatchdog.stop();
287 
288         getLogger().info("Connection closed");
289     }

321     /***
322      * This method parses NNTP commands read off the wire in handleConnection.
323      * Actual processing of the command (possibly including additional back and
324      * forth communication with the client) is delegated to one of a number of
325      * command specific handler methods.  The primary purpose of this method is
326      * to parse the raw command string to determine exactly which handler should
327      * be called.  It returns true if expecting additional commands, false otherwise.
328      *
329      * @param commandRaw the raw command string passed in over the socket
330      *
331      * @return whether additional commands are expected.
332      */
333     private boolean parseCommand(String commandRaw) {
334         if (commandRaw == null) {
335             return false;
336         }
337         if (getLogger().isDebugEnabled()) {
338             getLogger().debug("Command received: " + commandRaw);
339         }
340 
341         String command = commandRaw.trim();
342         String argument = null;
343         int spaceIndex = command.indexOf(" ");
344         if (spaceIndex >= 0) {
345             argument = command.substring(spaceIndex + 1);
346             command = command.substring(0, spaceIndex);
347         }
348         command = command.toUpperCase(Locale.US);
349 
350         boolean returnValue = true;
351         if (!isAuthorized(command) ) {
352             writeLoggedFlushedResponse("480 User is not authenticated");
353             getLogger().debug("Command not allowed.");
354             return returnValue;
355         }
356         if ((command.equals(COMMAND_MODE)) && (argument != null)) {
357             if (argument.toUpperCase(Locale.US).equals(MODE_TYPE_READER)) {
358                 doMODEREADER(argument);
359             } else if (argument.toUpperCase(Locale.US).equals(MODE_TYPE_STREAM)) {
360                 doMODESTREAM(argument);
361             } else {
362                 writeLoggedFlushedResponse("500 Command not understood");
363             }
364         } else if ( command.equals(COMMAND_LIST)) {
365             doLIST(argument);
366         } else if ( command.equals(COMMAND_GROUP) ) {
367             doGROUP(argument);
368         } else if ( command.equals(COMMAND_NEXT) ) {
369             doNEXT(argument);
370         } else if ( command.equals(COMMAND_LAST) ) {
371             doLAST(argument);
372         } else if ( command.equals(COMMAND_ARTICLE) ) {
373             doARTICLE(argument);
374         } else if ( command.equals(COMMAND_HEAD) ) {
375             doHEAD(argument);
376         } else if ( command.equals(COMMAND_BODY) ) {
377             doBODY(argument);
378         } else if ( command.equals(COMMAND_STAT) ) {
379             doSTAT(argument);
380         } else if ( command.equals(COMMAND_POST) ) {
381             doPOST(argument);
382         } else if ( command.equals(COMMAND_IHAVE) ) {
383             doIHAVE(argument);
384         } else if ( command.equals(COMMAND_QUIT) ) {
385             doQUIT(argument);
386             returnValue = false;
387         } else if ( command.equals(COMMAND_DATE) ) {
388             doDATE(argument);
389         } else if ( command.equals(COMMAND_HELP) ) {
390             doHELP(argument);
391         } else if ( command.equals(COMMAND_NEWGROUPS) ) {
392             doNEWGROUPS(argument);
393         } else if ( command.equals(COMMAND_NEWNEWS) ) {
394             doNEWNEWS(argument);
395         } else if ( command.equals(COMMAND_LISTGROUP) ) {
396             doLISTGROUP(argument);
397         } else if ( command.equals(COMMAND_OVER) ) {
398             doOVER(argument);
399         } else if ( command.equals(COMMAND_XOVER) ) {
400             doXOVER(argument);
401         } else if ( command.equals(COMMAND_HDR) ) {
402             doHDR(argument);
403         } else if ( command.equals(COMMAND_XHDR) ) {
404             doXHDR(argument);
405         } else if ( command.equals(COMMAND_AUTHINFO) ) {
406             doAUTHINFO(argument);
407         } else if ( command.equals(COMMAND_SLAVE) ) {
408             doSLAVE(argument);
409         } else if ( command.equals(COMMAND_PAT) ) {
410             doPAT(argument);
411         } else {
412             doUnknownCommand(command, argument);
413         }
414         return returnValue;
415     }

436     /***
437      * Implements only the originnal AUTHINFO.
438      * for simple and generic AUTHINFO, 501 is sent back. This is as
439      * per article 3.1.3 of RFC 2980
440      *
441      * @param argument the argument passed in with the AUTHINFO command
442      */
443     private void doAUTHINFO(String argument) {
444         String command = null;
445         String value = null;
446         if (argument != null) {
447             int spaceIndex = argument.indexOf(" ");
448             if (spaceIndex >= 0) {
449                 command = argument.substring(0, spaceIndex);
450                 value = argument.substring(spaceIndex + 1);
451             }
452         }
453         if (command == null) {
454             writeLoggedFlushedResponse("501 Syntax error");
455             return;
456         }
457         command = command.toUpperCase(Locale.US);
458         if ( command.equals(AUTHINFO_PARAM_USER) ) {
459             // Reject re-authentication
460             if ( isAlreadyAuthenticated ) {
461                 writeLoggedFlushedResponse("482 Already authenticated - rejecting new credentials");
462             }
463             // Reject doubly sent user
464             if (user != null) {
465                 user = null;
466                 password = null;
467                 isAlreadyAuthenticated = false;
468                 writeLoggedFlushedResponse("482 User already specified - rejecting new user");
469                 return;
470             }
471             user = value;
472             writeLoggedFlushedResponse("381 More authentication information required");
473         } else if ( command.equals(AUTHINFO_PARAM_PASS) ) {
474             // Reject password sent before user
475             if (user == null) {
476                 writeLoggedFlushedResponse("482 User not yet specified.  Rejecting user.");
477                 return;
478             }
479             // Reject doubly sent password
480             if (password != null) {
481                 user = null;
482                 password = null;
483                 isAlreadyAuthenticated = false;
484                 writeLoggedFlushedResponse("482 Password already specified - rejecting new password");
485                 return;
486             }
487             password = value;
488             isAlreadyAuthenticated = isAuthenticated();
489             if ( isAlreadyAuthenticated ) {
490                 writeLoggedFlushedResponse("281 Authentication accepted");
491             } else {
492                 writeLoggedFlushedResponse("482 Authentication rejected");
493                 // Clear bad authentication
494                 user = null;
495                 password = null;
496             }
497         } else {
498             writeLoggedFlushedResponse("501 Syntax error");
499             return;
500         }
501     }

503     /***
504      * Lists the articles posted since the date passed in as
505      * an argument.
506      *
507      * @param argument the argument passed in with the NEWNEWS command.
508      *                 Should be a wildmat followed by a date.
509      */
510     private void doNEWNEWS(String argument) {
511         // see section 11.4
512 
513         String wildmat = "*";
514 
515         if (argument != null) {
516             int spaceIndex = argument.indexOf(" ");
517             if (spaceIndex >= 0) {
518                 wildmat = argument.substring(0, spaceIndex);
519                 argument = argument.substring(spaceIndex + 1);
520             } else {
521                 getLogger().error("NEWNEWS had an invalid argument");
522                 writeLoggedFlushedResponse("501 Syntax error");
523                 return;
524             }
525         } else {
526             getLogger().error("NEWNEWS had a null argument");
527             writeLoggedFlushedResponse("501 Syntax error");
528             return;
529         }
530 
531         Date theDate = null;
532         try {
533             theDate = getDateFrom(argument);
534         } catch (NNTPException nntpe) {
535             getLogger().error("NEWNEWS had an invalid argument", nntpe);
536             writeLoggedFlushedResponse("501 Syntax error");
537             return;
538         }
539 
540         writeLoggedFlushedResponse("230 list of new articles by message-id follows");
541         Iterator groupIter = theConfigData.getNNTPRepository().getMatchedGroups(wildmat);
542         while ( groupIter.hasNext() ) {
543             Iterator articleIter = ((NNTPGroup)(groupIter.next())).getArticlesSince(theDate);
544             while (articleIter.hasNext()) {
545                 StringBuffer iterBuffer =
546                     new StringBuffer(64)
547                         .append(((NNTPArticle)articleIter.next()).getUniqueID());
548                 writeLoggedResponse(iterBuffer.toString());
549             }
550         }
551         writeLoggedFlushedResponse(".");
552     }

554     /***
555      * Lists the groups added since the date passed in as
556      * an argument.
557      *
558      * @param argument the argument passed in with the NEWGROUPS command.
559      *                 Should be a date.
560      */
561     private void doNEWGROUPS(String argument) {
562         // see section 11.3
563         // both draft-ietf-nntpext-base-15.txt and rfc977 have only group names 
564         // in response lines, but INN sends 
565         // '<group name> <last article> <first article> <posting allowed>'
566         // NOTE: following INN over either document.
567         //
568         // TODO: Check this.  Audit at http://www.academ.com/pipermail/ietf-nntp/2001-July/002185.html
569         // doesn't mention the supposed discrepancy.  Consider changing code to 
570         // be in line with spec.
571         Date theDate = null;
572         try {
573             theDate = getDateFrom(argument);
574         } catch (NNTPException nntpe) {
575             getLogger().error("NEWGROUPS had an invalid argument", nntpe);
576             writeLoggedFlushedResponse("501 Syntax error");
577             return;
578         }
579         Iterator iter = theConfigData.getNNTPRepository().getGroupsSince(theDate);
580         writeLoggedFlushedResponse("231 list of new newsgroups follows");
581         while ( iter.hasNext() ) {
582             NNTPGroup currentGroup = (NNTPGroup)iter.next();
583             StringBuffer iterBuffer =
584                 new StringBuffer(128)
585                     .append(currentGroup.getName())
586                     .append(" ")
587                     .append(currentGroup.getLastArticleNumber())
588                     .append(" ")
589                     .append(currentGroup.getFirstArticleNumber())
590                     .append(" ")
591                     .append((currentGroup.isPostAllowed()?"y":"n"));
592             writeLoggedResponse(iterBuffer.toString());
593         }
594         writeLoggedFlushedResponse(".");
595     }

637     /***
638      * Handles the LIST command and its assorted extensions.
639      *
640      * @param argument the argument passed in with the LIST command.
641      */
642     private void doLIST(String argument) {
643         // see section 9.4.1
644         String wildmat = "*";
645         boolean isListNewsgroups = false;
646 
647         String extension = argument;
648         if (argument != null) {
649             int spaceIndex = argument.indexOf(" ");
650             if (spaceIndex >= 0) {
651                 wildmat = argument.substring(spaceIndex + 1);
652                 extension = argument.substring(0, spaceIndex);
653             }
654             extension = extension.toUpperCase(Locale.US);
655         }
656 
657         if (extension != null) {
658             if (extension.equals("ACTIVE")) {
659                 isListNewsgroups = false;
660             } else if (extension.equals("NEWSGROUPS") ) {
661                 isListNewsgroups = true;
662             } else if (extension.equals("EXTENSIONS") ) {
663                 doLISTEXTENSIONS();
664                 return;
665             } else if (extension.equals("OVERVIEW.FMT") ) {
666                 doLISTOVERVIEWFMT();
667                 return;
668             } else if (extension.equals("ACTIVE.TIMES") ) {
669                 // not supported - 9.4.2.1, 9.4.3.1, 9.4.4.1
670                 writeLoggedFlushedResponse("503 program error, function not performed");
671                 return;
672             } else if (extension.equals("DISTRIBUTIONS") ) {
673                 // not supported - 9.4.2.1, 9.4.3.1, 9.4.4.1
674                 writeLoggedFlushedResponse("503 program error, function not performed");
675                 return;
676             } else if (extension.equals("DISTRIB.PATS") ) {
677                 // not supported - 9.4.2.1, 9.4.3.1, 9.4.4.1
678                 writeLoggedFlushedResponse("503 program error, function not performed");
679                 return;
680             } else {
681                 writeLoggedFlushedResponse("501 Syntax error");
682                 return;
683             }
684         }
685 
686         Iterator iter = theConfigData.getNNTPRepository().getMatchedGroups(wildmat);
687         writeLoggedFlushedResponse("215 list of newsgroups follows");
688         while ( iter.hasNext() ) {
689             NNTPGroup theGroup = (NNTPGroup)iter.next();
690             if (isListNewsgroups) {
691                 writeLoggedResponse(theGroup.getListNewsgroupsFormat());
692             } else {
693                 writeLoggedResponse(theGroup.getListFormat());
694             }
695         }
696         writeLoggedFlushedResponse(".");
697     }

1372     /***
1373      * Returns the date from @param input.
1374      * The input tokens are assumed to be in format date time [GMT|UTC] .
1375      * 'date' is in format [XX]YYMMDD. 'time' is in format 'HHMMSS'
1376      * NOTE: This routine could do with some format checks.
1377      *
1378      * @param argument the date string
1379      */
1380     private Date getDateFrom(String argument) {
1381         if (argument == null) {
1382             throw new NNTPException("Date argument was absent.");
1383         }
1384         StringTokenizer tok = new StringTokenizer(argument, " ");
1385         if (tok.countTokens() < 2) {
1386             throw new NNTPException("Date argument was ill-formed.");
1387         }
1388         String date = tok.nextToken();
1389         String time = tok.nextToken();
1390         boolean utc = ( tok.hasMoreTokens() );
1391         try {
1392             StringBuffer dateStringBuffer =
1393                 new StringBuffer(64)
1394                     .append(date)
1395                     .append(" ")
1396                     .append(time);
1397             Date dt = DF_RFC977.parse(dateStringBuffer.toString());
1398             if ( utc ) {
1399                 dt = new Date(dt.getTime()+UTC_OFFSET);
1400             }
1401             return dt;
1402         } catch ( ParseException pe ) {
1403             StringBuffer exceptionBuffer =
1404                 new StringBuffer(128)
1405                     .append("Date extraction failed: ")
1406                     .append(date)
1407                     .append(",")
1408                     .append(time)
1409                     .append(",")
1410                     .append(utc);
1411             throw new NNTPException(exceptionBuffer.toString());
1412         }
1413     }
