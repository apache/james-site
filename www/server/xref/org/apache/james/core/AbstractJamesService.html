190     /***
191      * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
192      */
193     public void configure(Configuration conf) throws ConfigurationException {
194         enabled = conf.getAttributeAsBoolean("enabled", true);
195         if (!enabled) {
196           getLogger().info(getServiceType() + " disabled by configuration");
197           return;
198         }
199 
200         Configuration handlerConfiguration = conf.getChild("handler");
201 
202         // Send the handler subconfiguration to the super class.  This 
203         // ensures that the handler config is passed to the handlers.
204         //
205         // TODO: This should be rationalized.  The handler element of the
206         //       server configuration doesn't really make a whole lot of 
207         //       sense.  We should modify the config to get rid of it.
208         //       Keeping it for now to maintain backwards compatibility.
209         super.configure(handlerConfiguration);
210 
211         port = conf.getChild("port").getValueAsInteger(getDefaultPort());
212 
213         Configuration serverSocketTypeConf = conf.getChild("serverSocketType", false);
214         String confSocketType = null;
215         if (serverSocketTypeConf != null ) {
216             confSocketType = serverSocketTypeConf.getValue();
217         }
218 
219         if (confSocketType == null) {
220             // Only load the useTLS parameter if a specific socket type has not
221             // been specified.  This maintains backwards compatibility while
222             // allowing us to have more complex (i.e. multiple SSL configuration)
223             // deployments
224             final boolean useTLS = conf.getChild("useTLS").getValueAsBoolean(isDefaultTLSEnabled());
225             if (useTLS) {
226               serverSocketType = "ssl";
227             }
228         } else {
229             serverSocketType = confSocketType;
230         }
231 
232         StringBuffer infoBuffer;
233         threadGroup = conf.getChild("threadGroup").getValue(null);
234         if (threadGroup != null) {
235             infoBuffer =
236                 new StringBuffer(64)
237                         .append(getServiceType())
238                         .append(" uses thread group: ")
239                         .append(threadGroup);
240             getLogger().info(infoBuffer.toString());
241         }
242         else {
243             getLogger().info(getServiceType() + " uses default thread group.");
244         }
245 
246         try {
247             final String bindAddress = conf.getChild("bind").getValue(null);
248             if( null != bindAddress ) {
249                 bindTo = InetAddress.getByName(bindAddress);
250                 infoBuffer =
251                     new StringBuffer(64)
252                             .append(getServiceType())
253                             .append(" bound to: ")
254                             .append(bindTo);
255                 getLogger().info(infoBuffer.toString());
256             }
257         }
258         catch( final UnknownHostException unhe ) {
259             throw new ConfigurationException( "Malformed bind parameter in configuration of service " + getServiceType(), unhe );
260         }
261 
262         configureHelloName(handlerConfiguration);
263 
264         timeout = handlerConfiguration.getChild(TIMEOUT_NAME).getValueAsInteger(DEFAULT_TIMEOUT);
265 
266         infoBuffer =
267             new StringBuffer(64)
268                     .append(getServiceType())
269                     .append(" handler connection timeout is: ")
270                     .append(timeout);
271         getLogger().info(infoBuffer.toString());
272 
273         backlog = conf.getChild(BACKLOG_NAME).getValueAsInteger(DEFAULT_BACKLOG);
274 
275         infoBuffer =
276                     new StringBuffer(64)
277                     .append(getServiceType())
278                     .append(" connection backlog is: ")
279                     .append(backlog);
280         getLogger().info(infoBuffer.toString());
281 
282         String connectionLimitString = conf.getChild("connectionLimit").getValue(null);
283         if (connectionLimitString != null) {
284             try {
285                 connectionLimit = new Integer(connectionLimitString);
286             } catch (NumberFormatException nfe) {
287                 getLogger().error("Connection limit value is not properly formatted.", nfe);
288             }
289             if (connectionLimit.intValue() < 0) {
290                 getLogger().error("Connection limit value cannot be less than zero.");
291                 throw new ConfigurationException("Connection limit value cannot be less than zero.");
292             }
293         } else {
294             connectionLimit = new Integer(connectionManager.getMaximumNumberOfOpenConnections());
295         }
296         infoBuffer = new StringBuffer(128)
297             .append(getServiceType())
298             .append(" will allow a maximum of ")
299             .append(connectionLimit.intValue())
300             .append(" connections.");
301         getLogger().info(infoBuffer.toString());
302     }

304     private void configureHelloName(Configuration handlerConfiguration) {
305         StringBuffer infoBuffer;
306         String hostName = null;
307         try {
308             hostName = InetAddress.getLocalHost().getHostName();
309         } catch (UnknownHostException ue) {
310             hostName = "localhost";
311         }
312 
313         infoBuffer =
314             new StringBuffer(64)
315                     .append(getServiceType())
316                     .append(" is running on: ")
317                     .append(hostName);
318         getLogger().info(infoBuffer.toString());
319 
320         Configuration helloConf = handlerConfiguration.getChild(HELLO_NAME);
321         boolean autodetect = helloConf.getAttributeAsBoolean("autodetect", true);
322         if (autodetect) {
323             helloName = hostName;
324         } else {
325             helloName = helloConf.getValue("localhost");
326         }
327         infoBuffer =
328             new StringBuffer(64)
329                     .append(getServiceType())
330                     .append(" handler hello name is: ")
331                     .append(helloName);
332         getLogger().info(infoBuffer.toString());
333     }

336      * @see org.apache.avalon.framework.activity.Initializable#initialize()
337      */
338     public void initialize() throws Exception {
339         if (!isEnabled()) {
340             getLogger().info(getServiceType() + " Disabled");
341             System.out.println(getServiceType() + " Disabled");
342             return;
343         }
344         getLogger().debug(getServiceType() + " init...");
345 
346         // keeping these looked up services locally, because they are only needed beyond initialization
347         ThreadManager threadManager = (ThreadManager) componentManager.lookup(ThreadManager.ROLE);
348         SocketManager socketManager = (SocketManager) componentManager.lookup(SocketManager.ROLE);
349         
350         initializeThreadPool(threadManager);
351 
352         initializeServerSocket(socketManager);
353 
354         getLogger().debug(getServiceType() + " ...init end");
355 
356         initializeHandlerPool();
357         
358         // do avalon specific preparations
359         ContainerUtil.enableLogging(theHandlerPool, getLogger());
360         ContainerUtil.initialize(theHandlerPool);
361 
362         theWatchdogFactory = getWatchdogFactory();
363 
364     }

366     private void initializeThreadPool(ThreadManager threadManager) {
367         if (threadGroup != null) {
368             threadPool = threadManager.getThreadPool(threadGroup);
369         } else {
370             threadPool = threadManager.getDefaultThreadPool();
371         }
372     }

374     private void initializeServerSocket(SocketManager socketManager) throws Exception {
375         ServerSocketFactory factory = socketManager.getServerSocketFactory(serverSocketType);
376         ServerSocket serverSocket = factory.createServerSocket(port, backlog, bindTo);
377 
378         if (null == connectionName) {
379             final StringBuffer sb = new StringBuffer();
380             sb.append(serverSocketType);
381             sb.append(':');
382             sb.append(port);
383 
384             if (null != bindTo) {
385                 sb.append('/');
386                 sb.append(bindTo);
387             }
388             connectionName = sb.toString();
389         }
390 
391         if ((connectionLimit != null)) {
392             if (null != threadPool) {
393                 connectionManager.connect(connectionName, serverSocket, this, threadPool, connectionLimit.intValue());
394             } else {
395                 connectionManager.connect(connectionName, serverSocket, this, connectionLimit.intValue()); // default pool
396             }
397         } else {
398             if (null != threadPool) {
399                 connectionManager.connect(connectionName, serverSocket, this, threadPool);
400             } else {
401                 connectionManager.connect(connectionName, serverSocket, this); // default pool
402             }
403         }
404     }

406     private void initializeHandlerPool() throws Exception {
407         StringBuffer logBuffer =
408                 new StringBuffer(64)
409                         .append(getServiceType())
410                         .append(" started ")
411                         .append(connectionName);
412         String logString = logBuffer.toString();
413         System.out.println(logString);
414         getLogger().info(logString);
415 
416         if (connectionLimit != null) {
417             theHandlerPool = new HardResourceLimitingPool((ObjectFactory) this, 5, connectionLimit.intValue());
418             if (getLogger().isDebugEnabled()) {
419                 getLogger().debug("Using a bounded pool for "+getServiceType()+" handlers with upper limit " + connectionLimit.intValue());
420             }
421         } else {
422             // NOTE: The maximum here is not a real maximum.  The handler pool will continue to
423             //       provide handlers beyond this value.
424             theHandlerPool = new DefaultPool((ObjectFactory) this, null, 5, 30);
425             getLogger().debug("Using an unbounded pool for "+getServiceType()+" handlers.");
426         }
427     }

429     /***
430      * @see org.apache.avalon.framework.activity.Disposable#dispose()
431      */
432     public void dispose() {
433 
434         if (!isEnabled()) {
435             return;
436         }
437 
438         if( m_disposed )
439         {
440             if( getLogger().isWarnEnabled() )
441             {
442                 getLogger().warn( "ignoring disposal request - already disposed" );
443             }
444             return;
445         }
446 
447         if( getLogger().isDebugEnabled() )
448         {
449             getLogger().debug( "disposal" );
450         }
451 
452         m_disposed = true;
453         if( getLogger().isDebugEnabled() )
454         {
455             StringBuffer infoBuffer =
456                new StringBuffer(64).append(getServiceType()).append(
457                    " dispose... ").append(connectionName);
458             getLogger().debug(infoBuffer.toString());
459         }
460 
461         try {
462             connectionManager.disconnect(connectionName, true);
463         } catch (final Exception e) {
464             StringBuffer warnBuffer =
465                 new StringBuffer(64)
466                         .append("Error disconnecting ")
467                         .append(getServiceType())
468                         .append(": ");
469             getLogger().warn(warnBuffer.toString(), e);
470         }
471 
472         componentManager = null;
473 
474         connectionManager = null;
475         threadPool = null;
476 
477         // This is needed to make sure sockets are promptly closed on Windows 2000
478         // TODO: Check this - shouldn't need to explicitly gc to force socket closure
479         System.gc();
480 
481         getLogger().debug(getServiceType() + " ...dispose end");
482     }

513     protected ConnectionHandler newHandler()
514             throws Exception {
515         AbstractJamesHandler theHandler = (AbstractJamesHandler)theHandlerPool.get();
516         
517         if (getLogger().isDebugEnabled()) {
518             getLogger().debug("Getting Handler from pool.");
519         }
520 
521         Watchdog theWatchdog = theWatchdogFactory.getWatchdog(theHandler.getWatchdogTarget());
522 
523         theHandler.setConfigurationData(getConfigurationData());
524         theHandler.setWatchdog(theWatchdog);
525         return theHandler;

563     /***
564      * This method returns the type of service provided by this server.
565      * This should be invariant over the life of the class.
566      *
567      * Subclasses may override this implementation.  This implementation
568      * parses the complete class name and returns the undecorated class
569      * name.
570      *
571      * @return description of this server
572      */
573     public String getServiceType() {
574         String name = getClass().getName();
575         int p = name.lastIndexOf(".");
576         if (p > 0 && p < name.length() - 2) {
577             name = name.substring(p + 1);
578         }
579         return name;
580     }
