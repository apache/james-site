105     /***
106      * @see org.apache.avalon.framework.service.Serviceable#service(ServiceManager)
107      */
108     public void service( final ServiceManager componentManager )
109         throws ServiceException {
110         super.service(componentManager);
111         MailServer mailServer = (MailServer)componentManager.lookup(MailServer.ROLE );
112         setMailServer(mailServer);
113         Store store = (Store)componentManager.
114             lookup( "org.apache.avalon.cornerstone.services.store.Store" );
115         setStore(store);
116         UsersStore usersStore = (UsersStore)componentManager. lookup(UsersStore.ROLE );
117         setUsersStore(usersStore);
118         UsersRepository users = (UsersRepository) componentManager.lookup(UsersRepository.ROLE);
119         if (users == null) {
120             throw new ServiceException("","The user repository could not be found.");
121         }
122         setUsers(users);
123         SpoolManagementService spoolManagement = 
124             (SpoolManagementService) componentManager.lookup(SpoolManagementService.ROLE);
125         setSpoolManagement(spoolManagement);
126     }

128     /***
129      * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
130      */
131     public void configure( final Configuration configuration )
132         throws ConfigurationException {
133 
134         super.configure(configuration);
135         if (isEnabled()) {
136             Configuration handlerConfiguration = configuration.getChild("handler");
137             Configuration admin = handlerConfiguration.getChild( "administrator_accounts" );
138             Configuration[] accounts = admin.getChildren( "account" );
139             for ( int i = 0; i < accounts.length; i++ ) {
140                 adminAccounts.put( accounts[ i ].getAttribute( "login" ),
141                                    accounts[ i ].getAttribute( "password" ) );
142             }
143             Configuration promtConfiguration = handlerConfiguration.getChild("prompt", false);
144             if (promtConfiguration != null) prompt = promtConfiguration.getValue();
145             if (prompt == null) prompt = ""; 
146             else if (!prompt.equals("") && !prompt.endsWith(" ")) prompt += " "; 
147         }
148     }
