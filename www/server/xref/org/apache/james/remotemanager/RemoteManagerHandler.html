170     /***
171      * @see org.apache.avalon.cornerstone.services.connection.ConnectionHandler#handleConnection(Socket)
172      */
173     protected void handleProtocol() throws IOException {
174         writeLoggedResponse("JAMES Remote Administration Tool " + Constants.SOFTWARE_VERSION );
175         writeLoggedResponse("Please enter your login and password");
176         String login = null;
177         String password = null;
178         do {
179             if (login != null) {
180                 final String message = "Login failed for " + login;
181                 writeLoggedFlushedResponse(message);
182             }
183             writeLoggedFlushedResponse("Login id:");
184             login = inReader.readLine().trim();
185             writeLoggedFlushedResponse("Password:");
186             password = inReader.readLine().trim();
187         } while (!password.equals(theConfigData.getAdministrativeAccountData().get(login)) || password.length() == 0);
188 
189         StringBuffer messageBuffer =
190             new StringBuffer(64)
191                     .append("Welcome ")
192                     .append(login)
193                     .append(". HELP for a list of commands");
194         out.println( messageBuffer.toString() );
195         out.flush();
196         if (getLogger().isInfoEnabled()) {
197             StringBuffer infoBuffer =
198                 new StringBuffer(128)
199                         .append("Login for ")
200                         .append(login)
201                         .append(" successful");
202             getLogger().info(infoBuffer.toString());
203         }
204 
205         try {
206             out.print(theConfigData.getPrompt());
207             out.flush();
208             theWatchdog.start();
209             while (parseCommand(inReader.readLine())) {
210                 theWatchdog.reset();
211                 out.print(theConfigData.getPrompt());
212                 out.flush();
213             }
214             theWatchdog.stop();
215         } catch (IOException ioe) {
216             //We can cleanly ignore this as it's probably a socket timeout
217         } catch (Throwable thr) {
218             System.out.println("Exception: " + thr.getMessage());
219             getLogger().error("Encountered exception in handling the remote manager connection.", thr);
220         }
221         StringBuffer infoBuffer =
222             new StringBuffer(64)
223                     .append("Logout for ")
224                     .append(login)
225                     .append(".");
226         getLogger().info(infoBuffer.toString());
227 
228     }

260     /***
261      * <p>This method parses and processes RemoteManager commands read off the
262      * wire in handleConnection.  It returns true if expecting additional
263      * commands, false otherwise.</p>
264      *
265      * @param rawCommand the raw command string passed in over the socket
266      *
267      * @return whether additional commands are expected.
268      */
269     private boolean parseCommand( String rawCommand ) {
270         if (rawCommand == null) {
271             return false;
272         }
273         String command = rawCommand.trim();
274         String argument = null;
275         int breakIndex = command.indexOf(" ");
276         if (breakIndex > 0) {
277             argument = command.substring(breakIndex + 1);
278             command = command.substring(0, breakIndex);
279         }
280         command = command.toUpperCase(Locale.US);
281         if (command.equals(COMMAND_ADDUSER)) {
282             doADDUSER(argument);
283         } else if (command.equals(COMMAND_SETPASSWORD)) {
284             return doSETPASSWORD(argument);
285         } else if (command.equals(COMMAND_DELUSER)) {
286             return doDELUSER(argument);
287         } else if (command.equals(COMMAND_LISTUSERS)) {
288             return doLISTUSERS(argument);
289         } else if (command.equals(COMMAND_COUNTUSERS)) {
290             return doCOUNTUSERS(argument);
291         } else if (command.equals(COMMAND_VERIFY)) {
292             return doVERIFY(argument);
293         } else if (command.equals(COMMAND_HELP)) {
294             return doHELP(argument);
295         } else if (command.equals(COMMAND_SETALIAS)) {
296             return doSETALIAS(argument);
297         } else if (command.equals(COMMAND_SETFORWARDING)) {
298             return doSETFORWARDING(argument);
299         } else if (command.equals(COMMAND_SHOWALIAS)) {
300             return doSHOWALIAS(argument);
301         } else if (command.equals(COMMAND_SHOWFORWARDING)) {
302             return doSHOWFORWARDING(argument);
303         } else if (command.equals(COMMAND_UNSETALIAS)) {
304             return doUNSETALIAS(argument);
305         } else if (command.equals(COMMAND_UNSETFORWARDING)) {
306             return doUNSETFORWARDING(argument);
307         } else if (command.equals(COMMAND_USER)) {
308             return doUSER(argument);
309         } else if (command.equals(COMMAND_LISTSPOOL)) {
310             return doLISTSPOOL(argument);
311         } else if (command.equals(COMMAND_FLUSHSPOOL)) {
312             return doFLUSHSPOOL(argument);
313         } else if (command.equals(COMMAND_DELETESPOOL)) {
314             return doDELETESPOOL(argument);
315         } else if (command.equals(COMMAND_QUIT)) {
316             return doQUIT(argument);
317         } else if (command.equals(COMMAND_SHUTDOWN)) {
318             return doSHUTDOWN(argument);
319         } else {
320             return doUnknownCommand(rawCommand);
321         }
322         return true;
323     }

325     /***
326      * Handler method called upon receipt of an ADDUSER command.
327      * Returns whether further commands should be read off the wire.
328      *
329      * @param argument the argument passed in with the command
330      */
331     private boolean doADDUSER(String argument) {
332         int breakIndex = -1;
333         if ((argument == null) ||
334             (argument.equals("")) ||
335             ((breakIndex = argument.indexOf(" ")) < 0)) {
336             writeLoggedFlushedResponse("Usage: adduser [username] [password]");
337             return true;
338         }
339         String username = argument.substring(0,breakIndex);
340         String passwd = argument.substring(breakIndex + 1);
341         if (username.equals("") || passwd.equals("")) {
342             writeLoggedFlushedResponse("Usage: adduser [username] [password]");
343             return true;
344         }
345 
346         boolean success = false;
347         if (users.contains(username)) {
348             StringBuffer responseBuffer =
349                 new StringBuffer(64)
350                         .append("User ")
351                         .append(username)
352                         .append(" already exists");
353             String response = responseBuffer.toString();
354             writeLoggedResponse(response);
355         } else {
356             success = users.addUser(username, passwd);
357         }
358         if ( success ) {
359             StringBuffer responseBuffer =
360                 new StringBuffer(64)
361                         .append("User ")
362                         .append(username)
363                         .append(" added");
364             String response = responseBuffer.toString();
365             out.println(response);
366             getLogger().info(response);
367         } else {
368             out.println("Error adding user " + username);
369             getLogger().error("Error adding user " + username);
370         }
371         out.flush();
372         return true;
373     }

564      */
565     private boolean doSETALIAS(String argument) {
566         int breakIndex = -1;
567         if ((argument == null) ||
568             (argument.equals("")) ||
569             ((breakIndex = argument.indexOf(" ")) < 0)) {
570             writeLoggedFlushedResponse("Usage: setalias [username] [emailaddress]");
571             return true;
572         }
573         String username = argument.substring(0,breakIndex);
574         String alias = argument.substring(breakIndex + 1);
575         if (username.equals("") || alias.equals("")) {
576             writeLoggedFlushedResponse("Usage: setalias [username] [alias]");
577             return true;
578         }
579 
580         User baseuser = users.getUserByName(username);
581         if (baseuser == null) {
582             writeLoggedFlushedResponse("No such user " + username);
583             return true;
584         }
585         if (! (baseuser instanceof JamesUser ) ) {
586             writeLoggedFlushedResponse("Can't set alias for this user type.");
587             return true;
588         }
589 
590         JamesUser user = (JamesUser) baseuser;
591         JamesUser aliasUser = (JamesUser) users.getUserByName(alias);
592         if (aliasUser == null) {
593             writeLoggedFlushedResponse("Alias unknown to server - create that user first.");
594             return true;
595         }
596 
597         boolean success = user.setAlias(alias);
598         if (success) {
599             user.setAliasing(true);
600             users.updateUser(user);
601             StringBuffer responseBuffer =
602                 new StringBuffer(64)
603                         .append("Alias for ")
604                         .append(username)
605                         .append(" set to:")
606                         .append(alias);
607             String response = responseBuffer.toString();
608             out.println(response);
609             getLogger().info(response);
610         } else {
611             out.println("Error setting alias");
612             getLogger().error("Error setting alias");
613         }
614         out.flush();
615         return true;
616     }

621      *
622      * @param argument the argument passed in with the command
623      */
624     private boolean doSETFORWARDING(String argument) {
625         int breakIndex = -1;
626         if ((argument == null) ||
627             (argument.equals("")) ||
628             ((breakIndex = argument.indexOf(" ")) < 0)) {
629             writeLoggedFlushedResponse("Usage: setforwarding [username] [emailaddress]");
630             return true;
631         }
632         String username = argument.substring(0,breakIndex);
633         String forward = argument.substring(breakIndex + 1);
634         if (username.equals("") || forward.equals("")) {
635            writeLoggedFlushedResponse("Usage: setforwarding [username] [emailaddress]");
636            return true;
637         }
638         // Verify user exists
639         User baseuser = users.getUserByName(username);
640         if (baseuser == null) {
641             writeLoggedFlushedResponse("No such user " + username);
642             return true;
643         } else if (! (baseuser instanceof JamesUser ) ) {
644             writeLoggedFlushedResponse("Can't set forwarding for this user type.");
645             return true;
646         }
647         JamesUser user = (JamesUser)baseuser;
648         // Verify acceptable email address
649         MailAddress forwardAddr;
650         try {
651              forwardAddr = new MailAddress(forward);
652         } catch(ParseException pe) {
653             writeLoggedResponse("Parse exception with that email address: " + pe.getMessage());
654             writeLoggedFlushedResponse("Forwarding address not added for " + username);
655             return true;
656         }
657 
658         boolean success = user.setForwardingDestination(forwardAddr);
659         if (success) {
660             user.setForwarding(true);
661             users.updateUser(user);
662             StringBuffer responseBuffer =
663                 new StringBuffer(64)
664                         .append("Forwarding destination for ")
665                         .append(username)
666                         .append(" set to:")
667                         .append(forwardAddr.toString());
668             String response = responseBuffer.toString();
669             out.println(response);
670             getLogger().info(response);
671         } else {
672             out.println("Error setting forwarding");
673             getLogger().error("Error setting forwarding");
674         }
675         out.flush();
676         return true;
677     }

681      * Returns whether further commands should be read off the wire.
682      *
683      * @param username the user name
684      */
685     private boolean doSHOWALIAS(String username) {
686         if ( username == null || username.equals("") ) {
687             writeLoggedFlushedResponse("Usage: showalias [username]");
688             return true;
689         }
690 
691 
692         User baseuser = users.getUserByName(username);
693         if (baseuser == null) {
694             writeLoggedFlushedResponse("No such user " + username);
695             return true;
696         } else if (! (baseuser instanceof JamesUser ) ) {
697             writeLoggedFlushedResponse("Can't show aliases for this user type.");
698             return true;
699         }
700 
701         JamesUser user = (JamesUser)baseuser;
702         if ( user == null ) {
703             writeLoggedFlushedResponse("No such user " + username);
704             return true;
705         }
706 
707         if ( !user.getAliasing() ) {
708             writeLoggedFlushedResponse("User " + username + " does not currently have an alias");
709             return true;
710         }
711 
712         String alias = user.getAlias();
713 
714         if ( alias == null || alias.equals("") ) {    //  defensive programming -- neither should occur
715             String errmsg = "For user " + username + ", the system indicates that aliasing is set but no alias was found";
716             out.println(errmsg);
717             getLogger().error(errmsg);
718             return true;
719         }
720 
721         writeLoggedFlushedResponse("Current alias for " + username + " is: " + alias);
722         return true;
723     }
