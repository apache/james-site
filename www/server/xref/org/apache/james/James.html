307     private void initializeServernamesAndPostmaster() throws ConfigurationException, ParseException {
308         String hostName = null;
309         try {
310             hostName = InetAddress.getLocalHost().getHostName();
311         } catch  (UnknownHostException ue) {
312             hostName = "localhost";
313         }
314 
315         context = new DefaultContext();
316         context.put("HostName", hostName);
317         getLogger().info("Local host is: " + hostName);
318 
319         // Get the domains and hosts served by this instance
320         serverNames = new HashSet();
321         Configuration serverConf = conf.getChild("servernames");
322         if (serverConf.getAttributeAsBoolean("autodetect") && (!hostName.equals("localhost"))) {
323             serverNames.add(hostName.toLowerCase(Locale.US));
324         }
325 
326         final Configuration[] serverNameConfs =
327                 conf.getChild( "servernames" ).getChildren( "servername" );
328         for ( int i = 0; i < serverNameConfs.length; i++ ) {
329             serverNames.add( serverNameConfs[i].getValue().toLowerCase(Locale.US));
330 
331             if (serverConf.getAttributeAsBoolean("autodetectIP", true)) {
332                 try {
333                     /* This adds the IP address(es) for each host to support
334                     * support <user@address-literal> - RFC 2821, sec 4.1.3.
335                     * It might be proper to use the actual IP addresses
336                     * available on this server, but we can't do that
337                     * without NetworkInterface from JDK 1.4.  Because of
338                     * Virtual Hosting considerations, we may need to modify
339                     * this to keep hostname and IP associated, rather than
340                     * just both in the set.
341                     */
342                     InetAddress[] addrs = InetAddress.getAllByName(serverNameConfs[i].getValue());
343                     for (int j = 0; j < addrs.length ; j++) {
344                         serverNames.add(addrs[j].getHostAddress());
345                     }
346                 }
347                 catch(Exception genericException) {
348                     getLogger().error("Cannot get IP address(es) for " + serverNameConfs[i].getValue());
349                 }
350             }
351         }
352         if (serverNames.isEmpty()) {
353             throw new ConfigurationException( "Fatal configuration error: no servernames specified!");
354         }
355 
356         if (getLogger().isInfoEnabled()) {
357             for (Iterator i = serverNames.iterator(); i.hasNext(); ) {
358                 getLogger().info("Handling mail for: " + i.next());
359             }
360         }
361 
362         String defaultDomain = (String) serverNames.iterator().next();
363         context.put(Constants.DEFAULT_DOMAIN, defaultDomain);
364         attributes.put(Constants.DEFAULT_DOMAIN, defaultDomain);
365 
366         // Get postmaster
367         String postMasterAddress = conf.getChild("postmaster").getValue("postmaster").toLowerCase(Locale.US);
368         // if there is no @domain part, then add the first one from the
369         // list of supported domains that isn't localhost.  If that
370         // doesn't work, use the hostname, even if it is localhost.
371         if (postMasterAddress.indexOf('@') < 0) {
372             String domainName = null;    // the domain to use
373             // loop through candidate domains until we find one or exhaust the list
374             for ( int i = 0; domainName == null && i < serverNameConfs.length ; i++ ) {
375                 String serverName = serverNameConfs[i].getValue().toLowerCase(Locale.US);
376                 if (!("localhost".equals(serverName))) {
377                     domainName = serverName;    // ok, not localhost, so use it
378                 }
379             }
380             // if we found a suitable domain, use it.  Otherwise fallback to the host name.
381             postMasterAddress = postMasterAddress + "@" + (domainName != null ? domainName : hostName);
382         }
383         this.postmaster = new MailAddress( postMasterAddress );
384         context.put( Constants.POSTMASTER, postmaster );
385 
386         if (!isLocalServer(postmaster.getHost())) {
387             StringBuffer warnBuffer
388                     = new StringBuffer(320)
389                     .append("The specified postmaster address ( ")
390                     .append(postmaster)
391                     .append(" ) is not a local address.  This is not necessarily a problem, but it does mean that emails addressed to the postmaster will be routed to another server.  For some configurations this may cause problems.");
392             getLogger().warn(warnBuffer.toString());
393         }
394     }

424     /***
425      * Place a mail on the spool for processing
426      *
427      * @param message the message to send
428      *
429      * @throws MessagingException if an exception is caught while placing the mail
430      *                            on the spool
431      */
432     public void sendMail(MimeMessage message) throws MessagingException {
433         MailAddress sender = new MailAddress((InternetAddress)message.getFrom()[0]);
434         Collection recipients = new HashSet();
435         Address addresses[] = message.getAllRecipients();
436         if (addresses != null) {
437             for (int i = 0; i < addresses.length; i++) {
438                 // Javamail treats the "newsgroups:" header field as a
439                 // recipient, so we want to filter those out.
440                 if ( addresses[i] instanceof InternetAddress ) {
441                     recipients.add(new MailAddress((InternetAddress)addresses[i]));
442                 }
443             }
444         }
445         sendMail(sender, recipients, message);
446     }

481     /***
482      * Place a mail on the spool for processing
483      *
484      * @param sender the sender of the mail
485      * @param recipients the recipients of the mail
486      * @param msg an <code>InputStream</code> containing the message
487      *
488      * @throws MessagingException if an exception is caught while placing the mail
489      *                            on the spool
490      */
491     public void sendMail(MailAddress sender, Collection recipients, InputStream msg)
492             throws MessagingException {
493         // parse headers
494         MailHeaders headers = new MailHeaders(msg);
495 
496         // if headers do not contains minimum REQUIRED headers fields throw Exception
497         if (!headers.isValid()) {
498             throw new MessagingException("Some REQURED header field is missing. Invalid Message");
499         }
500         ByteArrayInputStream headersIn = new ByteArrayInputStream(headers.toByteArray());
501         sendMail(new MailImpl(getId(), sender, recipients, new SequenceInputStream(headersIn, msg)));
502     }

504     /***
505      * Place a mail on the spool for processing
506      *
507      * @param mail the mail to place on the spool
508      *
509      * @throws MessagingException if an exception is caught while placing the mail
510      *                            on the spool
511      */
512     public void sendMail(Mail mail) throws MessagingException {
513         try {
514             spool.store(mail);
515         } catch (Exception e) {
516             getLogger().error("Error storing message: " + e.getMessage(),e);
517             try {
518                 spool.remove(mail);
519             } catch (Exception ignored) {
520                 getLogger().error("Error removing message after an error storing it: " + e.getMessage(),e);
521             }
522             throw new MessagingException("Exception spooling message: " + e.getMessage(), e);
523         }
524         if (getLogger().isDebugEnabled()) {
525             StringBuffer logBuffer =
526                 new StringBuffer(64)
527                         .append("Mail ")
528                         .append(mail.getName())
529                         .append(" pushed in spool");
530             getLogger().debug(logBuffer.toString());
531         }
532     }

534     /***
535      * <p>Retrieve the mail repository for a user</p>
536      *
537      * <p>For POP3 server only - at the moment.</p>
538      *
539      * @param userName the name of the user whose inbox is to be retrieved
540      *
541      * @return the POP3 inbox for the user
542      */
543     public synchronized MailRepository getUserInbox(String userName) {
544         MailRepository userInbox = null;
545 
546         userInbox = (MailRepository) mailboxes.get(userName);
547 
548         if (userInbox != null) {
549             return userInbox;
550         } else if (mailboxes.containsKey(userName)) {
551             // we have a problem
552             getLogger().error("Null mailbox for non-null key");
553             throw new RuntimeException("Error in getUserInbox.");
554         } else {
555             // need mailbox object
556             if (getLogger().isDebugEnabled()) {
557                 getLogger().debug("Retrieving and caching inbox for " + userName );
558             }
559             StringBuffer destinationBuffer =
560                 new StringBuffer(192)
561                         .append(inboxRootURL)
562                         .append(userName)
563                         .append("/");
564             String destination = destinationBuffer.toString();
565             try {
566                 // Copy the inboxRepository configuration and modify the destinationURL
567                 DefaultConfiguration mboxConf = new DefaultConfiguration(conf
568                         .getChild("inboxRepository").getChild("repository"));
569                 mboxConf.setAttribute("destinationURL", destination);
570 
571                 userInbox = (MailRepository) store.select(mboxConf);
572                 if (userInbox!=null) {
573                     mailboxes.put(userName, userInbox);
574                 }
575             } catch (Exception e) {
576                 if (getLogger().isErrorEnabled()) {
577                     getLogger().error("Cannot open user Mailbox",e);
578                 }
579                 throw new RuntimeException("Error in getUserInbox.",e);
580             }
581             return userInbox;
582         }
583     }

661     /***
662      * This generates a response to the Return-Path address, or the
663      * address of the message's sender if the Return-Path is not
664      * available.  Note that this is different than a mail-client's
665      * reply, which would use the Reply-To or From header.
666      *
667      * Bounced messages are attached in their entirety (headers and
668      * content) and the resulting MIME part type is "message/rfc822".
669      *
670      * The attachment to the subject of the original message (or "No
671      * Subject" if there is no subject in the original message)
672      *
673      * There are outstanding issues with this implementation revolving
674      * around handling of the return-path header.
675      *
676      * MIME layout of the bounce message:
677      *
678      * multipart (mixed)/
679      *     contentPartRoot (body) = mpContent (alternative)/
680      *           part (body) = message
681      *     part (body) = original
682      *
683      */
684 
685     public void bounce(Mail mail, String message, MailAddress bouncer) throws MessagingException {
686         if (mail.getSender() == null) {
687             if (getLogger().isInfoEnabled())
688                 getLogger().info("Mail to be bounced contains a null (<>) reverse path.  No bounce will be sent.");
689             return;
690         } else {
691             // Bounce message goes to the reverse path, not to the Reply-To address
692             if (getLogger().isInfoEnabled())
693                 getLogger().info("Processing a bounce request for a message with a reverse path of " + mail.getSender().toString());
694         }
695 
696         MailImpl reply = rawBounce(mail,message);
697         //Change the sender...
698         reply.getMessage().setFrom(bouncer.toInternetAddress());
699         reply.getMessage().saveChanges();
700         //Send it off ... with null reverse-path
701         reply.setSender(null);
702         sendMail(reply);
703     }

706      * Generates a bounce mail that is a bounce of the original message.
707      *
708      * @param bounceText the text to be prepended to the message to describe the bounce condition
709      *
710      * @return the bounce mail
711      *
712      * @throws MessagingException if the bounce mail could not be created
713      */
714     private MailImpl rawBounce(Mail mail, String bounceText) throws MessagingException {
715         //This sends a message to the james component that is a bounce of the sent message
716         MimeMessage original = mail.getMessage();
717         MimeMessage reply = (MimeMessage) original.reply(false);
718         reply.setSubject("Re: " + original.getSubject());
719         reply.setSentDate(new Date());
720         Collection recipients = new HashSet();
721         recipients.add(mail.getSender());
722         InternetAddress addr[] = { new InternetAddress(mail.getSender().toString())};
723         reply.setRecipients(Message.RecipientType.TO, addr);
724         reply.setFrom(new InternetAddress(mail.getRecipients().iterator().next().toString()));
725         reply.setText(bounceText);
726         reply.setHeader(RFC2822Headers.MESSAGE_ID, "replyTo-" + mail.getName());
727         return new MailImpl(
728             "replyTo-" + mail.getName(),
729             new MailAddress(mail.getRecipients().iterator().next().toString()),
730             recipients,
731             reply);
732     }
