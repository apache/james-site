337     /***
338      * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
339      */
340     protected void configure(Configuration conf) throws ConfigurationException
341     {
342         setHost(conf.getChild("host").getValue());
343 
344         setFetchTaskName(conf.getAttribute("name"));
345         setJavaMailProviderName(
346             conf.getChild("javaMailProviderName").getValue());
347         setJavaMailFolderName(conf.getChild("javaMailFolderName").getValue());
348         setRecurse(conf.getChild("recursesubfolders").getValueAsBoolean());
349 
350         Configuration recipientNotFound = conf.getChild("recipientnotfound");
351         setDeferRecipientNotFound(
352             recipientNotFound.getAttributeAsBoolean("defer"));
353         setRejectRecipientNotFound(
354             recipientNotFound.getAttributeAsBoolean("reject"));
355         setLeaveRecipientNotFound(
356             recipientNotFound.getAttributeAsBoolean("leaveonserver"));
357         setMarkRecipientNotFoundSeen(
358             recipientNotFound.getAttributeAsBoolean("markseen"));
359 
360         Configuration defaultDomainName = conf.getChild("defaultdomain", false);
361         if (null != defaultDomainName)
362             setDefaultDomainName(defaultDomainName.getValue());
363 
364         setFetchAll(conf.getChild("fetchall").getValueAsBoolean());
365 
366         Configuration fetched = conf.getChild("fetched");
367         setLeave(fetched.getAttributeAsBoolean("leaveonserver"));
368         setMarkSeen(fetched.getAttributeAsBoolean("markseen"));
369 
370         Configuration remoterecipient = conf.getChild("remoterecipient");
371         setRejectRemoteRecipient(
372             remoterecipient.getAttributeAsBoolean("reject"));
373         setLeaveRemoteRecipient(
374             remoterecipient.getAttributeAsBoolean("leaveonserver"));
375         setMarkRemoteRecipientSeen(
376             remoterecipient.getAttributeAsBoolean("markseen"));
377 
378         Configuration blacklist = conf.getChild("blacklist");
379         setBlacklist(blacklist.getValue(""));
380         setRejectBlacklisted(blacklist.getAttributeAsBoolean("reject"));
381         setLeaveBlacklisted(blacklist.getAttributeAsBoolean("leaveonserver"));
382         setMarkBlacklistedSeen(blacklist.getAttributeAsBoolean("markseen"));
383 
384         Configuration userundefined = conf.getChild("userundefined");
385         setRejectUserUndefined(userundefined.getAttributeAsBoolean("reject"));
386         setLeaveUserUndefined(
387             userundefined.getAttributeAsBoolean("leaveonserver"));
388         setMarkUserUndefinedSeen(
389             userundefined.getAttributeAsBoolean("markseen"));
390 
391         Configuration undeliverable = conf.getChild("undeliverable");
392         setLeaveUndeliverable(
393             undeliverable.getAttributeAsBoolean("leaveonserver"));
394         setMarkUndeliverableSeen(
395             undeliverable.getAttributeAsBoolean("markseen"));
396 
397         Configuration remotereceivedheader = conf.getChild("remotereceivedheader", false);
398         if (null != remotereceivedheader)
399         {
400             setRemoteReceivedHeaderIndex(
401                 remotereceivedheader.getAttributeAsInteger("index"));
402             setRejectRemoteReceivedHeaderInvalid(
403                 remotereceivedheader.getAttributeAsBoolean("reject"));
404             setLeaveRemoteReceivedHeaderInvalid(
405                 remotereceivedheader.getAttributeAsBoolean("leaveonserver"));
406             setMarkRemoteReceivedHeaderInvalidSeen(
407                 remotereceivedheader.getAttributeAsBoolean("markseen"));
408         }            
409 
410         Configuration maxmessagesize = conf.getChild("maxmessagesize", false);
411         if (null != maxmessagesize)
412         {
413             setMaxMessageSizeLimit(
414                 maxmessagesize.getAttributeAsInteger("limit") * 1024);
415             setRejectMaxMessageSizeExceeded(
416                 maxmessagesize.getAttributeAsBoolean("reject"));
417             setLeaveMaxMessageSizeExceeded(
418                 maxmessagesize.getAttributeAsBoolean("leaveonserver"));
419             setMarkMaxMessageSizeExceededSeen(
420                 maxmessagesize.getAttributeAsBoolean("markseen"));
421         }
422 
423         if (getLogger().isDebugEnabled())
424         {
425             getLogger().info(
426                 "Configured FetchMail fetch task " + getFetchTaskName());
427         }
428     }

488             && !isMarkSeen()
489             && isLeaveBlacklisted()
490             && !isMarkBlacklistedSeen()
491             && isLeaveRemoteRecipient()
492             && !isMarkRemoteRecipientSeen()
493             && isLeaveUserUndefined()
494             && !isMarkUserUndefinedSeen()
495             && isLeaveUndeliverable()
496             && !isMarkUndeliverableSeen()
497             && isLeaveMaxMessageSizeExceeded()
498             && !isMarkMaxMessageSizeExceededSeen()              
499             && isLeaveRemoteReceivedHeaderInvalid()
500             && !isMarkRemoteReceivedHeaderInvalidSeen()

695     /***
696      * Sets the blacklist.
697      * @param blacklist The blacklist to set
698      */
699     protected void setBlacklist(String blacklistValue)
700         throws ConfigurationException
701     {
702         StringTokenizer st = new StringTokenizer(blacklistValue, ", \t", false);
703         Set blacklist = new HashSet();
704         String token = null;
705         while (st.hasMoreTokens())
706         {
707             try
708             {
709                 token = st.nextToken();
710                 blacklist.add(new MailAddress(token));
711             }
712             catch (ParseException pe)
713             {
714                 throw new ConfigurationException(
715                     "Invalid blacklist mail address specified: " + token);
716             }
717         }
718         setBlacklist(blacklist);
719     }

872     /***
873      * Validates the defaultDomainName.
874      * @param defaultDomainName The defaultDomainName to validate
875      */
876     protected void validateDefaultDomainName(String defaultDomainName) throws ConfigurationException
877     {
878         if (!getServer().isLocalServer(defaultDomainName))
879         {
880             throw new ConfigurationException(
881                 "Default domain name is not a local server: "
882                     + defaultDomainName);
883         }

897             InetAddress addr1 = java.net.InetAddress.getLocalHost();
