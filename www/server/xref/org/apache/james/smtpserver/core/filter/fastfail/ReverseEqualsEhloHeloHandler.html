18  package org.apache.james.smtpserver.core.filter.fastfail;
19  
20  import org.apache.avalon.framework.configuration.Configurable;
21  import org.apache.avalon.framework.configuration.Configuration;
22  import org.apache.avalon.framework.configuration.ConfigurationException;
23  import org.apache.avalon.framework.logger.AbstractLogEnabled;
24  import org.apache.avalon.framework.service.ServiceException;
25  import org.apache.avalon.framework.service.ServiceManager;
26  import org.apache.avalon.framework.service.Serviceable;
27  import org.apache.james.services.DNSServer;
28  import org.apache.james.smtpserver.CommandHandler;
29  import org.apache.james.smtpserver.SMTPSession;
30  import org.apache.james.util.mail.dsn.DSNStatus;
31  
32  import java.net.UnknownHostException;
33  import java.util.ArrayList;
34  import java.util.Collection;
35  
36  public class ReverseEqualsEhloHeloHandler extends AbstractLogEnabled
37          implements CommandHandler, Configurable, Serviceable {
38  
39      private boolean checkAuthNetworks = false;
40  
41      private DNSServer dnsServer = null;
42  
43      /***
44       * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
45       */
46      public void configure(Configuration handlerConfiguration)
47              throws ConfigurationException {
48          Configuration configRelay = handlerConfiguration.getChild(
49                  "checkAuthNetworks", false);
50          if (configRelay != null) {
51              setCheckAuthNetworks(configRelay.getValueAsBoolean(false));
52          }
53      }
54  
55      /***
56       * @see org.apache.avalon.framework.service.Serviceable#service(ServiceManager)
57       */
58      public void service(ServiceManager serviceMan) throws ServiceException {
59          setDnsServer((DNSServer) serviceMan.lookup(DNSServer.ROLE));
60      }
61  
62      /***
63       * Set to true if AuthNetworks should be included in the EHLO check
64       * 
65       * @param checkAuthNetworks
66       *            Set to true to enable
67       */
68      public void setCheckAuthNetworks(boolean checkAuthNetworks) {
69          this.checkAuthNetworks = checkAuthNetworks;
70      }
71  
72      /***
73       * Set the DNSServer
74       * 
75       * @param dnsServer
76       *            The DNSServer
77       */
78      public void setDnsServer(DNSServer dnsServer) {
79          this.dnsServer = dnsServer;
80      }
81  
82      /***
83       * @see org.apache.james.smtpserver.CommandHandler#onCommand(SMTPSession)
84       */
85      public void onCommand(SMTPSession session) {
86          String argument = session.getCommandArgument();
87          String responseString = null;
88  
89          /***
90           * don't check if the ip address is allowed to relay. Only check if it
91           * is set in the config. ed.
92           */
93          if (!session.isRelayingAllowed() || checkAuthNetworks) {
94              try {
95                  // get reverse entry
96                  String reverse = dnsServer.getByName(
97                          session.getRemoteIPAddress()).getHostName();
98  
99                  if (!argument.equals(reverse)) {
100                     responseString = "501 "
101                             + DSNStatus.getStatus(DSNStatus.PERMANENT,
102                                     DSNStatus.DELIVERY_INVALID_ARG)
103                             + " Provided EHLO " + argument
104                             + " not equal reverse of "
105                             + session.getRemoteIPAddress();
106 
107                     session.writeResponse(responseString);
108                     getLogger().info(responseString);
109 
110                     // After this filter match we should not call any other handler!
111                     session.setStopHandlerProcessing(true);
112                 }
113             } catch (UnknownHostException e) {
114                 responseString = "501 "
115                         + DSNStatus.getStatus(DSNStatus.PERMANENT,
116                                 DSNStatus.DELIVERY_INVALID_ARG) + " Ipaddress "
117                         + session.getRemoteIPAddress() + " can not resolved";
118 
119                 session.writeResponse(responseString);
120                 getLogger().info(responseString);
121 
122                 // After this filter match we should not call any other handler!
123                 session.setStopHandlerProcessing(true);
124             }
125         }
126     }
127     
128     /***
129      * @see org.apache.james.smtpserver.CommandHandler#getImplCommands()
130      */
131     public Collection getImplCommands() {
132         Collection implCommands = new ArrayList();
133         implCommands.add("EHLO");
134         implCommands.add("HELO");
135         
136         return implCommands;
137     }
