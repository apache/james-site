34  /***
35   * This MessageHandler could be used to check message against spamd before
36   * accept the email. So its possible to reject a message on smtplevel if a
37   * configured hits amount is reached. The handler add the follow attributes to
38   * the mail object:<br>
39   * org.apache.james.spamassassin.status - Holds the status
40   * org.apache.james.spamassassin.flag - Holds the flag <br>
41   * 
42   * Sample Configuration: <br>
43   * <br>
44   * &lt;handler class="org.apache.james.smtpserver.SpamAssassinHandler"&gt;
45   * &lt;spamdHost&gt;localhost&lt;/spamdHost&gt;
46   * &lt;spamdPort&gt;783&lt;/spamdPort&gt; <br>
47   * &lt;spamdRejectionHits&gt;15.0&lt;/spamdRejectionHits&gt;
48   * &lt;checkAuthNetworks&gt;false&lt;/checkAuthNetworks&gt; &lt;/handler&gt;
49   */
50  public class SpamAssassinHandler extends AbstractLogEnabled implements
51          MessageHandler, Configurable {
52  
53      /***
54       * The port spamd is listen on
55       */
56      private int spamdPort = 783;
57  
58      /***
59       * The host spamd is runnin on
60       */
61      private String spamdHost = "localhost";
62  
63      /***
64       * The hits on which the message get rejected
65       */
66      private double spamdRejectionHits = 0.0;
67  
68      private boolean checkAuthNetworks = false;
69  
70      /***
71       * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
72       */
73      public void configure(Configuration arg0) throws ConfigurationException {
74          Configuration spamdHostConf = arg0.getChild("spamdHost", false);
75          if (spamdHostConf != null) {
76              setSpamdHost(spamdHostConf.getValue("localhost"));
77          }
78  
79          Configuration spamdPortConf = arg0.getChild("spamdPort", false);
80          if (spamdPortConf != null) {
81              setSpamdPort(spamdPortConf.getValueAsInteger(783));
82          }
83  
84          Configuration spamdRejectionHitsConf = arg0.getChild(
85                  "spamdRejectionHits", false);
86          if (spamdRejectionHitsConf != null) {
87              setSpamdRejectionHits(spamdRejectionHitsConf.getValueAsDouble(0.0));
88          }
89  
90          Configuration configRelay = arg0.getChild("checkAuthNetworks", false);
91          if (configRelay != null) {
92              setCheckAuthNetworks(configRelay.getValueAsBoolean(false));
93          }
94  
95      }
96  
97      /***
98       * Set to true if AuthNetworks should be included in the EHLO check
99       * 
100      * @param checkAuthNetworks
101      *            Set to true to enable
102      */
103     public void setCheckAuthNetworks(boolean checkAuthNetworks) {
104         this.checkAuthNetworks = checkAuthNetworks;
105     }
106 
107     /***
108      * Set the host the spamd daemon is running at
109      * 
110      * @param spamdHost
111      *            The spamdHost
112      */
113     public void setSpamdHost(String spamdHost) {
114         this.spamdHost = spamdHost;
115     }
116 
117     /***
118      * Set the port the spamd damon is listen on
119      * 
120      * @param spamdPort
121      *            the spamdPort
122      */
123     public void setSpamdPort(int spamdPort) {
124         this.spamdPort = spamdPort;
125     }
126 
127     /***
128      * Set the hits on which the message will be rejected.
129      * 
130      * @param spamdRejectionHits
131      *            The hits
132      */
133     public void setSpamdRejectionHits(double spamdRejectionHits) {
134         this.spamdRejectionHits = spamdRejectionHits;
135 
136     }
137 
138     /***
139      * @see org.apache.james.smtpserver.MessageHandler#onMessage(SMTPSession)
140      */
141     public void onMessage(SMTPSession session) {
142 
143         // Not scan the message if relaying allowed
144         if (session.isRelayingAllowed() && !checkAuthNetworks) {
145             return;
146         }
147 
148         try {
149             Mail mail = session.getMail();
150             MimeMessage message = mail.getMessage();
151             SpamAssassinInvoker sa = new SpamAssassinInvoker(spamdHost,
152                     spamdPort);
153             sa.scanMail(message);
154 
155             Iterator headers = sa.getHeadersAsAttribute().keySet().iterator();
156 
157             // Add the headers
158             while (headers.hasNext()) {
159                 String key = headers.next().toString();
160 
161                 mail.setAttribute(key, (String) sa.getHeadersAsAttribute().get(
162                         key));
163             }
164 
165             // Check if rejectionHits was configured
166             if (spamdRejectionHits > 0) {
167                 try {
168                     double hits = Double.parseDouble(sa.getHits());
169 
170                     // if the hits are bigger the rejectionHits reject the
171                     // message
172                     if (spamdRejectionHits <= hits) {
173                         String responseString = "554 "
174                                 + DSNStatus.getStatus(DSNStatus.PERMANENT,
175                                         DSNStatus.SECURITY_OTHER)
176                                 + " This message reach the spam hits treshold. Please contact the Postmaster if the email is not SPAM. Message rejected";
177                         StringBuffer buffer = new StringBuffer(256).append(
178                                 "Rejected message from ").append(
179                                 session.getState().get(SMTPSession.SENDER)
180                                         .toString()).append(" from host ")
181                                 .append(session.getRemoteHost()).append(" (")
182                                 .append(session.getRemoteIPAddress()).append(
183                                         ") " + responseString).append(
184                                         ". Required rejection hits: "
185                                                 + spamdRejectionHits
186                                                 + " hits: " + hits);
187                         getLogger().info(buffer.toString());
188                         session.writeResponse(responseString);
189 
190                         // Message reject .. abort it!
191                         session.abortMessage();
192                     }
193                 } catch (NumberFormatException e) {
194                     // hits unknown
195                 }
196             }
197         } catch (MessagingException e) {
198             getLogger().error(e.getMessage());
199         }
200 
201     }
