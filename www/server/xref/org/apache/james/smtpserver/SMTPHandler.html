173     /***
174      * @see org.apache.james.core.AbstractJamesHandler#handleProtocol()
175      */
176     protected void handleProtocol() throws IOException {
177         smtpID = random.nextInt(1024) + "";
178         relayingAllowed = theConfigData.isRelayingAllowed(remoteIP);
179         authRequired = theConfigData.isAuthRequired(remoteIP);
180         heloEhloEnforcement = theConfigData.useHeloEhloEnforcement();
181         sessionEnded = false;
182         smtpGreeting = theConfigData.getSMTPGreeting();
183         resetState();
184         resetConnectionState();
185 
186         // if no greeting was configured use a default
187         if (smtpGreeting == null) {
188             // Initially greet the connector
189             // Format is:  Sat, 24 Jan 1998 13:16:09 -0500
190 
191             responseBuffer.append("220 ")
192                           .append(theConfigData.getHelloName())
193                           .append(" SMTP Server (")
194                           .append(SOFTWARE_TYPE)
195                           .append(") ready ")
196                           .append(rfc822DateFormat.format(new Date()));
197         } else {
198             responseBuffer.append("220 ")
199                           .append(smtpGreeting);
200         }
201         String responseString = clearResponseBuffer();
202         writeLoggedFlushedResponse(responseString);
203 
204         //the core in-protocol handling logic
205         //run all the connection handlers, if it fast fails, end the session
206         //parse the command command, look up for the list of command handlers
207         //Execute each of the command handlers. If any command handlers writes
208         //response then, End the subsequent command handler processing and
209         //start parsing new command. Once the message is received, run all
210         //the message handlers. The message handlers can either terminate
211         //message or terminate session
212 
213         //At the beginning
214         //mode = command_mode
215         //once the commandHandler writes response, the mode is changed to RESPONSE_MODE.
216         //This will cause to skip the subsequent command handlers configured for that command.
217         //For instance:
218         //There are 2 commandhandlers MailAddressChecker and MailCmdHandler for
219         //MAIL command. If MailAddressChecker validation of the MAIL FROM
220         //address is successful, the MailCmdHandlers will be executed.
221         //Incase it fails, it has to write response. Once we write response
222         //there is no need to execute the MailCmdHandler.
223         //Next, Once MAIL message is received the DataCmdHandler and any other
224         //equivalent commandHandler will call setMail method. this will change
225         //he mode to MAIL_RECEIVED_MODE. This mode will trigger the message
226         //handlers to be execute. Message handlers can abort message. In that case,
227         //message will not spooled.
228 
229         //Session started - RUN all connect handlers
230         List connectHandlers = handlerChain.getConnectHandlers();
231         if(connectHandlers != null) {
232             int count = connectHandlers.size();
233             for(int i = 0; i < count; i++) {
234                 ((ConnectHandler)connectHandlers.get(i)).onConnect(this);
235                 if(sessionEnded) {
236                     break;
237                 }
238             }
239         }
240 
241         theWatchdog.start();
242         while(!sessionEnded) {
243           //Reset the current command values
244           curCommandName = null;
245           curCommandArgument = null;
246           mode = COMMAND_MODE;
247 
248           //parse the command
249           String cmdString =  readCommandLine();
250           if (cmdString == null) {
251               break;
252           }
253           int spaceIndex = cmdString.indexOf(" ");
254           if (spaceIndex > 0) {
255               curCommandName = cmdString.substring(0, spaceIndex);
256               curCommandArgument = cmdString.substring(spaceIndex + 1);
257           } else {
258               curCommandName = cmdString;
259           }
260           curCommandName = curCommandName.toUpperCase(Locale.US);
261 
262           //fetch the command handlers registered to the command
263           List commandHandlers = handlerChain.getCommandHandlers(curCommandName);
264           if(commandHandlers == null) {
265               //end the session
266               break;
267           } else {
268               int count = commandHandlers.size();
269               for(int i = 0; i < count; i++) {
270                   setStopHandlerProcessing(false);
271                   ((CommandHandler)commandHandlers.get(i)).onCommand(this);
272                   
273                   theWatchdog.reset();
274                   
275                   //if the response is received, stop processing of command handlers
276                   if(mode != COMMAND_MODE || getStopHandlerProcessing()) {
277                       break;
278                   }
279               }
280 
281           }
282 
283           //handle messages
284           if(mode == MESSAGE_RECEIVED_MODE) {
285               getLogger().info("executing message handlers");
286               List messageHandlers = handlerChain.getMessageHandlers();
287               int count = messageHandlers.size();
288               for(int i =0; i < count; i++) {
289                   ((MessageHandler)messageHandlers.get(i)).onMessage(this);
290                   //if the response is received, stop processing of command handlers
291                   if(mode == MESSAGE_ABORT_MODE) {
292                       break;
293                   }
294               }
295           }
296 
297           //do the clean up
298           if(mail != null) {
299               ContainerUtil.dispose(mail);
300               
301               // remember the ehlo mode
302               Object currentHeloMode = state.get(CURRENT_HELO_MODE);
303               
304               mail = null;
305               resetState();
306 
307               // start again with the old helo mode
308               if (currentHeloMode != null) {
309                   state.put(CURRENT_HELO_MODE,currentHeloMode);
310               }
311           }
312 
313         }
314         theWatchdog.stop();
315         getLogger().debug("Closing socket.");
316     }

492     /***
493      * @see org.apache.james.smtpserver.SMTPSession#readCommandLine()
494      */
495     public final String readCommandLine() throws IOException {
496         for (;;) try {
497             String commandLine = inReader.readLine();
498             if (commandLine != null) {
499                 commandLine = commandLine.trim();
500             }
501             return commandLine;
502         } catch (CRLFTerminatedReader.TerminationException te) {
503             writeLoggedFlushedResponse("501 Syntax error at character position " + te.position() + ". CR and LF must be CRLF paired.  See RFC 2821 #2.7.1.");
504         } catch (CRLFTerminatedReader.LineLengthExceededException llee) {
505             writeLoggedFlushedResponse("500 Line length exceeded. See RFC 2821 #4.5.3.1.");
506         }
507     }
