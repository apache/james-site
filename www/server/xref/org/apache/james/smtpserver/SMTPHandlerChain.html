74      /***
75       * loads the various handlers from the configuration
76       * 
77       * @param configuration
78       *            configuration under handlerchain node
79       */
80      public void configure(Configuration configuration)
81              throws ConfigurationException {
82          addToMap(UnknownCmdHandler.UNKNOWN_COMMAND, unknownHandler);
83          if (configuration == null
84                  || configuration.getChildren("handler") == null
85                  || configuration.getChildren("handler").length == 0) {
86              configuration = new DefaultConfiguration("handlerchain");
87              Properties cmds = new Properties();
88              cmds.setProperty("Default CoreCmdFilterHandlerLoader",
89                      CoreFilterCmdHandlerLoader.class.getName());
90              cmds.setProperty("Default CoreCmdHandlerLoader", CoreCmdHandlerLoader.class
91                      .getName());
92              cmds.setProperty("Default SendMailHandler", SendMailHandler.class
93                      .getName());
94              Enumeration e = cmds.keys();
95              while (e.hasMoreElements()) {
96                  String cmdName = (String) e.nextElement();
97                  String className = cmds.getProperty(cmdName);
98                  ((DefaultConfiguration) configuration).addChild(addHandler(
99                          cmdName, className));
100             }
101         }
102         if (configuration != null) {
103             Configuration[] children = configuration.getChildren("handler");
104             ClassLoader classLoader = getClass().getClassLoader();
105 
106             // load the BaseFilterCmdHandler
107             loadClass(classLoader, CoreFilterCmdHandlerLoader.class.getName(),
108                     addHandler(null, CoreFilterCmdHandlerLoader.class.getName()));
109 
110             // load the configured handlers
111             if (children != null) {
112                 for (int i = 0; i < children.length; i++) {
113                     String className = children[i].getAttribute("class");
114                     if (className != null) {
115 
116                         // ignore base handlers.
117                         if (!className.equals(CoreFilterCmdHandlerLoader.class
118                                 .getName())
119                                 && !className.equals(CoreCmdHandlerLoader.class
120                                         .getName())
121                                 && !className.equals(SendMailHandler.class
122                                         .getName())) {
123 
124                             // load the handler
125                             loadClass(classLoader, className, children[i]);
126                         }
127                     }
128                 }
129 
130                 // load the BaseCmdHandler and SendMailHandler
131                 loadClass(classLoader, CoreCmdHandlerLoader.class.getName(),
132                         addHandler(null, CoreCmdHandlerLoader.class.getName()));
133                 loadClass(classLoader, SendMailHandler.class.getName(),
134                         addHandler(null, SendMailHandler.class.getName()));
135             }
136         }
137 
138         // the size must be greater than 1 because we added UnknownCmdHandler to
139         // the map
140 
141         if (commandHandlerMap.size() < 2) {
142             if (getLogger().isErrorEnabled()) {
143                 getLogger().error("No commandhandlers configured");
144             }
145             throw new ConfigurationException("No commandhandlers configured");
146         } else {
147             boolean found = true;
148             for (int i = 0; i < mandatoryCommands.length; i++) {
149                 if (!commandHandlerMap.containsKey(mandatoryCommands[i])) {
150                     if (getLogger().isErrorEnabled()) {
151                         getLogger().error(
152                                 "No commandhandlers configured for the command:"
153                                         + mandatoryCommands[i]);
154                     }
155                     found = false;
156                     break;
157                 }
158             }
159 
160             if (!found) {
161                 throw new ConfigurationException(
162                         "No commandhandlers configured for mandatory commands");
163             }
164 
165             if (messageHandlers.size() == 0) {
166                 if (getLogger().isErrorEnabled()) {
167                     getLogger()
168                             .error(
169                                     "No messageHandler configured. Check that SendMailHandler is configured in the SMTPHandlerChain");
170                 }
171                 throw new ConfigurationException("No messageHandler configured");
172             }
173 
174         }
175 
176     }

178     /***
179      * @see org.apache.avalon.framework.activity.Initializable#initialize()
180      */
181     public void initialize() throws Exception {
182         Iterator h = commandHandlerMap.keySet().iterator();
183     
184         while(h.hasNext()) {
185             List handlers = (List) commandHandlerMap.get(h.next());
186             Iterator h2 = handlers.iterator();
187             while (h2.hasNext()) {
188               ContainerUtil.initialize(h2.next());
189             }
190         }
191     }

193     /***
194      * Load and add the classes to the handler map
195      * 
196      * @param classLoader The classLoader to use
197      * @param className The class name 
198      * @param config The configuration 
199      * @throws ConfigurationException Get thrown on error
200      */
201     private void loadClass(ClassLoader classLoader, String className,
202             Configuration config) throws ConfigurationException {
203         try {
204             Object handler = classLoader.loadClass(className).newInstance();
205 
206             // enable logging
207             ContainerUtil.enableLogging(handler, getLogger());
208 
209             ContainerUtil.contextualize(handler, context);
210 
211             // servicing the handler
212             ContainerUtil.service(handler, serviceManager);
213 
214             // configure the handler
215             ContainerUtil.configure(handler, config);
216 
217             // if it is a connect handler add it to list of connect handlers
218             if (handler instanceof ConnectHandler) {
219                 connectHandlers.add((ConnectHandler) handler);
220                 if (getLogger().isInfoEnabled()) {
221                     getLogger().info("Added ConnectHandler: " + className);
222                 }
223             }
224 
225             // if it is a commands handler add it to the map with key as command
226             // name
227             if (handler instanceof CommandsHandler) {
228                 Map c = ((CommandsHandler) handler).getCommands();
229 
230                 Iterator cmdKeys = c.keySet().iterator();
231 
232                 while (cmdKeys.hasNext()) {
233                     String commandName = cmdKeys.next().toString();
234                     String cName = c.get(commandName).toString();
235 
236                     DefaultConfiguration cmdConf = new DefaultConfiguration(
237                             "handler");
238                     cmdConf.setAttribute("command", commandName);
239                     cmdConf.setAttribute("class", cName);
240 
241                     loadClass(classLoader, cName, cmdConf);
242                 }
243 
244             }
245 
246             // if it is a command handler add it to the map with key as command
247             // name
248             if (handler instanceof CommandHandler) {
249                 String commandName = config.getAttribute("command");
250                 String cmds[] = commandName.split(",");
251                 Collection implCmds = ((CommandHandler) handler).getImplCommands();
252 
253                 for (int i = 0; i < cmds.length; i++) {
254                     commandName = cmds[i].trim().toUpperCase(Locale.US);
255 
256                     // Check if the commandHandler implement the configured command
257                     if (implCmds.contains(commandName)) {
258                         addToMap(commandName, (CommandHandler) handler);
259                         if (getLogger().isInfoEnabled()) {
260                             getLogger().info(
261                                     "Added Commandhandler: " + className);
262                         }
263                     } else {
264                         // The Configured command is not implemented. Throw an exception
265                         throw new ConfigurationException("Commandhandler "
266                                 + className + " not implement the command "
267                                 + commandName);
268                     }
269 
270                 }
271 
272             }
273 
274             // if it is a message handler add it to list of message handlers
275             if (handler instanceof MessageHandler) {
276                 messageHandlers.add((MessageHandler) handler);
277                 if (getLogger().isInfoEnabled()) {
278                     getLogger().info("Added MessageHandler: " + className);
279                 }
280             }
281         } catch (ClassNotFoundException ex) {
282             if (getLogger().isErrorEnabled()) {
283                 getLogger().error("Failed to add Commandhandler: " + className,
284                         ex);
285             }
286             throw new ConfigurationException("Failed to add Commandhandler: "
287                     + className, ex);
288         } catch (IllegalAccessException ex) {
289             if (getLogger().isErrorEnabled()) {
290                 getLogger().error("Failed to add Commandhandler: " + className,
291                         ex);
292             }
293             throw new ConfigurationException("Failed to add Commandhandler: "
294                     + className, ex);
295         } catch (InstantiationException ex) {
296             if (getLogger().isErrorEnabled()) {
297                 getLogger().error("Failed to add Commandhandler: " + className,
298                         ex);
299             }
300             throw new ConfigurationException("Failed to add Commandhandler: "
301                     + className, ex);
302         } catch (ServiceException e) {
303             if (getLogger().isErrorEnabled()) {
304                 getLogger().error(
305                         "Failed to service Commandhandler: " + className, e);
306             }
307             throw new ConfigurationException("Failed to add Commandhandler: "
308                     + className, e);
309         } catch (ContextException e) {
310             if (getLogger().isErrorEnabled()) {
311                 getLogger().error(
312                         "Failed to service Commandhandler: " + className, e);
313             }
314             throw new ConfigurationException("Failed to add Commandhandler: "
315                     + className, e);
316         }
317     }

319     /***
320      * Return a DefaultConfiguration build on the given command name and classname
321      * 
322      * @param cmdName The command name
323      * @param className The class name
324      * @return DefaultConfiguration
325      */
326     private DefaultConfiguration addHandler(String cmdName, String className) {
327         DefaultConfiguration cmdConf = new DefaultConfiguration("handler");
328         cmdConf.setAttribute("command",cmdName);
329         cmdConf.setAttribute("class",className);
330         return cmdConf;
331     }

333     /***
334      * Add it to map (key as command name, value is an array list of commandhandlers)
335      *
336      * @param commandName the command name which will be key
337      * @param cmdHandler The commandhandler object
338      */
339     private void addToMap(String commandName, CommandHandler cmdHandler) {
340         ArrayList handlers = (ArrayList)commandHandlerMap.get(commandName);
341         if(handlers == null) {
342             handlers = new ArrayList();
343             commandHandlerMap.put(commandName, handlers);
344         }
345         handlers.add(cmdHandler);
346     }

348     /***
349      * Returns all the configured commandhandlers for the specified command
350      *
351      * @param commandName the command name which will be key
352      * @return List of commandhandlers
353      */
354     List getCommandHandlers(String command) {
355         if (command == null) {
356             return null;
357         }
358         if (getLogger().isDebugEnabled()) {
359             getLogger().debug("Lookup command handler for command: " + command);
360         }
361         List handlers =  (List)commandHandlerMap.get(command);
362         if(handlers == null) {
363             handlers = (List)commandHandlerMap.get(UnknownCmdHandler.UNKNOWN_COMMAND);
364         }
365 
366         return handlers;
367     }
