100     public void initialize()
101         throws Exception
102     {
103         getLogger().info( "Init " + getClass().getName() + " Store" );
104 
105         m_name = "Repository";
106         String m_postfix = getExtensionDecorator();
107         m_extension = "." + m_name + m_postfix;
108         m_filter = new ExtensionFileFilter(m_extension);
109         //m_filter = new NumberedRepositoryFileFilter(getExtensionDecorator());
110 
111         final File directory = new File( m_path );
112         directory.mkdirs();
113 
114         getLogger().info( getClass().getName() + " opened in " + m_path );
115 
116         //We will look for all numbered repository files in this
117         //  directory and rename them to non-numbered repositories,
118         //  logging all the way.
119 
120         FilenameFilter num_filter = new NumberedRepositoryFileFilter(getExtensionDecorator());
121         final String[] names = directory.list( num_filter );
122 
123         try {
124             for( int i = 0; i < names.length; i++ ) {
125                 String origFilename = names[i];
126 
127                 //This needs to handle (skip over) the possible repository numbers
128                 int pos = origFilename.length() - m_postfix.length();
129                 while (pos >= 1 && Character.isDigit(origFilename.charAt(pos - 1))) {
130                     pos--;
131                 }
132                 pos -= ".".length() + m_name.length();
133                 String newFilename = origFilename.substring(0, pos) + m_extension;
134 
135                 File origFile = new File(directory, origFilename);
136                 File newFile = new File(directory, newFilename);
137 
138                 if (origFile.renameTo(newFile)) {
139                     getLogger().info("Renamed " + origFile + " to " + newFile);
140                 } else {
141                     getLogger().info("Unable to rename " + origFile + " to " + newFile);
142                 }
143             }
144         } catch (Exception e) {
145             e.printStackTrace();
146             throw e;
147         }
148 
149     }

151     protected void setDestination( final String destination )
152         throws ConfigurationException
153     {
154         if( !destination.startsWith( HANDLED_URL ) )
155         {
156             throw new ConfigurationException( "cannot handle destination " + destination );
157         }
158 
159         m_path = destination.substring( HANDLED_URL.length() );
160 
161         File directory;
162 
163         // Check for absolute path
164         if( m_path.startsWith( "/" ) )
165         {
166             directory = new File( m_path );
167         }
168         else
169         {
170             directory = new File( m_baseDirectory, m_path );
171         }
172 
173         try
174         {
175             directory = directory.getCanonicalFile();
176         }
177         catch( final IOException ioe )
178         {
179             throw new ConfigurationException( "Unable to form canonical representation of " +
180                                               directory );
181         }
182 
183         m_path = directory.toString();
184 
185         m_destination = destination;
186     }

194     public Repository getChildRepository( final String childName )
195     {
196         AbstractFileRepository child = null;
197 
198         try
199         {
200             child = createChildRepository();
201         }
202         catch( final Exception e )
203         {
204             throw new RuntimeException( "Cannot create child repository " +
205                                         childName + " : " + e );
206         }
207 
208         try
209         {
210             child.service( m_serviceManager );
211         }
212         catch( final ServiceException cme )
213         {
214             throw new RuntimeException( "Cannot service child " +
215                                         "repository " + childName +
216                                         " : " + cme );
217         }
218 
219         try
220         {
221             child.setDestination( m_destination + File.pathSeparatorChar +
222                                   childName + File.pathSeparator );
223         }
224         catch( final ConfigurationException ce )
225         {
226             throw new RuntimeException( "Cannot set destination for child child " +
227                                         "repository " + childName +
228                                         " : " + ce );
229         }
230 
231         try
232         {
233             child.initialize();
234         }
235         catch( final Exception e )
236         {
237             throw new RuntimeException( "Cannot initialize child " +
238                                         "repository " + childName +
239                                         " : " + e );
240         }
241 
242         if( DEBUG )
243         {
244             getLogger().debug( "Child repository of " + m_name + " created in " +
245                                m_destination + File.pathSeparatorChar +
246                                childName + File.pathSeparator );
247         }
248 
249         return child;
250     }

311     /***
312      * Returns the list of used keys.
313      */
314     public Iterator list()
315     {
316         final File storeDir = new File( m_path );
317         final String[] names = storeDir.list( m_filter );
318         final ArrayList list = new ArrayList();
319 
320         for( int i = 0; i < names.length; i++ )
321         {
322             String decoded = decode(names[i]);
323             list.add(decoded);
324         }
325 
326         return list.iterator();
327     }

329     /***
330      * Returns a String that uniquely identifies the object.
331      * <b>Note:</b> since this method uses the Object.toString()
332      * method, it's up to the caller to make sure that this method
333      * doesn't change between different JVM executions (like
334      * it may normally happen). For this reason, it's highly recommended
335      * (even if not mandated) that Strings be used as keys.
336      */
337     protected String encode( final String key )
338     {
339         final byte[] bytes = key.getBytes();
340         final char[] buffer = new char[ bytes.length << 1 ];
341 
342         for( int i = 0, j = 0; i < bytes.length; i++ )
343         {
344             final int k = bytes[ i ];
345             buffer[ j++ ] = HEX_DIGITS[ ( k >>> 4 ) & BYTE_MASK ];
346             buffer[ j++ ] = HEX_DIGITS[ k & BYTE_MASK ];
347         }
348 
349         StringBuffer result = new StringBuffer();
350         result.append( m_path );
351         result.append( File.separator );
352         result.append( buffer );
353         result.append( m_extension );
354         return result.toString();
355     }
356 
357     /***
358      * Inverse of encode exept it do not use path.
359      * So decode(encode(s) - m_path) = s.
360      * In other words it returns a String that can be used as key to retive
361      * the record contained in the 'filename' file.
362      */
363     protected String decode( String filename )
364     {
365         filename = filename.substring( 0, filename.length() - m_extension.length() );
366         final int size = filename.length();
367         final byte[] bytes = new byte[ size >>> 1 ];
368 
369         for( int i = 0, j = 0; i < size; j++ )
370         {
371             bytes[ j ] = Byte.parseByte( filename.substring( i, i + 2 ), 16 );
372             i +=2;
373         }
374 
375         return new String( bytes );
376     }
