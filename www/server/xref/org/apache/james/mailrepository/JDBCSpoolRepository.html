32  /***
33   * Implementation of a SpoolRepository on a database.
34   *
35   * <p>Requires a configuration element in the .conf.xml file of the form:
36   *  <br>&lt;repository destinationURL="town://path"
37   *  <br>            type="MAIL"
38   *  <br>            model="SYNCHRONOUS"/&gt;
39   *  <br>            &lt;driver&gt;sun.jdbc.odbc.JdbcOdbcDriver&lt;/conn&gt;
40   *  <br>            &lt;conn&gt;jdbc:odbc:LocalDB&lt;/conn&gt;
41   *  <br>            &lt;table&gt;Message&lt;/table&gt;
42   *  <br>&lt;/repository&gt;
43   * <p>destinationURL specifies..(Serge??)
44   * <br>Type can be SPOOL or MAIL
45   * <br>Model is currently not used and may be dropped
46   * <br>conn is the location of the ...(Serge)
47   * <br>table is the name of the table in the Database to be used
48   *
49   * <p>Requires a logger called MailRepository.
50   *
51   * <p>Approach for spool manager:
52   *
53   * PendingMessage inner class
54   *
55   * accept() is called....
56   * checks whether needs to load PendingMessages()
57   * tries to get a message()
58   * if none, wait 60
59   *
60   * accept(long) is called
61   * checks whether needs to load PendingMessages
62   * tries to get a message(long)
63   * if none, wait accordingly
64   *
65   * sync checkswhetherneedstoloadPendingMessages()
66   * if pending messages has messages in immediate process, return immediately
67   * if run query in last WAIT_LIMIT time, return immediately
68   * query and build 2 vectors of Pending messages.
69   *  Ones that need immediate processing
70   *  Ones that are delayed.  put them in time order
71   * return
72   *
73   * get_a_message()
74   * loop through immediate messages.
75   *  - remove top message
76   *  - try to lock.  if successful, return.  otherwise loop.
77   * if nothing, return null
78   *
79   * get_a_message(long)
80   * try get_a_message()
81   * check top message in pending.  if ready, then remove, try to lock, return if lock.
82   * return null.
83   *
84   *
85   * @version 1.0.0, 24/04/1999
86   */
87  public class JDBCSpoolRepository extends JDBCMailRepository implements SpoolRepository {

90       * How long a thread should sleep when there are no messages to process.

92      private static int WAIT_LIMIT = 60000;
93      /***
94       * How long we have to wait before reloading the list of pending messages
95       */
96      private static int LOAD_TIME_MININUM = 1000;
97      /***
98       * A queue in memory of messages that need processing

104     private long pendingMessagesLoadTime = 0;

106      * Maximum size of the pendingMessages queue

133     /***
134      * Return a message that's ready to process.  If a message is of type "error"
135      * then check the last updated time, and don't try it until the long 'delay' parameter
136      * milliseconds has passed.
137      */
138     public synchronized Mail accept(final long delay) throws InterruptedException {
139         return accept (new SpoolRepository.AcceptFilter () {
140             long sleepUntil = 0;
141                 
142                 public boolean accept (String key, String state, long lastUpdated, String errorMessage) {
143                     if (Mail.ERROR.equals(state)) {
144                     //if it's an error message, test the time
145                         long processingTime = delay + lastUpdated;
146                     if (processingTime < System.currentTimeMillis()) {
147                         //It's time to process
148                             return true;
149                     } else {
150                         //We don't process this, but we want to possibly reduce the amount of time
151                         //  we sleep so we wake when this message is ready.
152                         if (sleepUntil == 0 || processingTime < sleepUntil) {
153                             sleepUntil = processingTime;
154                         }
155                             return false;
156                     }
157                 } else {
158                         return true;
159                     }
160                 }
161                 
162 
163                 public long getWaitTime () {
164                     if (sleepUntil == 0) {
165                         // in AvalonSpoolRepository we return 0: why do we change sleepUntil?
166                         // sleepUntil = System.currentTimeMillis();
167                         return 0;
168                     }
169                     long waitTime = sleepUntil - System.currentTimeMillis();
170                     sleepUntil = 0;
171                     return waitTime <= 0 ? 1 : waitTime;
172                 }
173                 
174             });
175     }

177     /***
178      * Returns an arbitrarily selected mail deposited in this Repository for
179      * which the supplied filter's accept method returns true.
180      * Usage: RemoteDeliverySpool calls accept(filter) with some a filter which determines
181      * based on number of retries if the mail is ready for processing.
182      * If no message is ready the method will block until one is, the amount of time to block is
183      * determined by calling the filters getWaitTime method.
184      *
185      * @return  the mail
186      */
187     public synchronized Mail accept(SpoolRepository.AcceptFilter filter) throws InterruptedException {
188         while (!Thread.currentThread().isInterrupted()) {
189             //Loop through until we are either out of pending messages or have a message
190             // that we can lock
191             PendingMessage next = null;
192             while ((next = getNextPendingMessage()) != null && !Thread.currentThread().isInterrupted()) {
193                 //Check whether this is time to expire
194                 
195                 boolean shouldProcess = filter.accept (next.key, next.state, next.lastUpdated, next.errorMessage);
196                 
197                 if (shouldProcess && lock(next.key)) {
198                     try {
199                         Mail mail = retrieve(next.key);
200                         // Retrieve can return null if the mail is no longer on the spool
201                         // (i.e. another thread has gotten to it first).
202                         // In this case we simply continue to the next key
203                         if (mail == null) {
204                             unlock(next.key);
205                             continue;
206                         }
207                         return mail;
208                     } catch (javax.mail.MessagingException e) {
209                         unlock(next.key);
210                         getLogger().error("Exception during retrieve -- skipping item " + next.key, e);
211                     }
212                 }
213             }
214             //Nothing to do... sleep!
215             long wait_time = filter.getWaitTime();
216             if (wait_time <= 0) {
217                 wait_time = WAIT_LIMIT;
218             }
219             try {
220                 wait (wait_time);
221             } catch (InterruptedException ex) {
222                 throw ex;
223             }
224         }
225         throw new InterruptedException();
226     }

239     /***
240      * If not empty, gets the next pending message.  Otherwise checks
241      * checks the last time pending messages was loaded and load if
242      * it's been more than 1 second (should be configurable).
243      */
244     private PendingMessage getNextPendingMessage() {
245         synchronized (pendingMessages) {
246             if (pendingMessages.size() == 0 && pendingMessagesLoadTime < System.currentTimeMillis()) {
247                 pendingMessagesLoadTime = LOAD_TIME_MININUM + System.currentTimeMillis();
248                 loadPendingMessages();
249             }
250 
251             if (pendingMessages.size() == 0) {
252                 return null;
253             } else {
254                 return (PendingMessage)pendingMessages.removeFirst();
255             }
256         }
257     }

259     /***
260      * Retrieves the pending messages that are in the database
261      */
262     private void loadPendingMessages() {
263         //Loads a vector with PendingMessage objects
264         synchronized (pendingMessages) {
265             pendingMessages.clear();
266 
267             Connection conn = null;
268             PreparedStatement listMessages = null;
269             ResultSet rsListMessages = null;
270             try {
271                 conn = datasource.getConnection();
272                 listMessages =
273                     conn.prepareStatement(sqlQueries.getSqlString("listMessagesSQL", true));
274                 listMessages.setString(1, repositoryName);
275                 listMessages.setMaxRows(maxPendingMessages);
276                 rsListMessages = listMessages.executeQuery();
277                 // Continue to have it loop through the list of messages until we hit
278                 // a possible message, or we retrieve maxPendingMessages messages.
279                 // This maxPendingMessages cap is to avoid loading thousands or
280                 // hundreds of thousands of messages when the spool is enourmous.
281                 while (rsListMessages.next() && pendingMessages.size() < maxPendingMessages && !Thread.currentThread().isInterrupted()) {
282                     String key = rsListMessages.getString(1);
283                     String state = rsListMessages.getString(2);
284                     long lastUpdated = rsListMessages.getTimestamp(3).getTime();
285                     String errorMessage = rsListMessages.getString(4);
286                     pendingMessages.add(new PendingMessage(key, state, lastUpdated, errorMessage));
287                 }
288             } catch (SQLException sqle) {
289                 //Log it and avoid reloading for a bit
290                 getLogger().error("Error retrieving pending messages", sqle);
291                 pendingMessagesLoadTime = LOAD_TIME_MININUM * 10 + System.currentTimeMillis();
292             } finally {
293                 theJDBCUtil.closeJDBCResultSet(rsListMessages);
294                 theJDBCUtil.closeJDBCStatement(listMessages);
295                 theJDBCUtil.closeJDBCConnection(conn);
296             }
297         }
298     }
