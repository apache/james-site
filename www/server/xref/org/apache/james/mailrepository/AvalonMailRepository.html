91       * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
92       */
93      public void configure(Configuration conf) throws ConfigurationException {
94          destination = conf.getAttribute("destinationURL");
95          if (getLogger().isDebugEnabled()) {
96              getLogger().debug("AvalonMailRepository.destinationURL: " + destination);
97          }
98          String checkType = conf.getAttribute("type");
99          if (! (checkType.equals("MAIL") || checkType.equals("SPOOL")) ) {
100             String exceptionString = "Attempt to configure AvalonMailRepository as " +
101                                      checkType;
102             if (getLogger().isWarnEnabled()) {
103                 getLogger().warn(exceptionString);
104             }
105             throw new ConfigurationException(exceptionString);
106         }
107         fifo = conf.getAttributeAsBoolean("FIFO", false);
108         cacheKeys = conf.getAttributeAsBoolean("CACHEKEYS", true);
109         // ignore model
110     }

112     /***
113      * @see org.apache.avalon.framework.activity.Initializable#initialize()
114      */
115     public void initialize()
116             throws Exception {
117         try {
118             objectRepository = (ObjectRepository) selectRepository(store, "OBJECT");
119             streamRepository = (StreamRepository) selectRepository(store, "STREAM");
120 
121             lock = new Lock();
122             if (cacheKeys) keys = Collections.synchronizedSet(new HashSet());
123 
124             //Finds non-matching pairs and deletes the extra files
125             HashSet streamKeys = new HashSet();
126             for (Iterator i = streamRepository.list(); i.hasNext(); ) {
127                 streamKeys.add(i.next());
128             }
129             HashSet objectKeys = new HashSet();
130             for (Iterator i = objectRepository.list(); i.hasNext(); ) {
131                 objectKeys.add(i.next());
132             }
133 
134             Collection strandedStreams = (Collection)streamKeys.clone();
135             strandedStreams.removeAll(objectKeys);
136             for (Iterator i = strandedStreams.iterator(); i.hasNext(); ) {
137                 String key = (String)i.next();
138                 remove(key);
139             }
140 
141             Collection strandedObjects = (Collection)objectKeys.clone();
142             strandedObjects.removeAll(streamKeys);
143             for (Iterator i = strandedObjects.iterator(); i.hasNext(); ) {
144                 String key = (String)i.next();
145                 remove(key);
146             }
147 
148             if (keys != null) {
149                 // Next get a list from the object repository
150                 // and use that for the list of keys
151                 keys.clear();
152                 for (Iterator i = objectRepository.list(); i.hasNext(); ) {
153                     keys.add(i.next());
154                 }
155             }
156             if (getLogger().isDebugEnabled()) {
157                 StringBuffer logBuffer =
158                     new StringBuffer(128)
159                             .append(getClass().getName())
160                             .append(" created in ")
161                             .append(destination);
162                 getLogger().debug(logBuffer.toString());
163             }
164         } catch (Exception e) {
165             final String message = "Failed to retrieve Store component:" + e.getMessage();
166             getLogger().error( message, e );
167             throw e;
168         }
169     }

237     /***
238      * Stores a message in this repository. Shouldn't this return the key
239      * under which it is stored?
240      *
241      * @param mc the mail message to store
242      */
243     public void store(Mail mc) throws MessagingException {
244         try {
245             String key = mc.getName();
246             //Remember whether this key was locked
247             boolean wasLocked = true;
248             synchronized (this) {
249                 wasLocked = lock.isLocked(key);
250     
251                 if (!wasLocked) {
252                     //If it wasn't locked, we want a lock during the store
253                     lock(key);
254                 }
255             }
256             try {
257                 if (keys != null && !keys.contains(key)) {
258                     keys.add(key);
259                 }
260                 boolean saveStream = true;
261 
262                 MimeMessage message = mc.getMessage();
263                 // if the message is a Copy on Write proxy we check the wrapped message
264                 // to optimize the behaviour in case of MimeMessageWrapper
265                 if (message instanceof MimeMessageCopyOnWriteProxy) {
266                     MimeMessageCopyOnWriteProxy messageCow = (MimeMessageCopyOnWriteProxy) message;
267                     message = messageCow.getWrappedMessage();
268                 }
269                 if (message instanceof MimeMessageWrapper) {
270                     MimeMessageWrapper wrapper = (MimeMessageWrapper) message;
271                     if (DEEP_DEBUG) {
272                         System.out.println("Retrieving from: " + wrapper.getSourceId());
273                         StringBuffer debugBuffer =
274                             new StringBuffer(64)
275                                     .append("Saving to:       ")
276                                     .append(destination)
277                                     .append("/")
278                                     .append(mc.getName());
279                         System.out.println(debugBuffer.toString());
280                         System.out.println("Modified: " + wrapper.isModified());
281                     }
282                     StringBuffer destinationBuffer =
283                         new StringBuffer(128)
284                             .append(destination)
285                             .append("/")
286                             .append(mc.getName());
287                     if (destinationBuffer.toString().equals(wrapper.getSourceId()) && !wrapper.isModified()) {
288                         //We're trying to save to the same place, and it's not modified... we shouldn't save.
289                         //More importantly, if we try to save, we will create a 0-byte file since we're
290                         //retrying to retrieve from a file we'll be overwriting.
291                         saveStream = false;
292                     }
293                 }
294                 if (saveStream) {
295                     OutputStream out = null;
296                     try {
297                         out = streamRepository.put(key);
298                         mc.getMessage().writeTo(out);
299                     } finally {
300                         if (out != null) out.close();
301                     }
302                 }
303                 //Always save the header information
304                 objectRepository.put(key, mc);
305             } finally {
306                 if (!wasLocked) {
307                     // If it wasn't locked, we need to unlock now
308                     unlock(key);
309                     synchronized (this) {
310                         notify();
311                     }
312                 }
313             }
314 
315             if ((DEEP_DEBUG) && (getLogger().isDebugEnabled())) {
316                 StringBuffer logBuffer =
317                     new StringBuffer(64)
318                             .append("Mail ")
319                             .append(key)
320                             .append(" stored.");
321                 getLogger().debug(logBuffer.toString());
322             }
323 
324         } catch (Exception e) {
325             getLogger().error("Exception storing mail: " + e, e);
326             throw new MessagingException("Exception caught while storing Message Container: " + e);
327         }
328     }

330     /***
331      * Retrieves a message given a key. At the moment, keys can be obtained
332      * from list() in superinterface Store.Repository
333      *
334      * @param key the key of the message to retrieve
335      * @return the mail corresponding to this key, null if none exists
336      */
337     public Mail retrieve(String key) throws MessagingException {
338         if ((DEEP_DEBUG) && (getLogger().isDebugEnabled())) {
339             getLogger().debug("Retrieving mail: " + key);
340         }
341         try {
342             Mail mc = null;
343             try {
344                 mc = (Mail) objectRepository.get(key);
345             } 
346             catch (RuntimeException re){
347                 StringBuffer exceptionBuffer = new StringBuffer(128);
348                 if(re.getCause() instanceof Error){
349                     exceptionBuffer.append("Error when retrieving mail, not deleting: ")
350                             .append(re.toString());
351                 }else{
352                     exceptionBuffer.append("Exception retrieving mail: ")
353                             .append(re.toString())
354                             .append(", so we're deleting it.");
355                     remove(key);
356                 }
357                 getLogger().warn(exceptionBuffer.toString());
358                 return null;
359             }
360             MimeMessageAvalonSource source = new MimeMessageAvalonSource(streamRepository, destination, key);
361             mc.setMessage(new MimeMessageCopyOnWriteProxy(source));
362 
363             return mc;
364         } catch (Exception me) {
365             getLogger().error("Exception retrieving mail: " + me);
366             throw new MessagingException("Exception while retrieving mail: " + me.getMessage());
367         }
368     }

393     /***
394      * Removes a message identified by key.
395      *
396      * @param key the key of the message to be removed from the repository
397      */
398     public void remove(String key) throws MessagingException {
399         if (lock(key)) {
400             try {
401                 if (keys != null) keys.remove(key);
402                 streamRepository.remove(key);
403                 objectRepository.remove(key);
404             } finally {
405                 unlock(key);
406             }
407         } else {
408             StringBuffer exceptionBuffer =
409                 new StringBuffer(64)
410                         .append("Cannot lock ")
411                         .append(key)
412                         .append(" to remove it");
413             throw new MessagingException(exceptionBuffer.toString());
414         }
415     }

417     /***
418      * List string keys of messages in repository.
419      *
420      * @return an <code>Iterator</code> over the list of keys in the repository
421      *
422      */
423     public Iterator list() {
424         // Fix ConcurrentModificationException by cloning 
425         // the keyset before getting an iterator
426         final ArrayList clone;
427         if (keys != null) synchronized(keys) {
428             clone = new ArrayList(keys);
429         } else {
430             clone = new ArrayList();
431             for (Iterator i = objectRepository.list(); i.hasNext(); ) {
432                 clone.add(i.next());
433             }
434         }
435         if (fifo) Collections.sort(clone); // Keys is a HashSet; impose FIFO for apps that need it
436         return clone.iterator();
437     }
