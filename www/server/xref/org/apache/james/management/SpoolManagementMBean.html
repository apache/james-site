19  package org.apache.james.management;
20  
21  /***
22   * Expose spool management functionality through JMX.
23   * 
24   * @phoenix:mx-topic name="SpoolAdministration"
25   */
26  public interface SpoolManagementMBean {
27  
28      /***
29       * List mails on the spool matching the given criteria
30       *
31       * @phoenix:mx-operation
32       * @phoenix:mx-description List mails on the spool matching the given criteria
33       *
34       * @param spoolRepositoryURL specifies the spool
35       * @param state only mails in the given state are processed, or ALL if NULL
36       * @param header the header whose value should be checked
37       * @param headerValueRegex regular expression matched against header value. only matching mails are processed
38       * @return number of removed items
39       * 
40       * @throws SpoolManagementException
41       */
42      String[] listSpoolItems(String spoolRepositoryURL, String state, String header, String headerValueRegex) 
43              throws SpoolManagementException;
44      
45      /***
46       * Removes one specific or all mails from the given spool repository matching the given criteria
47       *
48       * @phoenix:mx-operation
49       * @phoenix:mx-description Removes one specific or all mails from the given spool repository matching 
50       * the given criteria
51       *
52       * @param spoolRepositoryURL specifies the spool
53       * @param key identifies the item to be removed. if NULL, all items are removed
54       * @param state only mails in the given state are processed, or ALL if NULL
55       * @param header the header whose value should be checked
56       * @param headerValueRegex regular expression matched against header value. only matching mails are processed
57       * @return number of removed items
58       */
59      int removeSpoolItems(String spoolRepositoryURL, String key, String state, String header, String headerValueRegex) 
60              throws SpoolManagementException;
61  
62      /***
63       * (Re-)tries to send one specific or all mails in the given spool repository matching the given criteria
64       *
65       * @phoenix:mx-operation
66       * @phoenix:mx-description (Re-)tries to send one specific or all mails in the given spool repository
67       * matching the given criteria
68       *
69       * @param spoolRepositoryURL specifies the spool
70       * @param key identifies the item to be sent. if NULL, all items with status ERROR are sent
71       * @param state only mails in the given state are processed, or ALL if NULL
72       * @param header the header whose value should be checked
73       * @param headerValueRegex regular expression matched against header value. only matching mails are processed
74       * @return number of processed items
75       */
76      int resendSpoolItems(String spoolRepositoryURL, String key, String state, String header, String headerValueRegex) 
77              throws SpoolManagementException;
78  
79  }
