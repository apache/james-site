36  public class MockMailServer implements MailServer {

38      private final MockUsersRepository m_users = new MockUsersRepository();

43      private final ArrayList mails = new ArrayList();

51      public void sendMail(MailAddress sender, Collection recipients, MimeMessage msg) throws MessagingException {
52          Object[] mailObjects = new Object[]{sender, recipients, new MimeMessageCopyOnWriteProxy(msg)};
53          mails.add(mailObjects);
54      }
55  
56      public void sendMail(MailAddress sender, Collection recipients, InputStream msg) throws MessagingException {
57          Object[] mailObjects = new Object[]{sender, recipients, msg};
58          mails.add(mailObjects);
59      }
60  
61      public void sendMail(Mail mail) throws MessagingException {
62          int bodySize = mail.getMessage().getSize();
63          try {
64              if (m_maxMessageSizeBytes != 0 && m_maxMessageSizeBytes < bodySize) throw new MessageSizeException();
65          } catch (MessageSizeException e) {
66              throw new MessagingException("message size exception is nested", e);
67          }
68          sendMail(mail.getSender(), mail.getRecipients(), mail.getMessage());
69      }
70  
71      public void sendMail(MimeMessage message) throws MessagingException {
72          // taken from class org.apache.james.James 
73          MailAddress sender = new MailAddress((InternetAddress)message.getFrom()[0]);
74          Collection recipients = new HashSet();
75          Address addresses[] = message.getAllRecipients();
76          if (addresses != null) {
77              for (int i = 0; i < addresses.length; i++) {
78                  // Javamail treats the "newsgroups:" header field as a
79                  // recipient, so we want to filter those out.
80                  if ( addresses[i] instanceof InternetAddress ) {
81                      recipients.add(new MailAddress((InternetAddress)addresses[i]));
82                  }
83              }
84          }
85          sendMail(sender, recipients, message);
86      }
