80      public void verifyLastMail(String sender, String recipient, MimeMessage msg) throws IOException, MessagingException {
81          Object[] mailData = m_mailServer.getLastMail();
82          assertNotNull("mail received by mail server", mailData);
83  
84          if (sender == null && recipient == null && msg == null) fail("no verification can be done with all arguments null");
85  
86          if (sender != null) assertEquals("sender verfication", sender, ((MailAddress)mailData[0]).toString());
87          if (recipient != null) assertTrue("recipient verfication", ((Collection) mailData[1]).contains(new MailAddress(recipient)));
88          if (msg != null) {
89              ByteArrayOutputStream bo1 = new ByteArrayOutputStream();
90              msg.writeTo(bo1);
91              ByteArrayOutputStream bo2 = new ByteArrayOutputStream();
92              ((MimeMessage) mailData[2]).writeTo(bo2);
93              assertEquals(bo1.toString(),bo2.toString());
94              assertEquals("message verification", msg, ((MimeMessage) mailData[2]));
95          }
96      }

173     public void testSimpleMailSendWithEHLO() throws Exception {
174         finishSetUp(m_testConfiguration);
175         
176         SMTPClient smtpProtocol = new SMTPClient();
177         smtpProtocol.connect("127.0.0.1", m_smtpListenerPort);
178 
179         // no message there, yet
180         assertNull("no mail received by mail server", m_mailServer.getLastMail());
181 
182         smtpProtocol.sendCommand("EHLO "+InetAddress.getLocalHost());
183         String[] capabilityRes = smtpProtocol.getReplyStrings();
184         
185         List capabilitieslist = new ArrayList();
186         for (int i = 1; i < capabilityRes.length; i++) {
187             capabilitieslist.add(capabilityRes[i].substring(4));
188         }
189         
190         assertEquals("capabilities", 2, capabilitieslist.size());
191         assertTrue("capabilities present PIPELINING", capabilitieslist.contains("PIPELINING"));
192         assertTrue("capabilities present ENHANCEDSTATUSCODES", capabilitieslist.contains("ENHANCEDSTATUSCODES"));
193         //assertTrue("capabilities present 8BITMIME", capabilitieslist.contains("8BITMIME"));
194 
195         smtpProtocol.setSender("mail@localhost");
196         smtpProtocol.addRecipient("mail@localhost");
197 
198         smtpProtocol.sendShortMessageData("Subject: test\r\n\r\nBody\r\n\r\n.\r\n");
199         smtpProtocol.quit();
200         smtpProtocol.disconnect();
201 
202         // mail was propagated by SMTPServer
203         assertNotNull("mail received by mail server", m_mailServer.getLastMail());
204     }
