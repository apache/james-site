27  import javax.mail.MessagingException;
28  import java.io.UnsupportedEncodingException;

30  public class SetMailAttributeTest extends TestCase {
31  
32      private Mail mockedMail;
33  
34      private Mailet mailet;
35  
36      private final String ATTRIBUTE_NAME1 = "org.apache.james.junit1";
37  
38      private final String ATTRIBUTE_NAME2 = "org.apache.james.junit2";
39  
40      public SetMailAttributeTest(String arg0)
41              throws UnsupportedEncodingException {
42          super(arg0);
43      }
44  
45      private void setupMailet() throws MessagingException {
46          mailet = new SetMailAttribute();
47          MockMailetConfig mci = new MockMailetConfig("Test",
48                  new MockMailContext());
49          mci.setProperty(ATTRIBUTE_NAME1, "true");
50          mci.setProperty(ATTRIBUTE_NAME2, "true");
51  
52          mailet.init(mci);
53      }
54  
55      // test if the Header was add
56      public void testMailAttributeAdded() throws MessagingException {
57          mockedMail = Util.createMockMail2Recipients(null);
58          setupMailet();
59  
60          assertNull(mockedMail.getAttribute(ATTRIBUTE_NAME1));
61          assertNull(mockedMail.getAttribute(ATTRIBUTE_NAME2));
62          mailet.service(mockedMail);
63  
64          assertEquals("true", mockedMail.getAttribute(ATTRIBUTE_NAME1));
65          assertEquals("true", mockedMail.getAttribute(ATTRIBUTE_NAME2));
66      }
67  }
