18  package org.apache.james.transport.matchers;
19  
20  import java.io.UnsupportedEncodingException;
21  import java.util.Collection;
22  
23  import javax.mail.MessagingException;
24  
25  public class RemoteAddrInNetworkTest extends AbstractRemoteAddrInNetworkTest {
26  
27      private final String ALLOWED_NETWORK = "192.168.200.0/24";
28  
29      public RemoteAddrInNetworkTest(String arg0)
30              throws UnsupportedEncodingException {
31          super(arg0);
32      }
33  
34      // test if the recipients get returned as matched
35      public void testRemoteAddrInNetworkMatched() throws MessagingException {
36          setRemoteAddr("192.168.200.1");
37  
38          setupAll();
39  
40          Collection matchedRecipients = matcher.match(mockedMail);
41  
42          assertNotNull(matchedRecipients);
43          assertEquals(matchedRecipients.size(), mockedMail.getRecipients()
44                  .size());
45      }
46  
47      // test if no recipient get returned cause it not match
48      public void testRemoteAddrInNetworkNotMatch() throws MessagingException {
49          setRemoteAddr("192.168.1.1");
50  
51          setupAll();
52  
53          Collection matchedRecipients = matcher.match(mockedMail);
54  
55          assertNull(matchedRecipients);
56      }
57  
58      protected AbstractNetworkMatcher createMatcher() {
59          return new RemoteAddrInNetwork();
60      }
61  
62      protected String getConfigOption() {
63          return "AllowedNetworkIs=";
64      }
65  
66      protected String getAllowedNetworks() {
67          return ALLOWED_NETWORK;
68      }
69  }
