37  import junit.framework.TestCase;

39  public class HostIsTest extends TestCase {
40  
41      private MimeMessage mockedMimeMessage;
42  
43      private MockMail mockedMail;
44  
45      private Matcher matcher;
46  
47      private final String HOST_NAME = "james.apache.org";
48  
49      private MailAddress[] recipients;
50  
51      public HostIsTest(String arg0) throws UnsupportedEncodingException {
52          super(arg0);
53      }
54  
55      private void setRecipients(MailAddress[] recipients) {
56          this.recipients = recipients;
57      }
58  
59      private void setupMockedMimeMessage() throws MessagingException {
60          String sender = "test@james.apache.org";
61          String rcpt = "test2@james.apache.org";
62  
63          mockedMimeMessage = new MockMimeMessage();
64          mockedMimeMessage.setFrom(new InternetAddress(sender));
65          mockedMimeMessage.setRecipients(RecipientType.TO, rcpt);
66          mockedMimeMessage.setSubject("testmail");
67          mockedMimeMessage.setText("testtext");
68          mockedMimeMessage.saveChanges();
69  
70      }
71  
72      private void setupMockedMail(MimeMessage m) {
73          mockedMail = new MockMail();
74          mockedMail.setMessage(m);
75          mockedMail.setRecipients(Arrays.asList(recipients));
76  
77      }
78  
79      private void setupMatcher() throws MessagingException {
80          setupMockedMimeMessage();
81          matcher = new HostIs();
82          MockMatcherConfig mci = new MockMatcherConfig("HostIs=" + HOST_NAME,
83                  new MockMailContext());
84          matcher.init(mci);
85      }
86  
87      // test if all recipients get returned as matched
88      public void testHostIsMatchedAllRecipients() throws MessagingException {
89          setRecipients(new MailAddress[] {
90                  new MailAddress("test@james.apache.org"),
91                  new MailAddress("test2@james.apache.org") });
92  
93          setupMockedMimeMessage();
94          setupMockedMail(mockedMimeMessage);
95          setupMatcher();
96  
97          Collection matchedRecipients = matcher.match(mockedMail);
98  
99          assertNotNull(matchedRecipients);
100         assertEquals(matchedRecipients.size(), mockedMail.getRecipients()
101                 .size());
102     }
103 
104     // test if one recipients get returned as matched
105     public void testHostIsMatchedOneRecipient() throws MessagingException {
106         setRecipients(new MailAddress[] {
107                 new MailAddress("test@james2.apache.org"),
108                 new MailAddress("test2@james.apache.org") });
109 
110         setupMockedMimeMessage();
111         setupMockedMail(mockedMimeMessage);
112         setupMatcher();
113 
114         Collection matchedRecipients = matcher.match(mockedMail);
115 
116         assertNotNull(matchedRecipients);
117         assertEquals(matchedRecipients.size(), 1);
118     }
119 
120     // test if no recipient get returned cause it not match
121     public void testHostIsNotMatch() throws MessagingException {
122         setRecipients(new MailAddress[] {
123                 new MailAddress("test@james2.apache.org"),
124                 new MailAddress("test2@james2.apache.org") });
125

133     }
