81      private class MyMailet extends GenericMailet {
82  
83          public void service(Mail mail) throws MessagingException {
84        try {
85          MimeMessage message = mail.getMessage () ;
86          //Set the header name and value (supplied at init time).
87          String newText =  "new text "+counter++;
88          System.err.println("Setting body to "+newText);
89          message.addHeader("x-Header", newText);
90          message.setText(newText);
91          message.setSubject(newText);
92          message.saveChanges();
93        } catch (javax.mail.MessagingException me) {
94           log (me.getMessage());
95        }
96          }

130     public void testCopyOnWrite() throws IOException, MessagingException {
131         linearProcessor.setSpool(new MockSpoolRepository());
132         Matcher recipientIs = new RecipientIs();
133         recipientIs.init(new DummyMatcherConfig("rec1@domain.com"));
134         
135         Matcher all = new All();
136         all.init(new DummyMatcherConfig(""));
137         
138         Mailet changeBody = new MyMailet();
139         Mailet changeBody2 = new MyMailet();
140 
141         changeBody.init(mockMailetConfig);
142         changeBody2.init(mockMailetConfig);
143         
144         Mailet dumpSystemErr = new DumpSystemErr();
145         changeBody.init(mockMailetConfig);
146         
147         checkerMailet = new CheckerMailet();
148         linearProcessor.openProcessorList();
149         linearProcessor.add(recipientIs,changeBody);
150         linearProcessor.add(all,changeBody);
151         linearProcessor.add(all,dumpSystemErr);
152         linearProcessor.add(all,checkerMailet);
153         linearProcessor.closeProcessorLists();
154 
155         Collection recipients = new ArrayList();
156         recipients.add(new MailAddress("rec1","domain.com"));
157         recipients.add(new MailAddress("rec2","domain.com"));
158         try {
159             MailImpl m = new MailImpl("mail1",new MailAddress("sender","domain.com"),recipients,mimeMessage);
160             linearProcessor.service(m);
161             ArrayList a = checkerMailet.receivedMails;
162             assertEquals(2,a.size());
163             MimeMessage m1 = ((Mail) a.get(0)).getMessage();
164             MimeMessage m2 = ((Mail) a.get(1)).getMessage();
165             assertNotSame(m1,m2);
166             assertEquals(m1.getSubject(),"new text 1");
167             assertEquals(m2.getSubject(),"new text 2");
168         } catch (MessagingException e) {
169             // TODO Auto-generated catch block
170             e.printStackTrace();
171         }
172     }

174     public void testStateChange() throws IOException, MessagingException {
175         linearProcessor.setSpool(new MockSpoolRepository() {
176             public void store(Mail mc) throws MessagingException {
177                 assertEquals("MYSTATE",mc.getState());
178                 super.store(mc);
179             }
180         });
181         
182         Matcher recipientIs = new RecipientIs();
183         recipientIs.init(new DummyMatcherConfig("rec1@domain.com"));
184         
185         Matcher all = new All();
186         all.init(new DummyMatcherConfig(""));
187         
188         Mailet dumpSystemErr = new DumpSystemErr();
189         
190         checkerMailet = new CheckerMailet();
191         linearProcessor.openProcessorList();
192         linearProcessor.add(recipientIs,dumpSystemErr);
193         linearProcessor.add(all,dumpSystemErr);
194         linearProcessor.add(all,checkerMailet);
195         linearProcessor.closeProcessorLists();
196 
197         Collection recipients = new ArrayList();
198         recipients.add(new MailAddress("rec1","domain.com"));
199         recipients.add(new MailAddress("rec2","domain.com"));
200         try {
201             MailImpl m = new MailImpl("mail1",new MailAddress("sender","domain.com"),recipients,mimeMessage);
202             m.setState("MYSTATE");
203             linearProcessor.service(m);
204             ArrayList a = checkerMailet.receivedMails;
205             assertEquals(2,a.size());
206             MimeMessage m1 = ((Mail) a.get(0)).getMessage();
207             MimeMessage m2 = ((Mail) a.get(1)).getMessage();
208             assertNotSame(m1,m2);
209             assertEquals("MYSTATE",((Mail) a.get(0)).getState());
210             assertEquals("MYSTATE",((Mail) a.get(1)).getState());
211         } catch (MessagingException e) {
212             // TODO Auto-generated catch block
213             e.printStackTrace();
214         }
215     }
