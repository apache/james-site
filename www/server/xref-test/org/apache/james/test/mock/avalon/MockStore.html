19  package org.apache.james.test.mock.avalon;
20  
21  import org.apache.avalon.cornerstone.services.store.Store;
22  import org.apache.avalon.framework.configuration.Configuration;
23  import org.apache.avalon.framework.configuration.ConfigurationException;
24  import org.apache.avalon.framework.service.ServiceException;
25  
26  import java.util.HashMap;
27  import java.util.Map;
28  
29  public class MockStore implements Store {
30  
31      Map m_storedObjectMap = new HashMap();
32  
33      public void add(Object key, Object obj) {
34          m_storedObjectMap.put(key, obj);
35      }
36      
37      public Object select(Object object) throws ServiceException {
38          Object result = get(object);
39          return result;
40      }
41  
42      private Object get(Object object) {
43          return m_storedObjectMap.get(extractKeyObject(object));
44      }
45  
46      private Object extractKeyObject(Object object) {
47          if (object instanceof Configuration) {
48              Configuration repConf = (Configuration) object;
49              try {
50                  String type = repConf.getAttribute("type");
51                  String prefix = "";
52                  if (!"MAIL".equals(type) && !"SPOOL".equals(type)) {
53                      prefix = type+".";
54                  }
55                  String attribute = repConf.getAttribute("destinationURL");
56                  String[] strings = attribute.split("/");
57                  if (strings.length > 0) {
58                      return prefix+strings[strings.length-1];
59                  }
60              } catch (ConfigurationException e) {
61                  throw new RuntimeException("test configuration setup failed");
62              }
63              
64          }
65          return object;
66      }
67  
68      public boolean isSelectable(Object object) {
69          return get(object) != null;
70      }
71  
72      public void release(Object object) {
73          //trivial implementation
74      }
75  }
