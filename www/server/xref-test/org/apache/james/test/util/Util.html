37      private static final int PORT_RANGE_START =  8000; // the lowest possible port number assigned for testing
38      private static final int PORT_RANGE_END   = 11000; // the highest possible port number assigned for testing
39      private static int PORT_LAST_USED = PORT_RANGE_START;
40  
41      /***
42       * assigns a port from the range of test ports
43       * @return port number
44       */
45      public static int getNonPrivilegedPort() {
46          return getNextNonPrivilegedPort(); // uses sequential assignment of ports
47      }
48  
49      /***
50       * assigns a random port from the range of test ports
51       * @return port number
52       */
53      protected static int getRandomNonPrivilegedPortInt() {
54          return ((int)( Math.random() * (PORT_RANGE_END - PORT_RANGE_START) + PORT_RANGE_START));
55      }
56  
57      /***
58       * assigns ports sequentially from the range of test ports
59       * @return port number
60       */
61      protected synchronized static int getNextNonPrivilegedPort() {
62          PORT_LAST_USED++;
63          if (PORT_LAST_USED > PORT_RANGE_END) PORT_LAST_USED = PORT_RANGE_START;
64          return PORT_LAST_USED;
65      }

77      public static DefaultConfiguration createSMTPHandlerChainConfiguration() {
78          DefaultConfiguration handlerChainConfig = new DefaultConfiguration("handlerchain");
79          handlerChainConfig.addChild(createCommandHandlerConfiguration("HELO", HeloCmdHandler.class));
80          handlerChainConfig.addChild(createCommandHandlerConfiguration("EHLO", EhloCmdHandler.class));
81          handlerChainConfig.addChild(createCommandHandlerConfiguration("AUTH", AuthCmdHandler.class));
82          handlerChainConfig.addChild(createCommandHandlerConfiguration("VRFY", VrfyCmdHandler.class));
83          handlerChainConfig.addChild(createCommandHandlerConfiguration("EXPN", ExpnCmdHandler.class));
84          handlerChainConfig.addChild(createCommandHandlerConfiguration("MAIL", MailCmdHandler.class));
85          handlerChainConfig.addChild(createCommandHandlerConfiguration("RCPT", RcptCmdHandler.class));
86          handlerChainConfig.addChild(createCommandHandlerConfiguration("DATA", DataCmdHandler.class));
87          handlerChainConfig.addChild(createCommandHandlerConfiguration("RSET", RsetCmdHandler.class));
88          handlerChainConfig.addChild(createCommandHandlerConfiguration("HELP", HelpCmdHandler.class));
89          handlerChainConfig.addChild(createCommandHandlerConfiguration("QUIT", QuitCmdHandler.class));
90          // mail sender
91          handlerChainConfig.addChild(createCommandHandlerConfiguration(null, SendMailHandler.class));
92          return handlerChainConfig;
93      }
94  
95      private static DefaultConfiguration createCommandHandlerConfiguration(String command, Class commandClass) {
96          DefaultConfiguration cmdHandlerConfig = new DefaultConfiguration("handler");
97          if (command != null) {
98              cmdHandlerConfig.setAttribute("command", command);
99          }
100         String classname = commandClass.getName();
101         cmdHandlerConfig.setAttribute("class", classname);
102         return cmdHandlerConfig;
103     }

130     public static MockMimeMessage createMimeMessage(String headerName, String headerValue, String subject, int number) throws MessagingException {
131         String sender = "test@james.apache.org";
132         String rcpt = "test2@james.apache.org";
133 
134         MockMimeMessage mockedMimeMessage = new MockMimeMessage(number);
135         mockedMimeMessage.setFrom(new InternetAddress(sender));
136         mockedMimeMessage.setRecipients(MimeMessage.RecipientType.TO, rcpt);
137         if (headerName != null) mockedMimeMessage.setHeader(headerName, headerValue);
138         if (subject != null) mockedMimeMessage.setSubject(subject);
139         mockedMimeMessage.setText("testtext");
140         mockedMimeMessage.saveChanges();
141         return mockedMimeMessage;
142     }
