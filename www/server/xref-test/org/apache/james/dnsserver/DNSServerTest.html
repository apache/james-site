42      /***
43       * Please note that this is an hardcoded test that works because
44       * www.pippo.com. is an alias to pippo.com and pippo.com has
45       * "pippo.com.inbound.mxlogic.net." as its mx record.
46       * This is the first domain with a record proving a previous james bug.
47       * This test will be invalidated by any change in the pippo.com dns records
48       * 
49       * @param args
50       * @throws Exception
51       */
52      public void testINARecords() throws Exception {
53          Zone z = loadZone("pippo.com.");
54          dnsServer.setResolver(null);
55          dnsServer.setLookupper(new ZoneLookupper(z));
56          Collection records = dnsServer.findMXRecords("www.pippo.com.");
57          assertEquals(1, records.size());
58          assertEquals("pippo.com.inbound.mxlogic.net.", records.iterator()
59                  .next());
60      }

62      /***
63       * @throws Exception
64       */
65      public void testMXCatches() throws Exception {
66          Zone z = loadZone("test-zone.com.");
67          dnsServer.setResolver(null);
68          dnsServer.setLookupper(new ZoneLookupper(z));
69          Collection res = dnsServer.findMXRecords("test-zone.com.");
70          try {
71              res.add(new Object());
72              fail("MX Collection should not be modifiable");
73          } catch (UnsupportedOperationException e) {
74          }
75          assertEquals(1,res.size());
76          assertEquals("mail.test-zone.com.",res.iterator().next());
77      }

113     private Zone loadZone(String zoneName) throws IOException {
114         String zoneFilename = zoneName + "zone";
115         URL zoneResource = getClass().getResource(zoneFilename);
116         assertNotNull("test resource for zone could not be loaded: " + zoneFilename, zoneResource);
117         String zoneFile = zoneResource.getFile();
118         Zone zone = new Zone(Name.fromString(zoneName),zoneFile);
119         return zone;
120     }

122     private class ZoneLookupper implements Lookupper {
123         private final Zone z;
124 
125         private ZoneLookupper(Zone z) {
126             super();
127             this.z = z;
128         }
129 
130         public SetResponse lookup(Name name, int type) {
131             SetResponse s = z.findRecords(name,type);
132             System.out.println("Zone Lookup: "+name+" "+type+" = "+s);
133             return s; 
134         }
135     }

137     private interface Lookupper {
138         SetResponse lookup(Name name, int type);
139     }
140     
141     private final class TestableDNSServer extends DNSServer {
142         
143         private Lookupper lookupper;
144 
145         public void setLookupper(Lookupper l) {
146             this.lookupper = l;
147         }
148         
149         public Record[] lookup(String name, int type) {
150             if (lookupper != null) {
151                 try {
152                     SetResponse lookup = lookupper.lookup(Name.fromString(name), type);
153                     if (lookup != null && lookup.isSuccessful()) {
154                         return processSetResponse(lookup);
155                     } else {
156                         return null;
157                     }
158                 } catch (TextParseException e) {
159                     e.printStackTrace();
160                     return null;
161                 }
162             } else {
163                 return super.lookup(name, type);
164             }
165         }
166 
167         public void setResolver(Resolver r) {
168             resolver = r;
169         }
170 
171         public Resolver getResolver() {
172             return resolver;
173         }
174     }
