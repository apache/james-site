185         MimeMessage mm = getMultipartMessage();
186         
187 //        ByteArrayOutputStream out = new ByteArrayOutputStream();
188 //        mmCreated.writeTo(out,new String[] {"Message-ID"});
189 //        String messageSource = out.toString();

192         
193         MimeMultipart content1 = (MimeMultipart) mm.getContent();
194         BodyPart b1 = content1.getBodyPart(0);
195         b1.setContent("test\u20AC","text/plain; charset=Cp1252");
196         mm.setContent(content1,mm.getContentType());
197         //.setHeader(RFC2822Headers.CONTENT_TYPE,contentType);
198         mm.saveChanges();
199 
200         assertEquals(getMultipartMessageExpected1(),getCleanedMessageSource(mm));
201 
202         MimeMultipart content2 = (MimeMultipart) mm.getContent();
203         content2.addBodyPart(new MimeBodyPart(new InternetHeaders(new ByteArrayInputStream("Subject: test3\r\n".getBytes())),"second part".getBytes()));
204         mm.setContent(content2,mm.getContentType());
205         mm.saveChanges();
206 
207         assertEquals(getMultipartMessageExpected2(),getCleanedMessageSource(mm));
208 
209         mm.setContent("mynewco\u00F2\u00E0\u00F9ntent\u20AC\u00E0!","text/plain; charset=cp1252");
210         mm.setHeader(RFC2822Headers.CONTENT_TYPE,"binary/octet-stream");
211         //mm.setHeader("Content-Transfer-Encoding","8bit");
212         mm.saveChanges();
213         
214         assertEquals(getMultipartMessageExpected3(),getCleanedMessageSource(mm));
215         
216         ContainerUtil.dispose(mm);

261     protected String getCleanedMessageSource(MimeMessage mm) throws Exception {
262         ByteArrayOutputStream out2;
263         out2 = new ByteArrayOutputStream();
264         mm.writeTo(out2,new String[] {"Message-ID"});
265 
266         String res = out2.toString();
267 
268         int p = res.indexOf("\r\n\r\n");
269         if (p > 0) {
270             String head = res.substring(0,p);
271             String[] str = head.split("\r\n");
272             Arrays.sort(str);
273             StringBuffer outputHead = new StringBuffer();
274             for (int i = str.length-1; i >= 0; i--) {
275                 outputHead.append(str[i]);
276                 outputHead.append("\r\n");
277             }
278             outputHead.append(res.substring(p+2));
279             res = outputHead.toString();
280         }
281         
282         res = res.replaceAll("----=_Part_//d_//d+//.//d+","----=_Part_//0_XXXXXXXXXXX.XXXXXXXXXXX");
283         return res;
284     }
