35      public void testConstr1() throws MessagingException {
36          MailImpl mail = new MailImpl();
37  
38          helperTestInitialState(mail);
39          helperTestMessageSize(mail, 0); // MimeMessageWrapper default is 0

45      public void testConstr2() throws MessagingException {
46          ArrayList recepients = new ArrayList();
47          String name = new MockMailServer().getId();
48          String sender = "sender@localhost";
49          MailAddress senderMailAddress = new MailAddress(sender);
50          MailImpl mail = new MailImpl(name, senderMailAddress, recepients);
51  
52          helperTestInitialState(mail); // MimeMessageWrapper default is 0
53          helperTestMessageSize(mail, 0); // MimeMessageWrapper default is 0
54          assertNull("no initial message", mail.getMessage());
55          assertEquals("sender", sender, mail.getSender().toString());
56          assertEquals("name", name, mail.getName());
57  
58          mail.setMessage(new MockMimeMessage());
59          assertNotNull("message", mail.getMessage());
60      }

62      public void testConstr3() throws MessagingException {
63          ArrayList recepients = new ArrayList();
64          String name = new MockMailServer().getId();
65          String sender = "sender@localhost";
66          MailAddress senderMailAddress = new MailAddress(sender);
67          MockMimeMessage mimeMessage = new MockMimeMessage();
68          MailImpl mail = new MailImpl(name, senderMailAddress, recepients, mimeMessage);
69  
70          helperTestInitialState(mail); 
71          helperTestMessageSize(mail, 0);
72          assertEquals("initial message", mimeMessage.getMessageID(), mail.getMessage().getMessageID());
73          assertEquals("sender", sender, mail.getSender().toString());
74          assertEquals("name", name, mail.getName());
75      }

77      public void testDuplicate() throws MessagingException {
78          MailImpl mail = new MailImpl();
79          MailImpl duplicate = (MailImpl) mail.duplicate();

85      public void testDuplicateNewName() throws MessagingException {
86          String newName = "aNewName";
87          
88          MailImpl mail = new MailImpl();
89          assertFalse("before + after names differ", newName.equals(mail.getName()));
90          
91          MailImpl duplicate = (MailImpl) mail.duplicate(newName);
