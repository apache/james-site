229     public void testNonSpam() throws IOException, MessagingException {
230 
231         int port = Util.getNonPrivilegedPort();
232         MockSpamd spamd = new MockSpamd(port);
233         new Thread(spamd).start();
234 
235         SMTPSession session = setupMockedSMTPSession(setupMockedMail(setupMockedMimeMessage("test")));
236 
237         SpamAssassinHandler handler = new SpamAssassinHandler();
238 
239         ContainerUtil.enableLogging(handler, new MockLogger());
240         handler.setSpamdHost(SPAMD_HOST);
241         handler.setSpamdPort(port);
242         handler.setSpamdRejectionHits(200.0);
243         handler.onMessage(session);
244 
245         assertNull("Email was not rejected", getResponse());
246         assertEquals("email was not spam", session.getMail().getAttribute(
247                 SpamAssassinInvoker.FLAG_MAIL_ATTRIBUTE_NAME), "NO");
248         assertNotNull("spam hits", session.getMail().getAttribute(
249                 SpamAssassinInvoker.STATUS_MAIL_ATTRIBUTE_NAME));
250 
251     }
