101     protected String getLastLine(List list) {
102         if (list == null || list.isEmpty()) return null;
103         return (String)list.get(list.size()-1);
104     }

106     protected List readAnswer() {
107         return readAnswer(1);
108     }

110     protected List readAnswer(int numLines) {
111         List allAnswerLines = new ArrayList();
112         try {
113             if (numLines > 0) {
114                 for (int i = 0; i < numLines; i++) {
115                     allAnswerLines.add(m_reader.readLine());
116                 }
117             } else {
118                 String line = m_reader.readLine();
119                 allAnswerLines.add(line);
120                 
121                 while (m_reader.ready()) {
122                     allAnswerLines.add(m_reader.readLine());
123                 }
124             }
125             return allAnswerLines;
126         } catch (IOException e) {
127             return null;
128         }
129     }
130 
131     protected void sendCommand(String command) throws IOException {
132         m_writer.println(command);
133         m_writer.flush();
134     }

143         readAnswer(3);
144     }

281     public void testListUsers() throws IOException {
282         finishSetUp(m_testConfiguration);
283         connect();
284         login();
285 
286         String[] users = new String[] {"ccc", "aaa", "dddd", "bbbbb"};
287 
288         for (int i = 0; i < users.length; i++) {
289             String user = users[i];
290             sendCommand("adduser " + user + " test");
291             readAnswer(1);
292         }
293 
294         delay();
295 
296         sendCommand("listusers");
297         List list = readAnswer(5);
298 
299         assertEquals("user count line", "Existing accounts " + users.length, list.get(0));
300         
301         List readUserNames = new ArrayList();
302         for (Iterator iterator = list.iterator(); iterator.hasNext();) {
303             String answerLine = (String) iterator.next();
304             if (!answerLine.startsWith("user: ")) continue;
305             readUserNames.add(answerLine.substring(6));
306         }
307         assertEquals("user count", users.length, readUserNames.size());
308 
309         for (int i = 0; i < users.length; i++) {
310             String user = users[i];
311             assertTrue("name found", readUserNames.contains(user));
312         }
313     }
