119     /***
120      * Stores a message in this repository. Shouldn't this return the key
121      * under which it is stored?
122      *
123      * @param mc the mail message to store
124      */
125     public void store(Mail mc) throws MessagingException {
126         try {
127             String key = mc.getName();
128             //Remember whether this key was locked
129             boolean wasLocked = true;
130             synchronized (this) {
131                 wasLocked = lock.isLocked(key);
132     
133                 if (!wasLocked) {
134                     //If it wasn't locked, we want a lock during the store
135                     lock(key);
136                 }
137             }
138             try {
139                 spool.put(key,mc);
140             } finally {
141                 if (!wasLocked) {
142                     // If it wasn't locked, we need to unlock now
143                     unlock(key);
144                     synchronized (this) {
145                         notify();
146                     }
147                 }
148             }
149 
150             if ((DEEP_DEBUG) && (getLogger().isDebugEnabled())) {
151                 StringBuffer logBuffer =
152                     new StringBuffer(64)
153                             .append("Mail ")
154                             .append(key)
155                             .append(" stored.");
156                 getLogger().debug(logBuffer.toString());
157             }
158 
159         } catch (Exception e) {
160             getLogger().error("Exception storing mail: " + e);
161             throw new MessagingException("Exception caught while storing Message Container: ",e);
162         }
163     }

165     /***
166      * Retrieves a message given a key. At the moment, keys can be obtained
167      * from list() in superinterface Store.Repository
168      *
169      * @param key the key of the message to retrieve
170      * @return the mail corresponding to this key, null if none exists
171      */
172     public Mail retrieve(String key) throws MessagingException {
173         if ((DEEP_DEBUG) && (getLogger().isDebugEnabled())) {
174             getLogger().debug("Retrieving mail: " + key);
175         }
176         try {
177             Mail mc = null;
178             try {
179                 mc = (Mail) spool.get(key);
180             } 
181             catch (RuntimeException re){
182                 StringBuffer exceptionBuffer = new StringBuffer(128);
183                 if(re.getCause() instanceof Error){
184                     exceptionBuffer.append("Error when retrieving mail, not deleting: ")
185                             .append(re.toString());
186                 }else{
187                     exceptionBuffer.append("Exception retrieving mail: ")
188                             .append(re.toString())
189                             .append(", so we're deleting it.");
190                     remove(key);
191                 }
192                 getLogger().warn(exceptionBuffer.toString());
193                 return null;
194             }
195             return mc;
196         } catch (Exception me) {
197             getLogger().error("Exception retrieving mail: " + me);
198             throw new MessagingException("Exception while retrieving mail: " + me.getMessage());
199         }
200     }

225     /***
226      * Removes a message identified by key.
227      *
228      * @param key the key of the message to be removed from the repository
229      */
230     public void remove(String key) throws MessagingException {
231         if (lock(key)) {
232             try {
233                 if (spool != null) spool.remove(key);
234             } finally {
235                 unlock(key);
236             }
237         } else {
238             StringBuffer exceptionBuffer =
239                 new StringBuffer(64)
240                         .append("Cannot lock ")
241                         .append(key)
242                         .append(" to remove it");
243             throw new MessagingException(exceptionBuffer.toString());
244         }
245     }

252      */
253     public Iterator list() {
254         // Fix ConcurrentModificationException by cloning 
255         // the keyset before getting an iterator
256         final ArrayList clone;
257         synchronized(spool) {
258             clone = new ArrayList(spool.keySet());
259         }
260         return clone.iterator();

288     /***
289      * <p>Returns an arbitrarily selected mail deposited in this Repository that
290      * is either ready immediately for delivery, or is younger than it's last_updated plus
291      * the number of failed attempts times the delay time.
292      * Usage: RemoteDeliverySpool calls accept() with some delay and should block until an
293      * unprocessed mail is available.</p>
294      *
295      * <p>Synchronized to ensure thread safe access to the underlying spool.</p>
296      *
297      * @return the mail
298      */
299     public synchronized Mail accept(final long delay) throws InterruptedException
300     {
301         if ((DEEP_DEBUG) && (getLogger().isDebugEnabled())) {
302             getLogger().debug("Method accept(delay) called");
303         }
304         return accept(new SpoolRepository.AcceptFilter () {
305             long youngest = 0;
306                 
307                 public boolean accept (String key, String state, long lastUpdated, String errorMessage) {
308                     if (state.equals(Mail.ERROR)) {
309                         //Test the time...
310                         long timeToProcess = delay + lastUpdated;
311                 
312                         if (System.currentTimeMillis() > timeToProcess) {
313                             //We're ready to process this again
314                             return true;
315                         } else {
316                             //We're not ready to process this.
317                             if (youngest == 0 || youngest > timeToProcess) {
318                                 //Mark this as the next most likely possible mail to process
319                                 youngest = timeToProcess;
320                             }
321                             return false;
322                         }
323                     } else {
324                         //This mail is good to go... return the key
325                         return true;
326                     }
327                 }
328         
329                 public long getWaitTime () {
330                     if (youngest == 0) {
331                         return 0;
332                     } else {
333                         long duration = youngest - System.currentTimeMillis();
334                         youngest = 0; //get ready for next round
335                         return duration <= 0 ? 1 : duration;
336                     }
337                 }
338             });
339     }

342     /***
343      * Returns an arbitrarily select mail deposited in this Repository for
344      * which the supplied filter's accept method returns true.
345      * Usage: RemoteDeliverySpool calls accept(filter) with some a filter which determines
346      * based on number of retries if the mail is ready for processing.
347      * If no message is ready the method will block until one is, the amount of time to block is
348      * determined by calling the filters getWaitTime method.
349      *
350      * <p>Synchronized to ensure thread safe access to the underlying spool.</p>
351      *
352      * @return  the mail
353      */
354     public synchronized Mail accept(SpoolRepository.AcceptFilter filter) throws InterruptedException {
355         if ((DEEP_DEBUG) && (getLogger().isDebugEnabled())) {
356             getLogger().debug("Method accept(Filter) called");
357         }
358         while (!Thread.currentThread().isInterrupted()) try {
359             for (Iterator it = list(); it.hasNext(); ) {
360                 String s = it.next().toString();
361                 if ((DEEP_DEBUG) && (getLogger().isDebugEnabled())) {
362                     StringBuffer logBuffer =
363                         new StringBuffer(64)
364                                 .append("Found item ")
365                                 .append(s)
366                                 .append(" in spool.");
367                     getLogger().debug(logBuffer.toString());
368                 }
369                 if (lock(s)) {
370                     if ((DEEP_DEBUG) && (getLogger().isDebugEnabled())) {
371                         getLogger().debug("accept(Filter) has locked: " + s);
372                     }
373                     try {
374                         Mail mail = retrieve(s);
375                         // Retrieve can return null if the mail is no longer on the spool
376                         // (i.e. another thread has gotten to it first).
377                         // In this case we simply continue to the next key
378                         if (mail == null || !filter.accept (mail.getName(),
379                                                             mail.getState(),
380                                                             mail.getLastUpdated().getTime(),
381                                                             mail.getErrorMessage())) {
382                             unlock(s);
383                             continue;
384                         }
385                         return mail;
386                     } catch (javax.mail.MessagingException e) {
387                         unlock(s);
388                         getLogger().error("Exception during retrieve -- skipping item " + s, e);
389                     }
390                 }
391             }
392 
393             //We did not find any... let's wait for a certain amount of time
394             wait (filter.getWaitTime());
395         } catch (InterruptedException ex) {
396             throw ex;
397         } catch (ConcurrentModificationException cme) {
398             // Should never get here now that list methods clones keyset for iterator
399             getLogger().error("CME in spooler - please report to http://james.apache.org", cme);
400         }
401         throw new InterruptedException();
402     }
