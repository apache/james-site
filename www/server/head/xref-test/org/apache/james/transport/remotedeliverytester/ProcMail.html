88      public static String getKey(Message message, Address address) {
89          try {
90              ByteArrayOutputStream out = new ByteArrayOutputStream();
91              message.writeTo(out);
92              String msg = out.toString();
93              msg = msg.substring(msg.indexOf("\r\n\r\n") + 4);
94              String key = address.toString() + "_" + message.getSubject() + "_" + Math.abs(msg.hashCode());
95              return key;
96          } catch (IOException e) {
97              e.printStackTrace();
98          } catch (MessagingException e) {
99              e.printStackTrace();
100         }
101         return null;
102     }

104     public static String getKey(Mail mail, MailAddress recipient) {
105         try {
106             return getKey(mail.getMessage(), recipient.toInternetAddress());
107         } catch (MessagingException e) {
108             e.printStackTrace();
109         }
110         return null;

150     public void sent(String serverName, Exception e) {
151         state = e == null ? STATE_SENT : STATE_SENT_ERROR;
152         sendDates.add(new Date());
153         sendServers.add(serverName);
154         sendExceptions.add(e);
155         lastEventDate = new Date();
156 
157         // check server
158         String domain = owner.getDomainAssociated(serverName);
159         if (!recipient.getHost().equals(domain)) errorFlags = errorFlags | ERRORFLAG_WRONGSERVER; 
160     }
