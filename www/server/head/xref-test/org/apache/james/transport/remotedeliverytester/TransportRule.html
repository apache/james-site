57      /**
58       * Server rule string format:
59       * id-rule.[*-userrule].[match*-userrule].string
60       * 
61       * - id-rule are for "connect"-time exceptions
62       * - *-userrule are for "send"-time exceptions involving every mail (not depending on the recipients address)
63       * - match*-userrule are for "send"-time exceptions involving specific recipients (starting with match - ** to match every address)
64       * 
65       * Available rules:
66       * - me[v]: MessagingException
67       * - sfe[v]: SendFailedException
68       * - smtpafeXXX[v]: SMTPAddressFailedException with XXX returnCode
69       * - smtpaseXXX[v]: SMTPAddressSuccededException with XXX returnCode
70       * - smtpsfeXXX[v]: SMTPSendFailedException with XXX returnCode
71       * - null: NullPointerException
72       * - io: IOException (nested in a MessagingException)
73       * - rpt: repeat the previous rule pattern 
74       * - (*) if a "mail" rule (one involving addresses) ends with "v" resulting in a "valid" address (the failure is about some other error: valid unsent)
75       */
76      public class NameExpression extends Default {

87          /**
88           * Take a string like "name-action[-count]-[action2[-count2]...]"
89           * and convert it in a list "name, action, action, ..., action2, action2, ..., ..."
90           * @param data
91           * @return
92           */
93          private String[] getRules(String data) {
94              String[] pieces = data.split("-");
95              Vector actions = new Vector();
96              actions.add(pieces[0]);
97              for (int i = 1; i < pieces.length; i += 2) {
98                  int cnt = i + 1 < pieces.length ? Integer.parseInt(pieces[i + 1]) : 1;
99                  for (int j = 0; j < cnt; j++) actions.add(pieces[i]);
100             }
101             return (String[]) actions.toArray(new String[0]);
102         }

105             MessagingException result = null;
106             if (idx + 1 >= actions.length && actions[actions.length - 1].equals("rpt")) idx = 1 + (idx % actions.length - 2);

108             if (action.equals("me") || action.equals("ex")) {
109                 String flag = action.substring(2).trim();

111                 result = new MessagingException("000" + (valid ? "V " : "I ") + prefix + "#" + idx);
112             } else if (action.equals("sfe")) {
113                 String flag = action.substring(3).trim();

115                 result = new SendFailedException("000" + (valid ? "V " : "I ") + prefix + "#" + idx);
116             } else if (action.startsWith("smtpafe")) {
117                 String retcodes = action.substring(7, 10).trim();
118                 int retcode = 500;

123             } else if (action.startsWith("smtpase")) {

130             } else if (action.startsWith("smtpsfe")) {
131                 String retcodes = action.substring(7).trim();
132                 int retcode = 500;

137                 
138             } else if (action.startsWith("null")) {
139                 result = new NullMessagingException();
140                 
141             } else if (action.startsWith("io")) {
142                 result = new IOMessagingException();
143             }

148         private boolean isValid(Exception ex) {
149             return ex != null && ex.getMessage() != null && ex.getMessage().length() > 4 && ex.getMessage().charAt(3) == 'V';
150         }

152         /**
153          * Calculate the priority of an exception (in order to choose the right exception on multiple/conflicts)
154          * @param ex
155          * @return
156          */
157         private int getPrio(Exception ex) {
158             // Addresses exceptions are low priority
159             // Take care of exception hierarchy: parents are the last in this ifs list
160             if (ex instanceof NullMessagingException) return 7000;
161             if (ex instanceof IOMessagingException) return 6000;
162             if (ex instanceof SMTPSendFailedException) return 5000 + ((SMTPAddressFailedException)ex).getReturnCode();
163             if (ex instanceof SMTPAddressFailedException) return 2000 + ((SMTPAddressFailedException)ex).getReturnCode();
164             if (ex instanceof SMTPAddressSucceededException) return 1000 + ((SMTPAddressSucceededException)ex).getReturnCode();
165             if (ex instanceof SendFailedException) return 4000;
166             if (ex instanceof MessagingException) return 3000;
167             return -1;
168         }
169         
170         /**
171          * Returns the exception with the greatest priority.
172          * @param newex
173          * @param oldex
174          * @return
175          */
176         private MessagingException getMajor(MessagingException newex, MessagingException oldex) {
177             if (newex == null) return oldex;
178             if (oldex == null) return newex;
179             return getPrio(newex) >= getPrio(oldex) ? newex : oldex;
180         }

182         private void addException(HashMap map, MessagingException ex, Address recipient) {
183             if (map.get(recipient) != null) {
184                 MessagingException oldex = (MessagingException) map.get(recipient);
185                 ex = getMajor(ex, oldex);
186             }
187             map.put(recipient, ex);
188         }

190         /**
191          * Clone the exception, sets the "invalid" addresses as invalidAddress (in the clone)
192          * If the exception does not support "invalidaddress" declaration rethrow the original exception.
193          * @return
194          */
195         private MessagingException generateClone(Exception ex, Address[] validUnsent, Address[] invalid, MessagingException next) {
196             // Attenzione: le eccezioni devono essere in ordine di gerarchia (le + generiche - MessagingException - per ultime)  
197             if (ex instanceof SMTPSendFailedException) {
198                 return new SMTPSendFailedException(((SMTPSendFailedException)ex).getCommand(), ((SMTPSendFailedException)ex).getReturnCode(), ex.getMessage(), next, new Address[0], validUnsent, invalid);
199                 
200             } else if (ex instanceof SendFailedException) {
201                 return new SendFailedException(ex.getMessage(), next, new Address[0], validUnsent, invalid);
202                 
203             } else if ((ex instanceof MessagingException) || (ex instanceof SMTPAddressSucceededException) || (ex instanceof SMTPAddressFailedException)) {
204                 //if (next != null) ex.setNextException(next);
205                 return (MessagingException) ex;
206             } 
207             return null;
208         }

210         private MessagingException generateParent(Exception majorChildEx, Address[] validSent, Address[] validUnsent, Address[] invalid, MessagingException next) {
211             // NOTE: exceptions must be checked considering their hierarchy
212             if (majorChildEx instanceof SMTPAddressSucceededException) {
213                 // If the major exception is an address exception we generate a SendFailedException (non SMTP). If we generate an SMTP exception the return code must be 2xx (e.g: 250)
214                 // ERR return new SMTPSendFailedException(((SMTPAddressSucceededException)majorChildEx).getCommand(), ((SMTPAddressSucceededException)majorChildEx).getReturnCode(), majorChildEx.getMessage(), next, validSent, validUnsent, invalid);
215                 //return new SMTPSendFailedException(((SMTPAddressSucceededException)majorChildEx).getCommand(), 250, "250 "+majorChildEx.getMessage(), next, validSent, validUnsent, invalid);
216                 return new SendFailedException(majorChildEx.getMessage(), next, validSent, validUnsent, invalid);
217                 
218             } else if (majorChildEx instanceof SMTPAddressFailedException) {
219                 // ERR: return new SMTPSendFailedException(((SMTPAddressFailedException)majorChildEx).getCommand(), ((SMTPAddressFailedException)majorChildEx).getReturnCode(), majorChildEx.getMessage(), next, validSent, validUnsent, invalid);
220                 //return new SMTPSendFailedException(((SMTPAddressFailedException)majorChildEx).getCommand(), 250, "250 "+majorChildEx.getMessage(), next, validSent, validUnsent, invalid);
221                 return new SendFailedException(majorChildEx.getMessage(), next, validSent, validUnsent, invalid);
222                 
223             } else if (majorChildEx instanceof SMTPSendFailedException) {
224                 return new SMTPSendFailedException(((SMTPSendFailedException)majorChildEx).getCommand(), ((SMTPSendFailedException)majorChildEx).getReturnCode(), majorChildEx.getMessage(), next, validSent, validUnsent, invalid);
225                 
226             } else if (majorChildEx instanceof SendFailedException) {
227                 return new SendFailedException(majorChildEx.getMessage(), next, validSent, validUnsent, invalid);
228                 
229             } else if (majorChildEx instanceof MessagingException) {
230                 //if (next != null) majorChildEx.setNextException(next);
231                 return (MessagingException) majorChildEx;
232             } 
233             
234             return null;
235         }

237         private MessagingException getRootException(HashMap map) {
238             // x recipient: sfe (generate an sfe in root), smtpase (generate an smtafe), smtpafe (generate an smtafe), 
239             // x root: me, sfe, smtpafe
240             
241             MessagingException current = null, prev = null, major = null;
242             Vector invalid = new Vector();
243             Vector validSent = new Vector();
244             Vector validUnsent = new Vector();
245             Iterator i = map.keySet().iterator();
246             while (i.hasNext()) {
247                 Address address = (Address) i.next();
248                 if (address != null) {
249                     current = (MessagingException) map.get(address);
250                     if (current != null) {
251                         if (prev != null) current.setNextException(prev);
252                         if (major != null) major = getMajor(major, current);
253                         else major = current;
254                         prev = current;
255                         
256                         if (!(current instanceof SMTPAddressSucceededException)) {
257                             if (isValid(current)) validUnsent.add(address);
258                             else invalid.add(address);
259                         } 
260                         else if (propSendValid) validSent.add(address); else validUnsent.add(address);
261                     } 
262                     else if (propSendValid) validSent.add(address); else validUnsent.add(address);
263                 }
264             }
265             
266             // on generic exception (not a per-recipient exception) clone it with all "unsents" (validSent -> validUnsent)
267             MessagingException ex1 = (MessagingException) map.get(null);
268             if (ex1 != null) {
269                 validUnsent.addAll(validSent);
270                 ex1 = generateClone(ex1, (Address[]) validUnsent.toArray(new Address[0]), (Address[]) invalid.toArray(new Address[0]), prev);
271             }
272             
273             // Create a container exception for the above exceptions
274             MessagingException ex2 = major != null ? (MessagingException) generateParent(major, (Address[]) validSent.toArray(new Address[0]), (Address[]) validUnsent.toArray(new Address[0]), (Address[]) invalid.toArray(new Address[0]), prev) : null;
275             
276             return getMajor(ex1, ex2);
277             
278         }

280         public void onConnect(TestStatus status, String server) throws MessagingException {
281             String serverData = server;
282             if (serverData.indexOf("://") >= 0) serverData = serverData.substring(serverData.indexOf("://") + 3);
283             if (serverData.indexOf(":") >= 0) serverData = serverData.substring(0, serverData.indexOf(":"));
284             String[] pieces = serverData.split("\\.");
285             String[] actions = getRules(pieces[0]);
286             int idx = status.getTransportServerConnectionCount(server);
287             
288             MessagingException ex = getException(actions, idx, null, server);
289             if (ex != null) throwME(ex);
290         }

292         public void onSendMessage(TestStatus status, String server, Listing pmails) throws MessagingException, SendFailedException {
293             String serverData = server;
294             if (serverData.indexOf("://") >= 0) serverData = serverData.substring(serverData.indexOf("://") + 3);
295             if (serverData.indexOf(":") >= 0) serverData = serverData.substring(0, serverData.indexOf(":"));
296             String[] serverPieces = serverData.split("\\.");
297             
298             HashMap exceptionMap = new HashMap();
299             int idx = status.getTransportServerSendCount(server);
300             
301             for (int i = 1; i < serverPieces.length - 1; i++) {
302                 String[] actions = getRules(serverPieces[i]);
303                 System.out.println("Actions1!");
304                 for (int k = 0; k < actions.length; k++) System.out.println(actions[k]);
305                 if (actions[0].endsWith(STAR_MATCHER) && actions[0].length() > 1) {
306                     String match = actions[0].substring(0, actions[0].length() - STAR_MATCHER.length());
307                     System.out.println("Match!");
308                     System.out.println(match);
309                     for (int j = 0; j < pmails.size(); j++) if (pmails.get(j).getRecipient().toString().startsWith(match) || match.equals(STAR_MATCHER)) {
310                         MessagingException ex = getException(actions, idx, pmails.get(j), server);
311                         addException(exceptionMap, ex, pmails.get(j).getRecipient().toInternetAddress());
312                     } else addException(exceptionMap, null, pmails.get(j).getRecipient().toInternetAddress());
313                 }
314             }
315             
316             for (int i = 1; i < serverPieces.length - 1; i++) {
317                 String[] actions = getRules(serverPieces[i]);
318                 System.out.println("Actions2!");
319                 for (int k = 0; k < actions.length; k++) System.out.println(actions[k]);
320                 if (actions[0].equals(STAR_MATCHER)) {
321                     MessagingException ex = getException(actions, idx, null, server);
322                     addException(exceptionMap, ex, null);
323                 }
324             }
325             
326             MessagingException ex = getRootException(exceptionMap);
327             if (ex != null) throwME(ex);
328         }

330         public void throwME(MessagingException ex) throws MessagingException {
331             if (ex instanceof NullMessagingException) throw new NullPointerException();
332             if (ex instanceof IOMessagingException) {
333                 MessagingException me = new MessagingException();
334                 me.setNextException(new IOException());
335                 throw me;
336             }
337             throw ex;
338         }
