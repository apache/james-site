137     protected int waitEmptySpool(int maxWait) {
138         if (maxWait == 0) maxWait = -1;
139         while (outgoingSpool.size() > 0 && (maxWait > 0 || maxWait == -1)) {
140             synchronized (this) {
141                 try {
142                     wait(1000);
143                 } catch (InterruptedException e) {}
144             }
145             if (maxWait != -1) maxWait -= 1000;
146         }
147 
148         
149         if (outgoingSpool.size() > 0) {
150             Iterator i = outgoingSpool.list();
151             while (i.hasNext()) {
152                 String key = (String) i.next();
153                 Mail m = null;
154                 try {
155                     m = outgoingSpool.retrieve(key);
156                     System.err.println("Still in outgoing: "+key+" S:"+m.getState()+" E:"+m.getErrorMessage()+" F:"+m.getSender()+" T:"+m.getRecipients()+" M:"+m.getMessage().getContent());
157                 } catch (MessagingException e) {
158                     e.printStackTrace();
159                     System.err.println("Still in outgoing: "+key+" NULL");
160                 } catch (IOException e) {
161                     e.printStackTrace();
162                     System.err.println("Still in outgoing: "+key+" IOException");
163                 }
164             }
165         
166         }
167 
168         return outgoingSpool.size();
169     }

171     /**
172      * 
173      * @param status
174      * @param sends
175      *            Email send attempts number. if < 0 will take the minimum number.
176      * @param maxConnection
177      *            Max attempts to connect the server. If < 0 defaults to the minimum.
178      */
179     protected void assertWhole(Tester.TestStatus status, int sends, int maxConnection) {
180         if (sends >= 0)
181             assertEquals(sends, status.getTransportSendCount());
182         else
183             assertTrue(status.getTransportSendCount() >= -sends);
184         if (maxConnection >= 0) {
185             assertTrue(maxConnection >= status.getTransportConnectionCount());
186             assertTrue(maxConnection >= status.getTransportCloseCount());
187         } else {
188             assertTrue(-maxConnection <= status.getTransportConnectionCount());
189             assertTrue(-maxConnection <= status.getTransportCloseCount());
190         }
191         assertEquals(status.getTransportConnectionCount(), status
192                 .getTransportCloseCount());
193     }

195     /**
196      * 
197      * @param status
198      * @param server
199      * @param sends
200      *            Email send attempts number. if < 0 will take the minimum number.
201      * @param maxConnection
202      *            Max attempts to connect the server. If < 0 defaults to the minimum.
203      */
204     protected void assertServer(Tester.TestStatus status, String server, int sends, int maxConnection) {
205         if (sends >= 0)
206             assertEquals(sends, status.getTransportServerSendCount(server));
207         else
208             assertTrue(status.getTransportServerSendCount(server) >= -sends);
209         if (maxConnection >= 0) {
210             assertTrue(maxConnection >= status.getTransportServerConnectionCount(server));
211             assertTrue(maxConnection >= status.getTransportServerCloseCount(server));
212         } else {
213             assertTrue(-maxConnection <= status.getTransportServerConnectionCount(server));
214             assertTrue(-maxConnection <= status.getTransportServerCloseCount(server));
215         }
216         assertEquals(status.getTransportServerConnectionCount(server), status.getTransportServerCloseCount(server));
217     }

219     /**
220      * Assert procmail result.
221      * 
222      * @param pmail
223      * @param state
224      * @param sends
225      *            Email send attempts number. if < 0 will take the minimum number.
226      * @param minBounce
227      * @param lastServer
228      */
229     protected void assertProcMail(ProcMail pmail, int state, int sends, int minBounce, String lastServer) {
230         if (sends >= 0) assertEquals(sends, pmail.getSendCount());
231         else assertTrue(pmail.getSendCount() >= -sends);
232         assertTrue(pmail.getBounceCount() >= minBounce);
233         assertEquals(state, pmail.getState());
234         assertEquals(lastServer, pmail.getSendCount() > 0 ? pmail.getSendServer(pmail.getSendCount() - 1) : null);
235         assertEquals(0, pmail.getErrorFlags());
236     }

444     private void stackDump(AssertionFailedError e) throws AssertionFailedError {
445         // TEMPORARILY add a dump stack on failure to 
446         // see if we have a deadlock (unlikely) or simply the
447         // notification is not working properly. (see JAMES-850)
448         
449         // Use reflection to invoke java 1.5 stack functions.
450         try {
451             Method m = Thread.class.getMethod("getAllStackTraces", null);
452             Map stackDump = (Map) m.invoke(null, null);
453             Set threads = stackDump.keySet();
454             for (Iterator i = threads.iterator(); i.hasNext(); ) {
455                 Thread th = (Thread) i.next();
456                 System.out.println("Thread "+th);
457                 StackTraceElement[] stack = (StackTraceElement[]) stackDump.get(th);
458                 for (int k = 0; k < stack.length; k++) {
459                     System.out.println("  - "+stack[k]);
460                 }
461             }
462         } catch (Exception reflectException) {
463         }
464         throw e;
465     }

751     protected void doTestMulti(RemoteDeliveryTestable rd, Properties params) throws Exception {
752         // Number of attempts
753         int loopCount = 20;
754         // Wait time between attempts
755         int loopWait = 0;
756         // Max wait betwen attempts (for the randomization)
757         int loopWaitRandom = 100;
758         // Max number of recipients per mail
759         int maxRecipientsPerEmail = 3;
760         // Probability for the recipients to be in the same domain
761         int probRecipientsSameDomain = 30;
762 
763         // creazione tester
764         initEnvironment();
765         Tester tester = new Tester(rd);
766         try {
767             tester.init(serviceManager, params);
768             
769             // String[][] servers = 
770             addServers(tester, getTestMultiServers(), true);
771             String[][] emails = getTestMultiEmails();
772             
773             Random rnd = new Random();
774             ProcMail.Listing[] results = new ProcMail.Listing[loopCount];
775             for (int i = 0; i < loopCount; i++) {
776                 // Calcolo recipients
777                 ArrayList rcpts = new ArrayList();
778                 int rcptn = rnd.nextInt(maxRecipientsPerEmail) + 1;
779                 if (rnd.nextInt(100) < probRecipientsSameDomain) {
780                     // same domain
781                     int dom = rnd.nextInt(emails.length);
782                     // for the domain we don't have more email, takes other domains emails
783                     if (rcptn >= (emails[dom].length - 1) / 2) for (int j = 1; j < emails[dom].length; j += 2) rcpts.add(emails[dom][j]);
784                     else {
785                         boolean[] got = new boolean[(emails[dom].length - 1) / 2];
786                         int done = 0;
787                         while (done < rcptn) {
788                             int t = rnd.nextInt((emails[dom].length - 1) / 2);
789                             if (!got[t]) {
790                                 rcpts.add(emails[dom][t * 2 + 1]);
791                                 got[t] = true;
792                                 done++;
793                             }
794                         }
795                     }
796                 } else {
797                     // multiple domains
798                     boolean got[][] = new boolean[emails.length][];
799                     for (int j = 0; j < emails.length; j++) got[j] = new boolean[1 + (emails[j].length - 1) / 2];
800                     int done = 0;
801                     int gotdepth = 0;
802                     while (done < rcptn && gotdepth < got.length) {
803                         int dom;
804                         do {dom = rnd.nextInt(got.length);} while (got[dom][0]);
805                         int t;
806                         do {t = rnd.nextInt(got[dom].length - 1);} while (got[dom][t + 1]);
807                         rcpts.add(emails[dom][t * 2 + 1]);
808                         got[dom][t + 1] = true;
809                         for (int j = 1; j < got[dom].length && got[dom][j]; j++) if (j == got[dom].length - 1) got[dom][0] = true;
810                         done ++;
811                     }
812                 }
813                 
814                 System.out.println("------");
815                 for (int j = 0; j < rcpts.size(); j++) System.out.println(rcpts.get(j)); 
816                 
817                 results[i] = tester.service("M" + i, i + "@test.it", (String[]) rcpts.toArray(new String[0]), "Subject: test" + i + "\r\nContent-Transfer-Encoding: plain\r\n\r\nbody");
818                 synchronized(this) {
819                     if (loopWait > 0) wait(loopWait);
820                     if (loopWaitRandom > 0) wait(rnd.nextInt(loopWaitRandom)+1);
821                 }
822             }
823             
824             // Wait for empty spool
825             assertEquals(0, waitEmptySpool(30000));
826             
827             Hashtable emailsRes = new Hashtable();
828             for (int i = 0; i < emails.length; i ++) for (int j = 1; j < emails[i].length; j += 2) emailsRes.put(emails[i][j], emails[i][j + 1]);
829             
830             boolean error = false;
831             for (int i = 0; i < results.length; i++) {
832                 System.out.println("Call#" + i);
833                 for (int j = 0; j < results[i].size(); j++) {
834                     ProcMail pmail = results[i].get(j);
835                     System.out.print(pmail.getKey() + " status:" + (pmail.getState() == ProcMail.STATE_SENT ? "SENT" : pmail.getState() == ProcMail.STATE_SENT_ERROR ? "ERR" : "" + pmail.getState() ) + " sends:" + pmail.getSendCount());
836                     String res = (String) emailsRes.get(pmail.getRecipient().toString());
837                     // if (pmail.getState() == ProcMail.STATE_IDLE) pmail.setState(ProcMail.STATE_SENT_ERROR);
838                     if ("ID".equals(res)) {
839                         if (pmail.getState() != ProcMail.STATE_IDLE) {
840                             error = true;
841                             System.out.print(" <<< ERROR, expected STATE_IDLE");
842                         }
843                     } else if ("KO".equals(res)) {
844                         if (pmail.getState() != ProcMail.STATE_SENT_ERROR) {
845                             error = true;
846                             System.out.print(" <<< ERROR, expected STATE_SENT_ERROR");
847                         }
848                     } else if ("OK".equals(res)) {
849                         if (pmail.getState() != ProcMail.STATE_SENT) {
850                             error = true;
851                             System.out.print(" <<< ERROR, expected STATE_SENT");
852                         }
853                     } else if ("NA".equals(res)) {
854                         if (pmail.getState() != ProcMail.STATE_SENT && pmail.getState() != ProcMail.STATE_SENT_ERROR) {
855                             error = true;
856                             System.out.print(" <<< ERROR, expected STATE_SENT or STATE_SENT_ERROR");
857                         }
858                     } else {
859                         throw new IllegalStateException("wrong expectation: "+res+". Check the test code!");
860                     }
861                     System.out.print("\n");
862                 }
