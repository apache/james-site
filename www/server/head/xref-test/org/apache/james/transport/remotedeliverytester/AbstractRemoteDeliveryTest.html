137     protected int waitEmptySpool(int maxWait) {
138         if (maxWait == 0) maxWait = -1;
139         while (outgoingSpool.size() > 0 && (maxWait > 0 || maxWait == -1)) {
140             synchronized (this) {
141                 try {
142                     wait(1000);
143                 } catch (InterruptedException e) {}
144             }
145             if (maxWait != -1) maxWait -= 1000;
146         }
147 
148         
149         if (outgoingSpool.size() > 0) {
150             Iterator i = outgoingSpool.list();
151             while (i.hasNext()) {
152                 String key = (String) i.next();
153                 Mail m = null;
154                 try {
155                     m = outgoingSpool.retrieve(key);
156                     System.err.println("Still in outgoing: "+key+" S:"+m.getState()+" E:"+m.getErrorMessage()+" F:"+m.getSender()+" T:"+m.getRecipients()+" M:"+m.getMessage().getContent());
157                 } catch (MessagingException e) {
158                     e.printStackTrace();
159                     System.err.println("Still in outgoing: "+key+" NULL");
160                 } catch (IOException e) {
161                     e.printStackTrace();
162                     System.err.println("Still in outgoing: "+key+" IOException");
163                 }
164             }
165         
166         }
167  
168         try {
169             Thread.currentThread().sleep(3000);
170         } catch (InterruptedException e) {}
171 
172         return outgoingSpool.size();
173     }

175     /**
176      * 
177      * @param status
178      * @param sends
179      *            Email send attempts number. if < 0 will take the minimum number.
180      * @param maxConnection
181      *            Max attempts to connect the server. If < 0 defaults to the minimum.
182      */
183     protected void assertWhole(Tester.TestStatus status, int sends, int maxConnection) {
184         if (sends >= 0)
185             assertEquals(sends, status.getTransportSendCount());
186         else
187             assertTrue(status.getTransportSendCount() >= -sends);
188         if (maxConnection >= 0) {
189             assertTrue(maxConnection >= status.getTransportConnectionCount());
190             assertTrue(maxConnection >= status.getTransportCloseCount());
191         } else {
192             assertTrue(-maxConnection <= status.getTransportConnectionCount());
193             assertTrue(-maxConnection <= status.getTransportCloseCount());
194         }
195         assertEquals(status.getTransportConnectionCount(), status
196                 .getTransportCloseCount());
197     }

199     /**
200      * 
201      * @param status
202      * @param server
203      * @param sends
204      *            Email send attempts number. if < 0 will take the minimum number.
205      * @param maxConnection
206      *            Max attempts to connect the server. If < 0 defaults to the minimum.
207      */
208     protected void assertServer(Tester.TestStatus status, String server, int sends, int maxConnection) {
209         if (sends >= 0)
210             assertEquals(sends, status.getTransportServerSendCount(server));
211         else
212             assertTrue(status.getTransportServerSendCount(server) >= -sends);
213         if (maxConnection >= 0) {
214             assertTrue(maxConnection >= status.getTransportServerConnectionCount(server));
215             assertTrue(maxConnection >= status.getTransportServerCloseCount(server));
216         } else {
217             assertTrue(-maxConnection <= status.getTransportServerConnectionCount(server));
218             assertTrue(-maxConnection <= status.getTransportServerCloseCount(server));
219         }
220         assertEquals(status.getTransportServerConnectionCount(server), status.getTransportServerCloseCount(server));
221     }

223     /**
224      * Assert procmail result.
225      * 
226      * @param pmail
227      * @param state
228      * @param sends
229      *            Email send attempts number. if < 0 will take the minimum number.
230      * @param minBounce
231      * @param lastServer
232      */
233     protected void assertProcMail(ProcMail pmail, int state, int sends, int minBounce, String lastServer) {
234         if (sends >= 0) assertEquals(sends, pmail.getSendCount());
235         else assertTrue(pmail.getSendCount() >= -sends);
236         assertTrue(pmail.getBounceCount() >= minBounce);
237         assertEquals(state, pmail.getState());
238         assertEquals(lastServer, pmail.getSendCount() > 0 ? pmail.getSendServer(pmail.getSendCount() - 1) : null);
239         assertEquals(0, pmail.getErrorFlags());
240     }

448     private void stackDump(AssertionFailedError e) throws AssertionFailedError {
449         // TEMPORARILY add a dump stack on failure to 
450         // see if we have a deadlock (unlikely) or simply the
451         // notification is not working properly. (see JAMES-850)
452         
453         // Use reflection to invoke java 1.5 stack functions.
454         try {
455             Method m = Thread.class.getMethod("getAllStackTraces", null);
456             Map stackDump = (Map) m.invoke(null, null);
457             Set threads = stackDump.keySet();
458             for (Iterator i = threads.iterator(); i.hasNext(); ) {
459                 Thread th = (Thread) i.next();
460                 System.out.println("Thread "+th);
461                 StackTraceElement[] stack = (StackTraceElement[]) stackDump.get(th);
462                 for (int k = 0; k < stack.length; k++) {
463                     System.out.println("  - "+stack[k]);
464                 }
465             }
466         } catch (Exception reflectException) {
467         }
468         throw e;
469     }

755     protected void doTestMulti(RemoteDeliveryTestable rd, Properties params) throws Exception {
756         // Number of attempts
757         int loopCount = 20;
758         // Wait time between attempts
759         int loopWait = 0;
760         // Max wait betwen attempts (for the randomization)
761         int loopWaitRandom = 100;
762         // Max number of recipients per mail
763         int maxRecipientsPerEmail = 3;
764         // Probability for the recipients to be in the same domain
765         int probRecipientsSameDomain = 30;
766 
767         // creazione tester
768         initEnvironment();
769         Tester tester = new Tester(rd);
770         try {
771             tester.init(serviceManager, params);
772             
773             // String[][] servers = 
774             addServers(tester, getTestMultiServers(), true);
775             String[][] emails = getTestMultiEmails();
776             
777             Random rnd = new Random();
778             ProcMail.Listing[] results = new ProcMail.Listing[loopCount];
779             for (int i = 0; i < loopCount; i++) {
780                 // Calcolo recipients
781                 ArrayList rcpts = new ArrayList();
782                 int rcptn = rnd.nextInt(maxRecipientsPerEmail) + 1;
783                 if (rnd.nextInt(100) < probRecipientsSameDomain) {
784                     // same domain
785                     int dom = rnd.nextInt(emails.length);
786                     // for the domain we don't have more email, takes other domains emails
787                     if (rcptn >= (emails[dom].length - 1) / 2) for (int j = 1; j < emails[dom].length; j += 2) rcpts.add(emails[dom][j]);
788                     else {
789                         boolean[] got = new boolean[(emails[dom].length - 1) / 2];
790                         int done = 0;
791                         while (done < rcptn) {
792                             int t = rnd.nextInt((emails[dom].length - 1) / 2);
793                             if (!got[t]) {
794                                 rcpts.add(emails[dom][t * 2 + 1]);
795                                 got[t] = true;
796                                 done++;
797                             }
798                         }
799                     }
800                 } else {
801                     // multiple domains
802                     boolean got[][] = new boolean[emails.length][];
803                     for (int j = 0; j < emails.length; j++) got[j] = new boolean[1 + (emails[j].length - 1) / 2];
804                     int done = 0;
805                     int gotdepth = 0;
806                     while (done < rcptn && gotdepth < got.length) {
807                         int dom;
808                         do {dom = rnd.nextInt(got.length);} while (got[dom][0]);
809                         int t;
810                         do {t = rnd.nextInt(got[dom].length - 1);} while (got[dom][t + 1]);
811                         rcpts.add(emails[dom][t * 2 + 1]);
812                         got[dom][t + 1] = true;
813                         for (int j = 1; j < got[dom].length && got[dom][j]; j++) if (j == got[dom].length - 1) got[dom][0] = true;
814                         done ++;
815                     }
816                 }
817                 
818                 System.out.println("------");
819                 for (int j = 0; j < rcpts.size(); j++) System.out.println(rcpts.get(j)); 
820                 
821                 results[i] = tester.service("M" + i, i + "@test.it", (String[]) rcpts.toArray(new String[0]), "Subject: test" + i + "\r\nContent-Transfer-Encoding: plain\r\n\r\nbody");
822                 synchronized(this) {
823                     if (loopWait > 0) wait(loopWait);
824                     if (loopWaitRandom > 0) wait(rnd.nextInt(loopWaitRandom)+1);
825                 }
826             }
827             
828             // Wait for empty spool
829             assertEquals(0, waitEmptySpool(30000));
830             
831             Hashtable emailsRes = new Hashtable();
832             for (int i = 0; i < emails.length; i ++) for (int j = 1; j < emails[i].length; j += 2) emailsRes.put(emails[i][j], emails[i][j + 1]);
833             
834             boolean error = false;
835             for (int i = 0; i < results.length; i++) {
836                 System.out.println("Call#" + i);
837                 for (int j = 0; j < results[i].size(); j++) {
838                     ProcMail pmail = results[i].get(j);
839                     System.out.print(pmail.getKey() + " status:" + (pmail.getState() == ProcMail.STATE_SENT ? "SENT" : pmail.getState() == ProcMail.STATE_SENT_ERROR ? "ERR" : "" + pmail.getState() ) + " sends:" + pmail.getSendCount());
840                     String res = (String) emailsRes.get(pmail.getRecipient().toString());
841                     // if (pmail.getState() == ProcMail.STATE_IDLE) pmail.setState(ProcMail.STATE_SENT_ERROR);
842                     if ("ID".equals(res)) {
843                         if (pmail.getState() != ProcMail.STATE_IDLE) {
844                             error = true;
845                             System.out.print(" <<< ERROR, expected STATE_IDLE");
846                         }
847                     } else if ("KO".equals(res)) {
848                         if (pmail.getState() != ProcMail.STATE_SENT_ERROR) {
849                             error = true;
850                             System.out.print(" <<< ERROR, expected STATE_SENT_ERROR");
851                         }
852                     } else if ("OK".equals(res)) {
853                         if (pmail.getState() != ProcMail.STATE_SENT) {
854                             error = true;
855                             System.out.print(" <<< ERROR, expected STATE_SENT");
856                         }
857                     } else if ("NA".equals(res)) {
858                         if (pmail.getState() != ProcMail.STATE_SENT && pmail.getState() != ProcMail.STATE_SENT_ERROR) {
859                             error = true;
860                             System.out.print(" <<< ERROR, expected STATE_SENT or STATE_SENT_ERROR");
861                         }
862                     } else {
863                         throw new IllegalStateException("wrong expectation: "+res+". Check the test code!");
864                     }
