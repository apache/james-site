55   * Email lifecycle

57   * - another thread pool, in mailtet.run it is accepted from the outgoing spool and sent to the "deliver" method.
58   * - "deliver" calls transport.connect and transport.sendMessage from javaMail
59   * - on delivery error a call to "failMessage" is done. failMessage will call mailetContext.sendMail or mailetContext.bounce to send a delivery failure receipt.
60   * - Back to "run" results in outgoing.remove (permanent result) or outgoing.store (temporary failure)

162         try {
163             if (!((s.getTransport("smtp")) instanceof SMTPTransport))

210     public ProcMail.Listing service(Mail mail, TransportRule rule) throws MessagingException {
211         ProcMail.Listing listing = new ProcMail.Listing();
212         Iterator recipients = mail.getRecipients().iterator();
213         while (recipients.hasNext()) {
214             MailAddress address = (MailAddress)recipients.next();
215             ProcMail pmail = new ProcMail(this, mail, address);
216             pmail.setTransportRule(rule);
217             storeProcMail(pmail);
218             listing.add(pmail);
219         }
220         remoteDelivery.service(mail);
221         return listing;
222     }

286     public synchronized void onTransportConnect(SMTPTransport tester) throws MessagingException {
287         String server = getServerName(tester.getURLName());
288         log("TRANSPORT", "Connection to " + server);
289         
290         Exception ex = null;
291         try {
292             if (serverRules.containsKey(server)) ((TransportRule)serverRules.get(server)).onConnect(testStatus, server);
293             if (genericRule != null) genericRule.onConnect(testStatus, server);
294         } catch (MessagingException e) {
295             ex = e;
296         } catch (RuntimeException e) {
297             ex = e;
298         }
299 
300         testStatus.incTransportConnectionCount(1);
301         testStatus.incTransportServerConnectionCount(server, 1);
302 
303         if (ex != null) {
304             if (ex instanceof MessagingException) throw (MessagingException)ex;
305             if (ex instanceof RuntimeException) throw (RuntimeException)ex;
306         }

309     public synchronized void onTransportSendMessage(SMTPTransport tester, Message message, Address[] recipients) throws MessagingException, SendFailedException {
310         String server = getServerName(tester.getURLName());
311         ProcMail.Listing listing = new ProcMail.Listing();
312         Vector rules = new Vector();
313         for (int i = 0; i < recipients.length; i++) {
314             ProcMail pmail = getProcMail(message, recipients[i]);
315             log("TRANSPORT.SEND", "Preparing send " + pmail.getKey() + " to " + server);
316             
317             if (pmail == null) throw new IllegalStateException("Unknow mail for " + recipients[i]);
318             listing.add(pmail);
319             if (pmail.getTransportRule() != null && !rules.contains(pmail.getTransportRule())) rules.add(pmail.getTransportRule());
320         }
321         
322         Exception ex = null;
323         try {
324             
325             log("TRANSPORT.SEND", "Sending to " + server + ", applying " + rules.size() + " mail rules");
326             for (int i = 0; i < rules.size(); i++) ((TransportRule) rules.get(i)).onSendMessage(testStatus, server, listing); 
327             
328             if (serverRules.containsKey(server)) ((TransportRule)serverRules.get(server)).onSendMessage(testStatus, server, listing);
329             if (genericRule != null) genericRule.onSendMessage(testStatus, server, listing);
330             
331         } catch (SendFailedException e) {
332             ex = e;
333         } catch (MessagingException e) {
334             ex = e;
335         } catch (RuntimeException e) {
336             ex = e;
337         }
338         
339         if (ex != null) log("TRANSPORT.SEND", "Got Exception " + dumpException(ex));
340         
341         // If we got a sendFailed then the specific recipient could be OK
342         Address[] validAddresses = ex instanceof SendFailedException ? ((SendFailedException)ex).getValidSentAddresses() : null;
343         for (int i = 0; i < listing.size(); i++) {
344             ProcMail pmail = listing.get(i);
345             boolean valid = false;
346             if (validAddresses != null) 
347                 for (int j = 0; !valid && j < validAddresses.length; j++)
348                     valid = pmail.getRecipient().toString().equals(validAddresses[j].toString());
349             
350             pmail.sent(server, !valid ? ex : null);
351 //          System.out.println("--------------------\r\n");
352 //          System.out.println("0. " + server + " " + pmail + " " + pmail.getKey() + " " + pmail.getSendCount() + valid + "\r\n");
353 //          System.out.println("--------------------\r\n");
354 
355             testStatus.incTransportSendCount(1);
356             testStatus.incTransportServerSendCount(server, 1);
357             if (ex == null || valid) {
358                 testStatus.incTransportSendSuccessfullCount(1);
359                 testStatus.incTransportServerSendSuccessfullCount(server, 1);
360             } else {
361                 testStatus.incTransportSendFailedCount(1);
362                 testStatus.incTransportServerSendFailedCount(server, 1);
363             }
364         }
365 
366         if (ex != null) {
367             if (ex instanceof SendFailedException) throw (SendFailedException)ex;
368             if (ex instanceof MessagingException) throw (MessagingException)ex;
369             if (ex instanceof RuntimeException) throw (RuntimeException)ex;
370         }

374     private String dumpException(Exception ex) {
375         if (!(ex instanceof MessagingException)) return ex.toString();
376         
377         StringBuffer res = new StringBuffer();
378         String exs = ex.getMessage();
379         int p = exs.indexOf("nested exception is:");
380         if (p >= 0) exs = exs.substring(0, p - 3);
381         res.append(exs);
382         //res.append(ex.toString());
383         
384         while (ex != null && (ex instanceof MessagingException)) {
385             res.append(" > ");
386             if (ex instanceof SMTPSendFailedException) res.append("smtpsfe" + ((SMTPSendFailedException)ex).getReturnCode()); else
387             if (ex instanceof SMTPAddressFailedException) res.append("smtpafe" + ((SMTPAddressFailedException)ex).getReturnCode()); else
388             if (ex instanceof SMTPAddressSucceededException) res.append("smtpase" + ((SMTPAddressSucceededException)ex).getReturnCode()); else
389             if (ex instanceof SendFailedException) res.append("sfe"); else
390             if (ex instanceof NullPointerException) res.append("null"); else 
391             res.append("me");
392             ex = ((MessagingException)ex).getNextException();
393         }
394         
395         return res.toString();
396     }
