141             // success

144         try {
145             m_userManagement.setAlias("testNonExist1", "testAlias", null);
146             fail("user unknown to server");
147         } catch (UserManagementException e) {
148             // success
149         }
150 
151         assertTrue("user added", m_userManagement.addUser("testAlias2", "test", null));
152 
153         // regular alias
154         assertTrue("alias for testAlias1 set to:testAlias2", m_userManagement.setAlias("testAlias1", "testAlias2", null));
155 
156         //TODO: is this correct? even primitive circular aliasing allowed!
157         assertTrue("alias for testAlias2 set to:testAlias1", m_userManagement.setAlias("testAlias2", "testAlias1", null));
158 
159         // did first one persist?
160         assertEquals("Current alias for testAlias1 is: testAlias2", "testAlias2", m_userManagement.getAlias("testAlias1", null));
161 
162         //TODO: is this correct? setting self as alias!
163         assertTrue("alias for testAlias1 set to:testAlias1", m_userManagement.setAlias("testAlias1", "testAlias1", null));
164 
165         assertTrue("user added", m_userManagement.addUser("testAlias3", "test", null));
166 
167         // re-set, simply overwrites
168         assertTrue("alias for testAlias1 set to:testAlias3", m_userManagement.setAlias("testAlias1", "testAlias3", null));
169 
170         // check overwrite
171         assertEquals("Current alias for testAlias1 is: testAlias3", "testAlias3", m_userManagement.getAlias("testAlias1", null));

182     public void testForward() throws UserManagementException {
183         m_mockUsersRepository.setForceUseJamesUser();
184 
185         // do some tests when parameter users don't exist
186         try {
187             m_userManagement.setForwardAddress("testNonExist1", "testForward1@locahost", null);
188             fail("user unknown to server");
189         } catch (UserManagementException e) {
190             // success
191         }
192 
193         assertTrue("user added", m_userManagement.addUser("testForwardUser", "test", null));
194 
195         assertNull("no forward set", m_userManagement.getForwardAddress("testForwardUser", null));
196 
197         assertTrue(m_userManagement.setForwardAddress("testForwardUser", "testForward1@locahost", null));
198 
199         // did it persist?
200         String forwardAddress = m_userManagement.getForwardAddress("testForwardUser", null);
201         assertEquals("forward for testForwardUser is: testForward1@locahost", "testForward1@locahost", forwardAddress);
202 
203         // re-set, simply overwrites
204         assertTrue(m_userManagement.setForwardAddress("testForwardUser", "testForward2@locahost", null));
205 
206         // check overwrite
207         forwardAddress = m_userManagement.getForwardAddress("testForwardUser", null);
208         assertEquals("forward for testForwardUser is: testForward2@locahost", "testForward2@locahost", forwardAddress);
209 
210         // retreat
211         assertTrue("Forward for testForwardUser unset", m_userManagement.unsetForwardAddress("testForwardUser", null));
212 
213         // check removed forward
214         assertNull("no more forward set", m_userManagement.getForwardAddress("testForwardUser", null));
215 
216     }

218     public void testSetPassword() throws UserManagementException {
219 
220         assertTrue("user added", m_userManagement.addUser("testPwdUser", "pwd1", null));
221 
222         assertTrue("initial password", m_mockUsersRepository.test("testPwdUser", "pwd1"));
223 
224         // set empty pwd
225         assertTrue("changed to empty password", m_userManagement.setPassword("testPwdUser", "", null));
226         assertTrue("password changed to empty", m_mockUsersRepository.test("testPwdUser", ""));
227 
228         // change pwd
229         assertTrue("changed password", m_userManagement.setPassword("testPwdUser", "pwd2", null));
230         assertTrue("password not changed to pwd2", m_mockUsersRepository.test("testPwdUser", "pwd2"));
231 
232         // assure case sensitivity
233         assertTrue("changed password", m_userManagement.setPassword("testPwdUser", "pWD2", null));
234         assertFalse("password no longer pwd2", m_mockUsersRepository.test("testPwdUser", "pwd2"));
235         assertTrue("password changed to pWD2", m_mockUsersRepository.test("testPwdUser", "pWD2"));
