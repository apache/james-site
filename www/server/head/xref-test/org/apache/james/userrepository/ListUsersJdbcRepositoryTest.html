72          configuration.addChild(new AttrValConfiguration("sqlFile","file://conf/sqlResources.xml"));
73          res.enableLogging(new ConsoleLogger());
74          res.configure(configuration );
75          res.initialize();
76      }

83          ContainerUtil.dispose(this.usersRepository);
