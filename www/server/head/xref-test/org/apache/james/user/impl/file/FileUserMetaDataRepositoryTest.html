20  package org.apache.james.user.impl.file;

22  import java.io.File;
23  
24  import junit.framework.TestCase;
25  
26  import org.apache.commons.io.FileUtils;
27  
28  public class FileUserMetaDataRepositoryTest extends TestCase {
29  
30      private static final String SYMBOLIC_KEY = "\\/><'@~ #][}{()=+.,| !`%$3\" exit(0)";
31  
32      private static final String KEY = "key";
33  
34      private static final String USER = "user";
35      private static final String ANOTHER_USER = "another user";
36  
37      private static final String TEST_DIRECTORY = "target/testusermetadata";
38      
39      FileUserMetaDataRepository repository;
40      
41      protected void setUp() throws Exception {
42          super.setUp();
43          final File directory = new File(TEST_DIRECTORY);
44          if (directory.exists()) {
45              FileUtils.deleteDirectory(directory);
46          }
47          repository = new FileUserMetaDataRepository(TEST_DIRECTORY);
48      }
49  
50      public void testClear() throws Exception {
51          final Long value = new Long(99);
52          repository.setAttribute(USER, value, KEY);
53          assertEquals(value, repository.getAttribute(USER, KEY));
54          final Long anotherValue = new Long(199);
55          repository.setAttribute(ANOTHER_USER, anotherValue, KEY);
56          assertEquals(anotherValue, repository.getAttribute(ANOTHER_USER, KEY));
57          repository.clear(USER);
58          assertNull(repository.getAttribute(USER, KEY));
59          assertEquals(anotherValue, repository.getAttribute(ANOTHER_USER, KEY));
60      }
61  
62      public void testSetGetAttribute() throws Exception {
63          final Long value = new Long(99);
64          repository.setAttribute(USER, value, KEY);
65          assertEquals(value, repository.getAttribute(USER, KEY));
66      }
67  
68      public void testUnsafeUserNames() throws Exception {
69          final Long valueOne = new Long(99);
70          final String userOne = USER + '\u25B2';
71          repository.setAttribute(userOne, valueOne, KEY);
72          final Long valueTwo = new Long(90);
73          final String userTwo = USER + '\u25B3';
74          repository.setAttribute(userTwo, valueTwo, KEY);
75          final Long valueThree = new Long(80);
76          final String userThree = USER + '\u25B4';
77          repository.setAttribute(userThree, valueThree, KEY);
78          final Long valueFour = new Long(199);
79          final String userFour = USER + '\uA48A';
80          repository.setAttribute(userFour, valueFour, KEY);
81          final Long valueFive = new Long(190);
82          final String userFive = USER + '\uA485';
83          repository.setAttribute(userFive, valueFive, KEY);
84          final Long valueSix = new Long(180);
85          final String userSix = USER + '\uA486';
86          repository.setAttribute(userSix, valueSix, KEY);
87          final Long valueSeven = new Long(290);
88          final String userSeven = USER + '\uFFFE';
89          repository.setAttribute(userSeven, valueSeven, KEY);
90          final Long valueEight = new Long(280);
91          final String userEight = USER + '\uFFFF';
92          repository.setAttribute(userEight, valueEight, KEY);
93          
94          assertEquals(valueOne, repository.getAttribute(userOne, KEY));
95          assertEquals(valueTwo, repository.getAttribute(userTwo, KEY));
96          assertEquals(valueThree, repository.getAttribute(userThree, KEY));
97          assertEquals(valueFour, repository.getAttribute(userFour, KEY));
98          assertEquals(valueFive, repository.getAttribute(userFive, KEY));
99          assertEquals(valueSix, repository.getAttribute(userSix, KEY));
100         assertEquals(valueSeven, repository.getAttribute(userSeven, KEY));
101         assertEquals(valueEight, repository.getAttribute(userEight, KEY));
102     }
103
