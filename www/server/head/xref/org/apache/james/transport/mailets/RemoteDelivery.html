77  /**
78   * Receives a MessageContainer from JamesSpoolManager and takes care of delivery
79   * the message to remote hosts. If for some reason mail can't be delivered
80   * store it in the "outgoing" Repository and set an Alarm. After the next "delayTime" the
81   * Alarm will wake the servlet that will try to send it again. After "maxRetries"
82   * the mail will be considered undeliverable and will be returned to sender.
83   *
84   * TO DO (in priority):
85   * 1. Support a gateway (a single server where all mail will be delivered) (DONE)
86   * 2. Provide better failure messages (DONE)
87   * 3. More efficiently handle numerous recipients
88   * 4. Migrate to use Phoenix for the delivery threads
89   *
90   * You really want to read the JavaMail documentation if you are
91   * working in here, and you will want to view the list of JavaMail
92   * attributes, which are documented here:
93   *
94   * http://java.sun.com/products/javamail/1.3/docs/javadocs/com/sun/mail/smtp/package-summary.html
95   *
96   * as well as other places.
97   *
98   * @version CVS $Revision: 717869 $ $Date: 2008-11-15 15:56:18 +0000 (Sat, 15 Nov 2008) $
99   */

129     /**
130      * This filter is used in the accept call to the spool.
131      * It will select the next mail ready for processing according to the mails
132      * retrycount and lastUpdated time
133      **/
134     private class MultipleDelayFilter implements SpoolRepository.AcceptFilter {
135         /**
136          * holds the time to wait for the youngest mail to get ready for processing
137          **/
138         long youngest = 0;
139 
140         /**
141          * Uses the getNextDelay to determine if a mail is ready for processing based on the delivered parameters
142          * errorMessage (which holds the retrycount), lastUpdated and state
143          * @param key the name/key of the message
144          * @param state the mails state
145          * @param lastUpdated the mail was last written to the spool at this time.
146          * @param errorMessage actually holds the retrycount as a string (see failMessage below)
147          **/
148         public boolean accept (String key, String state, long lastUpdated, String errorMessage) {
149             if (state.equals(Mail.ERROR)) {
150                 //Test the time...
151                 int retries = 0;
152                 
153                 try {
154                     retries = Integer.parseInt(errorMessage);
155                 } catch (NumberFormatException e) {
156                     // Something strange was happen with the errorMessage.. Ignore the Exception and try 
157                     // to deliver it now!
158                 }
159                 
160                 // If the retries count is 0 we should try to send the mail now!
161                 if (retries == 0) return true;
162                 
163                 long delay = getNextDelay (retries);
164                 long timeToProcess = delay + lastUpdated;
165 
166                 
167                 if (System.currentTimeMillis() > timeToProcess) {
168                     //We're ready to process this again
169                     return true;
170                 } else {
171                     //We're not ready to process this.
172                     if (youngest == 0 || youngest > timeToProcess) {
173                         //Mark this as the next most likely possible mail to process
174                         youngest = timeToProcess;
175                     }
176                     return false;
177                 }
178             } else {
179                 //This mail is good to go... return the key
180                 return true;
181             }
182         }
183 
184         /**
185          * @return the optimal time the SpoolRepository.accept(AcceptFilter) method should wait before
186          * trying to find a mail ready for processing again.
187          **/
188         public long getWaitTime () {
189             if (youngest == 0) {
190                 return 0;
191             } else {
192                 long duration = youngest - System.currentTimeMillis();
193                 youngest = 0; //get ready for next run
194                 return duration <= 0 ? 1 : duration;
195             }
196         }
197     }

269         // Set isDebug flag.
270         isDebug = (getInitParameter("debug") == null) ? false : new Boolean(getInitParameter("debug")).booleanValue();
271 
272         // Create list of Delay Times.
273         ArrayList delayTimesList = new ArrayList();
274         try {
275             if (getInitParameter("delayTime") != null) {
276                 delayTimeMatcher = new Perl5Matcher();
277                 String delayTimesParm = getInitParameter("delayTime");
278 
279                 // Split on commas
280                 StringTokenizer st = new StringTokenizer (delayTimesParm,",");
281                 while (st.hasMoreTokens()) {
282                     String delayTime = st.nextToken();
283                     delayTimesList.add (new Delay(delayTime));
284                 }
285             } else {
286                 // Use default delayTime.
287                 delayTimesList.add (new Delay());
288             }
289         } catch (Exception e) {
290             log("Invalid delayTime setting: " + getInitParameter("delayTime"));
291         }
292         
293         try {
294             // Get No. of Max Retries.
295             if (getInitParameter("maxRetries") != null) {
296                 maxRetries = Integer.parseInt(getInitParameter("maxRetries"));
297             }
298             
299             // Check consistency of 'maxRetries' with delayTimesList attempts.
300             int totalAttempts = calcTotalAttempts(delayTimesList);
301 
302             // If inconsistency found, fix it.
303             if (totalAttempts > maxRetries) {
304                 log("Total number of delayTime attempts exceeds maxRetries specified. "
305                         + " Increasing maxRetries from "
306                         + maxRetries
307                         + " to "
308                         + totalAttempts);
309                 maxRetries = totalAttempts;
310             } else {
311                 int extra = maxRetries - totalAttempts;
312                 if (extra != 0) {
313                     log("maxRetries is larger than total number of attempts specified.  "
314                             + "Increasing last delayTime with "
315                             + extra
316                             + " attempts ");
317 
318                     // Add extra attempts to the last delayTime.
319                     if (delayTimesList.size() != 0) { 
320                         // Get the last delayTime.
321                         Delay delay = (Delay) delayTimesList.get(delayTimesList
322                                 .size() - 1);
323 
324                         // Increase no. of attempts.
325                         delay.setAttempts(delay.getAttempts() + extra);
326                         log("Delay of " + delay.getDelayTime()
327                                 + " msecs is now attempted: " + delay.getAttempts()
328                                 + " times");
329                     } else {
330                         throw new MessagingException(
331                                 "No delaytimes, cannot continue");
332                     }
333                 }
334             }
335             delayTimes = expandDelays(delayTimesList);
336             
337         } catch (Exception e) {
338             log("Invalid maxRetries setting: " + getInitParameter("maxRetries"));
339         }
340 
341         ServiceManager compMgr = (ServiceManager) getMailetContext()
342                 .getAttribute(Constants.AVALON_COMPONENT_MANAGER);
343         
344         // Get the path for the 'Outgoing' repository. This is the place on the
345         // file system where Mail objects will be saved during the 'delivery'
346         // processing. This can be changed to a repository on a database (e.g.
347         // db://maildb/spool/retry).
348         String workRepositoryPath = getInitParameter("outgoing");
349         if (workRepositoryPath == null) {
350             workRepositoryPath = "file:///../var/mail/outgoing";
351         }
352 
353         try {
354             // Instantiate a MailRepository for mails to be processed.
355             Store mailstore = (Store) compMgr.lookup(Store.ROLE);

357             DefaultConfiguration spoolConf = new DefaultConfiguration(
358                     "repository", "generated:RemoteDelivery");
359             spoolConf.setAttribute("destinationURL", workRepositoryPath);

361             workRepository = (SpoolRepository) mailstore.select(spoolConf);
362         } catch (ServiceException cnfe) {
363             log("Failed to retrieve Store component:" + cnfe.getMessage());
364             throw new MessagingException("Failed to retrieve Store component",
365                     cnfe);
366         }
367         
368         // Start Workers Threads.
369         workersThreadCount = Integer.parseInt(getInitParameter("deliveryThreads"));
370         for (int i = 0; i < workersThreadCount; i++) {
371             String threadName = "Remote delivery thread (" + i + ")";
372             Thread t = new Thread(this, threadName);
373             t.start();
374             workersThreads.add(t);
375         }
376 
377         try {
378             if (getInitParameter("timeout") != null) {
379                 smtpTimeout = Integer.parseInt(getInitParameter("timeout"));
380             }
381         } catch (Exception e) {
382             log("Invalid timeout setting: " + getInitParameter("timeout"));
383         }
384 
385         try {
386             if (getInitParameter("connectiontimeout") != null) {
387                 connectionTimeout = Integer.parseInt(getInitParameter("connectiontimeout"));
388             }
389         } catch (Exception e) {
390             log("Invalid timeout setting: " + getInitParameter("timeout"));
391         }
392 
393         sendPartial = (getInitParameter("sendpartial") == null) ? false : new Boolean(getInitParameter("sendpartial")).booleanValue();
394 
395         bounceProcessor = getInitParameter("bounceProcessor");
396 
397         String gateway = getInitParameter("gateway");
398         String gatewayPort = getInitParameter("gatewayPort");
399 
400         if (gateway != null) {
401             gatewayServer = new ArrayList();
402             StringTokenizer st = new StringTokenizer(gateway, ",") ;
403             while (st.hasMoreTokens()) {
404                 String server = st.nextToken().trim() ;
405                 if (server.indexOf(':') < 0 && gatewayPort != null) {
406                     server += ":";
407                     server += gatewayPort;
408                 }
409 
410                 if (isDebug) log("Adding SMTP gateway: " + server) ;
411                 gatewayServer.add(server);
412             }
413             authUser = getInitParameter("gatewayUsername");
414             // backward compatibility with 2.3.x
415             if (authUser == null) {
416                 authUser = getInitParameter("gatewayusername");
417             }
418             authPass = getInitParameter("gatewayPassword");
419         }
420 
421         // Instantiate the DNSService for DNS queries lookup
422         try {
423             dnsServer = (DNSService) compMgr.lookup(DNSService.ROLE);
424         } catch (ServiceException e1) {
425             log("Failed to retrieve DNSService" + e1.getMessage());
426         }
427 
428         bindAddress = getInitParameter("bind");
429         isBindUsed = bindAddress != null;
430         try {
431             if (isBindUsed) RemoteDeliverySocketFactory.setBindAdress(bindAddress);
432         } catch (UnknownHostException e) {
433             log("Invalid bind setting (" + bindAddress + "): " + e.toString());
434         }
435 
436         // deal with <mail.*> attributes, passing them to javamail
437         Iterator i = getInitParameterNames();
438         while (i.hasNext()) {
439             String name = (String) i.next();
440             if (name.startsWith("mail.")) {
441                 defprops.put(name,getInitParameter(name));
442             }
443             
444         }
445         
446         String dnsRetry = getInitParameter("maxDnsProblemRetries");
447         if (dnsRetry != null && !dnsRetry.equals("")) {
448             dnsProblemRetry = Integer.parseInt(dnsRetry); 
449         }

452     /**
453      * Calculates Total no. of attempts for the specified delayList.
454      * 
455      * @param delayList
456      *            list of 'Delay' objects
457      * @return total no. of retry attempts
458      */
459     private int calcTotalAttempts (ArrayList delayList) {
460         int sum = 0;
461         Iterator i = delayList.iterator();
462         while (i.hasNext()) {
463             Delay delay = (Delay)i.next();
464             sum += delay.getAttempts();
465         }
466         return sum;
467     }

469     /**
470      * This method expands an ArrayList containing Delay objects into an array holding the
471      * only delaytime in the order.<p>
472      * So if the list has 2 Delay objects the first having attempts=2 and delaytime 4000
473      * the second having attempts=1 and delaytime=300000 will be expanded into this array:<p>
474      * long[0] = 4000<p>
475      * long[1] = 4000<p>
476      * long[2] = 300000<p>
477      * @param list the list to expand
478      * @return the expanded list
479      **/
480     private long[] expandDelays (ArrayList list) {
481         long[] delays = new long [calcTotalAttempts(list)];
482         Iterator i = list.iterator();
483         int idx = 0;
484         while (i.hasNext()) {
485             Delay delay = (Delay)i.next();
486             for (int j=0; j<delay.getAttempts(); j++) {
487                 delays[idx++]= delay.getDelayTime();
488             }            
489         }
490         return delays;
491     }

493     /**
494      * This method returns, given a retry-count, the next delay time to use.
495      * @param retry_count the current retry_count.
496      * @return the next delay time to use, given the retry count
497      **/
498     private long getNextDelay (int retry_count) {
499         if (retry_count > delayTimes.length) {
500             return DEFAULT_DELAY_TIME;
501         } 
502         return delayTimes[retry_count-1];
503     }

505     /**
506      * This class is used to hold a delay time and its corresponding number of
507      * retries.
508      **/
509     private class Delay {
510         private int attempts = 1;
511 
512         private long delayTime = DEFAULT_DELAY_TIME;
513 
514         /**
515          * This constructor expects Strings of the form
516          * "[attempt\*]delaytime[unit]".
517          * <p>
518          * The optional attempt is the number of tries this delay should be used
519          * (default = 1). The unit, if present, must be one of
520          * (msec,sec,minute,hour,day). The default value of unit is 'msec'.
521          * <p>
522          * The constructor multiplies the delaytime by the relevant multiplier
523          * for the unit, so the delayTime instance variable is always in msec.
524          * 
525          * @param initString
526          *            the string to initialize this Delay object from
527          **/
528         public Delay(String initString) throws MessagingException {
529             // Default unit value to 'msec'.
530             String unit = "msec";
531 
532             if (delayTimeMatcher.matches(initString, PATTERN)) {
533                 MatchResult res = delayTimeMatcher.getMatch();
534 
535                 // The capturing groups will now hold:
536                 // at 1: attempts * (if present)
537                 // at 2: delaytime
538                 // at 3: unit (if present)
539                 if (res.group(1) != null && !res.group(1).equals("")) {
540                     // We have an attempt *
541                     String attemptMatch = res.group(1);
542 
543                     // Strip the * and whitespace.
544                     attemptMatch = attemptMatch.substring(0,
545                             attemptMatch.length() - 1).trim();
546                     attempts = Integer.parseInt(attemptMatch);
547                 }
548 
549                 delayTime = Long.parseLong(res.group(2));
550 
551                 if (!res.group(3).equals("")) {
552                     // We have a value for 'unit'.
553                     unit = res.group(3).toLowerCase(Locale.US);
554                 }
555             } else {
556                 throw new MessagingException(initString + " does not match "
557                         + PATTERN_STRING);
558             }
559 
560             // calculate delayTime.
561             try {
562                 delayTime = TimeConverter.getMilliSeconds(delayTime, unit);
563             } catch (NumberFormatException e) {
564                 throw new MessagingException(e.getMessage());
565             }
566         }
567 
568         /**
569          * This constructor makes a default Delay object with attempts = 1 and
570          * delayTime = DEFAULT_DELAY_TIME.
571          **/
572         public Delay() {
573         }
574 
575         /**
576          * @return the delayTime for this Delay
577          **/
578         public long getDelayTime() {
579             return delayTime;
580         }
581 
582         /**
583          * @return the number attempts this Delay should be used.
584          **/
585         public int getAttempts() {
586             return attempts;
587         }
588 
589         /**
590          * Set the number attempts this Delay should be used.
591          **/
592         public void setAttempts(int value) {
593             attempts = value;
594         }
595 
596         /**
597          * Pretty prints this Delay
598          **/
599         public String toString() {
600             String message = getAttempts() + "*" + getDelayTime() + "msecs";
601             return message;
602         }
603     }

609     /**
610      * For this message, we take the list of recipients, organize these into distinct
611      * servers, and duplicate the message for each of these servers, and then call
612      * the deliver (messagecontainer) method for each server-specific
613      * messagecontainer ... that will handle storing it in the outgoing queue if needed.
614      *
615      * @param mail org.apache.mailet.Mail
616      */
617     public void service(Mail mail) throws MessagingException{
618         // Do I want to give the internal key, or the message's Message ID
619         if (isDebug) {
620             log("Remotely delivering mail " + mail.getName());
621         }
622         Collection recipients = mail.getRecipients();
623 
624         if (gatewayServer == null) {
625             // Must first organize the recipients into distinct servers (name made case insensitive)
626             Hashtable targets = new Hashtable();
627             for (Iterator i = recipients.iterator(); i.hasNext();) {
628                 MailAddress target = (MailAddress)i.next();
629                 String targetServer = target.getHost().toLowerCase(Locale.US);
630                 Collection temp = (Collection)targets.get(targetServer);
631                 if (temp == null) {
632                     temp = new ArrayList();
633                     targets.put(targetServer, temp);
634                 }
635                 temp.add(target);
636             }
637 
638             //We have the recipients organized into distinct servers... put them into the
639             //delivery store organized like this... this is ultra inefficient I think...
640 
641             // Store the new message containers, organized by server, in the outgoing mail repository
642             String name = mail.getName();
643             for (Iterator i = targets.keySet().iterator(); i.hasNext(); ) {
644                 String host = (String) i.next();
645                 Collection rec = (Collection) targets.get(host);
646                 if (isDebug) {
647                     StringBuffer logMessageBuffer =
648                         new StringBuffer(128)
649                                 .append("Sending mail to ")
650                                 .append(rec)
651                                 .append(" on host ")
652                                 .append(host);
653                     log(logMessageBuffer.toString());
654                 }
655                 mail.setRecipients(rec);
656                 StringBuffer nameBuffer =
657                     new StringBuffer(128)
658                             .append(name)
659                             .append("-to-")
660                             .append(host);
661                 mail.setName(nameBuffer.toString());
662                 workRepository.store(mail);
663                 //Set it to try to deliver (in a separate thread) immediately (triggered by storage)
664             }
665         } else {
666             // Store the mail unaltered for processing by the gateway server(s)
667             if (isDebug) {
668                 StringBuffer logMessageBuffer =
669                     new StringBuffer(128)
670                         .append("Sending mail to ")
671                         .append(mail.getRecipients())
672                         .append(" via ")
673                         .append(gatewayServer);
674                 log(logMessageBuffer.toString());
675             }
676 
677              //Set it to try to deliver (in a separate thread) immediately (triggered by storage)
678             workRepository.store(mail);
679         }
680         mail.setState(Mail.GHOST);
681     }

683     /**
684      * Stops all the worker threads that are waiting for messages. This method is
685      * called by the Mailet container before taking this Mailet out of service.
686      */
687     public synchronized void destroy() {
688         // Mark flag so threads from this Mailet stop themselves
689         destroyed = true;
690 
691         // Wake up all threads from waiting for an accept
692         for (Iterator i = workersThreads.iterator(); i.hasNext(); ) {
693             Thread t = (Thread)i.next();
694             t.interrupt();
695         }

805                             // Furthermore this mail should not be processed now
806                             // because we have a retry time scheduling.

808                         
809                         // Clear the object handle to make sure it recycles
810                         // this object.
811                         mail = null;
812                     } catch (Exception e) {
813                         // Prevent unexpected exceptions from causing looping by
814                         // removing message from outgoing.
815                         // DO NOT CHANGE THIS to catch Error! For example, if
816                         // there were an OutOfMemory condition caused because 
817                         // something else in the server was abusing memory, we would 
818                         // not want to start purging the retrying spool!
819                         ContainerUtil.dispose(mail);
820                         workRepository.remove(key);
821                         throw e;
822                     }
823                 } catch (Throwable e) {
824                     if (!destroyed) {
825                         log("Exception caught in RemoteDelivery.run()", e);
826                     }
827                 }
828             }
829         } finally {
830             // Restore the thread state to non-interrupted.
831             Thread.interrupted();
832         }

836     /**
837      * We can assume that the recipients of this message are all going to the same
838      * mail server.  We will now rely on the DNS server to do DNS MX record lookup
839      * and try to deliver to the multiple mail servers.  If it fails, it should
840      * throw an exception.
841      *
842      * Creation date: (2/24/00 11:25:00 PM)
843      * @param mail org.apache.james.core.MailImpl
844      * @param session javax.mail.Session
845      * @return boolean Whether the delivery was successful and the message can be deleted
846      */
847     private boolean deliver(Mail mail, Session session) {
848         try {
849             if (isDebug) {
850                 log("Attempting to deliver " + mail.getName());
851             }
852             MimeMessage message = mail.getMessage();
853 
854             //Create an array of the recipients as InternetAddress objects
855             Collection recipients = mail.getRecipients();
856             InternetAddress addr[] = new InternetAddress[recipients.size()];
857             int j = 0;
858             for (Iterator i = recipients.iterator(); i.hasNext(); j++) {
859                 MailAddress rcpt = (MailAddress)i.next();
860                 addr[j] = rcpt.toInternetAddress();
861             }
862 
863             if (addr.length <= 0) {
864                 log("No recipients specified... not sure how this could have happened.");
865                 return true;
866             }
867 
868             //Figure out which servers to try to send to.  This collection
869             //  will hold all the possible target servers
870             Iterator targetServers = null;
871             if (gatewayServer == null) {
872                 MailAddress rcpt = (MailAddress) recipients.iterator().next();
873                 String host = rcpt.getHost();
874 
875                 //Lookup the possible targets
876                 try {
877                     targetServers = dnsServer.getSMTPHostAddresses(host);
878                 } catch (TemporaryResolutionException e) {
879                     log("Temporary problem looking up mail server for host: " + host);
880                     StringBuffer exceptionBuffer =
881                         new StringBuffer(128)
882                         .append("Temporary problem looking up mail server for host: ")
883                         .append(host)
884                         .append(".  I cannot determine where to send this message.");
885                     
886                     // temporary problems
887                     return failMessage(mail, new MessagingException(exceptionBuffer.toString()), false);
888                 }
889                 if (!targetServers.hasNext()) {
890                     log("No mail server found for: " + host);
891                     StringBuffer exceptionBuffer =
892                         new StringBuffer(128)
893                         .append("There are no DNS entries for the hostname ")
894                         .append(host)
895                         .append(".  I cannot determine where to send this message.");
896                     
897                     int retry = 0;
898                     try {
899                         retry = Integer.parseInt(mail.getErrorMessage());
900                     } catch (NumberFormatException e) {
901                         // Unable to parse retryCount 
902                     }
903                     if (retry == 0 || retry > dnsProblemRetry) { 
904                         // The domain has no dns entry.. Return a permanent error
905                         return failMessage(mail, new MessagingException(exceptionBuffer.toString()), true);
906                     } else {
907                         return failMessage(mail, new MessagingException(exceptionBuffer.toString()), false);
908                     }
909                 }
910             } else {
911                 targetServers = getGatewaySMTPHostAddresses(gatewayServer);
912             }
913 
914             MessagingException lastError = null;
915 
916             while ( targetServers.hasNext()) {
917                 try {
918                     HostAddress outgoingMailServer = (HostAddress) targetServers.next();
919                     StringBuffer logMessageBuffer =
920                         new StringBuffer(256)
921                         .append("Attempting delivery of ")
922                         .append(mail.getName())
923                         .append(" to host ")
924                         .append(outgoingMailServer.getHostName())
925                         .append(" at ")
926                         .append(outgoingMailServer.getHost())
927                         .append(" for addresses ")
928                         .append(Arrays.asList(addr));
929                     log(logMessageBuffer.toString());
930 
931                     Properties props = session.getProperties();
932                     if (mail.getSender() == null) {
933                         props.put("mail.smtp.from", "<>");
934                     } else {
935                         String sender = mail.getSender().toString();
936                         props.put("mail.smtp.from", sender);
937                     }
938 
939                     //Many of these properties are only in later JavaMail versions
940                     //"mail.smtp.ehlo"  //default true
941                     //"mail.smtp.auth"  //default false
942                     //"mail.smtp.dsn.ret"  //default to nothing... appended as RET= after MAIL FROM line.
943                     //"mail.smtp.dsn.notify" //default to nothing...appended as NOTIFY= after RCPT TO line.
944 
945                     Transport transport = null;
946                     try {
947                         transport = session.getTransport(outgoingMailServer);
948                         try {
949                             if (authUser != null) {
950                                 transport.connect(outgoingMailServer.getHostName(), authUser, authPass);
951                             } else {
952                                 transport.connect();
953                             }
954                         } catch (MessagingException me) {
955                             // Any error on connect should cause the mailet to attempt
956                             // to connect to the next SMTP server associated with this
957                             // MX record.  Just log the exception.  We'll worry about
958                             // failing the message at the end of the loop.
959                             log(me.getMessage());
960                             continue;
961                         }
962                         // if the transport is a SMTPTransport (from sun) some
963                         // performance enhancement can be done.
964                         if (transport.getClass().getName().endsWith(".SMTPTransport"))  {
965                             boolean supports8bitmime = false;
966                             try {
967                                 Method supportsExtension = transport.getClass().getMethod("supportsExtension", new Class[] {String.class});
968                                 supports8bitmime = ((Boolean) supportsExtension.invoke(transport, new Object[] {"8BITMIME"})).booleanValue();
969                             } catch (NoSuchMethodException nsme) {
970                                 // An SMTPAddressFailedException with no getAddress method.
971                             } catch (IllegalAccessException iae) {
972                             } catch (IllegalArgumentException iae) {
973                             } catch (InvocationTargetException ite) {
974                                 // Other issues with getAddress invokation.
975                             }
976                            
977                             // if the message is alredy 8bit or binary and the
978                             // server doesn't support the 8bit extension it has
979                             // to be converted to 7bit. Javamail api doesn't perform
980                             // that conversion, but it is required to be a
981                             // rfc-compliant smtp server.
982                             
983                             // Temporarily disabled. See JAMES-638
984                             if (!supports8bitmime) {
985                                 try {
986                                     convertTo7Bit(message);
987                                 } catch (IOException e) {
988                                     // An error has occured during the 7bit conversion.
989                                     // The error is logged and the message is sent anyway.
990                                     
991                                     log("Error during the conversion to 7 bit.", e);
992                                 }
993                             }
994                         } else {
995                             // If the transport is not the one
996                             // developed by Sun we are not sure of how it
997                             // handles the 8 bit mime stuff,
998                             // so I convert the message to 7bit.
999                             try {
1000                                 convertTo7Bit(message);
1001                             } catch (IOException e) {
1002                                 log("Error during the conversion to 7 bit.", e);
1003                             }
1004                         }
1005                         transport.sendMessage(message, addr);
1006                     } finally {
1007                         if (transport != null) {
1008                             transport.close();
1009                             transport = null;
1010                         }
1011                     }
1012                     logMessageBuffer =
1013                                       new StringBuffer(256)
1014                                       .append("Mail (")
1015                                       .append(mail.getName())
1016                                       .append(") sent successfully to ")
1017                                       .append(outgoingMailServer.getHostName())
1018                                       .append(" at ")
1019                                       .append(outgoingMailServer.getHost())
1020                                       .append(" for ")
1021                                       .append(mail.getRecipients());
1022                     log(logMessageBuffer.toString());
1023                     return true;
1024                 } catch (SendFailedException sfe) {
1025                     logSendFailedException(sfe);
1026 
1027                     if (sfe.getValidSentAddresses() != null) {
1028                         Address[] validSent = sfe.getValidSentAddresses();
1029                         if (validSent.length > 0) {
1030                             StringBuffer logMessageBuffer =
1031                                 new StringBuffer(256)
1032                                 .append("Mail (")
1033                                 .append(mail.getName())
1034                                 .append(") sent successfully for ")
1035                                 .append(Arrays.asList(validSent));
1036                             log(logMessageBuffer.toString());
1037                         }
1038                     }
1039 
1040                     /* SMTPSendFailedException introduced in JavaMail 1.3.2, and provides detailed protocol reply code for the operation */
1041                     if (sfe.getClass().getName().endsWith(".SMTPSendFailedException")) {
1042                         try {
1043                             int returnCode = ((Integer) invokeGetter(sfe, "getReturnCode")).intValue();
1044                             // if 5xx, terminate this delivery attempt by re-throwing the exception.
1045                             if (returnCode >= 500 && returnCode <= 599) throw sfe;
1046                         } catch (ClassCastException cce) {
1047                         } catch (IllegalArgumentException iae) {
1048                         }
1049                     }
1050 
1051                     if (sfe.getValidUnsentAddresses() != null
1052                         && sfe.getValidUnsentAddresses().length > 0) {
1053                         if (isDebug) log("Send failed, " + sfe.getValidUnsentAddresses().length + " valid addresses remain, continuing with any other servers");
1054                         lastError = sfe;
1055                         continue;
1056                     } else {
1057                         // There are no valid addresses left to send, so rethrow
1058                         throw sfe;
1059                     }
1060                 } catch (MessagingException me) {
1061                     //MessagingException are horribly difficult to figure out what actually happened.
1062                     StringBuffer exceptionBuffer =
1063                         new StringBuffer(256)
1064                         .append("Exception delivering message (")
1065                         .append(mail.getName())
1066                         .append(") - ")
1067                         .append(me.getMessage());
1068                     log(exceptionBuffer.toString());
1069                     if ((me.getNextException() != null) &&
1070                           (me.getNextException() instanceof java.io.IOException)) {
1071                         //This is more than likely a temporary failure
1072 
1073                         // If it's an IO exception with no nested exception, it's probably
1074                         // some socket or weird I/O related problem.
1075                         lastError = me;
1076                         continue;
1077                     }
1078                     // This was not a connection or I/O error particular to one
1079                     // SMTP server of an MX set.  Instead, it is almost certainly
1080                     // a protocol level error.  In this case we assume that this
1081                     // is an error we'd encounter with any of the SMTP servers
1082                     // associated with this MX record, and we pass the exception
1083                     // to the code in the outer block that determines its severity.
1084                     throw me;
1085                 }
1086             } // end while
1087             //If we encountered an exception while looping through,
1088             //throw the last MessagingException we caught.  We only
1089             //do this if we were unable to send the message to any
1090             //server.  If sending eventually succeeded, we exit
1091             //deliver() though the return at the end of the try
1092             //block.
1093             if (lastError != null) {
1094                 throw lastError;
1095             }
1096         } catch (SendFailedException sfe) {
1097         logSendFailedException(sfe);
1098 
1099             Collection recipients = mail.getRecipients();
1100 
1101             boolean deleteMessage = false;
1102 
1103             /*
1104              * If you send a message that has multiple invalid
1105              * addresses, you'll get a top-level SendFailedException
1106              * that that has the valid, valid-unsent, and invalid
1107              * address lists, with all of the server response messages
1108              * will be contained within the nested exceptions.  [Note:
1109              * the content of the nested exceptions is implementation
1110              * dependent.]
1111              *
1112              * sfe.getInvalidAddresses() should be considered permanent.
1113              * sfe.getValidUnsentAddresses() should be considered temporary.
1114              *
1115              * JavaMail v1.3 properly populates those collections based
1116              * upon the 4xx and 5xx response codes to RCPT TO.  Some
1117              * servers, such as Yahoo! don't respond to the RCPT TO,
1118              * and provide a 5xx reply after DATA.  In that case, we
1119              * will pick up the failure from SMTPSendFailedException.
1120              *
1121              */
1122 
1123             /* SMTPSendFailedException introduced in JavaMail 1.3.2, and provides detailed protocol reply code for the operation */
1124             try {
1125                 if (sfe.getClass().getName().endsWith(".SMTPSendFailedException")) {
1126                     int returnCode = ((Integer) invokeGetter(sfe, "getReturnCode")).intValue();
1127                     // If we got an SMTPSendFailedException, use its RetCode to determine default permanent/temporary failure
1128                     deleteMessage = (returnCode >= 500 && returnCode <= 599);
1129                 } else {
1130                     // Sometimes we'll get a normal SendFailedException with nested SMTPAddressFailedException, so use the latter RetCode
1131                     MessagingException me = sfe;
1132                     Exception ne;
1133                     while ((ne = me.getNextException()) != null && ne instanceof MessagingException) {
1134                         me = (MessagingException)ne;
1135                         if (me.getClass().getName().endsWith(".SMTPAddressFailedException")) {
1136                             int returnCode = ((Integer) invokeGetter(me, "getReturnCode")).intValue();
1137                             deleteMessage = (returnCode >= 500 && returnCode <= 599);
1138                         }
1139                     }
1140                 }
1141             } catch (IllegalStateException ise) {
1142                 // unexpected exception (not a compatible javamail implementation)
1143             } catch (ClassCastException cce) {
1144                 // unexpected exception (not a compatible javamail implementation)
1145             }
1146 
1147             // log the original set of intended recipients
1148             if (isDebug) log("Recipients: " + recipients);
1149 
1150             if (sfe.getInvalidAddresses() != null) {
1151                 Address[] address = sfe.getInvalidAddresses();
1152                 if (address.length > 0) {
1153                     recipients.clear();
1154                     for (int i = 0; i < address.length; i++) {
1155                         try {
1156                             recipients.add(new MailAddress(address[i].toString()));
1157                         } catch (ParseException pe) {
1158                             // this should never happen ... we should have
1159                             // caught malformed addresses long before we
1160                             // got to this code.
1161                             log("Can't parse invalid address: " + pe.getMessage());
1162                         }
1163                     }
1164                     if (isDebug) log("Invalid recipients: " + recipients);
1165                     deleteMessage = failMessage(mail, sfe, true);
1166                 }
1167             }
1168 
1169             if (sfe.getValidUnsentAddresses() != null) {
1170                 Address[] address = sfe.getValidUnsentAddresses();
1171                 if (address.length > 0) {
1172                     recipients.clear();
1173                     for (int i = 0; i < address.length; i++) {
1174                         try {
1175                             recipients.add(new MailAddress(address[i].toString()));
1176                         } catch (ParseException pe) {
1177                             // this should never happen ... we should have
1178                             // caught malformed addresses long before we
1179                             // got to this code.
1180                             log("Can't parse unsent address: " + pe.getMessage());
1181                         }
1182                     }
1183                     if (isDebug) log("Unsent recipients: " + recipients);
1184                     if (sfe.getClass().getName().endsWith(".SMTPSendFailedException")) {
1185                         int returnCode = ((Integer) invokeGetter(sfe, "getReturnCode")).intValue();
1186                         deleteMessage = failMessage(mail, sfe, returnCode >= 500 && returnCode <= 599);
1187                     } else {
1188                         deleteMessage = failMessage(mail, sfe, false);
1189                     }
1190                 }
1191             }
1192 
1193             return deleteMessage;
1194         } catch (MessagingException ex) {
1195             // We should do a better job checking this... if the failure is a general
1196             // connect exception, this is less descriptive than more specific SMTP command
1197             // failure... have to lookup and see what are the various Exception
1198             // possibilities
1199 
1200             // Unable to deliver message after numerous tries... fail accordingly
1201 
1202             // We check whether this is a 5xx error message, which
1203             // indicates a permanent failure (like account doesn't exist
1204             // or mailbox is full or domain is setup wrong).
1205             // We fail permanently if this was a 5xx error
1206             return failMessage(mail, ex, ('5' == ex.getMessage().charAt(0)));
1207         } catch (Exception ex) {
1208             // Generic exception = permanent failure
1209             return failMessage(mail, ex, true);
1210         }
1211 
1212         /* If we get here, we've exhausted the loop of servers without
1213          * sending the message or throwing an exception.  One case
1214          * where this might happen is if we get a MessagingException on
1215          * each transport.connect(), e.g., if there is only one server
1216          * and we get a connect exception.
1217          */
1218         return failMessage(mail, new MessagingException("No mail server(s) available at this time."), false);
1219     }

1227      * Try to return a usefull logString created of the Exception which was given.
1228      * Return null if nothing usefull could be done
1229      * 
1230      * @param e The MessagingException to use
1231      * @return logString

1233     private String exceptionToLogString(Exception e) {
1234         if (e.getClass().getName().endsWith(".SMTPSendFailedException")) {
1235               return "RemoteHost said: " + e.getMessage();
1236         } else if (e instanceof SendFailedException) {
1237             SendFailedException exception  = (SendFailedException) e;
1238             
1239             // No error

1243              Exception ex;

1245              boolean smtpExFound = false;
1246              sb.append("RemoteHost said:");
1247 
1248              
1249              if (e instanceof MessagingException) while((ex = ((MessagingException) e).getNextException()) != null && ex instanceof MessagingException) {
1250                  e = ex;
1251                  if (ex.getClass().getName().endsWith(".SMTPAddressFailedException")) {
1252                      try {
1253                          InternetAddress ia = (InternetAddress) invokeGetter(ex, "getAddress");
1254                          sb.append(" ( " + ia + " - [" + ex.getMessage().replaceAll("\\n", "") + "] )");
1255                          smtpExFound = true;
1256                      } catch (IllegalStateException ise) {
1257                          // Error invoking the getAddress method
1258                      } catch (ClassCastException cce) {
1259                          // The getAddress method returned something different than InternetAddress
1260                      }
1261                  } 
1262              }
1263              if (!smtpExFound) {
1264                 boolean invalidAddr = false;
1265                 sb.append(" ( ");
1266             
1267                 if (exception.getInvalidAddresses().length > 0) {
1268                     sb.append(exception.getInvalidAddresses());
1269                     invalidAddr = true;
1270                 }
1271                 if (exception.getValidUnsentAddresses().length > 0) {
1272                     if (invalidAddr == true) sb.append(" " );
1273                     sb.append(exception.getValidUnsentAddresses());
1274                 }
1275                 sb.append(" - [");
1276                 sb.append(exception.getMessage().replaceAll("\\n", ""));
1277                 sb.append("] )");
1278              }
1279              return sb.toString();
1280         }
1281         return null;
1282     }

1284     /**
1285      * Utility method used to invoke getters for javamail implementation specific classes.
1286      * 
1287      * @param target the object whom method will be invoked
1288      * @param getter the no argument method name
1289      * @return the result object
1290      * @throws IllegalStateException on invocation error
1291      */
1292     private Object invokeGetter(Object target, String getter) {
1293         try {
1294             Method getAddress = target.getClass().getMethod(getter, null);
1295             return getAddress.invoke(target, null);
1296         } catch (NoSuchMethodException nsme) {
1297             // An SMTPAddressFailedException with no getAddress method.
1298         } catch (IllegalAccessException iae) {
1299         } catch (IllegalArgumentException iae) {
1300         } catch (InvocationTargetException ite) {
1301             // Other issues with getAddress invokation.
1302         }
1303         return new IllegalStateException("Exception invoking "+getter+" on a "+target.getClass()+" object");
1304     }

1306     /*
1307      * private method to log the extended SendFailedException introduced in JavaMail 1.3.2.
1308      */
1309     private void logSendFailedException(SendFailedException sfe) {
1310         if (isDebug) {
1311             MessagingException me = sfe;
1312             if (me.getClass().getName().endsWith(".SMTPSendFailedException")) {
1313                 try {
