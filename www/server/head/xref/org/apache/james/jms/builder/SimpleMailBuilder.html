43  /**
44   * <p>Basic implementation that builds {@link Mail} 
45   * from basic non-mime <code>ASCII</code> mail.
46   * Naively uses the <code>to</code> header
47   * to determine the recipent and <code>from</code>
48   * to determine the sender.
49   * </p><p>
50   * <strong>Note</strong> this implementation is too
51   * simple to cope well with wild emails. It may be useful
52   * when dealing with generated ASCII emails.
53   * </p>
54   */
55  public class SimpleMailBuilder implements MailBuilder {
56      
57      private static final String US_ASCII = "US-ASCII";
58  
59      private final IdGenerator generator;
60      
61      public SimpleMailBuilder() {
62          this(new IdGenerator() {
63              private final Random random = new Random();
64              public String getId() {
65                  return "SimpleMailBuilder#" + random.nextLong();
66              }
67          });
68      }
69      
70      public SimpleMailBuilder(final IdGenerator generator) {
71          this.generator = generator;
72      }
73      
74      public Mail build(String text) throws MessagingException {
75          final Collection recipients = new ArrayList();
76          MailAddress sender = null;
77          final int length = text.length();
78          int position = 0;
79          while (position < length) {
80              position = text.indexOf('\n', position);
81              if (position == -1) {
82                  position = length;
83              }
84              final char nextChar = text.charAt(++position);
85              
86              switch (nextChar) {
87                  case '\r':
88                      // end of headers
89                      position = length;
90                      break;
91                  case 'F':
92                  case 'f':
93                      // from header?
94                      sender = parseFrom(text, sender, length, position);
95                      break; 
96                  case 'T':
97                  case 't':
98                      parseTo(text, recipients, length, position);
99                      break; 
100               default:
101                   break;
102           }
103       }
104       final String key = generator.getId();
105       final InputStream input = toInputStream(text);
106       final Mail result = new MailImpl(key, sender, recipients, input);
107       return result;
108   }

110   private MailAddress parseFrom(final String text, final MailAddress defaultAddress, final int length, int position) throws ParseException {
111       MailAddress result = defaultAddress;
112       int eol = text.indexOf('\r', position);
113       if (eol == -1) {
114           eol = length;
115       }
116       final String line = text.substring(position, eol);
117       final Field parsedField = Field.parse(line);
118       if (parsedField.isFrom()) {
119           if (parsedField instanceof AddressListField) {
120               final AddressListField field = (AddressListField) parsedField; 
121               final MailboxList mailboxes = field.getAddressList().flatten();
122               result = firstMailAddress(mailboxes);
123           } else if (parsedField instanceof MailboxListField) {
124               final MailboxListField field = (MailboxListField) parsedField;
125               final MailboxList mailboxes = field.getMailboxList();
126               result = firstMailAddress(mailboxes);
127           }
128       }
129       return result;
130   }
131   
132   private void parseTo(final String text, Collection addresses, final int length, int position) throws ParseException {
133       int eol = text.indexOf('\r', position);
134       if (eol == -1) {
135           eol = length;
136       }
137       final String line = text.substring(position, eol);
138       final Field parsedField = Field.parse(line);
139       if (parsedField.isTo()) {
140           if (parsedField instanceof AddressListField) {
141               final AddressListField field = (AddressListField) parsedField; 
142               final MailboxList mailboxes = field.getAddressList().flatten();
143               addMailAddresses(addresses, mailboxes);
144           } else if (parsedField instanceof MailboxListField) {
145               final MailboxListField field = (MailboxListField) parsedField;
146               final MailboxList mailboxes = field.getMailboxList();
147               addMailAddresses(addresses, mailboxes);
148           }
149       }
150   }
151 
152   private void addMailAddresses(Collection addresses, final MailboxList mailboxes) throws ParseException {
153       int size = mailboxes.size();
154       for (int i=0;i< size;i++) {
155           final MailAddress address = toMailAddress(mailboxes.get(i));
156           addresses.add(address);
157       }
158   }
159 
160   private MailAddress firstMailAddress(final MailboxList mailboxes) throws ParseException {
161       MailAddress result = null;
162       if (mailboxes.size() > 0) {
163           final Mailbox first = mailboxes.get(0);
164           result = toMailAddress(first);
165       }
166       return result;
167   }
168 
169   private MailAddress toMailAddress(final Mailbox mailbox) throws ParseException {
170       MailAddress result;
171       final String domain = mailbox.getDomain();
172       final String local = mailbox.getLocalPart();
173       result = new MailAddress(local, domain);
174       return result;
175   }
176 
177   private InputStream toInputStream(String text) throws MessagingException {
178       try {
179           final InputStream result = IOUtils.toInputStream(text, US_ASCII);
180           return result;
181       } catch (IOException e) {
182           throw new MessagingException("Conversion to ASCII stream failed.", e);
183       }
184   }
