22  package org.apache.james.smtpserver.core;
23  
24  import java.util.ArrayList;
25  import java.util.Collection;
26  
27  import org.apache.avalon.framework.logger.AbstractLogEnabled;
28  import org.apache.james.dsn.DSNStatus;
29  import org.apache.james.smtpserver.CommandHandler;
30  import org.apache.james.smtpserver.SMTPSession;
31  import org.apache.mailet.MailAddress;
32  
33  /**
34    * Handles MAIL command
35    */
36  public class MailCmdHandler
37      extends AbstractLogEnabled
38      implements CommandHandler {
39  
40      
41      /**
42       * handles MAIL command
43       *
44       * @see org.apache.james.smtpserver.CommandHandler#onCommand(SMTPSession)
45       */
46      public void onCommand(SMTPSession session) {
47          doMAIL(session, session.getCommandArgument());
48      }
49  
50  
51      /**
52       * Handler method called upon receipt of a MAIL command.
53       * Sets up handler to deliver mail as the stated sender.
54       *
55       * @param session SMTP session object
56       * @param argument the argument passed in with the command by the SMTP client
57       */
58      private void doMAIL(SMTPSession session, String argument) {
59        
60          StringBuffer responseBuffer = session.getResponseBuffer();
61          String responseString = null;
62  
63          MailAddress sender = (MailAddress) session.getState().get(
64                  SMTPSession.SENDER);
65          responseBuffer.append(
66                  "250 "
67                          + DSNStatus.getStatus(DSNStatus.SUCCESS,
68                                  DSNStatus.ADDRESS_OTHER) + " Sender <").append(
69                  sender).append("> OK");
70          responseString = session.clearResponseBuffer();
71          session.writeResponse(responseString);
72          
73      }
74      
75      /**
76       * @see org.apache.james.smtpserver.CommandHandler#getImplCommands()
77       */
78      public Collection getImplCommands() {
79          Collection implCommands = new ArrayList();
80          implCommands.add("MAIL");
81          
82          return implCommands;
83      }
84  }
