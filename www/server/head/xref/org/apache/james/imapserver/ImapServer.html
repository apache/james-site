127     public void post(String url, MimeMessage mail)throws MessagingException {
128         final int endOfScheme = url.indexOf(':');
129         if (endOfScheme < 0) {
130             throw new MessagingException("Malformed URI");
131         } else {
132             final String scheme = url.substring(0, endOfScheme);
133             if ("mailbox".equals(scheme)) {
134                 final int startOfUser = endOfScheme + 3;
135                 final int endOfUser = url.indexOf('@', startOfUser);
136                 if (endOfUser < 0) {
137                     // TODO: when user missing, append to a default location
138                     throw new MessagingException("Shared mailbox is not supported");
139                 } else {
140                     final String user = url.substring(startOfUser, endOfUser);
141                     final int startOfHost = endOfUser + 1;
142                     final int endOfHost  = url.indexOf('/', startOfHost);
143                     final String host = url.substring(startOfHost, endOfHost);
144                     if (!"localhost".equals(host)) {
145                         //TODO: possible support for clustering?
146                         throw new MessagingException("Only local mailboxes are supported");
147                     } else {
148                         final String urlPath;
149                         final int length = url.length();
150                         if (endOfHost + 1 == length) {
151                             urlPath = "INBOX";
152                         } else {
153                             urlPath = url.substring(endOfHost, length);
154                         }
155                         // This allows Sieve scripts to use a standard delimiter regardless of mailbox implementation
156                         final String mailbox = urlPath.replace('/', MailboxManager.HIERARCHY_DELIMITER);
157                         postToMailbox(user, mail, mailbox);
158                     }
159                 }
160             } else {
161                 // TODO: add support for more protocols
162                 // TODO: for example mailto: for forwarding over SMTP
163                 // TODO: for example xmpp: for forwarding over Jabber
164                 throw new MessagingException("Unsupported protocol");
165             }
166         }
167     }

169     public void postToMailbox(String username, MimeMessage mail, String destination) throws MessagingException {
170         final MailboxManagerProvider mailboxManagerProvider = factory.getMailbox();
171         if (destination == null || "".equals(destination)) {
172             destination = "INBOX";
173         }
174         final String name = mailboxManagerProvider.getMailboxManager().resolve(username, destination);
175         final MailboxManager mailboxManager = mailboxManagerProvider.getMailboxManager();
176         final MailboxSession session = mailboxManager.createSession();
177         try
178         {
179             final Mailbox mailbox = mailboxManager.getMailbox(name, true);
180             
181             if (mailbox == null) {
182                 final String error = "Mailbox for user " + username
183                         + " was not found on this server.";
184                 throw new MessagingException(error);
185             }
186             mailbox.appendMessage(mail, new Date(), null, session);
187         }
188         finally 
189         {
190             session.close();   
191         }
192     }
