212     /**
213      * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
214      */
215     public void configure(Configuration conf) throws ConfigurationException {
216         enabled = conf.getAttributeAsBoolean("enabled", true);
217         final Logger logger = getLogger();
218         if (!enabled) {
219           logger.info(getServiceType() + " disabled by configuration");
220           return;
221         }
222 
223         Configuration handlerConfiguration = conf.getChild("handler");
224 
225         // Send the handler subconfiguration to the super class.  This 
226         // ensures that the handler config is passed to the handlers.
227         //
228         // TODO: This should be rationalized.  The handler element of the
229         //       server configuration doesn't really make a whole lot of 
230         //       sense.  We should modify the config to get rid of it.
231         //       Keeping it for now to maintain backwards compatibility.
232         super.configure(handlerConfiguration);
233         
234         
235         boolean streamdump=handlerConfiguration.getChild("streamdump").getAttributeAsBoolean("enabled", false);
236         String streamdumpDir=streamdump ? handlerConfiguration.getChild("streamdump").getAttribute("directory", null) : null;
237         setStreamDumpDir(streamdumpDir);
238 
239 
240         port = conf.getChild("port").getValueAsInteger(getDefaultPort());
241 
242         Configuration serverSocketTypeConf = conf.getChild("serverSocketType", false);
243         String confSocketType = null;
244         if (serverSocketTypeConf != null ) {
245             confSocketType = serverSocketTypeConf.getValue();
246         }
247 
248         if (confSocketType == null) {
249             // Only load the useTLS parameter if a specific socket type has not
250             // been specified.  This maintains backwards compatibility while
251             // allowing us to have more complex (i.e. multiple SSL configuration)
252             // deployments
253             final boolean useTLS = conf.getChild("useTLS").getValueAsBoolean(isDefaultTLSEnabled());
254             if (useTLS) {
255                 serverSocketType = "ssl";
256                 loadJCEProviders(conf, logger);
257             }
258         } else {
259             serverSocketType = confSocketType;
260         }
261 
262         StringBuffer infoBuffer;
263         threadGroup = conf.getChild("threadGroup").getValue(null);
264         if (threadGroup != null) {
265             infoBuffer =
266                 new StringBuffer(64)
267                         .append(getServiceType())
268                         .append(" uses thread group: ")
269                         .append(threadGroup);
270             logger.info(infoBuffer.toString());
271         }
272         else {
273             logger.info(getServiceType() + " uses default thread group.");
274         }
275 
276         try {
277             final String bindAddress = conf.getChild("bind").getValue(null);
278             if( null != bindAddress ) {
279                 bindTo = InetAddress.getByName(bindAddress);
280                 infoBuffer =
281                     new StringBuffer(64)
282                             .append(getServiceType())
283                             .append(" bound to: ")
284                             .append(bindTo);
285                 logger.info(infoBuffer.toString());
286             }
287         }
288         catch( final UnknownHostException unhe ) {
289             throw new ConfigurationException( "Malformed bind parameter in configuration of service " + getServiceType(), unhe );
290         }
291 
292         configureHelloName(handlerConfiguration);
293 
294         timeout = handlerConfiguration.getChild(TIMEOUT_NAME).getValueAsInteger(DEFAULT_TIMEOUT);
295 
296         infoBuffer =
297             new StringBuffer(64)
298                     .append(getServiceType())
299                     .append(" handler connection timeout is: ")
300                     .append(timeout);
301         logger.info(infoBuffer.toString());
302 
303         backlog = conf.getChild(BACKLOG_NAME).getValueAsInteger(DEFAULT_BACKLOG);
304 
305         infoBuffer =
306                     new StringBuffer(64)
307                     .append(getServiceType())
308                     .append(" connection backlog is: ")
309                     .append(backlog);
310         logger.info(infoBuffer.toString());
311 
312         String connectionLimitString = conf.getChild("connectionLimit").getValue(null);
313         if (connectionLimitString != null) {
314             try {
315                 connectionLimit = new Integer(connectionLimitString);
316             } catch (NumberFormatException nfe) {
317                 logger.error("Connection limit value is not properly formatted.", nfe);
318             }
319             if (connectionLimit.intValue() < 0) {
320                 logger.error("Connection limit value cannot be less than zero.");
321                 throw new ConfigurationException("Connection limit value cannot be less than zero.");
322             }
323         } else {
324             connectionLimit = new Integer(connectionManager.getMaximumNumberOfOpenConnections());
325         }
326         infoBuffer = new StringBuffer(128)
327             .append(getServiceType())
328             .append(" will allow a maximum of ")
329             .append(connectionLimit.intValue())
330             .append(" connections.");
331         logger.info(infoBuffer.toString());
332         
333         String connectionLimitPerIP = conf.getChild("connectionLimitPerIP").getValue(null);
334         if (connectionLimitPerIP != null) {
335             try {
336             connPerIP = new Integer(connectionLimitPerIP).intValue();
337             connPerIPConfigured = true;
338             } catch (NumberFormatException nfe) {
339                 logger.error("Connection limit per IP value is not properly formatted.", nfe);
340             }
341             if (connPerIP < 0) {
342                 logger.error("Connection limit per IP value cannot be less than zero.");
343                 throw new ConfigurationException("Connection limit value cannot be less than zero.");
344             }
345         } else {
346             connPerIP = connectionManager.getMaximumNumberOfOpenConnectionsPerIP();
347         }
348         infoBuffer = new StringBuffer(128)
349             .append(getServiceType())
350             .append(" will allow a maximum of ")
351             .append(connPerIP)
352             .append(" per IP connections for " +getServiceType());
353         logger.info(infoBuffer.toString());
354         
355     }

357     private void loadJCEProviders(Configuration conf, final Logger logger) throws ConfigurationException {
358         final Configuration [] providerConfiguration = conf.getChildren("provider");
359         for (int i = 0; i < providerConfiguration.length; i++) {
360             final String providerName = providerConfiguration[i].getValue();
361             loadProvider(logger, providerName);
362         }
363     }
364 
365     private void loadProvider(final Logger logger, final String providerName) {
366         if (providerName == null) {
367             logger.warn("Failed to specify provider. Continuing but JCE provider will not be loaded");   
368         } else {
369             try {
370                 logger.debug("Trying to load JCE provider '" + providerName + "'");
371                 Security.addProvider((Provider) Class.forName(providerName).newInstance());
372                 logger.info("Load JCE provider '" + providerName + "'");
373             } catch (IllegalAccessException e) {
374                 logJCELoadFailure(logger, providerName, e);
375             } catch (InstantiationException e) {
376                 logJCELoadFailure(logger, providerName, e);
377             } catch (ClassNotFoundException e) {
378                 logJCELoadFailure(logger, providerName, e);
379             } catch (RuntimeException e) {
380                 logJCELoadFailure(logger, providerName, e);
381             }
382         }
383     }
384 
385     private void logJCELoadFailure(final Logger logger, final String providerName, Exception e) {
386         logger.warn("Cannot load JCE provider" + providerName);
387         logger.debug(e.getMessage(), e);
388     }

404     private void configureHelloName(Configuration handlerConfiguration) {
405         StringBuffer infoBuffer;
406         String hostName = null;
407         try {
408             hostName = dnsServer.getHostName(dnsServer.getLocalHost());
409         } catch (UnknownHostException ue) {
410             hostName = "localhost";
411         }
412 
413         infoBuffer =
414             new StringBuffer(64)
415                     .append(getServiceType())
416                     .append(" is running on: ")
417                     .append(hostName);
418         getLogger().info(infoBuffer.toString());
419 
420         Configuration helloConf = handlerConfiguration.getChild(HELLO_NAME);
421  
422         if (helloConf != null) {
423             boolean autodetect = helloConf.getAttributeAsBoolean("autodetect", true);
424             if (autodetect) {
425                 helloName = hostName;
426             } else {
427                 // Should we use the defaultdomain here ?
428                 helloName = helloConf.getValue("localhost");
429             }
430         } else {
431             helloName = null;
432         }
433         infoBuffer =
434             new StringBuffer(64)
435                     .append(getServiceType())
436                     .append(" handler hello name is: ")
437                     .append(helloName);
438         getLogger().info(infoBuffer.toString());
439     }

442      * @see org.apache.avalon.framework.activity.Initializable#initialize()
443      */
444     public void initialize() throws Exception {
445         if (!isEnabled()) {
446             getLogger().info(getServiceType() + " Disabled");
447             System.out.println(getServiceType() + " Disabled");
448             return;
449         }
450         getLogger().debug(getServiceType() + " init...");
451 
452         // keeping these looked up services locally, because they are only needed beyond initialization
453         ThreadManager threadManager = (ThreadManager) componentManager.lookup(ThreadManager.ROLE);
454         SocketManager socketManager = (SocketManager) componentManager.lookup(SocketManager.ROLE);
455         
456         initializeThreadPool(threadManager);
457 
458         initializeServerSocket(socketManager);
459 
460         getLogger().debug(getServiceType() + " ...init end");
461 
462         initializeHandlerPool();
463         
464         // do avalon specific preparations
465         ContainerUtil.enableLogging(theHandlerPool, getLogger());
466         ContainerUtil.initialize(theHandlerPool);
467 
468         theWatchdogFactory = getWatchdogFactory();
469 
470     }

472     private void initializeThreadPool(ThreadManager threadManager) {
473         if (threadGroup != null) {
474             threadPool = threadManager.getThreadPool(threadGroup);
475         } else {
476             threadPool = threadManager.getDefaultThreadPool();
477         }
478     }

480     private void initializeServerSocket(SocketManager socketManager) throws Exception {
481         try {
482             initializeServerSocketWorker(socketManager);
483         } catch (BindException e) {
484             // handle a common exception and give detailed error message
485             String errorMessage = getBindingErrorMessage(e);
486             System.out.println("------------------------------");
487             System.out.println(errorMessage);
488             System.out.println("------------------------------");
489             getLogger().fatalError(errorMessage);
490             throw e;
491         }       
492     }

494     private String getBindingErrorMessage(BindException e) {
495         // general info about binding error
496         StringBuffer errorMessage = new StringBuffer();
497         errorMessage.append("FATAL ERROR when starting service '").append(getServiceType()).append("'! ");
498         errorMessage.append("could not bind to ");
499         errorMessage.append(bindTo == null ? "0.0.0.0" : bindTo.toString());
500         errorMessage.append(":").append(port).append(". ");
501         
502         // try to deliver more specific information 
503         if (e.getMessage().indexOf("Address already in use") != -1) {
504             errorMessage.append("Port is already exclusively in use by another application.");
505         } else if (e.getMessage().indexOf("Permission denied") != -1) {
506             errorMessage.append("The user account James is running under has not enough privileges to bind to this ");
507             if (port < 1024) errorMessage.append("privileged ");
508             errorMessage.append("port.");
509         } else {
510             errorMessage.append(e.getMessage());
511         }
512         return errorMessage.toString();
513     }

515     private void initializeServerSocketWorker(SocketManager socketManager) throws Exception {
516         ServerSocketFactory factory = socketManager.getServerSocketFactory(serverSocketType);
517         ServerSocket serverSocket = factory.createServerSocket(port, backlog, bindTo);
518 
519         if (null == connectionName) {
520             final StringBuffer sb = new StringBuffer();
521             sb.append(serverSocketType);
522             sb.append(':');
523             sb.append(port);
524 
525             if (null != bindTo) {
526                 sb.append('/');
527                 sb.append(bindTo);
528             }
529             connectionName = sb.toString();
530         }
531 
532         if ((connectionLimit != null)) {
533             if (null != threadPool) {
534             if (connPerIPConfigured) {
535                     connectionManager.connect(connectionName, serverSocket, this, threadPool, connectionLimit.intValue(),connPerIP);
536             } else {
537                 connectionManager.connect(connectionName, serverSocket, this, threadPool, connectionLimit.intValue());
538             }
539             } else {
540             if (connPerIPConfigured) {
541                     connectionManager.connect(connectionName, serverSocket, this, connectionLimit.intValue(),connPerIP); // default pool
542                 } else {
543                     connectionManager.connect(connectionName, serverSocket, this, connectionLimit.intValue());
544                 }
545             }
546         } else {
547             if (null != threadPool) {
548             if (connPerIPConfigured) {
549                     connectionManager.connect(connectionName, serverSocket, this, threadPool);
550             } else {
551                 connectionManager.connect(connectionName, serverSocket, this, threadPool, 0, connPerIP);
552             }
553             } else {
554             if (connPerIPConfigured) {
555                     connectionManager.connect(connectionName, serverSocket, this); // default pool
556             } else {
557                     connectionManager.connect(connectionName, serverSocket, this, 0, connPerIP);
558             }
559             }
560         }
561     }

563     private void initializeHandlerPool() throws Exception {
564         StringBuffer logBuffer =
565                 new StringBuffer(64)
566                         .append(getServiceType())
567                         .append(" started ")
568                         .append(connectionName);
569         String logString = logBuffer.toString();
570         System.out.println(logString);
571         getLogger().info(logString);
572 
573         if (connectionLimit != null) {
574             theHandlerPool = new HardResourceLimitingPool(this, 5, connectionLimit.intValue());
575             if (getLogger().isDebugEnabled()) {
576                 getLogger().debug("Using a bounded pool for "+getServiceType()+" handlers with upper limit " + connectionLimit.intValue());
577             }
578         } else {
579             // NOTE: The maximum here is not a real maximum.  The handler pool will continue to
580             //       provide handlers beyond this value.
581             theHandlerPool = new DefaultPool(this, null, 5, 30);
582             getLogger().debug("Using an unbounded pool for "+getServiceType()+" handlers.");
583         }
584     }

586     /**
587      * @see org.apache.avalon.framework.activity.Disposable#dispose()
588      */
589     public void dispose() {
590 
591         if (!isEnabled()) {
592             return;
593         }
594 
595         if( m_disposed )
596         {
597             if( getLogger().isWarnEnabled() )
598             {
599                 getLogger().warn( "ignoring disposal request - already disposed" );
600             }
601             return;
602         }
603 
604         if( getLogger().isDebugEnabled() )
605         {
606             getLogger().debug( "disposal" );
607         }
608 
609         m_disposed = true;
610         if( getLogger().isDebugEnabled() )
611         {
612             StringBuffer infoBuffer =
613                new StringBuffer(64).append(getServiceType()).append(
614                    " dispose... ").append(connectionName);
615             getLogger().debug(infoBuffer.toString());
616         }
617 
618         try {
619             connectionManager.disconnect(connectionName, true);
620         } catch (final Exception e) {
621             StringBuffer warnBuffer =
622                 new StringBuffer(64)
623                         .append("Error disconnecting ")
624                         .append(getServiceType())
625                         .append(": ");
626             getLogger().warn(warnBuffer.toString(), e);
627         }
628 
629         componentManager = null;
630 
631         connectionManager = null;
632         threadPool = null;
633 
634         // This is needed to make sure sockets are promptly closed on Windows 2000
635         // TODO: Check this - shouldn't need to explicitly gc to force socket closure
636         System.gc();
637 
638         getLogger().debug(getServiceType() + " ...dispose end");
639     }

665      * Overide this method to create actual instance of connection handler.
666      *
667      * @return the new ConnectionHandler
668      * @exception Exception if an error occurs
669      */
670     protected ConnectionHandler newHandler()
671             throws Exception {
672         AbstractJamesHandler theHandler = (AbstractJamesHandler)theHandlerPool.get();
673         
674         if (getLogger().isDebugEnabled()) {
675             getLogger().debug("Getting Handler from pool.");
676         }
677 
678         Watchdog theWatchdog = theWatchdogFactory.getWatchdog(theHandler.getWatchdogTarget());
679 
680         theHandler.setConfigurationData(getConfigurationData());
681         theHandler.setStreamDumpDir(streamDumpDir);
682         theHandler.setWatchdog(theWatchdog);
683         return theHandler;
684     }

721     /**
722      * This method returns the type of service provided by this server.
723      * This should be invariant over the life of the class.
724      *
725      * Subclasses may override this implementation.  This implementation
726      * parses the complete class name and returns the undecorated class
727      * name.
728      *
729      * @return description of this server
730      */
731     public String getServiceType() {
732         String name = getClass().getName();
733         int p = name.lastIndexOf(".");
734         if (p > 0 && p < name.length() - 2) {
735             name = name.substring(p + 1);
736         }
737         return name;
738     }
