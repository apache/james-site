137     /**
138      * Helper method for accepting connections.
139      * This MUST be called in the specializations.
140      *
141      * @param connection The Socket which belongs to the connection 
142      * @throws IOException get thrown if an IO error is detected
143      */
144     protected void initHandler( Socket connection ) throws IOException {
145         this.socket = connection;
146         remoteIP = socket.getInetAddress().getHostAddress();
147         remoteHost = dnsServer.getHostName(socket.getInetAddress());
148 
149         try {
150             synchronized (this) {
151                 handlerThread = Thread.currentThread();
152             }
153             in = new BufferedInputStream(socket.getInputStream(), DEFAULT_INPUT_BUFFER_SIZE);
154             outs = new BufferedOutputStream(socket.getOutputStream(), DEFAULT_OUTPUT_BUFFER_SIZE);
155             // enable tcp dump for debug
156             if (tcplogprefix != null) {
157                 outs = new SplitOutputStream(outs, new FileOutputStream(tcplogprefix+"out"));
158                 in = new CopyInputStream(in, new FileOutputStream(tcplogprefix+"in"));
159             }
160             
161             // An ASCII encoding can be used because all transmissions other
162             // that those in the message body command are guaranteed
163             // to be ASCII
164             inReader = new CRLFTerminatedReader(in, "ASCII");
165             
166             out = new InternetPrintWriter(outs, true);
167         } catch (RuntimeException e) {
168             StringBuffer exceptionBuffer = 
169                 new StringBuffer(256)
170                     .append("Unexpected exception opening from ")
171                     .append(remoteHost)
172                     .append(" (")
173                     .append(remoteIP)
174                     .append("): ")
175                     .append(e.getMessage());
176             String exceptionString = exceptionBuffer.toString();
177             getLogger().error(exceptionString, e);
178             throw e;
179         } catch (IOException e) {
180             StringBuffer exceptionBuffer = 
181                 new StringBuffer(256)
182                     .append("Cannot open connection from ")
183                     .append(remoteHost)
184                     .append(" (")
185                     .append(remoteIP)
186                     .append("): ")
187                     .append(e.getMessage());
188             String exceptionString = exceptionBuffer.toString();
189             getLogger().error(exceptionString, e);
190             throw e;
191         }
192         
193         if (getLogger().isInfoEnabled()) {
194             StringBuffer infoBuffer =
195                 new StringBuffer(128)
196                         .append("Connection from ")
197                         .append(remoteHost)
198                         .append(" (")
199                         .append(remoteIP)
200                         .append(")");
201             getLogger().info(infoBuffer.toString());
202         }
203     }

205     /**
206      * The method clean up and close the allocated resources
207      */
208     private void cleanHandler() {
209         // Clear the Watchdog
210         if (theWatchdog != null) {
211             ContainerUtil.dispose(theWatchdog);
212             theWatchdog = null;
213         }
214 
215         // Clear the streams
216         try {
217             if (inReader != null) {
218                 inReader.close();
219             }
220         } catch (IOException ioe) {
221             getLogger().warn("Handler: Unexpected exception occurred while closing reader: " + ioe);
222         } finally {
223             inReader = null;
224         }
225 
226         in = null;
227 
228         if (out != null) {
229             out.close();
230             out = null;
231         }
232         outs = null;
233 
234         try {
235             if (socket != null) {
236                 socket.close();
237             }
238         } catch (IOException ioe) {
239             getLogger().warn("Handler: Unexpected exception occurred while closing socket: " + ioe);
240         } finally {
241             socket = null;
242         }
243         
244         remoteIP = null;
245         remoteHost = null;
246 
247         synchronized (this) {
248             handlerThread = null;
249         }
250     }

252     /**
253      * @see org.apache.avalon.cornerstone.services.connection.ConnectionHandler#handleConnection(java.net.Socket)
254      */
255     public void handleConnection(Socket connection) throws IOException {
256         initHandler(connection);
257 
258         final Logger logger = getLogger();
259         try {
260             
261             // Do something:
262             handleProtocol();
263             
264             logger.debug("Closing socket.");
265         } catch (SocketException se) {
266             // Indicates a problem at the underlying protocol level
267             if (logger.isWarnEnabled()) {
268                 String message =
269                     new StringBuffer(64)
270                         .append("Socket to ")
271                         .append(remoteHost)
272                         .append(" (")
273                         .append(remoteIP)
274                         .append("): ")
275                         .append(se.getMessage()).toString();
276                 logger.warn(message);
277                 logger.debug(se.getMessage(), se);
278             }
279         } catch ( InterruptedIOException iioe ) {
280             if (logger.isErrorEnabled()) {
281                 StringBuffer errorBuffer =
282                     new StringBuffer(64)
283                         .append("Socket to ")
284                         .append(remoteHost)
285                         .append(" (")
286                         .append(remoteIP)
287                         .append(") timeout.");
288                 logger.error( errorBuffer.toString(), iioe );
289             }
290         } catch ( IOException ioe ) {
291             if (logger.isWarnEnabled()) {
292                 String message =
293                     new StringBuffer(256)
294                             .append("Exception handling socket to ")
295                             .append(remoteHost)
296                             .append(" (")
297                             .append(remoteIP)
298                             .append(") : ")
299                             .append(ioe.getMessage()).toString();
300                 logger.warn(message);
301                 logger.debug( ioe.getMessage(), ioe );
302             }
303         } catch (RuntimeException e) {
304             errorHandler(e);
305         } finally {
306             //Clear all the session state variables
307             cleanHandler();
308             resetHandler();
309         }
310     }

357     /**
358      * Idle out this connection
359      */
360     void idleClose() {
361         if (getLogger() != null) {
362             getLogger().error("Service Connection has idled out.");
363         }
364         try {
365             if (socket != null) {
366                 socket.close();
367             }
368         } catch (Exception e) {
369             // ignored
370         } finally {
371             socket = null;
372         }
373 
374         synchronized (this) {
375             // Interrupt the thread to recover from internal hangs
376             if (handlerThread != null) {
377                 handlerThread.interrupt();
378                 handlerThread = null;
379             }
380         }
381 
382     }
