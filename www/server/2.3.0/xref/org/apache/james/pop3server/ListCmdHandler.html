18  package org.apache.james.pop3server;
19  
20  import org.apache.mailet.Mail;
21  
22  import javax.mail.MessagingException;
23  
24  import java.util.Iterator;
25  
26  /***
27    * Handles LIST command
28    */
29  public class ListCmdHandler implements CommandHandler {
30  
31      /***
32       * @see org.apache.james.pop3server.CommandHandler#onCommand(POP3Session)
33       */
34      public void onCommand(POP3Session session) {
35          doLIST(session,session.getCommandArgument());
36      }
37  
38      /***
39       * Handler method called upon receipt of a LIST command.
40       * Returns the number of messages in the mailbox and its
41       * aggregate size, or optionally, the number and size of
42       * a single message.
43       *
44       * @param command the command parsed by the parseCommand method
45       * @param argument the first argument parsed by the parseCommand method
46       */
47      private void doLIST(POP3Session session,String argument) {
48          String responseString = null;
49          if (session.getHandlerState() == POP3Handler.TRANSACTION) {
50              if (argument == null) {
51                  long size = 0;
52                  int count = 0;
53                  try {
54                      for (Iterator i = session.getUserMailbox().iterator(); i.hasNext(); ) {
55                          Mail mc = (Mail) i.next();
56                          if (mc != POP3Handler.DELETED) {
57                              size += mc.getMessageSize();
58                              count++;
59                          }
60                      }
61                      StringBuffer responseBuffer =
62                          new StringBuffer(32)
63                                  .append(POP3Handler.OK_RESPONSE)
64                                  .append(" ")
65                                  .append(count)
66                                  .append(" ")
67                                  .append(size);
68                      responseString = responseBuffer.toString();
69                      session.writeResponse(responseString);
70                      count = 0;
71                      for (Iterator i = session.getUserMailbox().iterator(); i.hasNext(); count++) {
72                          Mail mc = (Mail) i.next();
73  
74                          if (mc != POP3Handler.DELETED) {
75                              responseBuffer =
76                                  new StringBuffer(16)
77                                          .append(count)
78                                          .append(" ")
79                                          .append(mc.getMessageSize());
80                              session.writeResponse(responseBuffer.toString());
81                          }
82                      }
83                      session.writeResponse(".");
84                  } catch (MessagingException me) {
85                      responseString = POP3Handler.ERR_RESPONSE;
86                      session.writeResponse(responseString);
87                  }
88              } else {
89                  int num = 0;
90                  try {
91                      num = Integer.parseInt(argument);
92                      Mail mc = (Mail) session.getUserMailbox().get(num);
93                      if (mc != POP3Handler.DELETED) {
94                          StringBuffer responseBuffer =
95                              new StringBuffer(64)
96                                      .append(POP3Handler.OK_RESPONSE)
97                                      .append(" ")
98                                      .append(num)
99                                      .append(" ")
100                                     .append(mc.getMessageSize());
101                         responseString = responseBuffer.toString();
102                         session.writeResponse(responseString);
103                     } else {
104                         StringBuffer responseBuffer =
105                             new StringBuffer(64)
106                                     .append(POP3Handler.ERR_RESPONSE)
107                                     .append(" Message (")
108                                     .append(num)
109                                     .append(") already deleted.");
110                         responseString = responseBuffer.toString();
111                         session.writeResponse(responseString);
112                     }
113                 } catch (IndexOutOfBoundsException npe) {
114                     StringBuffer responseBuffer =
115                         new StringBuffer(64)
116                                 .append(POP3Handler.ERR_RESPONSE)
117                                 .append(" Message (")
118                                 .append(num)
119                                 .append(") does not exist.");
120                     responseString = responseBuffer.toString();
121                     session.writeResponse(responseString);
122                 } catch (NumberFormatException nfe) {
123                     StringBuffer responseBuffer =
124                         new StringBuffer(64)
125                                 .append(POP3Handler.ERR_RESPONSE)
126                                 .append(" ")
127                                 .append(argument)
128                                 .append(" is not a valid number");
129                     responseString = responseBuffer.toString();
130                     session.writeResponse(responseString);
131                 } catch (MessagingException me) {
132                     responseString = POP3Handler.ERR_RESPONSE;
133                     session.writeResponse(responseString);
134                }
135             }
136         } else {
137             responseString = POP3Handler.ERR_RESPONSE;
138             session.writeResponse(responseString);
139         }
140     }
