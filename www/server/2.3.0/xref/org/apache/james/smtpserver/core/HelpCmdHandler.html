24  import org.apache.james.smtpserver.SMTPSession;
25  import org.apache.james.util.mail.dsn.DSNStatus;

27  /***
28    * Handles HELP command
29    */
30  public class HelpCmdHandler implements CommandHandler {
31      /***
32       * The name of the command handled by the command handler
33       */
34      private final static String COMMAND_NAME = "HELP";
35  
36  
37      /*
38       * handles HELP command
39       *
40       * @see org.apache.james.smtpserver.CommandHandler#onCommand(SMTPSession)
41      **/
42      public void onCommand(SMTPSession session) {
43          String responseString = "502 "+DSNStatus.getStatus(DSNStatus.PERMANENT,DSNStatus.SYSTEM_NOT_CAPABLE)+" HELP is not supported";
44          session.writeResponse(responseString);
45      }
46  
47      /***
48       * @see org.apache.james.smtpserver.CommandHandler#getImplCommands()
49       */
50      public Collection getImplCommands() {
51          Collection implCommands = new ArrayList();
52          implCommands.add("HELP");
53          
54          return implCommands;
55      }
56  }
