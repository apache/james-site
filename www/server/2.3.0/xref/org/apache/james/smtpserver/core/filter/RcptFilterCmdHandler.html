49      /***
50       * @param session SMTP session object
51       * @param argument the argument passed in with the command by the SMTP client
52       */
53      private void doRCPT(SMTPSession session, String argument) {
54          String responseString = null;
55          
56          String recipient = null;
57          if ((argument != null) && (argument.indexOf(":") > 0)) {
58              int colonIndex = argument.indexOf(":");
59              recipient = argument.substring(colonIndex + 1);
60              argument = argument.substring(0, colonIndex);
61          }
62          if (!session.getState().containsKey(SMTPSession.SENDER)) {
63              responseString = "503 "+DSNStatus.getStatus(DSNStatus.PERMANENT,DSNStatus.DELIVERY_OTHER)+" Need MAIL before RCPT";
64              session.writeResponse(responseString);
65              
66              // After this filter match we should not call any other handler!
67              session.setStopHandlerProcessing(true);
68              
69          } else if (argument == null || !argument.toUpperCase(Locale.US).equals("TO")
70                     || recipient == null) {
71              responseString = "501 "+DSNStatus.getStatus(DSNStatus.PERMANENT,DSNStatus.DELIVERY_SYNTAX)+" Usage: RCPT TO:<recipient>";
72              session.writeResponse(responseString);
73              
74              // After this filter match we should not call any other handler!
75              session.setStopHandlerProcessing(true);
76              
77          } else {
78              Collection rcptColl = (Collection) session.getState().get(SMTPSession.RCPT_LIST);
79              if (rcptColl == null) {
80                  rcptColl = new ArrayList();
81              }
82              recipient = recipient.trim();
83              int lastChar = recipient.lastIndexOf('>');
84              // Check to see if any options are present and, if so, whether they are correctly formatted
85              // (separated from the closing angle bracket by a ' ').
86              String rcptOptionString = null;
87              if ((lastChar > 0) && (recipient.length() > lastChar + 2) && (recipient.charAt(lastChar + 1) == ' ')) {
88                  rcptOptionString = recipient.substring(lastChar + 2);
89  
90                  // Remove the options from the recipient
91                  recipient = recipient.substring(0, lastChar + 1);
92              }
93              if (!recipient.startsWith("<") || !recipient.endsWith(">")) {
94                  responseString = "501 "+DSNStatus.getStatus(DSNStatus.PERMANENT,DSNStatus.DELIVERY_SYNTAX)+" Syntax error in parameters or arguments";
95                  session.writeResponse(responseString);
96                  if (getLogger().isErrorEnabled()) {
97                      StringBuffer errorBuffer =
98                          new StringBuffer(192)
99                                  .append("Error parsing recipient address: ")
100                                 .append("Address did not start and end with < >")
101                                 .append(getContext(session,null,recipient));
102                     getLogger().error(errorBuffer.toString());
103                 }
104                 
105                 // After this filter match we should not call any other handler!
106                 session.setStopHandlerProcessing(true);
107                 
108                 return;
109             }
110             MailAddress recipientAddress = null;
111             //Remove < and >
112             recipient = recipient.substring(1, recipient.length() - 1);
113             if (recipient.indexOf("@") < 0) {
114                 recipient = recipient + "@localhost";
115             }
116             
117             try {
118                 recipientAddress = new MailAddress(recipient);
119             } catch (Exception pe) {
120                 /*
121                  * from RFC2822;
122                  * 553 Requested action not taken: mailbox name not allowed
123                  *     (e.g., mailbox syntax incorrect)
124                  */
125                 responseString = "553 "+DSNStatus.getStatus(DSNStatus.PERMANENT,DSNStatus.ADDRESS_SYNTAX)+" Syntax error in recipient address";
126                 session.writeResponse(responseString);
127 
128                 if (getLogger().isErrorEnabled()) {
129                     StringBuffer errorBuffer =
130                         new StringBuffer(192)
131                                 .append("Error parsing recipient address: ")
132                                 .append(getContext(session,recipientAddress,recipient))
133                                 .append(pe.getMessage());
134                     getLogger().error(errorBuffer.toString());
135                 }
136                 
137                 // After this filter match we should not call any other handler!
138                 session.setStopHandlerProcessing(true);
139                 
140                 return;
141             }
142 
143             
144 
145             if (session.isAuthRequired() && !session.isRelayingAllowed()) {
146                 // Make sure the mail is being sent locally if not
147                 // authenticated else reject.
148                 if (session.getUser() == null) {
149                     String toDomain = recipientAddress.getHost();
150                     if (!session.getConfigurationData().getMailServer().isLocalServer(toDomain)) {
151                         responseString = "530 "+DSNStatus.getStatus(DSNStatus.PERMANENT,DSNStatus.SECURITY_AUTH)+" Authentication Required";
152                         session.writeResponse(responseString);
153                         StringBuffer sb = new StringBuffer(128);
154                         sb.append("Rejected message - authentication is required for mail request");
155                         sb.append(getContext(session,recipientAddress,recipient));
156                         getLogger().error(sb.toString());
157                         
158                         // After this filter match we should not call any other handler!
159                         session.setStopHandlerProcessing(true);
160                         
161                         return;
162                     }
163                 } else {
164                     // Identity verification checking
165                     if (session.getConfigurationData().isVerifyIdentity()) {
166                         String authUser = (session.getUser()).toLowerCase(Locale.US);
167                         MailAddress senderAddress = (MailAddress) session.getState().get(SMTPSession.SENDER);
168 
169                         if ((senderAddress == null) || (!authUser.equals(senderAddress.getUser())) ||
170                             (!session.getConfigurationData().getMailServer().isLocalServer(senderAddress.getHost()))) {
171                             responseString = "503 "+DSNStatus.getStatus(DSNStatus.PERMANENT,DSNStatus.SECURITY_AUTH)+" Incorrect Authentication for Specified Email Address";
172                             session.writeResponse(responseString);
173                             if (getLogger().isErrorEnabled()) {
174                                 StringBuffer errorBuffer =
175                                     new StringBuffer(128)
176                                         .append("User ")
177                                         .append(authUser)
178                                         .append(" authenticated, however tried sending email as ")
179                                         .append(senderAddress)
180                                         .append(getContext(session,recipientAddress,recipient));
181                                 getLogger().error(errorBuffer.toString());
182                             }
183                             
184                             // After this filter match we should not call any other handler!
185                             session.setStopHandlerProcessing(true);
186                             
187                             return;
188                         }
189                     }
190                 }
191             } else if (!session.isRelayingAllowed()) {
192                 String toDomain = recipientAddress.getHost();
193                 if (!session.getConfigurationData().getMailServer().isLocalServer(toDomain)) {
194                     responseString = "550 "+DSNStatus.getStatus(DSNStatus.PERMANENT,DSNStatus.SECURITY_AUTH)+" Requested action not taken: relaying denied";
195                     session.writeResponse(responseString);
196                     StringBuffer errorBuffer = new StringBuffer(128)
197                         .append("Rejected message - ")
198                         .append(session.getRemoteIPAddress())
199                         .append(" not authorized to relay to ")
200                         .append(toDomain)
201                         .append(getContext(session,recipientAddress,recipient));
202                     getLogger().error(errorBuffer.toString());
203                     
204                     // After this filter match we should not call any other handler!
205                     session.setStopHandlerProcessing(true);
206                     
207                     return;
208                 }
209             }
210             if (rcptOptionString != null) {
211 
212               StringTokenizer optionTokenizer = new StringTokenizer(rcptOptionString, " ");
213               while (optionTokenizer.hasMoreElements()) {
214                   String rcptOption = optionTokenizer.nextToken();
215                   int equalIndex = rcptOption.indexOf('=');
216                   String rcptOptionName = rcptOption;
217                   String rcptOptionValue = "";
218                   if (equalIndex > 0) {
219                       rcptOptionName = rcptOption.substring(0, equalIndex).toUpperCase(Locale.US);
220                       rcptOptionValue = rcptOption.substring(equalIndex + 1);
221                   }
222                   // Unexpected option attached to the RCPT command
223                   if (getLogger().isDebugEnabled()) {
224                       StringBuffer debugBuffer =
225                           new StringBuffer(128)
226                               .append("RCPT command had unrecognized/unexpected option ")
227                               .append(rcptOptionName)
228                               .append(" with value ")
229                               .append(rcptOptionValue)
230                               .append(getContext(session,recipientAddress,recipient));
231                       getLogger().debug(debugBuffer.toString());
232                   }
233                   
234                   // After this filter match we should not call any other handler!
235                   session.setStopHandlerProcessing(true);
236                   
237               }
238               optionTokenizer = null;
239             }
240     
241             session.getState().put(SMTPSession.CURRENT_RECIPIENT,recipientAddress);
242         }
243     }

246     private String getContext(SMTPSession session, MailAddress recipientAddress, String recipient){
247         StringBuffer sb = new StringBuffer(128);
248         if(null!=recipientAddress) {
249             sb.append(" [to:" + (recipientAddress).toInternetAddress().getAddress() + "]");
250         } else if(null!=recipient) {
251             sb.append(" [to:" + recipient + "]");
252         }
253         if (null!=session.getState().get(SMTPSession.SENDER)) {
254             sb.append(" [from:" + ((MailAddress)session.getState().get(SMTPSession.SENDER)).toInternetAddress().getAddress() + "]");
255         }
256         return sb.toString();
257     }
