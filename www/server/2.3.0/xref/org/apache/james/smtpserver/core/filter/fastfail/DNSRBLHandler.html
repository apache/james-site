60      /***
61       * @see org.apache.avalon.framework.configuration.Configurable#configure(Configuration)
62       */
63      public void configure(Configuration handlerConfiguration) throws ConfigurationException {
64          boolean validConfig = false;
65  
66          Configuration rblserverConfiguration = handlerConfiguration.getChild("rblservers", false);
67          if ( rblserverConfiguration != null ) {
68              ArrayList rblserverCollection = new ArrayList();
69              Configuration[] children = rblserverConfiguration.getChildren("whitelist");
70              if ( children != null ) {
71                  for ( int i = 0 ; i < children.length ; i++ ) {
72                      String rblServerName = children[i].getValue();
73                      rblserverCollection.add(rblServerName);
74                      if (getLogger().isInfoEnabled()) {
75                          getLogger().info("Adding RBL server to whitelist: " + rblServerName);
76                      }
77                  }
78                  if (rblserverCollection != null && rblserverCollection.size() > 0) {
79                      setWhitelist((String[]) rblserverCollection.toArray(new String[rblserverCollection.size()]));
80                      rblserverCollection.clear();
81                      validConfig = true;
82                  }
83              }
84              children = rblserverConfiguration.getChildren("blacklist");
85              if ( children != null ) {
86                  for ( int i = 0 ; i < children.length ; i++ ) {
87                      String rblServerName = children[i].getValue();
88                      rblserverCollection.add(rblServerName);
89                      if (getLogger().isInfoEnabled()) {
90                          getLogger().info("Adding RBL server to blacklist: " + rblServerName);
91                      }
92                  }
93                  if (rblserverCollection != null && rblserverCollection.size() > 0) {
94                      setBlacklist((String[]) rblserverCollection.toArray(new String[rblserverCollection.size()]));
95                      rblserverCollection.clear();
96                      validConfig = true;
97                  }
98              }
99          }
100         
101         // Throw an ConfiigurationException on invalid config
102         if (validConfig == false){
103             throw new ConfigurationException("Please configure whitelist or blacklist");
104         }
105         
106         Configuration configuration = handlerConfiguration.getChild("getDetail",false);
107         if(configuration != null) {
108            getDetail = configuration.getValueAsBoolean();
109         }
110 
111     }

165     /***
166      *
167      * This checks DNSRBL whitelists and blacklists.  If the remote IP is whitelisted
168      * it will be permitted to send e-mail, otherwise if the remote IP is blacklisted,
169      * the sender will only be permitted to send e-mail to postmaster (RFC 2821) or
170      * abuse (RFC 2142), unless authenticated.
171      * 
172      * @see org.apache.james.smtpserver.SMTPHandlerConfigurationData#checkDNSRBL(Socket)
173      */
174 
175     public void checkDNSRBL(SMTPSession session, String ipAddress) {
176         
177         /*
178          * don't check against rbllists if the client is allowed to relay..
179          * This whould make no sense.
180          */
181         if (session.isRelayingAllowed()) {
182             getLogger().info("Ipaddress " + session.getRemoteIPAddress() + " is allowed to relay. Don't check it");
183             session.getConnectionState().put(RBL_BLOCKLISTED_MAIL_ATTRIBUTE_NAME, "false");
184             return;
185         }
186         
187         if (whitelist != null || blacklist != null) {
188             StringBuffer sb = new StringBuffer();
189             StringTokenizer st = new StringTokenizer(ipAddress, " .", false);
190             while (st.hasMoreTokens()) {
191                 sb.insert(0, st.nextToken() + ".");
192             }
193             String reversedOctets = sb.toString();
194 
195             if (whitelist != null) {
196                 String[] rblList = whitelist;
197                 for (int i = 0 ; i < rblList.length ; i++) try {
198                     dnsServer.getByName(reversedOctets + rblList[i]);
199                     if (getLogger().isInfoEnabled()) {
200                         getLogger().info("Connection from " + ipAddress + " whitelisted by " + rblList[i]);
201                     }
202                     
203                     session.getConnectionState().put(RBL_BLOCKLISTED_MAIL_ATTRIBUTE_NAME, "false");
204                     return;
205                 } catch (java.net.UnknownHostException uhe) {
206                     if (getLogger().isDebugEnabled()) {
207                         getLogger().debug("IpAddress " + session.getRemoteIPAddress() + " not listed on " + rblList[i]);
208                     }
209                 }
210             }
211 
212             if (blacklist != null) {
213                 String[] rblList = blacklist;
214                 for (int i = 0 ; i < rblList.length ; i++) try {
215                     dnsServer.getByName(reversedOctets + rblList[i]);
216                     if (getLogger().isInfoEnabled()) {
217                         getLogger().info("Connection from " + ipAddress + " restricted by " + rblList[i] + " to SMTP AUTH/postmaster/abuse.");
218                     }
219                     
220                     // we should try to retrieve details
221                     if (getDetail) {
222                         Collection txt = dnsServer.findTXTRecords(reversedOctets + rblList[i]);
223                         
224                         // Check if we found a txt record
225                         if (!txt.isEmpty()) {
226                             // Set the detail
227                             String blocklistedDetail = txt.iterator().next().toString();
228                             
229                             session.getConnectionState().put(RBL_DETAIL_MAIL_ATTRIBUTE_NAME, blocklistedDetail);
230                         }
231                     }
232                     
233                     session.getConnectionState().put(RBL_BLOCKLISTED_MAIL_ATTRIBUTE_NAME, "true");
234                     return;
235                 } catch (java.net.UnknownHostException uhe) {
236                     // if it is unknown, it isn't blocked
237                     if (getLogger().isDebugEnabled()) {
238                         getLogger().debug("unknown host exception thrown:" + rblList[i]);
239                     }
240                 }
241             }
242         }
243         // default state
244         session.getConnectionState().put(RBL_BLOCKLISTED_MAIL_ATTRIBUTE_NAME, "false");
245     }

256     /***
257      * @see org.apache.james.smtpserver.CommandHandler#onCommand(SMTPSession)
258      */
259     public void onCommand(SMTPSession session) {
260         String responseString = null;
261         String blocklisted = (String) session.getConnectionState().get(RBL_BLOCKLISTED_MAIL_ATTRIBUTE_NAME);
262         MailAddress recipientAddress = (MailAddress) session.getState().get(
263                 SMTPSession.CURRENT_RECIPIENT);
264 
265         if (blocklisted.equals("true") && // was found in the RBL
266                 !(session.isAuthRequired() && session
267                         .getUser() != null) && // Not (SMTP AUTH is enabled and
268                                                 // not authenticated)
269                 !(recipientAddress.getUser().equalsIgnoreCase("postmaster") || recipientAddress
270                         .getUser().equalsIgnoreCase("abuse"))) {
271 
272             // trying to send e-mail to other than postmaster or abuse
273             if (blocklistedDetail != null) {
274                 responseString = "530 "
275                         + DSNStatus.getStatus(DSNStatus.PERMANENT,
276                                 DSNStatus.SECURITY_AUTH) + " "
277                         + blocklistedDetail;
278             } else {
279                 responseString = "530 "
280                         + DSNStatus.getStatus(DSNStatus.PERMANENT,
281                                 DSNStatus.SECURITY_AUTH)
282                         + " Rejected: unauthenticated e-mail from "
283                         + session.getRemoteIPAddress()
284                         + " is restricted.  Contact the postmaster for details.";
285             }
286             session.writeResponse(responseString);
287 
288             // After this filter match we should not call any other handler!
289             session.setStopHandlerProcessing(true);
290         }
