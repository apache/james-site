18  package org.apache.james.smtpserver.core;
19  
20  import java.util.ArrayList;
21  import java.util.Collection;
22  
23  import org.apache.avalon.framework.logger.AbstractLogEnabled;
24  import org.apache.james.smtpserver.CommandHandler;
25  import org.apache.james.smtpserver.SMTPSession;
26  import org.apache.james.util.mail.dsn.DSNStatus;
27  import org.apache.mailet.MailAddress;
28  
29  /***
30    * Handles MAIL command
31    */
32  public class MailCmdHandler
33      extends AbstractLogEnabled
34      implements CommandHandler {
35  
36      
37      /***
38       * handles MAIL command
39       *
40       * @see org.apache.james.smtpserver.CommandHandler#onCommand(SMTPSession)
41       */
42      public void onCommand(SMTPSession session) {
43          doMAIL(session, session.getCommandArgument());
44      }
45  
46  
47      /***
48       * Handler method called upon receipt of a MAIL command.
49       * Sets up handler to deliver mail as the stated sender.
50       *
51       * @param session SMTP session object
52       * @param argument the argument passed in with the command by the SMTP client
53       */
54      private void doMAIL(SMTPSession session, String argument) {
55        
56          StringBuffer responseBuffer = session.getResponseBuffer();
57          String responseString = null;
58  
59          MailAddress sender = (MailAddress) session.getState().get(
60                  SMTPSession.SENDER);
61          responseBuffer.append(
62                  "250 "
63                          + DSNStatus.getStatus(DSNStatus.SUCCESS,
64                                  DSNStatus.ADDRESS_OTHER) + " Sender <").append(
65                  sender).append("> OK");
66          responseString = session.clearResponseBuffer();
67          session.writeResponse(responseString);
68          
69      }
70      
71      /***
72       * @see org.apache.james.smtpserver.CommandHandler#getImplCommands()
73       */
74      public Collection getImplCommands() {
75          Collection implCommands = new ArrayList();
76          implCommands.add("MAIL");
77          
78          return implCommands;
79      }
80  }
