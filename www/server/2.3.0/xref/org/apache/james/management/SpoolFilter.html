19  package org.apache.james.management;
20  
21  import org.apache.oro.text.regex.MalformedPatternException;
22  import org.apache.oro.text.regex.Pattern;
23  import org.apache.oro.text.regex.Perl5Compiler;
24  
25  import java.util.HashMap;
26  import java.util.Iterator;
27  import java.util.Map;
28  
29  /***
30   * immutual collection of filters used to specify which mail should be processed by SpoolManagement
31   * criterias are:
32   * exact state match
33   * headerValue regular expressions match all related headers 
34   */
35  public class SpoolFilter {
36  
37      public static final SpoolFilter ERRORMAIL_FILTER = new SpoolFilter("error");
38  
39      private String state = null;
40  
41      /***
42       * Map<String headerName, String headerValueRegex>
43       */
44      private final Map headerFilters = new HashMap();
45  
46      /***
47       * Map<String headerName, Pattern>
48       */
49      private final Map headerFiltersCompiled = new HashMap();
50  
51      public SpoolFilter(String state, String header, String headerValueRegex) {
52          this.state = state;
53          if (header != null) headerFilters.put(header, headerValueRegex);
54      }
55  
56      public SpoolFilter(String state) {
57          this.state = state;
58      }
59  
60      public SpoolFilter(String header, String headerValueRegex) {
61          this.state = null;
62          if (header != null) headerFilters.put(header, headerValueRegex);
63      }
64  
65      /***
66       * @param state
67       * @param headerFilters Map<String headerName, String headerValueRegex>
68       */
69      public SpoolFilter(String state, Map headerFilters) {
70          this.state = state;
71          this.headerFilters.putAll(headerFilters);
72          this.headerFilters.remove(null); // NULL is not acceptable here
73      }
74  
75      public boolean doFilter() {
76          return doFilterHeader() || doFilterState();
77      }
78  
79      public boolean doFilterState() {
80          return state != null;
81      }
82  
83      public boolean doFilterHeader() {
84          return headerFilters.size() > 0;
85      }
86  
87      public String getState() {
88          return state;
89      }
90  
91      public Iterator getHeaders() {
92          return headerFilters.keySet().iterator();
93      }
94  
95      public String getHeaderValueRegex(String header) {
96          return (String) headerFilters.get(header);
97      }
98  
99      public Pattern getHeaderValueRegexCompiled(String header) throws SpoolManagementException {
100         Perl5Compiler compiler = new Perl5Compiler();
101 
102         // try to reuse cached pattern
103         if (headerFiltersCompiled.get(header) != null) return (Pattern)headerFiltersCompiled.get(header); 
104 
105         String headerValueRegex = getHeaderValueRegex(header);
106         if (headerValueRegex == null) return null;
107         try {
108             Pattern pattern = compiler.compile(headerValueRegex, Perl5Compiler.READ_ONLY_MASK);
109             headerFiltersCompiled.put(header, pattern); // cache
110             return pattern;
111         } catch (MalformedPatternException e) {
112             throw new SpoolManagementException("failed to compile regular expression", e);
113         }
114     }
115     
116 }
