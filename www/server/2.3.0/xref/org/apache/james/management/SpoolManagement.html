89      /***
90       * @param mail
91       * @param filter
92       * @return TRUE, if given mail matches all given filter criteria
93       * @throws SpoolManagementException
94       */
95      protected boolean filterMatches(Mail mail, SpoolFilter filter) throws SpoolManagementException {
96          if (filter == null || !filter.doFilter()) return true;
97  
98          if (filter.doFilterState() && !mail.getState().equalsIgnoreCase(filter.getState())) return false;
99          
100         if (filter.doFilterHeader()) {
101 
102             Perl5Matcher matcher = new Perl5Matcher();
103             
104             // check, if there is a match for every header/regex pair
105             Iterator headers = filter.getHeaders();
106             while (headers.hasNext()) {
107                 String header = (String) headers.next();
108                 
109                 String[] headerValues;
110                 try {
111                     headerValues = mail.getMessage().getHeader(header);
112                     if (headerValues == null) {
113                         // some headers need special retrieval
114                         if (header.equalsIgnoreCase("to")) {
115                             headerValues = addressesToStrings(mail.getMessage().getRecipients(MimeMessage.RecipientType.TO));
116                         }
117                         else if (header.equalsIgnoreCase("cc")) { 
118                             headerValues = addressesToStrings(mail.getMessage().getRecipients(MimeMessage.RecipientType.CC));
119                         }
120                         else if (header.equalsIgnoreCase("bcc")) { 
121                             headerValues = addressesToStrings(mail.getMessage().getRecipients(MimeMessage.RecipientType.BCC));
122                         }
123                         else if (header.equalsIgnoreCase("from")) { 
124                             headerValues = new String[]{mail.getMessage().getSender().toString()};
125                         }
126                     }
127                 } catch (MessagingException e) {
128                     throw new SpoolManagementException("could not filter mail by headers", e);
129                 }
130                 if (headerValues == null) return false; // no header for this criteria
131 
132                 Pattern pattern = filter.getHeaderValueRegexCompiled(header);
133 
134                 // the regex must match at least one entry for the header
135                 boolean matched = false;
136                 for (int i = 0; i < headerValues.length; i++) {
137                     String headerValue = headerValues[i];
138                     if (matcher.matches(headerValue, pattern)) {
139                         matched = true;
140                         break;
141                     }
142                 }
143                 if (!matched) return false;
144             }
145         }
146             
147         return true;
148     }

150     private String[] addressesToStrings(Address[] addresses) {
151         if (addresses == null) return null;
152         if (addresses.length == 0) return new String[]{};
153         String[] addressStrings = new String[addresses.length];
154         for (int i = 0; i < addresses.length; i++) {
155             addressStrings[i] = addresses[i].toString();
156         }
157         return addressStrings;
158     }

160     /***
161      * @param spoolRepositoryURL
162      * @param filter
163      * @return List<Mail> all matching mails from the given spool
164      * @throws SpoolManagementException
165      */
166     public List getSpoolItems(String spoolRepositoryURL, SpoolFilter filter)
167             throws ServiceException, MessagingException, SpoolManagementException {
168         SpoolRepository spoolRepository = getSpoolRepository(spoolRepositoryURL);
169 
170         List items = new ArrayList();
171 
172         // get an iterator of all keys
173         Iterator spoolR = spoolRepository.list();
174         while (spoolR.hasNext()) {
175             String key = spoolR.next().toString();
176             Mail m = spoolRepository.retrieve(key);
177 
178             if (filterMatches(m, filter)) {
179                 StringBuffer itemInfo = new StringBuffer();
180                 itemInfo.append("key: ").append(key).append(" sender: ").append(m.getSender()).append(" recipient:");
181                 Collection recipients = m.getRecipients();
182                 for (Iterator iterator = recipients.iterator(); iterator.hasNext();) {
183                     MailAddress mailAddress = (MailAddress) iterator.next();
184                     itemInfo.append(" ").append(mailAddress);
185                 }
186                 items.add(itemInfo.toString());
187             }
188         }
189 
190         return items;
191     }

215      * Removes all mails from the given repository matching the filter 
216      * @param spoolRepositoryURL the spool whose item are listed
217      * @param key ID of the mail to be removed. if not NULL, all other filters are ignored
218      * @param lockingFailures is populated with a list of mails which could not be processed because
219      * a lock could not be obtained
220      * @param filter the criteria against which all mails are matched. only applied if key is NULL.
221      * @return number of removed mails
222      */
223     public int removeSpoolItems(String spoolRepositoryURL, String key, List lockingFailures, SpoolFilter filter) throws ServiceException, MessagingException {
224         int count = 0;
225         SpoolRepository spoolRepository = getSpoolRepository(spoolRepositoryURL);
226 
227         if (key != null) {
228             count = removeMail(spoolRepository, key, count, lockingFailures, null);
229         } else {
230             Iterator spoolR = spoolRepository.list();
231 
232             while (spoolR.hasNext()) {
233                 key = (String)spoolR.next();
234                 count = removeMail(spoolRepository, key, count, lockingFailures, filter);
235             }
236         }
237         return count;
238     }

240     private int removeMail(SpoolRepository spoolRepository, String key, int count, List lockingFailures, SpoolFilter filter) throws MessagingException {
241         try {
242             if (removeMail(spoolRepository, key, filter)) count++;
243         } catch (IllegalStateException e) {
244             lockingFailures.add(key);
245         } catch (SpoolManagementException e) {
246             return count;
247         }
248         return count;
249     }

273     /***
274      * Tries to resend all mails from the given repository matching the given filter criteria 
275      * @param spoolRepositoryURL the spool whose item are about to be resend
276      * @param key ID of the mail to be resend. if not NULL, all other filters are ignored
277      * @param lockingFailures is populated with a list of mails which could not be processed because
278      * a lock could not be obtained
279      * @param filter the criteria against which all mails are matched. only applied if key is NULL.
280      * @return int number of resent mails 
281      * @throws SpoolManagementException
282      */
283     public int resendSpoolItems(String spoolRepositoryURL, String key, List lockingFailures, SpoolFilter filter)
284             throws ServiceException, MessagingException, SpoolManagementException {
285         int count = 0;
286         SpoolRepository spoolRepository = getSpoolRepository(spoolRepositoryURL);
287 
288         // check if an key was given as argument
289         if (key != null) {
290             try {
291                 if (resendMail(spoolRepository, key, filter)) count++;
292             } catch (IllegalStateException e) {
293                 if (lockingFailures != null) lockingFailures.add(key);
294             }
295         } else {
296             // get an iterator of all keys
297             Iterator spoolR = spoolRepository.list();
298 
299             while (spoolR.hasNext()) {
300                 key = spoolR.next().toString();
301                 try {
302                     if (resendMail(spoolRepository, key, filter)) count++;
303                 } catch (IllegalStateException e) {
304                     if (lockingFailures != null) lockingFailures.add(key);
305                 }
306             }
307         }
308         return count;
309     }

311     /***
312      * Resent the mail that belongs to the given key and spoolRepository 
313      * 
314      * @param spoolRepository The spoolRepository
315      * @param key The message key
316      * @param filter
317      * @return true or false
318      * @throws MessagingException Get thrown if there happen an error on modify the mail
319      */
320     private boolean resendMail(SpoolRepository spoolRepository, String key, SpoolFilter filter)
321             throws MessagingException, IllegalStateException, SpoolManagementException {
322         if (!spoolRepository.lock(key)) throw new IllegalStateException("locking failure");
323 
324         // get the mail and set the error_message to "0" that will force the spoolmanager to try to deliver it now!
325         Mail m = spoolRepository.retrieve(key);
326 
327         if (filterMatches(m, filter)) {
328 
329             // this will force Remotedelivery to try deliver the mail now!
330             m.setLastUpdated(new Date(0));
331 
332             // store changes
333             spoolRepository.store(m);
334             spoolRepository.unlock(key);
335 
336             synchronized (spoolRepository) {
337                 spoolRepository.notify();
338             }
339             return true;
340         } else {
341             spoolRepository.unlock(key);
342             return false;
343         }
344     }

346     /***
347      * Remove the mail that belongs to the given key and spoolRepository 
348      * @param spoolRepository The spoolRepository
349      * @param key The message key
350      * @param filter
351      * @return true or false
352      * @throws MessagingException Get thrown if there happen an error on modify the mail
353      */
354     private boolean removeMail(SpoolRepository spoolRepository, String key, SpoolFilter filter) 
355             throws MessagingException, SpoolManagementException {
356         if (!spoolRepository.lock(key)) throw new IllegalStateException("locking failure");
357 
358         Mail m = spoolRepository.retrieve(key);
359         if (m == null) throw new SpoolManagementException("mail not available having key " + key);
360         if (!filterMatches(m, filter)) return false;
361 
362         spoolRepository.remove(key);
363         return true;
364     }
