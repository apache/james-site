18  package org.apache.james.transport.mailets;
19  
20  import java.util.HashMap;
21  import java.util.Iterator;
22  import java.util.StringTokenizer;
23  
24  import javax.mail.MessagingException;
25  import javax.mail.internet.MimeMessage;
26  
27  import org.apache.mailet.GenericMailet;
28  import org.apache.mailet.Mail;
29  
30  /***
31   * Convert attributes to headers
32   * 
33   * Sample configuration:
34   * 
35   * &lt;mailet match="All" class="MailAttributesToMimeHeaders"&gt;
36   * &lt;simplemapping&gt;org.apache.james.attribute1;
37   * headerName1&lt;/simplemapping&gt;
38   * &lt;simplemapping&gt;org.apache.james.attribute2;
39   * headerName2&lt;/simplemapping&gt; &lt;/mailet&gt;
40   * 
41   */
42  public class MailAttributesToMimeHeaders extends GenericMailet {
43  
44      /***
45       * HashMap which holds the attributeName and headerName
46       */
47      private HashMap map = new HashMap();
48  
49      /***
50       * @see org.apache.mailet.GenericMailet#init()
51       */
52      public void init() throws MessagingException {
53          String simplemappings = getInitParameter("simplemapping");
54          if (simplemappings != null) {
55  
56              StringTokenizer st = new StringTokenizer(simplemappings, ",");
57              while (st.hasMoreTokens()) {
58  
59                  String parameters[] = st.nextToken().split(";");
60  
61                  // Check if we have a valid config
62                  if (parameters.length > 2 || parameters.length < 2) {
63                      throw new MessagingException(
64                              "Invalid config. Please use \"attributeName; headerName\"");
65                  } else {
66                      // Add it to the map
67                      map.put(parameters[0].trim(), parameters[1].trim());
68                  }
69              }
70          } else {
71              throw new MessagingException(
72                      "Invalid config. Please use \"attributeName; headerName\"");
73          }
74      }
75  
76      /***
77       * @see org.apache.mailet.GenericMailet#service(Mail)
78       */
79      public void service(Mail mail) {
80          MimeMessage message;
81          try {
82              message = mail.getMessage();
83  
84              Iterator keys = map.keySet().iterator();
85  
86              while (keys.hasNext()) {
87                  String key = keys.next().toString();
88                  String value = (String) mail.getAttribute(key);
89                  String headerName = map.get(key).toString();
90  
91                  // Check if we have all needed values
92                  if (headerName != null && value != null) {
93                      // Add the header
94                      message.setHeader(headerName, value);
95                  }
96              }
97              message.saveChanges();
98          } catch (MessagingException e) {
99              log(e.getMessage());
100         }
101     }
