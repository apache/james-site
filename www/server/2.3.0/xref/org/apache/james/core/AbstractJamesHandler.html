105     /***
106      * Helper method for accepting connections.
107      * This MUST be called in the specializations.
108      */
109     protected void initHandler( Socket connection ) throws IOException {
110         this.socket = connection;
111         remoteIP = socket.getInetAddress().getHostAddress();
112         remoteHost = socket.getInetAddress().getHostName();
113         try {
114             synchronized (this) {
115                 handlerThread = Thread.currentThread();
116             }
117             in = new BufferedInputStream(socket.getInputStream(), 1024);
118             // An ASCII encoding can be used because all transmissions other
119             // that those in the message body command are guaranteed
120             // to be ASCII
121             inReader = new CRLFTerminatedReader(in, "ASCII");
122             outs = new BufferedOutputStream(socket.getOutputStream(), 1024);
123             out = new InternetPrintWriter(outs, true);
124         } catch (RuntimeException e) {
125             StringBuffer exceptionBuffer = 
126                 new StringBuffer(256)
127                     .append("Unexpected exception opening from ")
128                     .append(remoteHost)
129                     .append(" (")
130                     .append(remoteIP)
131                     .append("): ")
132                     .append(e.getMessage());
133             String exceptionString = exceptionBuffer.toString();
134             getLogger().error(exceptionString, e);
135             throw e;
136         } catch (IOException e) {
137             StringBuffer exceptionBuffer = 
138                 new StringBuffer(256)
139                     .append("Cannot open connection from ")
140                     .append(remoteHost)
141                     .append(" (")
142                     .append(remoteIP)
143                     .append("): ")
144                     .append(e.getMessage());
145             String exceptionString = exceptionBuffer.toString();
146             getLogger().error(exceptionString, e);
147             throw e;
148         }
149         
150         if (getLogger().isInfoEnabled()) {
151             StringBuffer infoBuffer =
152                 new StringBuffer(128)
153                         .append("Connection from ")
154                         .append(remoteHost)
155                         .append(" (")
156                         .append(remoteIP)
157                         .append(")");
158             getLogger().info(infoBuffer.toString());
159         }
160     }

162     /***
163      * The method clean up and close the allocated resources
164      */
165     private void cleanHandler() {
166         // Clear the Watchdog
167         if (theWatchdog != null) {
168             ContainerUtil.dispose(theWatchdog);
169             theWatchdog = null;
170         }
171 
172         // Clear the streams
173         try {
174             if (inReader != null) {
175                 inReader.close();
176             }
177         } catch (IOException ioe) {
178             getLogger().warn("Handler: Unexpected exception occurred while closing reader: " + ioe);
179         } finally {
180             inReader = null;
181         }
182 
183         in = null;
184 
185         if (out != null) {
186             out.close();
187             out = null;
188         }
189         outs = null;
190 
191         try {
192             if (socket != null) {
193                 socket.close();
194             }
195         } catch (IOException ioe) {
196             getLogger().warn("Handler: Unexpected exception occurred while closing socket: " + ioe);
197         } finally {
198             socket = null;
199         }
200         
201         remoteIP = null;
202         remoteHost = null;
203 
204         synchronized (this) {
205             handlerThread = null;
206         }
207     }

209     /***
210      * @see org.apache.avalon.cornerstone.services.connection.ConnectionHandler#handleConnection(java.net.Socket)
211      */
212     public void handleConnection(Socket connection) throws IOException {
213         initHandler(connection);
214 
215         try {
216             
217             // Do something:
218             handleProtocol();
219             
220             getLogger().debug("Closing socket.");
221         } catch (SocketException se) {
222             if (getLogger().isErrorEnabled()) {
223                 StringBuffer errorBuffer =
224                     new StringBuffer(64)
225                         .append("Socket to ")
226                         .append(remoteHost)
227                         .append(" (")
228                         .append(remoteIP)
229                         .append(") closed remotely.");
230                 getLogger().error(errorBuffer.toString(), se );
231             }
232         } catch ( InterruptedIOException iioe ) {
233             if (getLogger().isErrorEnabled()) {
234                 StringBuffer errorBuffer =
235                     new StringBuffer(64)
236                         .append("Socket to ")
237                         .append(remoteHost)
238                         .append(" (")
239                         .append(remoteIP)
240                         .append(") timeout.");
241                 getLogger().error( errorBuffer.toString(), iioe );
242             }
243         } catch ( IOException ioe ) {
244             if (getLogger().isErrorEnabled()) {
245                 StringBuffer errorBuffer =
246                     new StringBuffer(256)
247                             .append("Exception handling socket to ")
248                             .append(remoteHost)
249                             .append(" (")
250                             .append(remoteIP)
251                             .append(") : ")
252                             .append(ioe.getMessage());
253                 getLogger().error( errorBuffer.toString(), ioe );
254             }
255         } catch (RuntimeException e) {
256             errorHandler(e);
257         } finally {
258             //Clear all the session state variables
259             cleanHandler();
260             resetHandler();
261         }
262     }

300     /***
301      * Idle out this connection
302      */
303     void idleClose() {
304         if (getLogger() != null) {
305             getLogger().error("Service Connection has idled out.");
306         }
307         try {
308             if (socket != null) {
309                 socket.close();
310             }
311         } catch (Exception e) {
312             // ignored
313         } finally {
314             socket = null;
315         }
316 
317         synchronized (this) {
318             // Interrupt the thread to recover from internal hangs
319             if (handlerThread != null) {
320                 handlerThread.interrupt();
321                 handlerThread = null;
322             }
323         }
324

327     /***
328      * This method logs at a "DEBUG" level the response string that
329      * was sent to the service client.  The method is provided largely

333      * @param responseString the response string sent to the client

336         if (getLogger().isDebugEnabled()) {

341     /***
