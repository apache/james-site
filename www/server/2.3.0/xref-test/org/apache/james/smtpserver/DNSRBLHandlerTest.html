276     // ip is blacklisted and has txt details
277     public void testBlackListedTextPresent() {
278         DNSRBLHandler rbl = new DNSRBLHandler();
279 
280         ContainerUtil.enableLogging(rbl, new MockLogger());
281 
282         setupMockedSMTPSession();
283         rbl.setDNSServer(mockedDnsServer);
284 
285         rbl.setBlacklist(new String[] { "bl.spamcop.net" });
286         rbl.setGetDetail(true);
287         rbl.onConnect(mockedSMTPSession);
288         assertEquals("Blocked - see http://www.spamcop.net/bl.shtml?127.0.0.2",
289                mockedSMTPSession.getConnectionState().get(RBL_DETAIL_MAIL_ATTRIBUTE_NAME));
290         assertEquals("true",mockedSMTPSession.getConnectionState().get(RBL_BLOCKLISTED_MAIL_ATTRIBUTE_NAME));
291     }
