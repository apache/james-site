67      private final static char[] hexchars = { '0', '1', '2', '3', '4', '5', '6',
68              '7', '8', '9', 'A', 'B', 'C', 'D', 'E', 'F' };
69  
70      public String debugString(String a, String charset)
71              throws UnsupportedEncodingException {
72          byte[] bytes = a.getBytes(charset);
73          StringBuffer res = new StringBuffer();
74          for (int i = 0; i < bytes.length; i++) {
75              if (i > 0)
76                  res.append("-");
77              res.append(hexchars[((bytes[i] + 256) % 256) / 16]);
78              res.append(hexchars[((bytes[i] + 256) % 256) % 16]);
79          }
80          res.append(" (");
81          res.append(MimeUtility.mimeCharset(charset));
82          res.append(" / ");
83          res.append(MimeUtility.javaCharset(charset));
84          res.append(")");
85          return res.toString();
86      }

284     private String processAddFooter(String asciisource, String footer)
285             throws MessagingException, IOException {
286         Mailet mailet = new AddFooter();
287 
288         MockMailetConfig mci = new MockMailetConfig("Test",new MockMailContext());
289         mci.setProperty("text",footer);
290 
291         mailet.init(mci);
292 
293         Mail mail = new MailImpl(new MimeMessage(Session
294                 .getDefaultInstance(new Properties()),
295                 new ByteArrayInputStream(asciisource.getBytes())));
296 
297         mailet.service(mail);
298 
299         ByteArrayOutputStream rawMessage = new ByteArrayOutputStream();
300         mail.getMessage().writeTo(
301                 rawMessage,
302                 new String[] { "Bcc", "Content-Length", "Message-ID" });
303         String res = rawMessage.toString();
304         return res;
305     }
