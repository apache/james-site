131     public void testCopyOnWrite() throws IOException, MessagingException {
132         t.setSpool(new MockSpoolRepository());
133         Matcher recipientIs = new RecipientIs();
134         recipientIs.init(new DummyMatcherConfig("rec1@domain.com"));
135         
136         Matcher all = new All();
137         all.init(new DummyMatcherConfig(""));
138         
139         Mailet changeBody = new MyMailet();
140         Mailet changeBody2 = new MyMailet();
141 
142         changeBody.init(mockMailetConfig);
143         changeBody2.init(mockMailetConfig);
144         
145         Mailet dumpSystemErr = new DumpSystemErr();
146         changeBody.init(mockMailetConfig);
147         
148         checkerMailet = new CheckerMailet();
149         t.add(recipientIs,changeBody);
150         t.add(all,changeBody);
151         t.add(all,dumpSystemErr);
152         t.add(all,checkerMailet);
153         t.closeProcessorLists();
154 
155         Collection recipients = new ArrayList();
156         recipients.add(new MailAddress("rec1","domain.com"));
157         recipients.add(new MailAddress("rec2","domain.com"));
158         try {
159             MailImpl m = new MailImpl("mail1",new MailAddress("sender","domain.com"),recipients,mw);
160             t.service(m);
161             ArrayList a = checkerMailet.receivedMails;
162             assertEquals(2,a.size());
163             MimeMessage m1 = ((Mail) a.get(0)).getMessage();
164             MimeMessage m2 = ((Mail) a.get(1)).getMessage();
165             assertNotSame(m1,m2);
166             assertEquals(m1.getSubject(),"new text 1");
167             assertEquals(m2.getSubject(),"new text 2");
168             m.dispose();
169         } catch (MessagingException e) {
170             // TODO Auto-generated catch block

172         }

176         t.setSpool(new MockSpoolRepository() {

184         recipientIs.init(new DummyMatcherConfig("rec1@domain.com"));

187         all.init(new DummyMatcherConfig(""));
188         
189         Mailet dumpSystemErr = new DumpSystemErr();
190         
191         checkerMailet = new CheckerMailet();
192         t.add(recipientIs,dumpSystemErr);
193         t.add(all,dumpSystemErr);
194         t.add(all,checkerMailet);
195         t.closeProcessorLists();
196 
197         Collection recipients = new ArrayList();
198         recipients.add(new MailAddress("rec1","domain.com"));
199         recipients.add(new MailAddress("rec2","domain.com"));
200         try {
201             MailImpl m = new MailImpl("mail1",new MailAddress("sender","domain.com"),recipients,mw);
202             m.setState("MYSTATE");
203             t.service(m);
204             ArrayList a = checkerMailet.receivedMails;
205             assertEquals(2,a.size());
206             MimeMessage m1 = ((Mail) a.get(0)).getMessage();
207             MimeMessage m2 = ((Mail) a.get(1)).getMessage();
208             assertNotSame(m1,m2);
209             assertEquals("MYSTATE",((Mail) a.get(0)).getState());
210             assertEquals("MYSTATE",((Mail) a.get(1)).getState());
211             m.dispose();
