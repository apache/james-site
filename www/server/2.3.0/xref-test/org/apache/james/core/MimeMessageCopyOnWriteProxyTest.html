238     private static String getReferences(MimeMessage m) {
239         StringBuffer ref = new StringBuffer("/");
240         while (m instanceof MimeMessageCopyOnWriteProxy) {
241             ref.append(((MimeMessageCopyOnWriteProxy) m).refCount.getReferenceCount()+"/");
242             m = ((MimeMessageCopyOnWriteProxy) m).getWrappedMessage();
243         }
244         if (m instanceof MimeMessageWrapper) {
245             ref.append("W");
246         } else if (m instanceof MimeMessage) {
247             ref.append("M");
248         } else {
249             ref.append(m.getClass());
250         }
251         return ref.toString();
252     }
253     
254     private static MimeMessage getWrappedMessage(MimeMessage m) {
255         while (m instanceof MimeMessageCopyOnWriteProxy) {
256           m = ((MimeMessageCopyOnWriteProxy) m).getWrappedMessage();
257         }
258         return m;
259     }
