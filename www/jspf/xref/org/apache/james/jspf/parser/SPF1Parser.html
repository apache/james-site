46  /***
47   * This class is used to parse SPF1-Records from their textual form to an
48   * SPF1Record object that is composed by 2 collections: directives and
49   * modifiers.
50   * 
51   * The parsing is modular and get informations from Mechanism and Modifiers
52   * classes declared in the org/apache/james/jspf/parser/jspf.default.terms file.
53   * 
54   * Each term implementation provide its own REGEX in the REGEX static public
55   * field. This parser simply join all the regexp in a single "alternative"
56   * pattern and count the number of catch groups (brackets) assigned to each
57   * regex fragment.
58   * 
59   * SO it creates a big regex and an array where it store what term is associated
60   * to each catch group of the big regex.
61   * 
62   * If the regex matches the input vspf1 record then it start looking for the
63   * matched group (not null) and lookup the term that created that part of the
64   * regex.
65   * 
66   * With this informations it creates a new instance of the term and, if the term
67   * is Configurable it calls the config() method passing to it only the specific
68   * subset of the MatchResult (using the MatchResultSubset).
69   * 
70   * TODO doubts about the specification - redirect or exp with no domain-spec are
71   * evaluated as an unknown-modifiers according to the current spec (it does not
72   * make too much sense) - top-label is defined differently in various specs.
73   * We'll have to review the code. -
74   * http://data.iana.org/TLD/tlds-alpha-by-domain.txt (we should probably beeter
75   * use and alpha sequence being at least 2 chars - Somewhere is defined as "."
76   * TLD [ "." ] - Otherwise defined as ( *alphanum ALPHA *alphanum ) / (
77   * 1*alphanum "-" *( * alphanum / "-" ) alphanum )
78   * 
79   * @see org.apache.james.jspf.core.SPF1Record
80   * 
81   */
82  public class SPF1Parser {

106      * ABNF: macro-string = *( macro-expand / macro-literal )

114     private static final String QUALIFIER_PATTERN = "[" + "//"
115             + SPF1Constants.PASS + "//" + SPF1Constants.FAIL + "//"
116             + SPF1Constants.NEUTRAL + "//" + SPF1Constants.SOFTFAIL + "]";

118     /***
119      * ABNF: toplabel = ( *alphanum ALPHA *alphanum ) / ( 1*alphanum "-" *(
120      * alphanum / "-" ) alphanum ) ; LDH rule plus additional TLD restrictions ;
121      * (see [RFC3696], Section 2)
122      */
123     private static final String TOP_LABEL_REGEX = "(?:"
124             + SPF1Parser.ALPHA_DIGIT_PATTERN + "*" + SPF1Parser.ALPHA_PATTERN
125             + "{1}" + SPF1Parser.ALPHA_DIGIT_PATTERN + "*|(?:"
126             + SPF1Parser.ALPHA_DIGIT_PATTERN + "+" + "//-" + "(?:"
127             + SPF1Parser.ALPHA_DIGIT_PATTERN + "|//-)*"
128             + SPF1Parser.ALPHA_DIGIT_PATTERN + "))";

131      * ABNF: domain-end = ( "." toplabel [ "." ] ) / macro-expand

162     private class TermDef {
163         private Pattern pattern;
164 
165         private Class termDef;
166 
167         private int matchSize = 0;
168 
169         public TermDef(Class tClass) throws IllegalArgumentException,
170                 SecurityException, IllegalAccessException, NoSuchFieldException {
171             String pString = (String) tClass.getField("REGEX").get(null);
172             pattern = Pattern.compile(pString);
173             termDef = tClass;
174             calcGroups(pString);
175         }
176 
177         /***
178          * This method should be done differently. We currently don't hanlde the
179          * escaping at all.
180          * 
181          * @param pString
182          */
183         private void calcGroups(String pString) {
184             int i = 0;
185             int c = 0;
186             while (true) {
187                 int p1 = pString.indexOf("(", i);
188                 int p2 = pString.indexOf("(?:", i);
189                 if (p1 < 0)
190                     break;
191                 if (p1 != p2)
192                     c++;
193                 i = p1 + 1;
194             }
195             matchSize = c;
196         }
197 
198         public Pattern getPattern() {
199             return pattern;
200         }
201 
202         public Class getTermDef() {
203             return termDef;
204         }
205 
206         public int getMatchSize() {
207             return matchSize;
208         }

211     /***
212      * Constructor. Creates all the values needed to run the parsing
213      */
214     public SPF1Parser(Logger logger) {
215         log = logger;
216         try {
217             InputStream is = Thread.currentThread().getContextClassLoader()
218                     .getResourceAsStream(termFile);
219             Properties p = new Properties();
220             p.load(is);
221             String mechs = p.getProperty("mechanisms");
222             String mods = p.getProperty("modifiers");
223             String[] classes;
224             classes = mechs.split(",");
225             Class[] knownMechanisms = new Class[classes.length];
226             for (int i = 0; i < classes.length; i++) {
227                 log.debug("Add following class as known mechanismn: "
228                         + classes[i]);
229                 knownMechanisms[i] = Thread.currentThread()
230                         .getContextClassLoader().loadClass(classes[i]);
231             }
232             mechanismsCollection = createTermCollection(knownMechanisms);
233             classes = mods.split(",");
234             Class[] knownModifiers = new Class[classes.length];
235             for (int i = 0; i < classes.length; i++) {
236                 log.debug("Add following class as known modifier: "
237                         + classes[i]);
238                 knownModifiers[i] = Thread.currentThread()
239                         .getContextClassLoader().loadClass(classes[i]);
240             }
241             modifiersCollection = createTermCollection(knownModifiers);
242 
243         } catch (IOException e) {
244             throw new IllegalStateException(
245                     "Term configuration cannot be found");
246         } catch (ClassNotFoundException e) {
247             throw new IllegalStateException(
248                     "One configured class cannot be found");
249         }
250 
251         /***
252          * ABNF: mechanism = ( all / include / A / MX / PTR / IP4 / IP6 / exists )
253          */
254         String MECHANISM_REGEX = createRegex(mechanismsCollection);
255 
256         /***
257          * ABNF: modifier = redirect / explanation / unknown-modifier
258          */
259         String MODIFIER_REGEX = "(" + createRegex(modifiersCollection) + ")";
260 
261         /***
262          * ABNF: directive = [ qualifier ] mechanism
263          */
264         String DIRECTIVE_REGEX = "(" + QUALIFIER_PATTERN + "?)("
265                 + MECHANISM_REGEX + ")";
266 
267         /***
268          * ABNF: ( directive / modifier )
269          */
270         String TERM_REGEX = "(?:" + MODIFIER_REGEX + "|" + DIRECTIVE_REGEX
271                 + ")";
272 
273         /***
274          * ABNF: 1*SP
275          */
276         String TERMS_SEPARATOR_REGEX = "[ ]+";
277 
278         termsSeparatorPattern = Pattern.compile(TERMS_SEPARATOR_REGEX);
279         termPattern = Pattern.compile(TERM_REGEX);
280 
281         initializePositions();
282     }

284     /***
285      * Fill in the matchResultPositions ArrayList. This array simply map each
286      * regex matchgroup to the Term class that originated that part of the
287      * regex.
288      */
289     private void initializePositions() {
290         ArrayList matchResultPositions = new ArrayList();
291 
292         // FULL MATCH
293         int posIndex = 0;
294         matchResultPositions.ensureCapacity(posIndex + 1);
295         matchResultPositions.add(posIndex, null);
296 
297         Iterator i;
298 
299         TERM_STEP_REGEX_MODIFIER_POS = ++posIndex;
300         matchResultPositions.ensureCapacity(posIndex + 1);
301         matchResultPositions.add(TERM_STEP_REGEX_MODIFIER_POS, null);
302         i = modifiersCollection.iterator();
303         while (i.hasNext()) {
304             TermDef td = (TermDef) i.next();
305             int size = td.getMatchSize() + 1;
306             for (int k = 0; k < size; k++) {
307                 posIndex++;
308                 matchResultPositions.ensureCapacity(posIndex + 1);
309                 matchResultPositions.add(posIndex, td);
310             }
311         }
312 
313         TERM_STEP_REGEX_QUALIFIER_POS = ++posIndex;
314         matchResultPositions.ensureCapacity(posIndex + 1);
315         matchResultPositions.add(posIndex, null);
316 
317         TERM_STEP_REGEX_MECHANISM_POS = ++posIndex;
318         matchResultPositions.ensureCapacity(posIndex + 1);
319         matchResultPositions.add(TERM_STEP_REGEX_MECHANISM_POS, null);
320         i = mechanismsCollection.iterator();
321         while (i.hasNext()) {
322             TermDef td = (TermDef) i.next();
323             int size = td.getMatchSize() + 1;
324             for (int k = 0; k < size; k++) {
325                 posIndex++;
326                 matchResultPositions.ensureCapacity(posIndex + 1);
327                 matchResultPositions.add(posIndex, td);
328             }
329         }
330 
331         if (log.isDebugEnabled()) {
332             log.debug("Parsing catch group positions: Modifiers["
333                     + TERM_STEP_REGEX_MODIFIER_POS + "] Qualifier["
334                     + TERM_STEP_REGEX_QUALIFIER_POS + " Mechanism["
335                     + TERM_STEP_REGEX_MECHANISM_POS + "]");
336             for (int k = 0; k < matchResultPositions.size(); k++) {
337                 log
338                         .debug(k
339                                 + ") "
340                                 + (matchResultPositions.get(k) != null ? ((TermDef) matchResultPositions
341                                         .get(k)).getPattern()
342                                         : null));
343             }
344         }
345         
346         this.matchResultPositions = Collections.synchronizedList(matchResultPositions);
347     }

349     /***
350      * Loop the classes searching for a String static field named
351      * staticFieldName and create an OR regeex like this:
352      * (?:FIELD1|FIELD2|FIELD3)
353      * 
354      * @param classes
355      *            classes to analyze
356      * @param staticFieldName
357      *            static field to concatenate
358      * @return regex
359      */
360     private String createRegex(Collection commandMap) {
361         StringBuffer modifierRegex = new StringBuffer();
362         Iterator i = commandMap.iterator();
363         boolean first = true;
364         while (i.hasNext()) {
365             if (first) {
366                 modifierRegex.append("(?:(");
367                 first = false;
368             } else {
369                 modifierRegex.append(")|(");
370             }
371             Pattern pattern = ((TermDef) i.next()).getPattern();
372             modifierRegex.append(pattern.pattern());
373         }
374         modifierRegex.append("))");
375         return modifierRegex.toString();
376     }

378     /***
379      * @param classes
380      *            classes to analyze
381      * @param staticFieldName
382      *            static field to concatenate
383      * @return map <Class,Pattern>
384      */
385     private Collection createTermCollection(Class[] classes) {
386         Collection l = new ArrayList();
387         for (int j = 0; j < classes.length; j++) {
388             Class mechClass = classes[j];
389             try {
390                 l.add(new TermDef(mechClass));
391             } catch (Exception e) {
392                 log.debug("Unable to create the term collection", e);
393                 throw new IllegalStateException(
394                         "Unable to create the term collection");
395             }
396         }
397         return Collections.synchronizedCollection(l);
398     }

400     /***
401      * This Method parse the given spf record and checks for syntax
402      * 
403      * parse can be called by multiple concurrent threads.
404      * 
405      * @param spfRecord
406      *            The String which represent the spf record in dns
407      * @return result The SPF1Record
408      * @throws PermErrorException
409      *             Get thrown if an syntax error was detected
410      * @throws NoneException
411      *             Getthrown if no spf record could be found
412      */
413     public SPF1Record parse(String spfRecord) throws PermErrorException,
414             NoneException {
415 
416         log.debug("Start parsing SPF-Record: " + spfRecord);
417 
418         SPF1Record result = new SPF1Record();
419 
420         // check the version "header"
421         if (!spfRecord.startsWith(SPF1Constants.SPF_VERSION + " ")) {
422             throw new NoneException("No valid SPF Record: " + spfRecord);
423         }
424 
425         // extract terms
426         String[] terms = termsSeparatorPattern.split(spfRecord.replaceFirst(
427                 SPF1Constants.SPF_VERSION, ""));
428 
429         // cycle terms
430         for (int i = 0; i < terms.length; i++) {
431             if (terms[i].length() > 0) {
432                 Matcher termMatcher = termPattern.matcher(terms[i]);
433                 if (!termMatcher.matches()) {
434                     throw new PermErrorException("Term [" + terms[i]
435                             + "] is not syntactically valid: "
436                             + termPattern.pattern());
437                 }
438 
439                 // true if we matched a modifier, false if we matched a
440                 // directive
441                 String modifierString = termMatcher
442                         .group(TERM_STEP_REGEX_MODIFIER_POS);
443 
444                 if (modifierString != null) {
445                     // MODIFIER
446                     Modifier mod = (Modifier) lookupAndCreateTerm(termMatcher,
447                             TERM_STEP_REGEX_MODIFIER_POS);
448 
449                     if (mod.enforceSingleInstance()) {
450                         Iterator it = result.getModifiers().iterator();
451                         while (it.hasNext()) {
452                             if (it.next().getClass().equals(mod.getClass())) {
453                                 throw new PermErrorException("More than one "
454                                         + modifierString
455                                         + " found in SPF-Record");
456                             }
457                         }
458                     }
459 
460                     result.getModifiers().add(mod);
461 
462                 } else {
463                     // DIRECTIVE
464                     String qualifier = termMatcher
465                             .group(TERM_STEP_REGEX_QUALIFIER_POS);
466 
467                     Object mech = lookupAndCreateTerm(termMatcher,
468                             TERM_STEP_REGEX_MECHANISM_POS);
469 
470                     result.getDirectives().add(
471                             new Directive(qualifier, (Mechanism) mech));
472 
473                 }
474 
475             }
476         }
477 
478         return result;
479     }

481     /***
482      * @param res
483      *            the MatchResult
484      * @param start
485      *            the position where the terms starts
486      * @return
487      * @throws PermErrorException
488      */
489     private Object lookupAndCreateTerm(Matcher res, int start)
490             throws PermErrorException {
491         for (int k = start + 1; k < res.groupCount(); k++) {
492             if (res.group(k) != null && k != TERM_STEP_REGEX_QUALIFIER_POS) {
493                 TermDef c = (TermDef) matchResultPositions.get(k);
494                 ConfigurationMatch subres = new ConfigurationMatch(res, k, c
495                         .getMatchSize());
496                 try {
497                     Object term = c.getTermDef().newInstance();
498                     if (term instanceof LogEnabled) {
499                         ((LogEnabled) term).enableLogging(log);
500                     }
501                     if (term instanceof Configurable) {
502                         if (subres == null || subres.groupCount() == 0) {
503                             ((Configurable) term).config(null);
504                         } else {
505                             ((Configurable) term).config(subres);
506                         }
507                     }
508                     return term;
509                 } catch (IllegalAccessException e) {
510                     throw new IllegalStateException(
511                             "Unexpected error creating term: " + e.getMessage());
512                 } catch (InstantiationException e) {
513                     throw new IllegalStateException(
514                             "Unexpected error creating term: " + e.getMessage());
515                 }
516 
517             }
518         }
519         return null;
520     }
