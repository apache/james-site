34      /***
35       * Creates an byte[] based on an ipAddressString. No error handling is
36       * performed here.
37       */
38      public static byte[] createByteArrayFromIPAddressString(
39              String ipAddressString) {
40  
41          if (isValidIPV4Address(ipAddressString)) {
42              StringTokenizer tokenizer = new StringTokenizer(ipAddressString,
43                      ".");
44              String token = "";
45              int tempInt = 0;
46              byte[] byteAddress = new byte[4];
47              for (int i = 0; i < 4; i++) {
48                  token = tokenizer.nextToken();
49                  tempInt = Integer.parseInt(token);
50                  byteAddress[i] = (byte) tempInt;
51              }
52  
53              return byteAddress;
54          }
55  
56          if (ipAddressString.charAt(0) == '[') {
57              ipAddressString = ipAddressString.substring(1, ipAddressString
58                      .length() - 1);
59          }
60  
61          StringTokenizer tokenizer = new StringTokenizer(ipAddressString, ":.",
62                  true);
63          ArrayList hexStrings = new ArrayList();
64          ArrayList decStrings = new ArrayList();
65          String token = "";
66          String prevToken = "";
67          int doubleColonIndex = -1; // If a double colon exists, we need to
68          // insert 0s.
69  
70          // Go through the tokens, including the seperators ':' and '.'
71          // When we hit a : or . the previous token will be added to either
72          // the hex list or decimal list. In the case where we hit a ::
73          // we will save the index of the hexStrings so we can add zeros
74          // in to fill out the string
75          while (tokenizer.hasMoreTokens()) {
76              prevToken = token;
77              token = tokenizer.nextToken();
78  
79              if (token.equals(":")) {
80                  if (prevToken.equals(":")) {
81                      doubleColonIndex = hexStrings.size();
82                  } else if (!prevToken.equals("")) {
83                      hexStrings.add(prevToken);
84                  }
85              } else if (token.equals(".")) {
86                  decStrings.add(prevToken);
87              }
88          }
89  
90          if (prevToken.equals(":")) {
91              if (token.equals(":")) {
92                  doubleColonIndex = hexStrings.size();
93              } else {
94                  hexStrings.add(token);
95              }
96          } else if (prevToken.equals(".")) {
97              decStrings.add(token);
98          }
99  
100         // figure out how many hexStrings we should have
101         // also check if it is a IPv4 address
102         int hexStringsLength = 8;
103 
104         // If we have an IPv4 address tagged on at the end, subtract
105         // 4 bytes, or 2 hex words from the total
106         if (decStrings.size() > 0) {
107             hexStringsLength -= 2;
108         }
109 
110         // if we hit a double Colon add the appropriate hex strings
111         if (doubleColonIndex != -1) {
112             int numberToInsert = hexStringsLength - hexStrings.size();
113             for (int i = 0; i < numberToInsert; i++) {
114                 hexStrings.add(doubleColonIndex, "0");
115             }
116         }
117 
118         byte ipByteArray[] = new byte[16];
119 
120         // Finally convert these strings to bytes...
121         for (int i = 0; i < hexStrings.size(); i++) {
122             convertToBytes((String) hexStrings.get(i), ipByteArray, i * 2);
123         }
124 
125         // Now if there are any decimal values, we know where they go...
126         for (int i = 0; i < decStrings.size(); i++) {
127             ipByteArray[i + 12] = (byte) (Integer.parseInt((String) decStrings
128                     .get(i)) & 255);
129         }
130 
131         // now check to see if this guy is actually and IPv4 address
132         // an ipV4 address is ::FFFF:d.d.d.d
133         boolean ipV4 = true;
134         for (int i = 0; i < 10; i++) {
135             if (ipByteArray[i] != 0) {
136                 ipV4 = false;
137                 break;
138             }
139         }
140 
141         if (ipByteArray[10] != -1 || ipByteArray[11] != -1) {
142             ipV4 = false;
143         }
144 
145         if (ipV4) {
146             byte ipv4ByteArray[] = new byte[4];
147             for (int i = 0; i < 4; i++) {
148                 ipv4ByteArray[i] = ipByteArray[i + 12];
149             }
150             return ipv4ByteArray;
151         }
152 
153         return ipByteArray;
154 
155     }

157     /*** Converts a 4 character hex word into a 2 byte word equivalent */
158     public static void convertToBytes(String hexWord, byte ipByteArray[],
159             int byteIndex) {
160 
161         int hexWordLength = hexWord.length();
162         int hexWordIndex = 0;
163         ipByteArray[byteIndex] = 0;
164         ipByteArray[byteIndex + 1] = 0;
165         int charValue;
166 
167         // high order 4 bits of first byte
168         if (hexWordLength > 3) {
169             charValue = getIntValue(hexWord.charAt(hexWordIndex++));
170             ipByteArray[byteIndex] = (byte) (ipByteArray[byteIndex] | (charValue << 4));
171         }
172 
173         // low order 4 bits of the first byte
174         if (hexWordLength > 2) {
175             charValue = getIntValue(hexWord.charAt(hexWordIndex++));
176             ipByteArray[byteIndex] = (byte) (ipByteArray[byteIndex] | charValue);
177         }
178 
179         // high order 4 bits of second byte
180         if (hexWordLength > 1) {
181             charValue = getIntValue(hexWord.charAt(hexWordIndex++));
182             ipByteArray[byteIndex + 1] = (byte) (ipByteArray[byteIndex + 1] | (charValue << 4));
183         }
184 
185         // low order 4 bits of the first byte
186         charValue = getIntValue(hexWord.charAt(hexWordIndex));
187         ipByteArray[byteIndex + 1] = (byte) (ipByteArray[byteIndex + 1] | charValue & 15);
188     }

190     static int getIntValue(char c) {
191 
192         switch (c) {
193         case '0':
194             return 0;
195         case '1':
196             return 1;
197         case '2':
198             return 2;
199         case '3':
200             return 3;
201         case '4':
202             return 4;
203         case '5':
204             return 5;
205         case '6':
206             return 6;
207         case '7':
208             return 7;
209         case '8':
210             return 8;
211         case '9':
212             return 9;
213         }
214 
215         c = Character.toLowerCase(c);
216         switch (c) {
217         case 'a':
218             return 10;
219         case 'b':
220             return 11;
221         case 'c':
222             return 12;
223         case 'd':
224             return 13;
225         case 'e':
226             return 14;
227         case 'f':
228             return 15;
229         }
230         return 0;
231     }

233     public static boolean isValidIP6Address(String ipAddress) {
234         int length = ipAddress.length();
235         boolean doubleColon = false;
236         int numberOfColons = 0;
237         int numberOfPeriods = 0;
238         int numberOfPercent = 0;
239         String word = "";
240         char c = 0;
241         char prevChar = 0;
242         int offset = 0; // offset for [] ip addresses
243 
244         if (length < 2)
245             return false;
246 
247         for (int i = 0; i < length; i++) {
248             prevChar = c;
249             c = ipAddress.charAt(i);
250             switch (c) {
251 
252             // case for an open bracket [x:x:x:...x]
253             case '[':
254                 if (i != 0)
255                     return false; // must be first character
256                 if (ipAddress.charAt(length - 1) != ']')
257                     return false; // must have a close ]
258                 offset = 1;
259                 if (length < 4)
260                     return false;
261                 break;
262 
263             // case for a closed bracket at end of IP [x:x:x:...x]
264             case ']':
265                 if (i != length - 1)
266                     return false; // must be last charcter
267                 if (ipAddress.charAt(0) != '[')
268                     return false; // must have a open [
269                 break;
270 
271             // case for the last 32-bits represented as IPv4 x:x:x:x:x:x:d.d.d.d
272             case '.':
273                 numberOfPeriods++;
274                 if (numberOfPeriods > 3)
275                     return false;
276                 if (!isValidIP4Word(word))
277                     return false;
278                 if (numberOfColons != 6 && !doubleColon)
279                     return false;
280                 // a special case ::1:2:3:4:5:d.d.d.d allows 7 colons with an
281                 // IPv4 ending, otherwise 7 :'s is bad
282                 if (numberOfColons == 7 && ipAddress.charAt(0 + offset) != ':'
283                         && ipAddress.charAt(1 + offset) != ':')
284                     return false;
285                 word = "";
286                 break;
287 
288             case ':':
289                 // FIX "IP6 mechanism syntax #ip6-bad1"
290                 // An IPV6 address cannot start with a single ":".
291                 // Either it can starti with "::" or with a number.
292                 if (i == offset && (ipAddress.length() <= i || ipAddress.charAt(i+1) != ':')) {
293                     return false;
294                 }
295                 // END FIX "IP6 mechanism syntax #ip6-bad1"
296                 numberOfColons++;
297                 if (numberOfColons > 7)
298                     return false;
299                 if (numberOfPeriods > 0)
300                     return false;
301                 if (prevChar == ':') {
302                     if (doubleColon)
303                         return false;
304                     doubleColon = true;
305                 }
306                 word = "";
307                 break;
308             case '%':
309                 if (numberOfColons == 0)
310                     return false;
311                 numberOfPercent++;
312 
313                 // validate that the stuff after the % is valid
314                 if ((i + 1) >= length) {
315                     // in this case the percent is there but no number is
316                     // available
317                     return false;
318                 }
319                 try {
320                     Integer.parseInt(ipAddress.substring(i + 1));
321                 } catch (NumberFormatException e) {
322                     // right now we just support an integer after the % so if
323                     // this is not
324                     // what is there then return
325                     return false;
326                 }
327                 break;
328 
329             default:
330                 if (numberOfPercent == 0) {
331                     if (word.length() > 3)
332                         return false;
333                     if (!isValidHexChar(c))
334                         return false;
335                 }
336                 word += c;
337             }
338         }
339 
340         // Check if we have an IPv4 ending
341         if (numberOfPeriods > 0) {
342             if (numberOfPeriods != 3 || !isValidIP4Word(word))
343                 return false;
344         } else {
345             // If we're at then end and we haven't had 7 colons then there is a
346             // problem unless we encountered a doubleColon
347             if (numberOfColons != 7 && !doubleColon) {
348                 return false;
349             }
350 
351             // If we have an empty word at the end, it means we ended in either
352             // a : or a .
353             // If we did not end in :: then this is invalid
354             if (numberOfPercent == 0) {
355                 if (word == "" && ipAddress.charAt(length - 1 - offset) == ':'
356                         && ipAddress.charAt(length - 2 - offset) != ':') {
357                     return false;
358                 }
359             }
360         }
361 
362         return true;
363     }

365     public static boolean isValidIP4Word(String word) {
366         char c;
367         if (word.length() < 1 || word.length() > 3)
368             return false;
369         for (int i = 0; i < word.length(); i++) {
370             c = word.charAt(i);
371             if (!(c >= '0' && c <= '9'))
372                 return false;
373         }
374         if (Integer.parseInt(word) > 255)
375             return false;
376         return true;
377     }

379     static boolean isValidHexChar(char c) {
380 
381         return (c >= '0' && c <= '9') || (c >= 'A' && c <= 'F')
382                 || (c >= 'a' && c <= 'f');
383     }

385     /***
386      * Takes a string and parses it to see if it is a valid IPV4 address.
387      * 
388      * @return true, if the string represents an IPV4 address in dotted
389      *         notation, false otherwise
390      */
391     public static boolean isValidIPV4Address(String value) {
392 
393         int periods = 0;
394         int i = 0;
395         int length = value.length();
396 
397         if (length > 15)
398             return false;
399         char c = 0;
400         String word = "";
401         for (i = 0; i < length; i++) {
402             c = value.charAt(i);
403             if (c == '.') {
404                 periods++;
405                 if (periods > 3)
406                     return false;
407                 if (word == "")
408                     return false;
409                 if (Integer.parseInt(word) > 255)
410                     return false;
411                 word = "";
412             } else if (!(Character.isDigit(c)))
413                 return false;
414             else {
415                 if (word.length() > 2)
416                     return false;
417                 word += c;
418             }
419         }
420 
421         if (word == "" || Integer.parseInt(word) > 255)
422             return false;
423         if (periods != 3)
424             return false;
425         return true;
426     }
