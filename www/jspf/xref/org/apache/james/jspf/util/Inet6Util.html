31      /***
32       * Creates an byte[] based on an ipAddressString. No error handling is
33       * performed here.
34       */
35      public static byte[] createByteArrayFromIPAddressString(
36              String ipAddressString) {
37  
38          if (isValidIPV4Address(ipAddressString)) {
39              StringTokenizer tokenizer = new StringTokenizer(ipAddressString,
40                      ".");
41              String token = "";
42              int tempInt = 0;
43              byte[] byteAddress = new byte[4];
44              for (int i = 0; i < 4; i++) {
45                  token = tokenizer.nextToken();
46                  tempInt = Integer.parseInt(token);
47                  byteAddress[i] = (byte) tempInt;
48              }
49  
50              return byteAddress;
51          }
52  
53          if (ipAddressString.charAt(0) == '[') {
54              ipAddressString = ipAddressString.substring(1, ipAddressString
55                      .length() - 1);
56          }
57  
58          StringTokenizer tokenizer = new StringTokenizer(ipAddressString, ":.",
59                  true);
60          ArrayList hexStrings = new ArrayList();
61          ArrayList decStrings = new ArrayList();
62          String token = "";
63          String prevToken = "";
64          int doubleColonIndex = -1; // If a double colon exists, we need to
65          // insert 0s.
66  
67          // Go through the tokens, including the seperators ':' and '.'
68          // When we hit a : or . the previous token will be added to either
69          // the hex list or decimal list. In the case where we hit a ::
70          // we will save the index of the hexStrings so we can add zeros
71          // in to fill out the string
72          while (tokenizer.hasMoreTokens()) {
73              prevToken = token;
74              token = tokenizer.nextToken();
75  
76              if (token.equals(":")) {
77                  if (prevToken.equals(":")) {
78                      doubleColonIndex = hexStrings.size();
79                  } else if (!prevToken.equals("")) {
80                      hexStrings.add(prevToken);
81                  }
82              } else if (token.equals(".")) {
83                  decStrings.add(prevToken);
84              }
85          }
86  
87          if (prevToken.equals(":")) {
88              if (token.equals(":")) {
89                  doubleColonIndex = hexStrings.size();
90              } else {
91                  hexStrings.add(token);
92              }
93          } else if (prevToken.equals(".")) {
94              decStrings.add(token);
95          }
96  
97          // figure out how many hexStrings we should have
98          // also check if it is a IPv4 address
99          int hexStringsLength = 8;
100 
101         // If we have an IPv4 address tagged on at the end, subtract
102         // 4 bytes, or 2 hex words from the total
103         if (decStrings.size() > 0) {
104             hexStringsLength -= 2;
105         }
106 
107         // if we hit a double Colon add the appropriate hex strings
108         if (doubleColonIndex != -1) {
109             int numberToInsert = hexStringsLength - hexStrings.size();
110             for (int i = 0; i < numberToInsert; i++) {
111                 hexStrings.add(doubleColonIndex, "0");
112             }
113         }
114 
115         byte ipByteArray[] = new byte[16];
116 
117         // Finally convert these strings to bytes...
118         for (int i = 0; i < hexStrings.size(); i++) {
119             convertToBytes((String) hexStrings.get(i), ipByteArray, i * 2);
120         }
121 
122         // Now if there are any decimal values, we know where they go...
123         for (int i = 0; i < decStrings.size(); i++) {
124             ipByteArray[i + 12] = (byte) (Integer.parseInt((String) decStrings
125                     .get(i)) & 255);
126         }
127 
128         // now check to see if this guy is actually and IPv4 address
129         // an ipV4 address is ::FFFF:d.d.d.d
130         boolean ipV4 = true;
131         for (int i = 0; i < 10; i++) {
132             if (ipByteArray[i] != 0) {
133                 ipV4 = false;
134                 break;
135             }
136         }
137 
138         if (ipByteArray[10] != -1 || ipByteArray[11] != -1) {
139             ipV4 = false;
140         }
141 
142         if (ipV4) {
143             byte ipv4ByteArray[] = new byte[4];
144             for (int i = 0; i < 4; i++) {
145                 ipv4ByteArray[i] = ipByteArray[i + 12];
146             }
147             return ipv4ByteArray;
148         }
149 
150         return ipByteArray;
151 
152     }

154     /*** Converts a 4 character hex word into a 2 byte word equivalent */
155     public static void convertToBytes(String hexWord, byte ipByteArray[],
156             int byteIndex) {
157 
158         int hexWordLength = hexWord.length();
159         int hexWordIndex = 0;
160         ipByteArray[byteIndex] = 0;
161         ipByteArray[byteIndex + 1] = 0;
162         int charValue;
163 
164         // high order 4 bits of first byte
165         if (hexWordLength > 3) {
166             charValue = getIntValue(hexWord.charAt(hexWordIndex++));
167             ipByteArray[byteIndex] = (byte) (ipByteArray[byteIndex] | (charValue << 4));
168         }
169 
170         // low order 4 bits of the first byte
171         if (hexWordLength > 2) {
172             charValue = getIntValue(hexWord.charAt(hexWordIndex++));
173             ipByteArray[byteIndex] = (byte) (ipByteArray[byteIndex] | charValue);
174         }
175 
176         // high order 4 bits of second byte
177         if (hexWordLength > 1) {
178             charValue = getIntValue(hexWord.charAt(hexWordIndex++));
179             ipByteArray[byteIndex + 1] = (byte) (ipByteArray[byteIndex + 1] | (charValue << 4));
180         }
181 
182         // low order 4 bits of the first byte
183         charValue = getIntValue(hexWord.charAt(hexWordIndex));
184         ipByteArray[byteIndex + 1] = (byte) (ipByteArray[byteIndex + 1] | charValue & 15);
185     }

187     static int getIntValue(char c) {
188 
189         switch (c) {
190         case '0':
191             return 0;
192         case '1':
193             return 1;
194         case '2':
195             return 2;
196         case '3':
197             return 3;
198         case '4':
199             return 4;
200         case '5':
201             return 5;
202         case '6':
203             return 6;
204         case '7':
205             return 7;
206         case '8':
207             return 8;
208         case '9':
209             return 9;
210         }
211 
212         c = Character.toLowerCase(c);
213         switch (c) {
214         case 'a':
215             return 10;
216         case 'b':
217             return 11;
218         case 'c':
219             return 12;
220         case 'd':
221             return 13;
222         case 'e':
223             return 14;
224         case 'f':
225             return 15;
226         }
227         return 0;
228     }

230     public static boolean isValidIP6Address(String ipAddress) {
231         int length = ipAddress.length();
232         boolean doubleColon = false;
233         int numberOfColons = 0;
234         int numberOfPeriods = 0;
235         int numberOfPercent = 0;
236         String word = "";
237         char c = 0;
238         char prevChar = 0;
239         int offset = 0; // offset for [] ip addresses
240 
241         if (length < 2)
242             return false;
243 
244         for (int i = 0; i < length; i++) {
245             prevChar = c;
246             c = ipAddress.charAt(i);
247             switch (c) {
248 
249             // case for an open bracket [x:x:x:...x]
250             case '[':
251                 if (i != 0)
252                     return false; // must be first character
253                 if (ipAddress.charAt(length - 1) != ']')
254                     return false; // must have a close ]
255                 offset = 1;
256                 if (length < 4)
257                     return false;
258                 break;
259 
260             // case for a closed bracket at end of IP [x:x:x:...x]
261             case ']':
262                 if (i != length - 1)
263                     return false; // must be last charcter
264                 if (ipAddress.charAt(0) != '[')
265                     return false; // must have a open [
266                 break;
267 
268             // case for the last 32-bits represented as IPv4 x:x:x:x:x:x:d.d.d.d
269             case '.':
270                 numberOfPeriods++;
271                 if (numberOfPeriods > 3)
272                     return false;
273                 if (!isValidIP4Word(word))
274                     return false;
275                 if (numberOfColons != 6 && !doubleColon)
276                     return false;
277                 // a special case ::1:2:3:4:5:d.d.d.d allows 7 colons with an
278                 // IPv4 ending, otherwise 7 :'s is bad
279                 if (numberOfColons == 7 && ipAddress.charAt(0 + offset) != ':'
280                         && ipAddress.charAt(1 + offset) != ':')
281                     return false;
282                 word = "";
283                 break;
284 
285             case ':':
286                 numberOfColons++;
287                 if (numberOfColons > 7)
288                     return false;
289                 if (numberOfPeriods > 0)
290                     return false;
291                 if (prevChar == ':') {
292                     if (doubleColon)
293                         return false;
294                     doubleColon = true;
295                 }
296                 word = "";
297                 break;
298             case '%':
299                 if (numberOfColons == 0)
300                     return false;
301                 numberOfPercent++;
302 
303                 // validate that the stuff after the % is valid
304                 if ((i + 1) >= length) {
305                     // in this case the percent is there but no number is
306                     // available
307                     return false;
308                 }
309                 try {
310                     Integer.parseInt(ipAddress.substring(i + 1));
311                 } catch (NumberFormatException e) {
312                     // right now we just support an integer after the % so if
313                     // this is not
314                     // what is there then return
315                     return false;
316                 }
317                 break;
318 
319             default:
320                 if (numberOfPercent == 0) {
321                     if (word.length() > 3)
322                         return false;
323                     if (!isValidHexChar(c))
324                         return false;
325                 }
326                 word += c;
327             }
328         }
329 
330         // Check if we have an IPv4 ending
331         if (numberOfPeriods > 0) {
332             if (numberOfPeriods != 3 || !isValidIP4Word(word))
333                 return false;
334         } else {
335             // If we're at then end and we haven't had 7 colons then there is a
336             // problem unless we encountered a doubleColon
337             if (numberOfColons != 7 && !doubleColon) {
338                 return false;
339             }
340 
341             // If we have an empty word at the end, it means we ended in either
342             // a : or a .
343             // If we did not end in :: then this is invalid
344             if (numberOfPercent == 0) {
345                 if (word == "" && ipAddress.charAt(length - 1 - offset) == ':'
346                         && ipAddress.charAt(length - 2 - offset) != ':') {
347                     return false;
348                 }
349             }
350         }
351 
352         return true;
353     }

355     public static boolean isValidIP4Word(String word) {
356         char c;
357         if (word.length() < 1 || word.length() > 3)
358             return false;
359         for (int i = 0; i < word.length(); i++) {
360             c = word.charAt(i);
361             if (!(c >= '0' && c <= '9'))
362                 return false;
363         }
364         if (Integer.parseInt(word) > 255)
365             return false;
366         return true;
367     }

369     static boolean isValidHexChar(char c) {
370 
371         return (c >= '0' && c <= '9') || (c >= 'A' && c <= 'F')
372                 || (c >= 'a' && c <= 'f');
373     }

375     /***
376      * Takes a string and parses it to see if it is a valid IPV4 address.
377      * 
378      * @return true, if the string represents an IPV4 address in dotted
379      *         notation, false otherwise
380      */
381     public static boolean isValidIPV4Address(String value) {
382 
383         int periods = 0;
384         int i = 0;
385         int length = value.length();
386 
387         if (length > 15)
388             return false;
389         char c = 0;
390         String word = "";
391         for (i = 0; i < length; i++) {
392             c = value.charAt(i);
393             if (c == '.') {
394                 periods++;
395                 if (periods > 3)
396                     return false;
397                 if (word == "")
398                     return false;
399                 if (Integer.parseInt(word) > 255)
400                     return false;
401                 word = "";
402             } else if (!(Character.isDigit(c)))
403                 return false;
404             else {
405                 if (word.length() > 2)
406                     return false;
407                 word += c;
408             }
409         }
410 
411         if (word == "" || Integer.parseInt(word) > 255)
412             return false;
413         if (periods != 3)
414             return false;
415         return true;
416     }
