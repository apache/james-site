65      /***
66       * @see org.apache.james.jspf.core.DNSService#getSpfRecord(java.lang.String,
67       *      java.lang.String)
68       */
69      public String getSpfRecord(String hostname, String spfVersion)
70              throws PermErrorException, NoneException, TempErrorException {
71  
72          String returnValue = null;
73          ArrayList txtR = new ArrayList();
74  
75          // do DNS lookup for TXT
76          txtR = getTXTRecords(hostname);
77  
78          // process returned records
79          if (!txtR.isEmpty()) {
80  
81              Iterator all = txtR.iterator();
82  
83              while (all.hasNext()) {
84                  String compare = all.next().toString().trim();
85  
86                  // remove '"'
87                  compare = compare.toLowerCase().substring(1,
88                          compare.length() - 1);
89  
90                  if (compare.startsWith(spfVersion + " ")) {
91                      if (returnValue == null) {
92                          returnValue = compare;
93                      } else {
94                          throw new PermErrorException(
95                                  "More than 1 SPF record found");
96                      }
97                  }
98              }
99          }
100         if (returnValue == null) {
101             throw new NoneException("No SPF record found");
102         }
103         return returnValue;
104     }

106     /***
107      * Get an ArrayList of all TXT Records for a partical domain.
108      * 
109      * @param hostname
110      *            The hostname for which the TXT-Records should be retrieved
111      * @return TXT Records-which were found.
112      * @throws NoneException
113      *             if none TXT-Records were found.
114      * @throws TempErrorException
115      *             if the lookup result was "TRY_AGAIN"
116      * @throws PermErrorException
117      *             if an PermError should be returned
118      */
119     private ArrayList getTXTRecords(String hostname)
120             throws NoneException, TempErrorException {
121         ArrayList txtR = new ArrayList();
122         Record[] records;
123         try {
124 
125             log.debug("Start TXT-Record lookup for : " + hostname);
126 
127             Lookup.getDefaultResolver().setTimeout(timeOut);
128             Lookup query = new Lookup(hostname, Type.TXT);
129             records = query.run();
130             int queryResult = query.getResult();
131             if ((queryResult != Lookup.TRY_AGAIN)) {
132                 if (records != null) {
133 
134                     log.debug("Found " + records.length + " TXT-Records");
135 
136                     for (int i = 0; i < records.length; i++) {
137                         TXTRecord txt = (TXTRecord) records[i];
138 
139                         log
140                                 .debug("Add txt " + txt.rdataToString()
141                                         + " to list");
142 
143                         txtR.add(txt.rdataToString());
144                     }
145                 } else {
146                     throw new NoneException("No TXTRecord found for: "
147                             + hostname);
148                 }
149             } else {
150                 throw new TempErrorException("DNS Server returns RCODE: "
151                         + queryResult);
152             }
153         } catch (TextParseException e) {
154             // I think thats the best we could do
155             throw new NoneException("No TXTRecord found for: " + hostname);
156         }
157         return txtR;
158     }

164     public List getARecords(String strServer, int mask) throws NoneException,
165             PermErrorException, TempErrorException {
166 
167         ArrayList listTxtData = new ArrayList();
168 
169         if (IPAddr.isIPAddr(strServer)) {
170             IPAddr ipTest = IPAddr.getAddress(strServer);
171             // Address is already an IP address, so add it to list
172             listTxtData.add(ipTest);
173         } else {
174 
175             Record[] records;
176             try {
177 
178                 log.debug("Start A-Record lookup for : " + strServer);
179 
180                 Lookup.getDefaultResolver().setTimeout(timeOut);
181                 Lookup query = new Lookup(strServer, Type.A);
182                 records = query.run();
183                 int queryResult = query.getResult();
184 
185                 if ((queryResult != Lookup.TRY_AGAIN)) {
186                     if (records != null) {
187 
188                         log.debug("Found " + records.length + " A-Records");
189 
190                         for (int i = 0; i < records.length; i++) {
191                             ARecord a = (ARecord) records[i];
192 
193                             ArrayList ipArray = getIPList(a.getAddress()
194                                     .getHostAddress(), mask);
195                             Iterator ip = ipArray.iterator();
196 
197                             while (ip.hasNext()) {
198                                 Object ipA = ip.next();
199 
200                                 log.debug("Add ipAddress " + ipA + " to list");
201                                 listTxtData.add(ipA);
202                             }
203                         }
204                     } else {
205                         throw new NoneException("No A record found for: "
206                                 + strServer);
207                     }
208                 } else {
209                     throw new TempErrorException("DNS Server returns RCODE: "
210                             + queryResult);
211                 }
212             } catch (TextParseException e) {
213                 // i think this is the best we could do
214                 throw new NoneException("No A Record found for: " + strServer);
215             }
216         }
217         return listTxtData;
218     }

281     /***
282      * Convert list of DNS names to masked IPAddr
283      * 
284      * @param addressList
285      *            ArrayList of DNS names which should be converted to masked
286      *            IPAddresses
287      * @param maskLength
288      *            the networkmask
289      * @return ArrayList of the conversion
290      * @throws PermErrorException
291      *             if an PermError should be returned
292      */
293     private ArrayList getAList(ArrayList addressList, int maskLength)
294             throws PermErrorException {
295 
296         ArrayList listAddresses = new ArrayList();
297         String aValue;
298 
299         for (int i = 0; i < addressList.size(); i++) {
300             aValue = addressList.get(i).toString();
301             try {
302                 listAddresses.addAll(getARecords(aValue, maskLength));
303             } catch (Exception e) {
304                 // Carry on regardless?
305             }
306         }
307         return listAddresses;
308     }

329      */
330     public List getPTRRecords(String ipAddress) throws NoneException,
331             PermErrorException, TempErrorException {
332 
333         ArrayList ptrR = new ArrayList();
334         Record[] records;
335 
336         // do DNS lookup for TXT
337         IPAddr ip;
338 
339         ip = IPAddr.getAddress(ipAddress);
340 
341         try {
342 
343             log.debug("Start PTR-Record lookup for : " + ipAddress);
344 
345             Lookup.getDefaultResolver().setTimeout(timeOut);
346             Lookup query = new Lookup(ip.getReverseIP() + ".in-addr.arpa",
347                     Type.PTR);
348             records = query.run();
349             int queryResult = query.getResult();
350 
351             if ((queryResult != Lookup.TRY_AGAIN)) {
352                 if (records != null) {
353                     log.debug("Found " + records.length + " PTR-Records");
354 
355                     for (int i = 0; i < records.length; i++) {
356                         PTRRecord ptr = (PTRRecord) records[i];
357                         ptrR.add(IPAddr.stripDot(ptr.getTarget().toString()));
358                         log.debug("Add ipAddress "
359                                 + IPAddr.stripDot(ptr.getTarget().toString())
360                                 + " to list");
361                     }
362                 } else {
363                     throw new NoneException("No PTRRecord found for: "
364                             + ipAddress);
365                 }
366             } else {
367                 throw new TempErrorException("DNS Server returns RCODE: "
368                         + queryResult);
369             }
370         } catch (TextParseException e) {
371             // i think this is the best we could do
372             throw new NoneException("No PTRRecord found for: " + ipAddress);
373         }
374 
375         return ptrR;
376     }

407                 listIP.addAll(IPAddr.getAddresses(host, mask));

478     public List getLocalDomainNames() {
479         List names = new ArrayList();
480 
481         log.debug("Start Local ipaddress lookup");
482         try {
483             InetAddress ia[] = InetAddress.getAllByName(InetAddress
484                     .getLocalHost().getHostName());
485 
486             for (int i = 0; i < ia.length; i++) {
487                 String host = ia[i].getHostName();
488                 names.add(host);
489 
490                 log.debug("Add hostname " + host + " to list");
491             }
492         } catch (UnknownHostException e) {
493             // just ignore this..
494         }
495         return names;
