55      /***
56       * @see org.apache.james.jspf.core.DNSService#getSpfRecord(java.lang.String,
57       *      java.lang.String)
58       */
59      public String getSpfRecord(String hostname, String spfVersion)
60              throws PermErrorException, NoneException, TempErrorException {
61  
62          String returnValue = null;
63          ArrayList txtR = new ArrayList();
64  
65          // do DNS lookup for TXT
66          txtR = getTXTRecords(hostname);
67  
68          // process returned records
69          if (!txtR.isEmpty()) {
70  
71              Iterator all = txtR.iterator();
72  
73              while (all.hasNext()) {
74                  String compare = all.next().toString().trim();
75  
76                  // remove '"'
77                  compare = compare.toLowerCase().substring(1,
78                          compare.length() - 1);
79  
80                  if (compare.startsWith(spfVersion + " ")) {
81                      if (returnValue == null) {
82                          returnValue = compare;
83                      } else {
84                          throw new PermErrorException(
85                                  "More than 1 SPF record found");
86                      }
87                  }
88              }
89          }
90          if (returnValue == null) {
91              throw new NoneException("No SPF record found");
92          }
93          return returnValue;
94      }

96      /***
97       * Get an ArrayList of all TXT Records for a partical domain.
98       * 
99       * @param hostname
100      *            The hostname for which the TXT-Records should be retrieved
101      * @return TXT Records-which were found.
102      * @throws NoneException
103      *             if none TXT-Records were found.
104      * @throws TempErrorException
105      *             if the lookup result was "TRY_AGAIN"
106      * @throws PermErrorException
107      *             if an PermError should be returned
108      */
109     private static ArrayList getTXTRecords(String hostname)
110             throws NoneException, TempErrorException {
111         ArrayList txtR = new ArrayList();
112         Record[] records;
113         try {
114 
115             log.debug("Start TXT-Record lookup for : " + hostname);
116 
117             Lookup.getDefaultResolver().setTimeout(timeOut);
118             Lookup query = new Lookup(hostname, Type.TXT);
119             records = query.run();
120             int queryResult = query.getResult();
121             if ((queryResult != Lookup.TRY_AGAIN)) {
122                 if (records != null) {
123 
124                     log.debug("Found " + records.length + " TXT-Records");
125 
126                     for (int i = 0; i < records.length; i++) {
127                         TXTRecord txt = (TXTRecord) records[i];
128 
129                         log
130                                 .debug("Add txt " + txt.rdataToString()
131                                         + " to list");
132 
133                         txtR.add(txt.rdataToString());
134                     }
135                 } else {
136                     throw new NoneException("No TXTRecord found for: "
137                             + hostname);
138                 }
139             } else {
140                 throw new TempErrorException("DNS Server returns RCODE: "
141                         + queryResult);
142             }
143         } catch (TextParseException e) {
144             // I think thats the best we could do
145             throw new NoneException("No TXTRecord found for: " + hostname);
146         }
147         return txtR;
148     }

154     public List getARecords(String strServer, int mask) throws NoneException,
155             PermErrorException, TempErrorException {
156 
157         ArrayList listTxtData = new ArrayList();
158 
159         if (IPAddr.isIPAddr(strServer)) {
160             IPAddr ipTest = IPAddr.getAddress(strServer);
161             // Address is already an IP address, so add it to list
162             listTxtData.add(ipTest);
163         } else {
164 
165             Record[] records;
166             try {
167 
168                 log.debug("Start A-Record lookup for : " + strServer);
169 
170                 Lookup.getDefaultResolver().setTimeout(timeOut);
171                 Lookup query = new Lookup(strServer, Type.A);
172                 records = query.run();
173                 int queryResult = query.getResult();
174 
175                 if ((queryResult != Lookup.TRY_AGAIN)) {
176                     if (records != null) {
177 
178                         log.debug("Found " + records.length + " A-Records");
179 
180                         for (int i = 0; i < records.length; i++) {
181                             ARecord a = (ARecord) records[i];
182 
183                             ArrayList ipArray = getIPList(a.getAddress()
184                                     .getHostAddress(), mask);
185                             Iterator ip = ipArray.iterator();
186 
187                             while (ip.hasNext()) {
188                                 Object ipA = ip.next();
189 
190                                 log.debug("Add ipAddress " + ipA + " to list");
191                                 listTxtData.add(ipA);
192                             }
193                         }
194                     } else {
195                         throw new NoneException("No A record found for: "
196                                 + strServer);
197                     }
198                 } else {
199                     throw new TempErrorException("DNS Server returns RCODE: "
200                             + queryResult);
201                 }
202             } catch (TextParseException e) {
203                 // i think this is the best we could do
204                 throw new NoneException("No A Record found for: " + strServer);
205             }
206         }
207         return listTxtData;
208     }

271     /***
272      * Convert list of DNS names to masked IPAddr
273      * 
274      * @param addressList
275      *            ArrayList of DNS names which should be converted to masked
276      *            IPAddresses
277      * @param maskLength
278      *            the networkmask
279      * @return ArrayList of the conversion
280      * @throws PermErrorException
281      *             if an PermError should be returned
282      */
283     private ArrayList getAList(ArrayList addressList, int maskLength)
284             throws PermErrorException {
285 
286         ArrayList listAddresses = new ArrayList();
287         String aValue;
288 
289         for (int i = 0; i < addressList.size(); i++) {
290             aValue = addressList.get(i).toString();
291             try {
292                 listAddresses.addAll(getARecords(aValue, maskLength));
293             } catch (Exception e) {
294                 // Carry on regardless?
295             }
296         }
297         return listAddresses;
298     }

319      */
320     public List getPTRRecords(String ipAddress) throws NoneException,
321             PermErrorException, TempErrorException {
322 
323         ArrayList ptrR = new ArrayList();
324         Record[] records;
325 
326         // do DNS lookup for TXT
327         IPAddr ip;
328 
329         ip = IPAddr.getAddress(ipAddress);
330 
331         try {
332 
333             log.debug("Start PTR-Record lookup for : " + ipAddress);
334 
335             Lookup.getDefaultResolver().setTimeout(timeOut);
336             Lookup query = new Lookup(ip.getReverseIP() + ".in-addr.arpa",
337                     Type.PTR);
338             records = query.run();
339             int queryResult = query.getResult();
340 
341             if ((queryResult != Lookup.TRY_AGAIN)) {
342                 if (records != null) {
343                     log.debug("Found " + records.length + " PTR-Records");
344 
345                     for (int i = 0; i < records.length; i++) {
346                         PTRRecord ptr = (PTRRecord) records[i];
347                         ptrR.add(IPAddr.stripDot(ptr.getTarget().toString()));
348                         log.debug("Add ipAddress "
349                                 + IPAddr.stripDot(ptr.getTarget().toString())
350                                 + " to list");
351                     }
352                 } else {
353                     throw new NoneException("No PTRRecord found for: "
354                             + ipAddress);
355                 }
356             } else {
357                 throw new TempErrorException("DNS Server returns RCODE: "
358                         + queryResult);
359             }
360         } catch (TextParseException e) {
361             // i think this is the best we could do
362             throw new NoneException("No PTRRecord found for: " + ipAddress);
363         }
364 
365         return ptrR;
366     }

392

394

397                 listIP.addAll(IPAddr.getAddresses(host, mask));

468     public List getLocalDomainNames() {
469         List names = new ArrayList();
470 
471         log.debug("Start Local ipaddress lookup");
472         try {
473             InetAddress ia[] = InetAddress.getAllByName(InetAddress
474                     .getLocalHost().getHostName());
475 
476             for (int i = 0; i < ia.length; i++) {
477                 String host = ia[i].getHostName();
478                 names.add(host);
479 
480                 log.debug("Add hostname " + host + " to list");
481             }
482         } catch (UnknownHostException e) {
483             // just ignore this..
484         }
485         return names;
