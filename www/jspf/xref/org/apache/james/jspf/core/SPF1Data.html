75      /***
76       * Build the SPF1Data from the given parameters
77       * 
78       * @param mailFrom
79       *            The emailaddress of the sender
80       * @param heloDomain
81       *            The helo provided by the sender
82       * @param clientIP
83       *            The ipaddress of the client
84       * @param dnsProbe
85       *            The DNSService
86       * @throws PermErrorException
87       *             Get thrown if invalid data get passed
88       * @throws NoneException
89       *             Get thrown if no valid emailaddress get passed
90       */
91      public SPF1Data(String mailFrom, String heloDomain, String clientIP,
92              DNSService dnsProbe) throws PermErrorException, NoneException {
93          super();
94  
95          this.mailFrom = mailFrom.trim();
96          this.hostName = heloDomain.trim();
97          this.ipAddress = clientIP.trim();
98          this.dnsProbe = dnsProbe;
99  
100         try {
101             // get the in Address
102             inAddress = IPAddr.getInAddress(clientIP);
103         } catch (PermErrorException e) {
104             // throw an exception cause the ip was not rfc conform
105             throw new PermErrorException(e.getMessage());
106         }
107 
108         // setup the data!
109         setupData(mailFrom, hostName);
110     }

112     /***
113      * Setup the data which used to retrieve the SPF-Record
114      * 
115      * @param mailFrom
116      *            The emailaddress of the sender
117      * @param helo
118      *            The provided helo
119      * @throws NoneException
120      *             Get thrown if an invalid emailaddress get passed
121      */
122     private void setupData(String mailFrom, String helo) throws NoneException {
123 
124         // if nullsender is used postmaster@helo will be used as email
125         if (mailFrom.equals("")) {
126             this.currentSenderPart = "postmaster";
127             this.senderDomain = helo;
128             this.mailFrom = currentSenderPart + "@" + helo;
129         } else {
130             String[] fromParts = mailFrom.split("@");
131 
132             // should never be bigger as 2 !
133             if (fromParts.length > 2) {
134                 throw new NoneException("Not a valid email address " + mailFrom);
135             } else if (fromParts.length == 2) {
136                 this.currentSenderPart = fromParts[0];
137                 this.senderDomain = fromParts[1];
138             } else {
139                 this.currentSenderPart = "postmaster";
140                 this.senderDomain = mailFrom;
141             }
142         }
143         this.currentDomain = this.senderDomain;
144     }

181     /***
182      * @see org.apache.james.jspf.macro.MacroData#getClientDomain()
183      */
184     public String getClientDomain() {
185         if (clientDomain == null) {
186             List domains;
187             try {
188                 domains = dnsProbe.getPTRRecords(ipAddress);
189                 if (domains.size() > 0) {
190                     clientDomain = (String) domains.get(0);
191                 } else {
192                     clientDomain = ipAddress;
193                 }
194             } catch (Exception e) {
195                 clientDomain = ipAddress;
196             }
197         }
198         return clientDomain;
199     }

251     /***
252      * @see org.apache.james.jspf.macro.MacroData#getReceivingDomain()
253      */
254     public String getReceivingDomain() {
255         List dNames;
256 
257         if (receivingDomain.equals("unknown")) {
258             dNames = dnsProbe.getLocalDomainNames();
259 
260             for (int i = 0; i < dNames.size(); i++) {
261                 // check if the domainname is a FQDN
262                 if (SPF1Utils.checkFQDN(dNames.get(i).toString())) {
263                     receivingDomain = dNames.get(i).toString();
264                     return receivingDomain;
265                 }
266             }
267         }
268         return receivingDomain;
269     }
