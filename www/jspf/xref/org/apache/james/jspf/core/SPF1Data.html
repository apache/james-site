72      /***
73       * Build the SPF1Data from the given parameters
74       * 
75       * @param mailFrom
76       *            The emailaddress of the sender
77       * @param heloDomain
78       *            The helo provided by the sender
79       * @param clientIP
80       *            The ipaddress of the client
81       * @param dnsProbe
82       *            The DNSService
83       * @throws PermErrorException
84       *             Get thrown if invalid data get passed
85       * @throws NoneException
86       *             Get thrown if no valid emailaddress get passed
87       */
88      public SPF1Data(String mailFrom, String heloDomain, String clientIP,
89              DNSService dnsProbe) throws PermErrorException, NoneException {
90          super();
91  
92          this.mailFrom = mailFrom.trim();
93          this.hostName = heloDomain.trim();
94          this.ipAddress = clientIP.trim();
95          this.dnsProbe = dnsProbe;
96  
97          try {
98              // get the in Address
99              inAddress = IPAddr.getInAddress(clientIP);
100         } catch (PermErrorException e) {
101             // throw an exception cause the ip was not rfc conform
102             throw new PermErrorException(e.getMessage());
103         }
104 
105         // setup the data!
106         setupData(mailFrom, hostName);
107     }

109     /***
110      * Setup the data which used to retrieve the SPF-Record
111      * 
112      * @param mailFrom
113      *            The emailaddress of the sender
114      * @param helo
115      *            The provided helo
116      * @throws NoneException
117      *             Get thrown if an invalid emailaddress get passed
118      */
119     private void setupData(String mailFrom, String helo) throws NoneException {
120 
121         // if nullsender is used postmaster@helo will be used as email
122         if (mailFrom.equals("")) {
123             this.currentSenderPart = "postmaster";
124             this.senderDomain = helo;
125             this.mailFrom = currentSenderPart + "@" + helo;
126         } else {
127             String[] fromParts = mailFrom.split("@");
128 
129             // should never be bigger as 2 !
130             if (fromParts.length > 2) {
131                 throw new NoneException("Not a valid email address " + mailFrom);
132             } else if (fromParts.length == 2) {
133                 this.currentSenderPart = fromParts[0];
134                 this.senderDomain = fromParts[1];
135             } else {
136                 this.currentSenderPart = "postmaster";
137                 this.senderDomain = mailFrom;
138             }
139         }
140         this.currentDomain = this.senderDomain;
141     }

178     /***
179      * @see org.apache.james.jspf.macro.MacroData#getClientDomain()
180      */
181     public String getClientDomain() {
182         if (clientDomain == null) {
183             List domains;
184             try {
185                 domains = dnsProbe.getPTRRecords(ipAddress);
186                 if (domains.size() > 0) {
187                     clientDomain = (String) domains.get(0);
188                 } else {
189                     clientDomain = ipAddress;
190                 }
191             } catch (Exception e) {
192                 clientDomain = ipAddress;
193             }
194         }
195         return clientDomain;
196     }

248     /***
249      * @see org.apache.james.jspf.macro.MacroData#getReceivingDomain()
250      */
251     public String getReceivingDomain() {
252         List dNames;
253 
254         if (receivingDomain.equals("unknown")) {
255             dNames = dnsProbe.getLocalDomainNames();
256 
257             for (int i = 0; i < dNames.size(); i++) {
258                 // check if the domainname is a FQDN
259                 if (SPF1Utils.checkFQDN(dNames.get(i).toString())) {
260                     receivingDomain = dNames.get(i).toString();
261                     return receivingDomain;
262                 }
263             }
264         }
265         return receivingDomain;
266     }
