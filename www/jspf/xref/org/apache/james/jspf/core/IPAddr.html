141     /***
142      * create series of 16 bit masks for each ip block
143      * 
144      * @param maskLength
145      *            The netmask
146      */
147     public void setMask(int maskLength) {
148         int startMask;
149         int shift;
150         int maskSize;
151 
152         this.maskLength = maskLength;
153         if (ipLength == 4) {
154             if (!((maskLength > -1) && (maskLength < 33))) {
155                 maskLength = 32;
156             }
157             maskSize = 8;
158             startMask = (maskLength - 1) / maskSize;
159         } else {
160             if (!((maskLength > -1) && (maskLength < 129))) {
161                 maskLength = 128;
162             }
163             maskSize = 16;
164             startMask = (maskLength - 1) / maskSize;
165         }
166 
167         for (int i = 0; i < ipRun; i++) {
168             // full mask
169             if (i < startMask) {
170                 mask[i] = MASK16;
171                 // variable mask
172             } else if (i == startMask) {
173                 shift = ((i + 1) * maskSize) - maskLength;
174                 mask[i] = (MASK16 << shift) & MASK16;
175                 // no mask
176             } else {
177                 mask[i] = 0;
178             }
179         }
180     }

201     /***
202      * Convert ipAddress to a byte Array which represent the ipAddress
203      * 
204      * @param netAddress
205      *            The ipAddress we should convert
206      * @throws PermErrorException
207      *             on error
208      */
209     private void stringToInternal(String netAddress) throws PermErrorException {
210         netAddress = stripDot(netAddress);
211 
212         byte[] bytes = Inet6Util.createByteArrayFromIPAddressString(netAddress);
213 
214         if (bytes.length == 4) {
215             for (int i = 0; i < bytes.length; i++) {
216                 address[i] = bytes[i];
217             }
218         } else if (bytes.length == 16) {
219             setIP6Defaults();
220             for (int i = 0; i < bytes.length / 2; i++) {
221                 address[i] = unsigned(bytes[i * 2]) * 256
222                         + unsigned(bytes[i * 2 + 1]);
223             }
224         } else {
225             throw new PermErrorException("Not a valid address: " + netAddress);
226         }
227     }

249     /***
250      * Get ip Address from given int Array
251      * 
252      * @param addressData
253      *            The int Array
254      * @return ipAddress The ipAddress
255      */
256     private String getIPAddress(int[] addressData) {
257         StringBuffer createAddress = new StringBuffer();
258         int[] workingAddress;
259 
260         // convert internal address to 8 bit
261         if (ipLength == 4) {
262             workingAddress = get8BitAddress(addressData);
263             // create IP string
264             createAddress.append(workingAddress[0]);
265             for (int i = 1; i < ipRun; i++) {
266                 createAddress.append(ipJoiner + workingAddress[i]);
267             }
268             // leave internal address as 16 bit
269         } else {
270             workingAddress = addressData;
271             // create IP string
272             createAddress.append(getHex(workingAddress[0]));
273             for (int i = 1; i < ipRun; i++) {
274                 createAddress.append(ipJoiner + getHex(workingAddress[i]));
275             }
276         }
277 
278         return createAddress.toString();
279     }

303      * Return the NibbleFormat of the IPAddr
304      * 
305      * @return ipAddress The ipAddress in nibbleFormat 
306      */
307     public String getNibbleFormat() {
308         StringBuffer sb = new StringBuffer();
309         int[] ip = address;
310         for (int i = 0; i < ip.length; i++) {
311             String hex = getHex(ip[i]);
312             for (int j = 0; j < hex.length(); j++) {
313                 sb.append(hex.charAt(j));
314                 if (i != ip.length -1 || j != hex.length() -1) {
315                     sb.append(".");
316                 }
317             }
318         }
319         return sb.toString();
320     }

331     /***
332      * Converts 16 bit representation to 8 bit for IP4
333      * 
334      * @param addressData
335      *            The given int Array
336      * @return converted String
337      */
338     private int[] get8BitAddress(int[] addressData) {
339         int[] convertAddress = new int[4];
340         for (int i = 0; i < ipRun; i++) {
341             convertAddress[i] = addressData[i] & MASK8;
342         }
343         return convertAddress;

355     private int[] maskedAddress(int[] addressData, int[] maskData) {
356         int[] maskedAddress = new int[ipLength];
357 
358         for (int i = 0; i < ipRun; i++) {
359             maskedAddress[i] = addressData[i] & maskData[i];
360         }

364     /***
365      * Reverses internal address
366      * 
367      * @param addressData
368      *            The int array represent the ipAddress
369      * @return reverseIP
370      */
371     private int[] reverseIP(int[] addressData) {
372         int[] reverseIP = new int[ipLength];
373         int temp;
374         for (int i = 0; i < ipRun; i++) {
375             temp = addressData[i];
376             reverseIP[i] = addressData[(ipRun - 1) - i];
377             reverseIP[(ipRun - 1) - i] = temp;
378         }
379         return reverseIP;
380     }

400     /***
401      * This method return the InAddress for the given ip.
402      * 
403      * @param ipAddress -
404      *            ipAddress that should be processed
405      * @return the inAddress (in-addr or ip6)
406      * @throws PermErrorException
407      *             if the ipAddress is not valid (rfc conform)
408      */
409     public static String getInAddress(String ipAddress)
410             throws PermErrorException {
411         if (ipAddress == null) {
412             throw new PermErrorException(
413                     "IP is not a valid ipv4 or ipv6 address");
414         } else if (Inet6Util.isValidIPV4Address(ipAddress)) {
415             return "in-addr";
416         } else if (Inet6Util.isValidIP6Address(ipAddress)) {
417             return "ip6";
418         } else {
419             throw new PermErrorException(
420                     "IP is not a valid ipv4 or ipv6 address");
421         }
422     }

447     /***
448      * This method try to covnert an ip address to an easy readable ip. See
449      * http://java.sun.com/j2se/1.4.2/docs/api/java/net/Inet6Address.html for
450      * the format it returns. For ipv4 it make no convertion
451      * 
452      * @param ip
453      *            The ip which should be tried to convert
454      * @return ip The converted ip
455      */
456     public static String getReadableIP(String ip) {
457 
458         // Convert the ip if its an ipv6 ip. For ipv4 no conversion is needed
459         if (Inet6Util.isValidIP6Address(ip)) {
460             try {
461                 return Address.getByName(ip).getHostAddress();
462             } catch (UnknownHostException e) {
463                 // ignore this
464             }
465         }
466         return ip;
