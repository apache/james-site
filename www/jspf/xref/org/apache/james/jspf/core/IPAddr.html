118     /***
119      * create series of 16 bit masks for each ip block
120      * 
121      * @param maskLength
122      *            The netmask
123      */
124     public void setMask(int maskLength) {
125         int startMask;
126         int shift;
127         int maskSize;
128 
129         this.maskLength = maskLength;
130         if (ipLength == 4) {
131             if (!((maskLength > -1) && (maskLength < 33))) {
132                 maskLength = 32;
133             }
134             maskSize = 8;
135             startMask = (maskLength - 1) / maskSize;
136         } else {
137             if (!((maskLength > -1) && (maskLength < 129))) {
138                 maskLength = 128;
139             }
140             maskSize = 16;
141             startMask = (maskLength - 1) / maskSize;
142         }
143 
144         for (int i = 0; i < ipRun; i++) {
145             // full mask
146             if (i < startMask) {
147                 mask[i] = MASK16;
148                 // variable mask
149             } else if (i == startMask) {
150                 shift = ((i + 1) * maskSize) - maskLength;
151                 mask[i] = (MASK16 << shift) & MASK16;
152                 // no mask
153             } else {
154                 mask[i] = 0;
155             }
156         }
157     }

178     /***
179      * Convert ipAddress to a byte Array which represent the ipAddress
180      * 
181      * @param netAddress
182      *            The ipAddress we should convert
183      * @throws PermErrorException
184      *             on error
185      */
186     private void stringToInternal(String netAddress) throws PermErrorException {
187         netAddress = stripDot(netAddress);
188 
189         try {
190             byte[] bytes = Inet6Util.createByteArrayFromIPAddressString(netAddress);
191     
192             if (bytes.length == 4) {
193                 for (int i = 0; i < bytes.length; i++) {
194                     address[i] = bytes[i];
195                 }
196             } else if (bytes.length == 16) {
197                 setIP6Defaults();
198                 for (int i = 0; i < bytes.length / 2; i++) {
199                     address[i] = unsigned(bytes[i * 2]) * 256
200                             + unsigned(bytes[i * 2 + 1]);
201                 }
202             } else {
203                 throw new PermErrorException("Not a valid address: " + netAddress);
204             }
205         } catch (NumberFormatException e) {
206             throw new PermErrorException("Not a valid address: " + netAddress);
207         }
208     }

230     /***
231      * Get ip Address from given int Array
232      * 
233      * @param addressData
234      *            The int Array
235      * @return ipAddress The ipAddress
236      */
237     private String getIPAddress(int[] addressData) {
238         StringBuffer createAddress = new StringBuffer();
239         int[] workingAddress;
240 
241         // convert internal address to 8 bit
242         if (ipLength == 4) {
243             workingAddress = get8BitAddress(addressData);
244             // create IP string
245             createAddress.append(workingAddress[0]);
246             for (int i = 1; i < ipRun; i++) {
247                 createAddress.append(ipJoiner + workingAddress[i]);
248             }
249             // leave internal address as 16 bit
250         } else {
251             workingAddress = addressData;
252             // create IP string
253             createAddress.append(getHex(workingAddress[0]));
254             for (int i = 1; i < ipRun; i++) {
255                 createAddress.append(ipJoiner + getHex(workingAddress[i]));
256             }
257         }
258 
259         return createAddress.toString();
260     }

279     private String getNibbleFormat(int[] address) {
280         StringBuffer sb = new StringBuffer();
281         int[] ip = address;
282         for (int i = 0; i < ip.length; i++) {
283             String hex = getHex(ip[i]);
284             for (int j = 0; j < hex.length(); j++) {
285                 sb.append(hex.charAt(j));
286                 if (i != ip.length -1 || j != hex.length() -1) {
287                     sb.append(".");
288                 }
289             }
290         }
291         return sb.toString();
292     }

314     private int[] get8BitAddress(int[] addressData) {

340     /***
341      * Reverses internal address
342      * 
343      * @param addressData
344      *            The int array represent the ipAddress
345      * @return reverseIP
346      */
347     private int[] reverseIP(int[] addressData) {
348         int[] reverseIP = new int[ipLength];
349         int temp;
350         for (int i = 0; i < ipRun; i++) {
351             temp = addressData[i];
352             reverseIP[i] = addressData[(ipRun - 1) - i];
353             reverseIP[(ipRun - 1) - i] = temp;
354         }
355         return reverseIP;
356     }

432     public static String getReadableIP(String ip) {
433 
434         // Convert the ip if its an ipv6 ip. For ipv4 no conversion is needed
435         if (Inet6Util.isValidIP6Address(ip)) {
436             try {

450     /***
451      * This method convert the given ip to the proper format. Convertion will only done if the given ipAddress is ipv6 and ipv4-mapped
452      * 
453      * This must be done to correct handle IPv4-mapped-addresses. 
454      * See: http://java.sun.com/j2se/1.4.2/docs/api/java/net/Inet6Address.html
455      *
456      * Special IPv6 address:
457      *  IPv4-mapped address:     
458      *      Of the form::ffff:w.x.y.z, this IPv6 address is used to represent an IPv4 address. It allows 
459      *      the native program to use the same address data structure and also the same socket when 
460      *      communicating with both IPv4 and IPv6 nodes. In InetAddress and Inet6Address, it is used 
461      *      for internal representation; it has no functional role. Java will never return an IPv4-mapped address. 
462      *      These classes can take an IPv4-mapped address as input, both in byte array and text representation. 
463      *       However, it will be converted into an IPv4 address.
464      * @param ip the ipAddress to convert
465      * @return return converted ip
466      * @throws PermErrorException if the given ipAddress is invalid
467      */
468     public static String getProperIpAddress(String ip) throws PermErrorException {
469         if (isIPV6(ip) && isIPV4MappedIP(ip)) {
470             try {
471                 return getConvertedIP(ip);
472             } catch (UnknownHostException e) {
473                 throw new PermErrorException("Invalid ipAddress: " + ip);
474             }
475         }
476         return ip;
477         
478     }
