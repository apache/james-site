21  package org.apache.james.jspf.core;
22  
23  import org.apache.james.jspf.exceptions.NoneException;
24  import org.apache.james.jspf.exceptions.PermErrorException;
25  import org.apache.james.jspf.exceptions.TempErrorException;
26  
27  import java.util.List;
28  
29  /***
30   * Interface which should be used to access all necassary DNS-Records
31   *  
32   */
33  public interface DNSService {
34  
35      /***
36       * Get the SPF-Record for a server given it's version
37       * 
38       * TODO: support SPF Records too. This will be done if dnsjava support it!
39       * 
40       * @param hostname
41       *            The hostname for which we want to retrieve the SPF-Record
42       * @param spfVersion
43       *            The SPF-Version which should used.
44       * @return The SPF-Record if one is found.
45       * @throws PermErrorException
46       *             if more then one SPF-Record was found.
47       * @throws NoneException
48       *             if no SPF-Record was found.
49       * @throws TempErrorException
50       *             if the lookup result was "TRY_AGAIN"
51       */
52      public String getSpfRecord(String hostname, String spfVersion)
53              throws PermErrorException, NoneException, TempErrorException;
54  
55      /***
56       * Get a list of IPAddr's for a server using the mask length
57       * 
58       * @param strServer
59       *            The hostname or ipAddress whe should get the A-Records for
60       * @param mask
61       *            The netmask to use
62       * @return The ipAddresses
63       * @throws NoneException
64       *             if no A records was found
65       * @throws PermErrorException
66       *             if an PermError should be returned
67       * @throws TempErrorException
68       *             if the lookup result was "TRY_AGAIN"
69       */
70      public List getARecords(String strServer, int mask) throws NoneException,
71              PermErrorException, TempErrorException;
72  
73      /***
74       * Get a list of IPAddr's for a server using the mask length
75       * 
76       * @param strServer
77       *            The hostname or ipAddress whe should get the AAAA-Records for
78       * @param mask
79       *            The netmask to use
80       * @return The ipAddresses
81       * @throws NoneException
82       *             if no AAAA records was found
83       * @throws PermErrorException
84       *             if an PermError should be returned
85       * @throws TempErrorException
86       *             if the lookup result was "TRY_AGAIN"
87       */
88      public List getAAAARecords(String strServer, int mask)
89              throws NoneException, PermErrorException, TempErrorException;
90  
91      /***
92       * Get TXT records as a string
93       * 
94       * @param strServer
95       *            The hostname for which we want to retrieve the TXT-Record
96       * @return String which reflect the TXT-Record
97       * @throws NoneException
98       *             if no TXT-Record was found
99       * @throws PermErrorException
100      *             if the hostname is not resolvable
101      * @throws TempErrorException
102      *             if the lookup result was "TRY_AGAIN"
103      */
104     public String getTxtCatType(String strServer) throws NoneException,
105             PermErrorException, TempErrorException;
106 
107     /***
108      * Get reverse DNS records
109      * 
110      * @param ipAddress
111      *            The ipAddress for which we want to get the PTR-Record
112      * @return the PTR-Records
113      * @throws NoneException
114      *             if no PTR-Record was found
115      * @throws PermErrorException
116      *             if an PermError should be returned
117      * @throws TempErrorException
118      *             if the lookup result was "TRY_AGAIN"
119      */
120 
121     public List getPTRRecords(String ipAddress) throws PermErrorException,
122             NoneException, TempErrorException;
123 
124     /***
125      * Get a list of masked IPAddr MX-Records
126      * 
127      * @param domainName
128      *            The domainName or ipaddress we want to get the ips for
129      * @param mask
130      *            The netmask
131      * @return IPAddresses of the MX-Records
132      * @throws NoneException
133      *             if no MX-Record was found
134      * @throws PermErrorException
135      *             if an PermError should be returned
136      * @throws TempErrorException
137      *             if the lookup result was "TRY_AGAIN"
138      */
139     public List getMXRecords(String domainName, int mask)
140             throws PermErrorException, NoneException, TempErrorException;
141 
142     /***
143      * Try to get all domain names for the running host
144      * 
145      * @return names A List contains all domain names which could resolved
146      */
147     public List getLocalDomainNames();
148 
149     /***
150      * Set the timeout for DNS-Requests
151      * 
152      * @param timeOut
153      *            The timeout in seconds
154      */
155     public void setTimeOut(int timeOut);
156 }
