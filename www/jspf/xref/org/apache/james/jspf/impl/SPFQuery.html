20  package org.apache.james.jspf.impl;
21  
22  import org.apache.commons.cli.CommandLine;
23  import org.apache.commons.cli.CommandLineParser;
24  import org.apache.commons.cli.OptionBuilder;
25  import org.apache.commons.cli.Options;
26  import org.apache.commons.cli.ParseException;
27  import org.apache.commons.cli.PosixParser;
28  import org.apache.james.jspf.SPF;
29  import org.apache.james.jspf.SPF1Utils;
30  import org.apache.james.jspf.SPFResult;
31  import org.apache.log4j.ConsoleAppender;
32  import org.apache.log4j.Level;
33  import org.apache.log4j.Logger;
34  import org.apache.log4j.SimpleLayout;
35  
36  /***
37   * This class is used for commandline usage of JSPF
38   * 
39   */
40  public class SPFQuery {
41  
42      private final static int PASS_RCODE = 0;
43  
44      private final static int FAIL_RCODE = 1;
45  
46      private final static int SOFTFAIL_RCODE = 2;
47  
48      private final static int NEUTRAL_RCODE = 3;
49  
50      private final static int TEMP_ERROR_RCODE = 4;
51  
52      private final static int PERM_ERROR_RCODE = 5;
53  
54      private final static int NONE_RCODE = 6;
55  
56      private final static int UNKNOWN_RCODE = 255;
57  
58      private final static String CMD_IP = "ip";
59  
60      private final static String CMD_SENDER = "sender";
61  
62      private final static String CMD_HELO = "helo";
63  
64      private final static String CMD_DEBUG = "debug";
65  
66      private final static String CMD_VERBOSE = "verbose";
67  
68      private final static String CMD_DEFAULT_EXP = "defaultexplanation";
69  
70      private final static String CMD_BEST_GUESS = "usebestguess";
71      
72      private final static String CMD_TRUSTED_FORWARDER = "usetrusted";
73  
74      private static Logger logger = Logger.getRootLogger();
75  
76      /***
77       * @param args
78       *            The commandline arguments to parse
79       */
80      public static void main(String[] args) {
81  
82          String ip = null;
83          String sender = null;
84          String helo = null;
85          String defaultExplanation = null;
86          boolean useBestGuess = false;
87          boolean useTrustedForwarder = false;
88  
89          SimpleLayout layout = new SimpleLayout();
90          ConsoleAppender consoleAppender = new ConsoleAppender(layout);
91          logger.addAppender(consoleAppender);
92  
93          logger.setLevel(Level.ERROR);
94  
95          Options options = generateOptions();
96          CommandLineParser parser = new PosixParser();
97  
98          try {
99              CommandLine line = parser.parse(options, args);
100 
101             ip = line.getOptionValue(CMD_IP);
102             sender = line.getOptionValue(CMD_SENDER);
103             helo = line.getOptionValue(CMD_HELO);
104             defaultExplanation = line.getOptionValue(CMD_DEFAULT_EXP);
105             useBestGuess = line.hasOption(CMD_BEST_GUESS);
106             useTrustedForwarder = line.hasOption(CMD_TRUSTED_FORWARDER);
107             // check if all needed values was set
108             if (ip != null && sender != null && helo != null) {
109 
110                 if (line.hasOption(CMD_DEBUG))
111                     logger.setLevel(Level.DEBUG);
112                 if (line.hasOption(CMD_VERBOSE))
113                     logger.setLevel(Level.TRACE);
114 
115                 SPF spf = new DefaultSPF(new Log4JLogger(logger));
116 
117                 // Check if we should set a costum default explanation
118                 if (defaultExplanation != null) {
119                     spf.setDefaultExplanation(defaultExplanation);
120                 }
121 
122                 // Check if we should use best guess
123                 if (useBestGuess == true) {
124                     spf.setUseBestGuess(true);
125                 }
126                 
127                 if (useTrustedForwarder == true) {
128                     spf.setUseTrustedForwarder(true);
129                 }
130 
131                 SPFResult result = spf.checkSPF(ip, sender, helo);
132                 System.out.println(result.getResult());
133                 System.out.println(result.getHeader());
134                 System.exit(getReturnCode(result.getResult()));
135 
136             } else {
137                 usage();
138             }
139         } catch (ParseException e) {
140             usage();
141         }
142     }
143 
144     /***
145      * Return the generated Options
146      * 
147      * @return options
148      */
149     private static Options generateOptions() {
150         Options options = new Options();
151         options.addOption(OptionBuilder.withLongOpt(CMD_IP).withValueSeparator(
152                 '=').hasArg().create());
153         options.addOption(OptionBuilder.withLongOpt(CMD_SENDER)
154                 .withValueSeparator('=').hasArg().create());
155         options.addOption(OptionBuilder.withLongOpt(CMD_HELO)
156                 .withValueSeparator('=').hasArg().create());
157         options.addOption(OptionBuilder.withLongOpt(CMD_DEFAULT_EXP)
158                 .withValueSeparator('=').hasArg().create());
159         options.addOption(OptionBuilder.withLongOpt(CMD_BEST_GUESS).create());
160         options.addOption(OptionBuilder.withLongOpt(CMD_TRUSTED_FORWARDER).create());
161         options.addOption(OptionBuilder.withLongOpt(CMD_DEBUG).create());
162         options.addOption(OptionBuilder.withLongOpt(CMD_VERBOSE).create());
163         return options;
164     }
165 
166     /***
167      * Print out the usage
168      */
169     private static void usage() {
170         // TODO: Use HelpFormatter for printing usage
171         System.out.println("Usage: java -jar jspf-x.jar --" + CMD_IP
172                 + "=192.168.100.1 --" + CMD_SENDER + "=postmaster@foo.bar --"
173                 + CMD_HELO + "=foo.bar [--" + CMD_DEFAULT_EXP
174                 + "=\"explanation String\"] [--" + CMD_BEST_GUESS + "] [--"+ CMD_TRUSTED_FORWARDER +"]"
175                 + CMD_DEBUG + "] [--" + CMD_VERBOSE + "]");
176         System.exit(UNKNOWN_RCODE);
177     }
178 
179     /***
180      * Return the return code for the result
181      * 
182      * @param result
183      *            The result
184      * @return returnCode
185      */
186     private static int getReturnCode(String result) {
187 
188         if (result.equals(SPF1Utils.PASS_CONV)) {
189             return PASS_RCODE;
190         } else if (result.equals(SPF1Utils.FAIL_CONV)) {
191             return FAIL_RCODE;
192         } else if (result.equals(SPF1Utils.SOFTFAIL_CONV)) {
193             return SOFTFAIL_RCODE;
194         } else if (result.equals(SPF1Utils.NEUTRAL_CONV)) {
195             return NEUTRAL_RCODE;
196         } else if (result.equals(SPF1Utils.TEMP_ERROR_CONV)) {
197             return TEMP_ERROR_RCODE;
198         } else if (result.equals(SPF1Utils.PERM_ERROR_CONV)) {
199             return PERM_ERROR_RCODE;
200         } else if (result.equals(SPF1Utils.NONE_CONV)) {
201             return NONE_RCODE;
202         }
203 
204         return UNKNOWN_RCODE;
205     }
