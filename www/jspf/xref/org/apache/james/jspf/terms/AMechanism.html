43       * ABNF: A = "a" [ ":" domain-spec ] [ dual-cidr-length ]

55      /***
56       * 
57       * @see org.apache.james.jspf.core.GenericMechanism#run(org.apache.james.jspf.core.SPF1Data)
58       */
59      public boolean run(SPF1Data spfData) throws PermErrorException,
60              TempErrorException {
61          // update currentDepth
62          spfData.increaseCurrentDepth();
63  
64          // Get the right host.
65          String host = expandHost(spfData);
66  
67          // get the ipAddress
68          try {
69              if (Inet6Util.isValidIPV4Address(spfData.getIpAddress())) {
70  
71                  IPAddr checkAddress = IPAddr.getAddress(spfData.getIpAddress(),
72                          getIp4cidr());
73  
74                  List aRecords = getARecords(dnsService,host);
75       
76                  // no a records just return null
77                  if (aRecords == null) {
78                      return false;
79                  }
80  
81                  if (checkAddressList(checkAddress, aRecords, getIp4cidr())) {
82                      return true;
83                  }
84              } else {
85                  IPAddr checkAddress = IPAddr.getAddress(spfData.getIpAddress(),
86                          getIp6cidr());
87  
88                  List aaaaRecords = getAAAARecords(dnsService, host);
89                  
90                  // no aaaa records just return false
91                  if (aaaaRecords == null) {
92                      return false;
93                  }
94                  
95                  if (checkAddressList(checkAddress, aaaaRecords, getIp6cidr())) {
96                      return true;
97                  }
98  
99              }
100         // PermError / TempError
101         // TODO: Should we replace this with the "right" Exceptions ?
102         } catch (Exception e) {
103             log.debug("No valid ipAddress: ",e);
104             throw new PermErrorException("No valid ipAddress: "
105                     + spfData.getIpAddress());
106         }
107         // No match found
108         return false;
109     }

111     /***
112      * @see org.apache.james.jspf.terms.GenericMechanism#config(Configuration)
113      */
114     public synchronized void config(Configuration params) throws PermErrorException {
115         super.config(params);
116         if (params.groupCount() >= 2 && params.group(2) != null) {
117             ip4cidr = Integer.parseInt(params.group(2));
118             if (ip4cidr > 32) {
119                 throw new PermErrorException("Ivalid IP4 CIDR length");
120             }
121         } else {
122             ip4cidr = 32;
123         }
124         if (params.groupCount() >= 3 && params.group(3) != null) {
125             ip6cidr = Integer.parseInt(params.group(3).toString());
126             if (ip6cidr > 128) {
127                 throw new PermErrorException("Ivalid IP6 CIDR length");
128             }
129         } else {
130             ip6cidr = 128;
131         }
132     }

134     /***
135      * Check if the given ipaddress array contains the provided ip.
136      * 
137      * @param checkAddress
138      *            The ip wich should be contained in the given ArrayList
139      * @param addressList
140      *            The ip ArrayList.
141      * @return true or false
142      * @throws PermErrorException 
143      */
144     public boolean checkAddressList(IPAddr checkAddress, List addressList, int cidr) throws PermErrorException {
145 
146         for (int i = 0; i < addressList.size(); i++) {
147             String ip = (String) addressList.get(i);
148 
149             // Check for empty record
150             if (ip != null) {
151                 // set the mask in the address.
152                 // TODO should we use cidr from the parameters or the input checkAddress cidr?
153                 IPAddr ipAddr = IPAddr.getAddress(ip, checkAddress.getMaskLength());
154                 if (checkAddress.getMaskedIPAddress().equals(
155                         ipAddr.getMaskedIPAddress())) {
156                     return true;
157                 }
158             }
159         }
160         return false;
161     }

187     protected String toString(String mechKey) {
188         StringBuffer res = new StringBuffer();
189         res.append(mechKey);
190         if (getDomain() != null) {
191             res.append(":"+getDomain());
192         }
193         if (getIp4cidr() != 32) {
194             res.append("/"+getIp4cidr());
195         }
196         if (getIp6cidr() != 128) {
197             res.append("//"+getIp4cidr());
198         }
199         return res.toString();
200     }

203     /***
204      * Retrieve a list of AAAA records
205      */
206     public List getAAAARecords(DNSService dns, String strServer)
207             throws PermErrorException, TempErrorException {
208         List listAAAAData;
209         if (IPAddr.isIPV6(strServer)) {
210             // Address is already an IP address, so add it to list
211             listAAAAData = new ArrayList();
212             listAAAAData.add(strServer);
213         } else {
214             try {
215                 listAAAAData = dns.getRecords(strServer, DNSService.AAAA);
216             } catch (DNSService.TimeoutException e) {
217                 throw new TempErrorException("Timeout querying dns server");
218             }
219         }
220         return listAAAAData;
221     }

224     /***
225      * Get a list of IPAddr's for a server
226      * 
227      * @params dns the DNSService to query
228      * @param strServer
229      *            The hostname or ipAddress whe should get the A-Records for
230      * @return The ipAddresses
231      * @throws PermErrorException
232      *             if an PermError should be returned
233      * @throws TempErrorException
234      *             if the lookup result was "TRY_AGAIN"
235      */
236     public List getARecords(DNSService dns, String strServer) throws PermErrorException, TempErrorException {
237         List listAData;
238         if (IPAddr.isIPAddr(strServer)) {
239             listAData = new ArrayList();
240             listAData.add(strServer);
241         } else {
242             try {
243                 listAData = dns.getRecords(strServer, DNSService.A);
244             } catch (DNSService.TimeoutException e) {
245                 throw new TempErrorException("Timeout querying dns server");
246             }
247         }
248         return listAData;
249     }
