24  import org.apache.james.jspf.core.SPF1Constants;
25  import org.apache.james.jspf.core.SPF1Data;

27  import org.apache.james.jspf.exceptions.TempErrorException;
28  import org.apache.james.jspf.macro.MacroExpand;
29  import org.apache.james.jspf.util.SPFTermsRegexps;
30  import org.apache.james.jspf.wiring.DNSServiceEnabled;

32  import java.util.List;

34  /***
35   * This class represent the exp modifier
36   * 
37   */
38  public class ExpModifier extends GenericModifier implements DNSServiceEnabled {
39  
40      /***
41       * ABNF: explanation = "exp" "=" domain-spec
42       */
43      public static final String REGEX = "[eE][xX][pP]" + "//="
44              + SPFTermsRegexps.DOMAIN_SPEC_REGEX;
45  
46      private DNSService dnsService;
47  
48      /***
49       * Generate the explanation and set it in SPF1Data so it can be accessed
50       * easy later if needed
51       * 
52       * @param spfData
53       *            The SPF1Data which should used
54       * @throws PermErrorException 
55       */
56      protected void checkSPFLogged(SPF1Data spfData) throws PermErrorException {
57          String exp = null;
58          String host = getHost();
59  
60          // If we should ignore the explanation we don't have to run this class
61          if (spfData.ignoreExplanation() == true)
62              return;
63          
64          // If the currentResult is not fail we have no need to run all these
65          // methods!
66          if (spfData.getCurrentResult()== null || !spfData.getCurrentResult().equals(SPF1Constants.FAIL))
67              return;
68  
69          host = new MacroExpand(spfData, log).expandDomain(host);
70  
71          try {
72              try {
73                  exp = getTxtType(dnsService, host);
74              } catch (TempErrorException e) {
75                  // Nothing todo here.. just return null
76                  return;
77              }
78  
79              if ((exp != null) && (!exp.equals(""))) {
80                  spfData.setExplanation(new MacroExpand(spfData, log)
81                          .expandExplanation(exp));
82              } 
83          } catch (PermErrorException e) {
84              // TODO add logging here!
85              // Only catch the error and return null
86              return;
87          }
88          return;
89      }
90  
91  
92      /***
93       * Get TXT records as a string
94       * 
95       * @param dns The DNSService to query
96       * @param strServer
97       *            The hostname for which we want to retrieve the TXT-Record
98       * @return String which reflect the TXT-Record
99       * @throws PermErrorException
100      *             if more then one TXT-Record for explanation was found
101      * @throws TempErrorException
102      *             if the lookup result was "TRY_AGAIN"
103      */
104     public String getTxtType(DNSService dns, String strServer) throws TempErrorException, PermErrorException {
105         try {
106             List records = dns.getRecords(strServer, DNSService.TXT);
107         
108             if (records == null) {
109                 return null;
110             }
111     
112             // See SPF-Spec 6.2
113             //
114             // If domain-spec is empty, or there are any DNS processing errors (any RCODE other than 0), 
115             // or if no records are returned, or if more than one record is returned, or if there are syntax 
116             // errors in the explanation string, then proceed as if no exp modifier was given.   
117             if (records.size() > 1) {
118                 log.debug("More then one TXT-Record found for explanation");
119                 throw new PermErrorException("More then one TXT-Record for explanation");
120             } else {
121                 return (String) records.get(0);
122             }
123         } catch (DNSService.TimeoutException e) {
124             throw new TempErrorException("Timeout querying dns server");
125         }
126     }
127     
128     /***
129      * @see java.lang.Object#toString()
130      */
131     public String toString() {
132        return "exp="+getHost();
133     }
134 
135     /***
136      * @see org.apache.james.jspf.wiring.DNSServiceEnabled#enableDNSService(org.apache.james.jspf.core.DNSService)
137      */
138     public void enableDNSService(DNSService service) {
139         this.dnsService = service;
140     }
