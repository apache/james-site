21  package org.apache.james.jspf.terms;
22  
23  import org.apache.james.jspf.core.IPAddr;
24  import org.apache.james.jspf.core.SPF1Data;
25  import org.apache.james.jspf.exceptions.PermErrorException;
26  import org.apache.james.jspf.util.Inet6Util;
27  import org.apache.james.jspf.util.ConfigurationMatch;
28  
29  /***
30   * This class represent the ip4 mechanism
31   * 
32   */
33  public class IP4Mechanism extends GenericMechanism {
34  
35      /***
36       * ABNF: IP4 = "ip4" ":" ip4-network [ ip4-cidr-length ]
37       */
38      public static final String REGEX = "[iI][pP][4]" + "//:" + "([0-9.]+)"
39              + "(?:" + IP4_CIDR_LENGTH_REGEX + ")?";
40  
41      private IPAddr ip = null;
42  
43      /***
44       * 
45       * @see org.apache.james.jspf.core.GenericMechanism#run(org.apache.james.jspf.core.SPF1Data)
46       */
47      public boolean run(SPF1Data spfData) throws PermErrorException {
48          IPAddr originalIP;
49  
50          originalIP = IPAddr.getAddress(spfData.getIpAddress(), getIp()
51                  .getMaskLength());
52  
53          if (getIp().getMaskedIPAddress().equals(originalIP.getMaskedIPAddress())) {
54              return true;
55          } else {
56              // No match
57              return false;
58          }
59      }
60  
61      /***
62       * @see org.apache.james.jspf.terms.GenericMechanism#config(ConfigurationMatch)
63       */
64      public synchronized void config(ConfigurationMatch params) throws PermErrorException {
65          if (params.groupCount() == 0) {
66              throw new PermErrorException("Missing ip");
67          }
68          String ipString = params.group(1);
69          if (!isValidAddress(ipString)) {
70              throw new PermErrorException("Invalid Address: " + ipString);
71          }
72          int maskLength = getMaxCidr();
73          if (params.groupCount() >= 2 && params.group(2) != null) {
74              String maskLengthString = params.group(2).toString();
75              maskLength = Integer.parseInt(maskLengthString);
76  
77              if (maskLength > getMaxCidr() || (maskLengthString.length() > 1 && maskLengthString.startsWith("0"))) {
78                  throw new PermErrorException("Invalid CIDR: " + maskLengthString);
79              }
80          }
81          ip = IPAddr.getAddress(ipString, maskLength);
82      }
83  
84      /***
85       * @see org.apache.james.jspf.util.Inet6Util#isValidIPV4Address(String)
86       */
87      protected boolean isValidAddress(String ipString) {
88          return Inet6Util.isValidIPV4Address(ipString);
89      }
90  
91      /***
92       * Returns the max cidr for ip4
93       * 
94       * @return maxCidr The max cidr
95       */
96      protected int getMaxCidr() {
97          return 32;
98      }
99  
100     /***
101      * @return Returns the ip.
102      */
103     protected synchronized IPAddr getIp() {
104         return ip;
105     }
106 
107 }
