18  package org.apache.james.jspf.terms;
19  
20  import org.apache.james.jspf.SPF;
21  import org.apache.james.jspf.core.SPF1Data;
22  import org.apache.james.jspf.exceptions.NoneException;
23  import org.apache.james.jspf.exceptions.PermErrorException;
24  import org.apache.james.jspf.exceptions.TempErrorException;
25  import org.apache.james.jspf.macro.MacroExpand;
26  import org.apache.james.jspf.parser.SPF1Parser;
27  
28  /***
29   * This class represent the redirect modifier
30   * 
31   */
32  public class RedirectModifier extends GenericModifier {
33  
34      /***
35       * ABNF: redirect = "redirect" "=" domain-spec
36       */
37      public static final String REGEX = "[rR][eE][dD][iI][rR][eE][cC][tT]"
38              + "//=" + SPF1Parser.DOMAIN_SPEC_REGEX;
39  
40      /***
41       * Set the host which should be used for redirection and set it in SPF1Data
42       * so it can be accessed easy later if needed
43       * 
44       * @param spfData
45       *            The SPF1Data which should used
46       * @return host The host to which we shuld redirect
47       * @throws PermErrorException
48       *             if an error is in the redirect modifier
49       * @throws TempErrorException
50       *             if an DNS problem accurred
51       */
52      public String run(SPF1Data spfData) throws PermErrorException,
53              TempErrorException {
54          // the redirect modifier is used only when we had no previous matches
55          if (!spfData.isMatch()) {
56  
57              String host = getHost();
58  
59              // update currentDepth
60              spfData.setCurrentDepth(spfData.getCurrentDepth() + 1);
61  
62              try {
63                  host = new MacroExpand(spfData).expandDomain(host);
64              } catch (Exception e) {
65                  throw new PermErrorException("Error in redirect modifier: "
66                          + host);
67              }
68  
69              spfData.setCurrentDomain(host);
70  
71              String res = null;
72              try {
73                  res = new SPF(spfData.getDnsProbe()).checkSPF(spfData).getResultChar();
74                  
75  
76              } catch (NoneException e) {
77                  // no spf record assigned to the redirect domain
78                  throw new PermErrorException(
79                          "included checkSPF returned NoneException");
80              } finally {
81                  //After the redirect we should not use the explanation from the orginal record
82                  spfData.setIgnoreExplanation(true);
83              }
84  
85              return res;
86  
87          } else {
88              // return null if we should not use the redirect at all
89              return null;
90          }
91      }
92  
93      /***
94       * @see org.apache.james.jspf.core.Modifier#enforceSingleInstance()
95       */
96      public boolean enforceSingleInstance() {
97          return true;
98      }
99  
100 }
