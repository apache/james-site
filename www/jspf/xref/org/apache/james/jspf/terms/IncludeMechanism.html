21  import org.apache.james.jspf.core.Configurable;
22  import org.apache.james.jspf.core.Mechanism;
23  import org.apache.james.jspf.core.SPF1Constants;
24  import org.apache.james.jspf.core.SPF1Data;
25  import org.apache.james.jspf.exceptions.NoneException;
26  import org.apache.james.jspf.exceptions.PermErrorException;
27  import org.apache.james.jspf.exceptions.TempErrorException;
28  import org.apache.james.jspf.macro.MacroExpand;
29  import org.apache.james.jspf.parser.SPF1Parser;
30  import org.apache.james.jspf.util.ConfigurationMatch;
31  
32  /***
33   * This class represent the incude mechanism
34   * 
35   */
36  public class IncludeMechanism implements Mechanism, Configurable {
37  
38      /***
39       * ABNF: include = "include" ":" domain-spec
40       */
41      public static final String REGEX = "[iI][nN][cC][lL][uU][dD][eE]" + "//:"
42              + SPF1Parser.DOMAIN_SPEC_REGEX;
43  
44      private String host;
45  
46      /***
47       * Set the host which should be used for include
48       * 
49       * @param spfData
50       *            The SPF1Data which should used
51       * @return The host which should be included
52       * @throws PermErrorException
53       *             if an error is in the redirect modifier
54       * @throws TempErrorException 
55       * @throws NoneException 
56       */
57      public boolean run(SPF1Data spfData) throws PermErrorException, TempErrorException, NoneException {
58          String host = getHost();
59  
60          // update currentDepth
61          spfData.setCurrentDepth(spfData.getCurrentDepth() + 1);      
62          
63          try {
64              host = new MacroExpand(spfData).expandDomain(host);
65          } catch (Exception e) {
66              throw new PermErrorException("Error in include modifier: " + host);
67          }
68              
69          spfData.setCurrentDomain(host);
70          
71          // On includes we should not use the explanation of the included domain
72          spfData.setIgnoreExplanation(true);
73          
74          String res = null;
75          try {
76              res = new SPF(spfData.getDnsProbe()).checkSPF(spfData).getResultChar();
77          } catch (NoneException e) {
78              throw new PermErrorException("included checkSPF returned NoneException");
79          }
80          
81          // Reset the ignore
82          spfData.setIgnoreExplanation(false);
83          
84          if (res == null) {
85              throw new TempErrorException("included checkSPF returned null");
86          } else if (res.equals(SPF1Constants.PASS)) {
87              return true;
88          } else if (res.equals(SPF1Constants.FAIL) || res.equals(SPF1Constants.SOFTFAIL) || res.equals(SPF1Constants.NEUTRAL)) {
89              return false;
90          } else {
91              throw new TempErrorException("included checkSPF returned an Illegal result");
92          }
93              
94      }
95  
96      /***
97       * @see org.apache.james.jspf.core.Configurable#config(ConfigurationMatch)
98       */
99      public synchronized void config(ConfigurationMatch params) throws PermErrorException {
100         if (params.groupCount() == 0) {
101             throw new PermErrorException("Include mechanism without an host");
102         }
103         host = params.group(1);
104     }
105 
106     /***
107      * @return Returns the host.
108      */
109     protected synchronized String getHost() {
110         return host;
111     }
