38  /***
39   * This class represent the incude mechanism
40   * 
41   */
42  public class IncludeMechanism implements Mechanism, Configurable, LogEnabled {
43  
44      /***
45       * ABNF: include = "include" ":" domain-spec
46       */
47      public static final String REGEX = "[iI][nN][cC][lL][uU][dD][eE]" + "//:"
48              + SPF1Parser.DOMAIN_SPEC_REGEX;
49  
50      private String host;
51      
52      private Logger log;
53  
54      /***
55       * Set the host which should be used for include
56       * 
57       * @param spfData
58       *            The SPF1Data which should used
59       * @return The host which should be included
60       * @throws PermErrorException
61       *             if an error is in the redirect modifier
62       * @throws TempErrorException 
63       *             if the dns return a temp error
64       */
65      public boolean run(SPF1Data spfData) throws PermErrorException, TempErrorException {
66          String host = getHost();
67  
68          // update currentDepth
69          spfData.setCurrentDepth(spfData.getCurrentDepth() + 1);      
70          
71          try {
72              host = new MacroExpand(spfData, log).expandDomain(host);
73          } catch (PermErrorException e) {
74              throw new PermErrorException("Error in include modifier: " + host);
75          }
76              
77          spfData.setCurrentDomain(host);
78          
79          // On includes we should not use the explanation of the included domain
80          spfData.setIgnoreExplanation(true);
81          
82          String res = null;
83          try {
84               res = new SPF(spfData.getDnsProbe(),log).checkSPF(spfData).getResultChar();
85            
86          } catch (NoneException e) {
87              throw new PermErrorException("included checkSPF returned NoneException");
88          } catch (NeutralException e) {
89              throw new PermErrorException("included checkSPF returned NeutralException");
90          }
91          
92          // Reset the ignore
93          spfData.setIgnoreExplanation(false);
94          
95          if (res == null) {
96              throw new TempErrorException("included checkSPF returned null");
97          } else if (res.equals(SPF1Constants.PASS)) {
98              return true;
99          } else if (res.equals(SPF1Constants.FAIL) || res.equals(SPF1Constants.SOFTFAIL) || res.equals(SPF1Constants.NEUTRAL)) {
100             return false;
101         } else {
102             throw new TempErrorException("included checkSPF returned an Illegal result");
103         }
104             
105     }
106 
107     /***
108      * @see org.apache.james.jspf.core.Configurable#config(ConfigurationMatch)
109      */
110     public synchronized void config(ConfigurationMatch params) throws PermErrorException {
111         if (params.groupCount() == 0) {
112             throw new PermErrorException("Include mechanism without an host");
113         }
114         host = params.group(1);
115     }
116 
117     /***
118      * @return Returns the host.
119      */
120     protected synchronized String getHost() {
121         return host;
122     }
123 
124     /***
125      * @see org.apache.james.jspf.core.LogEnabled#enableLogging(org.apache.james.jspf.core.Logger)
126      */
127     public void enableLogging(Logger logger) {
128         this.log = logger;
129     }
