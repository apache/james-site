39  /***
40   * This class represent the incude mechanism
41   * 
42   */
43  public class IncludeMechanism implements Mechanism, Configurable, LogEnabled, SPFCheckEnabled {
44  
45      /***
46       * ABNF: include = "include" ":" domain-spec
47       */
48      public static final String REGEX = "[iI][nN][cC][lL][uU][dD][eE]" + "//:"
49              + SPFTermsRegexps.DOMAIN_SPEC_REGEX;
50  
51      protected String host;
52      
53      protected Logger log;
54  
55      private SPFChecker spfChecker;
56  
57      /***
58       * Set the host which should be used for include
59       * 
60       * @param spfData
61       *            The SPF1Data which should used
62       * @return The host which should be included
63       * @throws PermErrorException
64       *             if an error is in the redirect modifier
65       * @throws TempErrorException 
66       *             if the dns return a temp error
67       */
68      public boolean run(SPF1Data spfData) throws PermErrorException, TempErrorException {
69          String host = getHost();
70  
71          // update currentDepth
72          spfData.increaseCurrentDepth();      
73          
74          // throws a PermErrorException that we can pass through
75          host = new MacroExpand(spfData, log).expandDomain(host);
76  
77          String prevRes = spfData.getCurrentResult();
78          String prevHost = spfData.getCurrentDomain();
79          
80          try {
81      
82              spfData.setCurrentDomain(host);
83              
84              // On includes we should not use the explanation of the included domain
85              spfData.setIgnoreExplanation(true);
86              // set a null current result
87              spfData.setCurrentResult(null);
88              
89              try {
90                   spfChecker.checkSPF(spfData);
91                
92              } catch (NoneException e) {
93                  throw new PermErrorException("included checkSPF returned NoneException");
94              } catch (NeutralException e) {
95                  throw new PermErrorException("included checkSPF returned NeutralException");
96              }
97              
98              if (spfData.getCurrentResult() == null) {
99                  throw new TempErrorException("included checkSPF returned null");
100             } else if (spfData.getCurrentResult().equals(SPF1Constants.PASS)) {
101                 return true;
102             } else if (spfData.getCurrentResult().equals(SPF1Constants.FAIL) || spfData.getCurrentResult().equals(SPF1Constants.SOFTFAIL) || spfData.getCurrentResult().equals(SPF1Constants.NEUTRAL)) {
103                 return false;
104             } else {
105                 throw new TempErrorException("included checkSPF returned an Illegal result");
106             }
107         } finally {
108             // Reset the ignore
109             spfData.setIgnoreExplanation(false);
110             spfData.setCurrentDomain(prevHost);
111             spfData.setCurrentResult(prevRes);
112         }
113             
114     }
115 
116     /***
117      * @see org.apache.james.jspf.core.Configurable#config(Configuration)
118      */
119     public synchronized void config(Configuration params) throws PermErrorException {
120         if (params.groupCount() == 0) {
121             throw new PermErrorException("Include mechanism without an host");
122         }
123         host = params.group(1);
124     }
125 
126     /***
127      * @return Returns the host.
128      */
129     protected synchronized String getHost() {
130         return host;
131     }
132 
133     /***
134      * @see org.apache.james.jspf.wiring.LogEnabled#enableLogging(org.apache.james.jspf.core.Logger)
135      */
136     public void enableLogging(Logger logger) {
137         this.log = logger;
138     }
139 
140     /***
141      * @see java.lang.Object#toString()
142      */
143     public String toString() {
144         return "include:"+getHost();
145     }
146 
147     /***
148      * @see org.apache.james.jspf.wiring.SPFCheckEnabled#enableSPFChecking(org.apache.james.jspf.core.SPFChecker)
149      */
150     public void enableSPFChecking(SPFChecker checker) {
151         this.spfChecker = checker;
152     }
