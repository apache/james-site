69       * Uses passed logger and passed dnsServicer

72       * @param logger the logger to use
73       */
74      public SPF(DNSService dnsProbe, Logger logger) {
75          super();
76          this.dnsProbe = dnsProbe;

78          this.log = logger;

81      /***
82       * Run check for SPF with the given values.
83       * 
84       * @param ipAddress
85       *            The ipAddress the connection is comming from
86       * @param mailFrom
87       *            The mailFrom which was provided
88       * @param hostName
89       *            The hostname which was provided as HELO/EHLO
90       * @return result The SPFResult
91       */
92      public SPFResult checkSPF(String ipAddress, String mailFrom, String hostName) {
93          SPF1Data spfData = null;
94          String result = null;
95          String resultChar = null;
96          String explanation = null;
97  
98          try {
99              // Setup the data
100             spfData = new SPF1Data(mailFrom, hostName, ipAddress, dnsProbe);
101             SPFInternalResult res = checkSPF(spfData);
102             resultChar = res.getResultChar();
103             result = SPF1Utils.resultToName(resultChar);
104             explanation = res.getExplanation();
105         } catch (PermErrorException e) {
106             log.warn(e.getMessage(),e);
107             result = SPF1Utils.PERM_ERROR_CONV;
108         } catch (NoneException e) {
109             log.warn(e.getMessage(),e);
110             result = SPF1Utils.NONE_CONV;
111         } catch (NeutralException e) {
112             result = SPF1Utils.NEUTRAL_CONV;
113         } catch (TempErrorException e) {
114             log.warn(e.getMessage(),e);
115             result = SPF1Utils.TEMP_ERROR_CONV;
116         } catch (IllegalStateException e) {
117             // this should never happen at all. But anyway we will set the
118             // result to neutral. Safety first ..
119             log.error(e.getMessage(),e);
120             result = SPF1Constants.NEUTRAL;
121         }
122 
123         SPFResult ret = new SPFResult(result, resultChar, explanation, spfData);
124         
125         log.info("[ipAddress=" + ipAddress + "] [mailFrom=" + mailFrom
126                 + "] [helo=" + hostName + "] => " + ret.getResult());
127 
128         return ret;
129 
130     }

132     /***
133      * Run check for SPF with the given values.
134      * 
135      * @param spfData
136      *             The SPF1Data which should be used to run the check
137      * @return result 
138      *             The SPFInternalResult 
139      * @throws PermErrorException
140      *             Get thrown if an error was detected
141      * @throws NoneException
142      *             Get thrown if no Record was found
143      * @throws TempErrorException
144      *             Get thrown if a DNS problem was detected
145      * @throws NeutralException  
146      *             Get thrown if the result should be neutral
147      */
148     public SPFInternalResult checkSPF(SPF1Data spfData) throws PermErrorException,
149             NoneException, TempErrorException, NeutralException {
150         String result = SPF1Constants.NEUTRAL;
151         String explanation = null;
152 
153         // Get the raw dns txt entry which contains a spf entry
154         String spfDnsEntry = dnsProbe.getSpfRecord(spfData.getCurrentDomain(),
155                 SPF1Constants.SPF_VERSION);
156 
157         // logging
158         log.debug("Start parsing SPF-Record:" + spfDnsEntry);
159 
160         SPF1Record spfRecord = parser.parse(spfDnsEntry);
161 
162         String qualifier = null;
163         boolean hasCommand = false;
164 
165         // get all commands
166         Iterator com = spfRecord.getDirectives().iterator();
167         while (com.hasNext()) {
168 
169             // if we reach maximum calls we must throw a PermErrorException. See
170             // SPF-RFC Section 10.1. Processing Limits
171             if (spfData.getCurrentDepth() > spfData.getMaxDepth()) {
172                 throw new PermErrorException(
173                         "Maximum mechanism/modifier calls done: "
174                                 + spfData.getCurrentDepth());
175             }
176 
177             hasCommand = true;
178             Directive d = (Directive) com.next();
179 
180             // logging
181             log.debug("Processing directive: " + d.getQualifier()
182                     + d.getMechanism().toString());
183 
184             qualifier = d.run(spfData);
185 
186             // logging
187             log.debug("Processed directive: " + d.getQualifier()
188                     + d.getMechanism().toString() + " returned " + qualifier);
189 
190             if (qualifier != null) {
191                 if (qualifier.equals("")) {
192                     result = SPF1Constants.PASS;
193                 } else {
194                     result = qualifier;
195                 }
196 
197                 spfData.setCurrentResult(result);
198                 spfData.setMatch(true);
199 
200                 // If we have a match we should break the while loop
201                 break;
202             }
203         }
204 
205         Iterator mod = spfRecord.getModifiers().iterator();
206         while (mod.hasNext()) {
207             spfData.setCurrentDepth(spfData.getCurrentDepth() + 1);
208 
209             // if we reach maximum calls we must throw a PermErrorException. See
210             // SPF-RFC Section 10.1. Processing Limits
211             if (spfData.getCurrentDepth() > spfData.getMaxDepth()) {
212                 throw new PermErrorException(
213                         "Maximum mechanism/modifiers calls done: "
214                                 + spfData.getCurrentDepth());
215             }
216 
217             Modifier m = (Modifier) mod.next();
218 
219             log.debug("Processing modifier: " + m.toString());
220 
221             String q = m.run(spfData);
222 
223             log.debug("Processed modifier: " + m.toString() + " resulted in "
224                     + q);
225 
226             if (q != null) {
227                 qualifier = q;
228             }
229 
230             if (qualifier != null) {
231                 result = qualifier;
232 
233                 spfData.setCurrentResult(result);
234                 spfData.setMatch(true);
235             }
236         }
237 
238         // If no match was found set the result to neutral
239         if (!spfData.isMatch() && (hasCommand == true)) {
240             result = SPF1Constants.NEUTRAL;
241         } 
242         
243         if (result.equals(SPF1Constants.FAIL)) {  
244             if (spfData.getExplanation()==null || spfData.getExplanation().equals("")) {
245                 try {
246                     spfData.setExplanation(new MacroExpand(spfData, log)
247                             .expandExplanation(SPF1Utils.DEFAULT_EXPLANATION));
248                 } catch (PermErrorException e) {}
249             }
250             explanation = spfData.getExplanation();
251         }
252         
253         return new SPFInternalResult(result, explanation);
254     }
