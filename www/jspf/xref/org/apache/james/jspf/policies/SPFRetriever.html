57      /***
58       * Get the SPF-Record for a server
59       * 
60       * @param dns
61       *            The dns service to query
62       * @param hostname
63       *            The hostname for which we want to retrieve the SPF-Record
64       * @param spfVersion
65       *            The SPF-Version which should used.
66       * @return The SPF-Record if one is found.
67       * @throws PermErrorException
68       *             if more then one SPF-Record was found.
69       * @throws TempErrorException
70       *             if the lookup result was "TRY_AGAIN"
71       */
72      protected String retrieveSpfRecord(String hostname)
73              throws PermErrorException, TempErrorException {
74  
75          try {
76              // first check for SPF-Type records
77              List spfR = dns.getRecords(hostname, DNSService.SPF);
78              
79              if (spfR == null || spfR.isEmpty()) {
80                  // do DNS lookup for TXT
81                  spfR = dns.getRecords(hostname, DNSService.TXT);
82              }
83      
84              // process returned records
85              if (spfR != null && !spfR.isEmpty()) {
86                  return extractSPFRecord(spfR);
87              } else {
88                  return null;
89              }
90          } catch (DNSService.TimeoutException e) {
91              throw new TempErrorException("Timeout querying dns");
92          }
93      }
94      
95      /***
96       * Return the extracted SPF-Record 
97       *  
98       * @param spfR the List which holds TXT/SPF - Records
99       * @return returnValue the extracted SPF-Record
100      * @throws PermErrorException if more then one SPF - Record was found in the 
101      *                            given List.
102      */
103     protected String extractSPFRecord(List spfR) throws PermErrorException {
104        String returnValue = null;
105         Iterator all = spfR.iterator();
106            
107         while (all.hasNext()) {
108             // DO NOT trim the result!
109             String compare = all.next().toString();
110 
111             // TODO is this correct? we remove the first and last char if the
112             // result has an initial " 
113             // remove '"'
114             if (compare.charAt(0)=='"') {
115                 compare = compare.toLowerCase().substring(1,
116                         compare.length() - 1);
117             }
118 
119             // We trim the compare value only for the comparison
120             if (compare.toLowerCase().trim().startsWith(SPF1Constants.SPF_VERSION + " ") || compare.trim().equalsIgnoreCase(SPF1Constants.SPF_VERSION)) {
121                 if (returnValue == null) {
122                     returnValue = compare;
123                 } else {
124                     throw new PermErrorException(
125                             "More than 1 SPF record found");
126                 }
127             }
128         }
129         
130         return returnValue;
131     }
