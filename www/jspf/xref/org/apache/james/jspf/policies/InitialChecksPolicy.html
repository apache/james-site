20  package org.apache.james.jspf.policies;
21  
22  import org.apache.james.jspf.core.SPF1Record;
23  import org.apache.james.jspf.exceptions.NeutralException;
24  import org.apache.james.jspf.exceptions.NoneException;
25  import org.apache.james.jspf.exceptions.PermErrorException;
26  import org.apache.james.jspf.exceptions.TempErrorException;
27  
28  /***
29   * Run the checks on the validity of the domain
30   * This is an override filter to be executed as the first 
31   * so it should be added as the last filter.
32   */
33  public final class InitialChecksPolicy implements Policy {
34      
35      /***
36       * @see org.apache.james.jspf.policies.Policy#getSPFRecord(java.lang.String)
37       */
38      public SPF1Record getSPFRecord(String currentDomain) throws PermErrorException, TempErrorException, NoneException, NeutralException {
39          // Initial checks (spec 4.3)
40          if (currentDomain != null) {
41              String[] labels = currentDomain.split("//.");
42              for (int i = 0; i < labels.length; i++) {
43                  if (labels[i] != null && labels[i].length() > 63) {
44                      throw new NoneException("Domain "+currentDomain+" is malformed (label longer than 63 characters)");
45                  }
46              }
47          }
48          return null;
49      }
50  }
