20  package org.apache.james.jspf.policies;
21  
22  import org.apache.james.jspf.core.SPF1Record;
23  import org.apache.james.jspf.exceptions.NeutralException;
24  import org.apache.james.jspf.exceptions.NoneException;
25  import org.apache.james.jspf.exceptions.PermErrorException;
26  import org.apache.james.jspf.exceptions.TempErrorException;
27  
28  import java.util.Iterator;
29  import java.util.List;
30  
31  /***
32   * Composed policy: get a list of NestedPolicies and chain them
33   * in a single policy
34   */
35  public class ChainPolicy implements Policy {
36      
37      private List policies;
38      private List policyFilters;
39  
40      /***
41       * Create a new ChainPolicy
42       * @param policies an array of Polcy and NestedPolicy objects
43       */
44      public ChainPolicy(List policies, List policyFilters) {
45          this.policies = policies;
46          this.policyFilters = policyFilters;
47      }
48      
49      /***
50       * @see org.apache.james.jspf.policies.Policy#getSPFRecord(java.lang.String)
51       */
52      public SPF1Record getSPFRecord(String currentDomain)
53              throws PermErrorException, TempErrorException, NoneException,
54              NeutralException {
55          SPF1Record res = null;
56          Iterator i = policies.iterator();
57          while (i.hasNext() && res == null) {
58              Policy newP = (Policy) i.next();
59              res = newP.getSPFRecord(currentDomain);
60          }
61          Iterator i2 = policyFilters.iterator();
62          while (i2.hasNext()) {
63              PolicyPostFilter newP = (PolicyPostFilter) i2.next();
64              res = newP.getSPFRecord(currentDomain, res);
65          }
66          return res;
67      }

69  }
