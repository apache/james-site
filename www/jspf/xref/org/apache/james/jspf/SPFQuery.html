18  package org.apache.james.jspf;
19  
20  import org.apache.log4j.ConsoleAppender;
21  import org.apache.log4j.Level;
22  import org.apache.log4j.Logger;
23  import org.apache.log4j.SimpleLayout;
24  
25  /***
26   * This class is used for commandline usage of JSPF
27   * 
28   */
29  public class SPFQuery {
30  
31      private final static int PASS_RCODE = 0;
32      
33      private final static int FAIL_RCODE =  1;
34      
35      private final static int SOFTFAIL_RCODE = 2;
36      
37      private final static int NEUTRAL_RCODE = 3;
38      
39      private final static int TEMP_ERROR_RCODE = 4;
40      
41      private final static int PERM_ERROR_RCODE = 5;
42      
43      private final static int NONE_RCODE = 6;
44      
45      private final static int UNKNOWN_RCODE = 255;
46  
47      private static Logger logger = Logger.getRootLogger();
48  
49      /***
50       * @param args
51       *            The commandline arguments to parse
52       */
53      public static void main(String[] args) {
54  
55          String ip = null;
56          String sender = null;
57          String helo = null;
58  
59          SimpleLayout layout = new SimpleLayout();
60          ConsoleAppender consoleAppender = new ConsoleAppender(layout);
61          logger.addAppender(consoleAppender);
62  
63          logger.setLevel(Level.ERROR);
64  
65          // Parse the command line arguments
66          if (args.length < 3 || args.length > 4) {
67              usage();
68          } else {
69              for (int i = 0; i < args.length; i++) {
70                  String[] arguments = args[i].split("=");
71  
72                  if (arguments[0].equals("-ip")) {
73                      ip = arguments[1];
74                  } else if (arguments[0].equals("-sender")) {
75                      sender = arguments[1];
76                  } else if (arguments[0].equals("-helo")) {
77                      helo = arguments[1];
78                  } else if (arguments[0].equals("-debug")) {
79                      logger.setLevel(Level.DEBUG);
80                  } else if (arguments[0].equals("-verbose")) {
81                      logger.setLevel(Level.TRACE);
82                  } else {
83                      usage();
84                  }
85  
86              }
87  
88              // check if all needed values was set
89              if (ip != null && sender != null && helo != null) {
90                  SPF spf = new SPF();
91                  SPFResult result = spf.checkSPF(ip, sender, helo);
92                  System.out.println(result.getResult());
93                  System.out.println(result.getHeader());
94                  System.exit(getReturnCode(result.getResult()));
95              } else {
96                  usage();
97              }
98          }
99      }
100 
101     /***
102      * Print out the usage
103      */
104     private static void usage() {
105         System.out
106                 .println("Usage: java -jar jspf-x.jar -ip=192.168.100.1 -sender=postmaster@foo.bar -helo=foo.bar [-debug] [-verbose]");
107         System.exit(UNKNOWN_RCODE);
108     }
109     
110     
111     /***
112      * Return the return code for the result
113      * 
114      * @param result The result 
115      * @return returnCode
116      */
117     public static int getReturnCode(String result) {
118         if (result.equals(SPF1Utils.PASS_CONV)) {
119             return PASS_RCODE;
120         } else if (result.equals(SPF1Utils.FAIL_CONV)) {
121             return FAIL_RCODE;
122         } else if (result.equals(SPF1Utils.SOFTFAIL_CONV)) {
123             return SOFTFAIL_RCODE;
124         } else if (result.equals(SPF1Utils.NEUTRAL_CONV)) {
125             return NEUTRAL_RCODE;
126         } else if (result.equals(SPF1Utils.TEMP_ERROR_CONV)) {
127             return TEMP_ERROR_RCODE;
128         } else if (result.equals(SPF1Utils.PERM_ERROR_CONV)) {
129             return PERM_ERROR_RCODE;
130         } else if (result.equals(SPF1Utils.NONE_CONV)) {
131             return NONE_RCODE;
132         } 
133         return UNKNOWN_RCODE;
134     }
