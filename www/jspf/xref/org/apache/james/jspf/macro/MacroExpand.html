85      /***
86       * This method expand the given domain. So all known macros get replaced
87       * 
88       * @param input
89       *            The domain which should be expand
90       * @return expanded The domain with replaced macros
91       * @throws PermErrorException
92       *             This get thrown if invalid macros are used
93       */
94      public String expandDomain(String input) throws PermErrorException {
95  
96          log.debug("Start expand domain: " + input);
97  
98          isExplanation = false;
99          String domainName = expand(input);
100         // reduce to less than 255 characters, deleting subdomains from left
101         int split = 0;
102         while (domainName.length() > 255 && split > -1) {
103             split = domainName.indexOf(".");
104             domainName = domainName.substring(split + 1);
105         }
106         return domainName;
107     }

109     /***
110      * Expand the given String
111      * 
112      * @param input
113      *            The inputString which should get expanded
114      * @return expanded The expanded given String
115      * @throws PermErrorException
116      *             This get thrown if invalid macros are used
117      */
118     private String expand(String input) throws PermErrorException {
119 
120         input = replaceLiterals(input);
121 
122         StringBuffer decodedValue = new StringBuffer();
123         inputMatcher = inputPattern.matcher(input);
124         String macroCell;
125 
126         while (inputMatcher.find()) {
127             macroCell = input.substring(inputMatcher.start() + 2, inputMatcher
128                     .end() - 1);
129             inputMatcher
130                     .appendReplacement(decodedValue, replaceCell(macroCell));
131         }
132         inputMatcher.appendTail(decodedValue);
133 
134         return decodedValue.toString();
135     }

137     /***
138      * Replace the macros in given String
139      * 
140      * @param replaceValue
141      *            The String in which known macros should get replaced
142      * @return returnData The String with replaced macros
143      * @throws PermErrorException
144      *             Get thrown if an error in processing happen
145      */
146     private String replaceCell(String replaceValue) throws PermErrorException {
147 
148         String variable = "";
149         String domainNumber = "";
150         boolean isReversed = false;
151         String delimeters = ".";
152 
153         if (isExplanation) {
154             // Find command
155             cellPattern = Pattern.compile("[ctCT]");
156             cellMatcher = cellPattern.matcher(replaceValue);
157             while (cellMatcher.find()) {
158                 if (cellMatcher.group().toUpperCase().equals(
159                         cellMatcher.group())) {
160                     variable = encodeURL(matchMacro(cellMatcher.group()));
161                 } else {
162                     variable = matchMacro(cellMatcher.group());
163                 }
164             }
165         }
166         // Get only command character so that 'r' command and 'r' modifier don't
167         // clash
168         String commandCharacter = replaceValue.substring(0, 1);
169         // Find command
170         cellPattern = Pattern.compile("[lsodipvhrLSODIPVHR]");
171         cellMatcher = cellPattern.matcher(commandCharacter);
172         while (cellMatcher.find()) {
173             if (cellMatcher.group().toUpperCase().equals(cellMatcher.group())) {
174                 variable = encodeURL(matchMacro(cellMatcher.group()));
175             } else {
176                 variable = matchMacro(cellMatcher.group());
177             }
178         }
179         // Remove Macro code so that r macro code does not clash with r the
180         // reverse modifier
181         replaceValue = replaceValue.substring(1);
182 
183         // Find number of domains to use
184         cellPattern = Pattern.compile("//d+");
185         cellMatcher = cellPattern.matcher(replaceValue);
186         while (cellMatcher.find()) {
187             domainNumber = cellMatcher.group();
188             if (Integer.parseInt(domainNumber) == 0) {
189                 throw new PermErrorException(
190                         "Digit transformer must be non-zero");
191             }
192         }
193         // find if reversed
194         cellPattern = Pattern.compile("r");
195         cellMatcher = cellPattern.matcher(replaceValue);
196         while (cellMatcher.find()) {
197             isReversed = true;
198         }
199 
200         // find delimeters
201         cellPattern = Pattern.compile("[//.//-//+//,/////_//=]+");
202         cellMatcher = cellPattern.matcher(replaceValue);
203         while (cellMatcher.find()) {
204             delimeters = cellMatcher.group();
205         }
206 
207         // Reverse domains as necessary
208         ArrayList data = split(variable, delimeters);
209         if (isReversed) {
210             data = reverse(data);
211         }
212 
213         // Truncate domain name to number of sub sections
214         String returnData;
215         if (!domainNumber.equals("")) {
216             returnData = subset(data, Integer.parseInt(domainNumber));
217         } else {
218             returnData = subset(data);
219         }
220 
221         return returnData;
222 
223     }

225     /***
226      * Get the value for the given macro like descripted in the RFC
227      * 
228      * @param macro
229      *            The macro we want to get the value for
230      * @return rValue The value for the given macro
231      * @throws PermErrorException
232      *             Get thrown if the given variable is an unknown macro
233      * 
234      */
235     private String matchMacro(String macro) throws PermErrorException {
236 
237         String rValue = null;
238 
239         String variable = macro.toLowerCase();
240         if (variable.equalsIgnoreCase("i")) {
241             rValue = macroData.getMacroIpAddress();
242         } else if (variable.equalsIgnoreCase("s")) {
243             rValue = macroData.getMailFrom();
244         } else if (variable.equalsIgnoreCase("h")) {
245             rValue = macroData.getHostName();
246         } else if (variable.equalsIgnoreCase("l")) {
247             rValue = macroData.getCurrentSenderPart();
248         } else if (variable.equalsIgnoreCase("d")) {
249             rValue = macroData.getCurrentDomain();
250         } else if (variable.equalsIgnoreCase("v")) {
251             rValue = macroData.getInAddress();
252         } else if (variable.equalsIgnoreCase("t")) {
253             rValue = Long.toString(macroData.getTimeStamp());
254         } else if (variable.equalsIgnoreCase("c")) {
255             rValue = macroData.getReadableIP();
256         } else if (variable.equalsIgnoreCase("p")) {
257             rValue = macroData.getClientDomain();
258         } else if (variable.equalsIgnoreCase("o")) {
259             rValue = macroData.getSenderDomain();
260         } else if (variable.equalsIgnoreCase("r")) {
261             rValue = macroData.getReceivingDomain();
262         }
263 
264         if (rValue == null) {
265             throw new PermErrorException("Unknown command : " + variable);
266 
267         } else {
268             log.debug("Used macro: " + macro + " replaced with: " + rValue);
269 
270             return rValue;
271         }
272     }

274     /***
275      * Create an ArrayList by the given String. The String get splitted by given
276      * delimeters and one entry in the Array will be made for each splited
277      * String
278      * 
279      * @param data
280      *            The String we want to put in the Array
281      * @param delimeters
282      *            The delimeter we want to use to split the String
283      * @return ArrayList which contains the String parts
284      */
285     private ArrayList split(String data, String delimeters) {
286 
287         String currentChar;
288         StringBuffer element = new StringBuffer();
289         ArrayList splitParts = new ArrayList();
290 
291         for (int i = 0; i < data.length(); i++) {
292             currentChar = data.substring(i, i + 1);
293             if (delimeters.indexOf(currentChar) > -1) {
294                 splitParts.add(element.toString());
295                 element.setLength(0);
296             } else {
297                 element.append(currentChar);
298             }
299         }
300         splitParts.add(element.toString());
301         return splitParts;
302     }

304     /***
305      * Reverse an ArrayList
306      * 
307      * @param data
308      *            The ArrayList we want to get reversed
309      * @return reversed The reversed given ArrayList
310      */
311     private ArrayList reverse(ArrayList data) {
312 
313         ArrayList reversed = new ArrayList();
314         for (int i = 0; i < data.size(); i++) {
315             reversed.add(0, data.get(i));
316         }
317         return reversed;
318     }

323     private String subset(ArrayList data) {
324         return subset(data, data.size());

327     /***
328      * Convert a ArrayList to a String which holds the entries seperated by dots
329      * 
330      * @param data The ArrayList which should be converted
331      * @param length The ArrayLength
332      * @return A String which holds all entries seperated by dots
333      */
334     private String subset(ArrayList data, int length) {
335 
336         StringBuffer buildString = new StringBuffer();
337         if (data.size() < length) {
338             length = data.size();
339         }
340         int start = data.size() - length;
341         for (int i = start; i < data.size(); i++) {
342             if (buildString.length() > 0) {
343                 buildString.append(".");
344             }
345             buildString.append(data.get(i));
346         }
347         return buildString.toString();
348 
349     }

351     /***
352      * Replace all literals in the given String
353      * 
354      * @param data
355      *            The String we want to replace the literals
356      * @return given The given String with all literales replaced
357      */
358     private String replaceLiterals(String data) {
359 
360         log.debug("Replace literals on String: " + data);
361 
362         data = data.replaceAll("%%", "%");
363         data = data.replaceAll("%_", " ");
364         data = data.replaceAll("%-", "%20");
365 
366         log.debug("Replaced String: " + data);
367 
368         return data;
369     }

371     /***
372      * Encode the given URL to UTF-8
373      * 
374      * @param data
375      *            url to encode
376      * @return encoded URL
377      */
378     private String encodeURL(String data) {
379 
380         try {
381             // TODO URLEncoder method is not RFC2396 compatible, known
382             // difference
383             // is Space character gets converted to "+" rather than "%20"
384             // Is there anything else which is not correct with URLEncoder?
385             // Couldn't find a RFC2396 encoder
386             data = URLEncoder.encode(data, "UTF-8");
387         } catch (UnsupportedEncodingException e) {
388             // This shouldn't happen ignore it!
389         }
390 
391         // workaround for the above descripted problem
392         return data.replaceAll("//+", "%20");
393 
394     }
