79      /***
80       * This method expand the given domain. So all known macros get replaced
81       * 
82       * @param input
83       *            The domain which should be expand
84       * @return expanded The domain with replaced macros
85       * @throws PermErrorException
86       *             This get thrown if invalid macros are used
87       */
88      public String expandDomain(String input) throws PermErrorException {
89  
90          log.debug("Start expand domain: " + input);
91  
92          isExplanation = false;
93          String domainName = expand(input);
94          // reduce to less than 255 characters, deleting subdomains from left
95          int split = 0;
96          while (domainName.length() > 255 && split > -1) {
97              split = domainName.indexOf(".");
98              domainName = domainName.substring(split + 1);
99          }
100         return domainName;
101     }

103     /***
104      * Expand the given String
105      * 
106      * @param input
107      *            The inputString which should get expanded
108      * @return expanded The expanded given String
109      * @throws PermErrorException
110      *             This get thrown if invalid macros are used
111      */
112     private String expand(String input) throws PermErrorException {
113 
114         input = replaceLiterals(input);
115 
116         StringBuffer decodedValue = new StringBuffer();
117         inputMatcher = inputPattern.matcher(input);
118         String macroCell;
119 
120         while (inputMatcher.find()) {
121             macroCell = input.substring(inputMatcher.start() + 2, inputMatcher
122                     .end() - 1);
123             inputMatcher
124                     .appendReplacement(decodedValue, replaceCell(macroCell));
125         }
126         inputMatcher.appendTail(decodedValue);
127 
128         return decodedValue.toString();
129     }

131     /***
132      * Replace the macros in given String
133      * 
134      * @param replaceValue
135      *            The String in which known macros should get replaced
136      * @return returnData The String with replaced macros
137      * @throws PermErrorException
138      *             Get thrown if an error in processing happen
139      */
140     private String replaceCell(String replaceValue) throws PermErrorException {
141 
142         String variable = "";
143         String domainNumber = "";
144         boolean isReversed = false;
145         String delimeters = ".";
146 
147         if (isExplanation) {
148             // Find command
149             cellPattern = Pattern.compile("[ctCT]");
150             cellMatcher = cellPattern.matcher(replaceValue);
151             while (cellMatcher.find()) {
152                 if (cellMatcher.group().toUpperCase().equals(
153                         cellMatcher.group())) {
154                     variable = encodeURL(matchMacro(cellMatcher.group()));
155                 } else {
156                     variable = matchMacro(cellMatcher.group());
157                 }
158             }
159         }
160         // Get only command character so that 'r' command and 'r' modifier don't
161         // clash
162         String commandCharacter = replaceValue.substring(0, 1);
163         // Find command
164         cellPattern = Pattern.compile("[lsodipvhrLSODIPVHR]");
165         cellMatcher = cellPattern.matcher(commandCharacter);
166         while (cellMatcher.find()) {
167             if (cellMatcher.group().toUpperCase().equals(cellMatcher.group())) {
168                 variable = encodeURL(matchMacro(cellMatcher.group()));
169             } else {
170                 variable = matchMacro(cellMatcher.group());
171             }
172         }
173         // Remove Macro code so that r macro code does not clash with r the
174         // reverse modifier
175         replaceValue = replaceValue.substring(1);
176 
177         // Find number of domains to use
178         cellPattern = Pattern.compile("//d+");
179         cellMatcher = cellPattern.matcher(replaceValue);
180         while (cellMatcher.find()) {
181             domainNumber = cellMatcher.group();
182             if (Integer.parseInt(domainNumber) == 0) {
183                 throw new PermErrorException(
184                         "Digit transformer must be non-zero");
185             }
186         }
187         // find if reversed
188         cellPattern = Pattern.compile("r");
189         cellMatcher = cellPattern.matcher(replaceValue);
190         while (cellMatcher.find()) {
191             isReversed = true;
192         }
193 
194         // find delimeters
195         cellPattern = Pattern.compile("[//.//-//+//,/////_//=]+");
196         cellMatcher = cellPattern.matcher(replaceValue);
197         while (cellMatcher.find()) {
198             delimeters = cellMatcher.group();
199         }
200 
201         // Reverse domains as necessary
202         ArrayList data = split(variable, delimeters);
203         if (isReversed) {
204             data = reverse(data);
205         }
206 
207         // Truncate domain name to number of sub sections
208         String returnData;
209         if (!domainNumber.equals("")) {
210             returnData = subset(data, Integer.parseInt(domainNumber));
211         } else {
212             returnData = subset(data);
213         }
214 
215         return returnData;
216 
217     }

219     /***
220      * Get the value for the given macro like descripted in the RFC
221      * 
222      * @param macro
223      *            The macro we want to get the value for
224      * @return rValue The value for the given macro
225      * @throws PermErrorException
226      *             Get thrown if the given variable is an unknown macro
227      * 
228      */
229     private String matchMacro(String macro) throws PermErrorException {
230 
231         String rValue = null;
232 
233         String variable = macro.toLowerCase();
234         if (variable.equalsIgnoreCase("i")) {
235             rValue = spfData.getMacroIpAddress();
236         } else if (variable.equalsIgnoreCase("s")) {
237             rValue = spfData.getMailFrom();
238         } else if (variable.equalsIgnoreCase("h")) {
239             rValue = spfData.getHostName();
240         } else if (variable.equalsIgnoreCase("l")) {
241             rValue = spfData.getCurrentSenderPart();
242         } else if (variable.equalsIgnoreCase("d")) {
243             rValue = spfData.getCurrentDomain();
244         } else if (variable.equalsIgnoreCase("v")) {
245             rValue = spfData.getInAddress();
246         } else if (variable.equalsIgnoreCase("t")) {
247             rValue = Long.toString(spfData.getTimeStamp());
248         } else if (variable.equalsIgnoreCase("c")) {
249             rValue = spfData.getReadableIP();
250         } else if (variable.equalsIgnoreCase("p")) {
251             rValue = spfData.getClientDomain();
252         } else if (variable.equalsIgnoreCase("o")) {
253             rValue = spfData.getSenderDomain();
254         } else if (variable.equalsIgnoreCase("r")) {
255             rValue = spfData.getReceivingDomain();
256         }
257 
258         if (rValue == null) {
259             throw new PermErrorException("Unknown command : " + variable);
260 
261         } else {
262 
263             log.debug("Used macro: " + macro + " replaced with: " + rValue);
264 
265             return rValue;
266         }
267     }

269     /***
270      * Create an ArrayList by the given String. The String get splitted by given
271      * delimeters and one entry in the Array will be made for each splited
272      * String
273      * 
274      * @param data
275      *            The String we want to put in the Array
276      * @param delimeters
277      *            The delimeter we want to use to split the String
278      * @return ArrayList which contains the String parts
279      */
280     private ArrayList split(String data, String delimeters) {
281 
282         String currentChar;
283         StringBuffer element = new StringBuffer();
284         ArrayList splitParts = new ArrayList();
285 
286         for (int i = 0; i < data.length(); i++) {
287             currentChar = data.substring(i, i + 1);
288             if (delimeters.indexOf(currentChar) > -1) {
289                 splitParts.add(element.toString());
290                 element.setLength(0);
291             } else {
292                 element.append(currentChar);
293             }
294         }
295         splitParts.add(element.toString());
296         return splitParts;
297     }

299     /***
300      * Reverse an ArrayList
301      * 
302      * @param data
303      *            The ArrayList we want to get reversed
304      * @return reversed The reversed given ArrayList
305      */
306     private ArrayList reverse(ArrayList data) {
307 
308         ArrayList reversed = new ArrayList();
309         for (int i = 0; i < data.size(); i++) {
310             reversed.add(0, data.get(i));
311         }
312         return reversed;
313     }

316      * @see #subset(ArrayList, int)
317      */
318     private String subset(ArrayList data) {
319         return subset(data, data.size());

322     /***
323      * Convert a ArrayList to a String which holds the entries seperated by dots
324      * 
325      * @param data The ArrayList which should be converted
326      * @param length The ArrayLength
327      * @return A String which holds all entries seperated by dots
328      */
329     private String subset(ArrayList data, int length) {
330 
331         StringBuffer buildString = new StringBuffer();
332         if (data.size() < length) {
333             length = data.size();
334         }
335         int start = data.size() - length;
336         for (int i = start; i < data.size(); i++) {
337             if (buildString.length() > 0) {
338                 buildString.append(".");
339             }
340             buildString.append(data.get(i));
341         }
342         return buildString.toString();
343 
344     }

346     /***
347      * Replace all literals in the given String
348      * 
349      * @param data
350      *            The String we want to replace the literals
351      * @return given The given String with all literales replaced
352      */
353     private String replaceLiterals(String data) {
354 
355         log.debug("Replace literals on String: " + data);
356 
357         data = data.replaceAll("%%", "%");
358         data = data.replaceAll("%_", " ");
359         data = data.replaceAll("%-", "%20");
360 
361         log.debug("Replaced String: " + data);
362 
363         return data;
364     }

366     /***
367      * Encode the given URL to UTF-8
368      * 
369      * @param data
370      *            url to encode
371      * @return encoded URL
372      */
373     private String encodeURL(String data) {
374 
375         try {
376             // TODO URLEncoder method is not RFC2396 compatible, known
377             // difference
378             // is Space character gets converted to "+" rather than "%20"
379             // Is there anything else which is not correct with URLEncoder?
380             // Couldn't find a RFC2396 encoder
381             data = URLEncoder.encode(data, "UTF-8");
382         } catch (UnsupportedEncodingException e) {
383             // This shouldn't happen ignore it!
384         }
385 
386         // workaround for the above descripted problem
387         return data.replaceAll("//+", "%20");
388 
389     }
