75      /***
76       * This method expand the given domain. So all known macros get replaced
77       * 
78       * @param input
79       *            The domain which should be expand
80       * @return expanded The domain with replaced macros
81       * @throws PermErrorException
82       *             This get thrown if invalid macros are used
83       */
84      public String expandDomain(String input) throws PermErrorException {
85  
86          log.debug("Start expand domain: " + input);
87  
88          isExplanation = false;
89          String domainName = expand(input);
90          // reduce to less than 255 characters, deleting subdomains from left
91          int split = 0;
92          while (domainName.length() > 255 && split > -1) {
93              split = domainName.indexOf(".");
94              domainName = domainName.substring(split + 1);
95          }
96          return domainName;
97      }

99      /***
100      * Expand the given String
101      * 
102      * @param input
103      *            The inputString which should get expanded
104      * @return expanded The expanded given String
105      * @throws PermErrorException
106      *             This get thrown if invalid macros are used
107      */
108     private String expand(String input) throws PermErrorException {
109 
110         input = replaceLiterals(input);
111 
112         StringBuffer decodedValue = new StringBuffer();
113         inputMatcher = inputPattern.matcher(input);
114         String macroCell;
115 
116         while (inputMatcher.find()) {
117             macroCell = input.substring(inputMatcher.start() + 2, inputMatcher
118                     .end() - 1);
119             inputMatcher
120                     .appendReplacement(decodedValue, replaceCell(macroCell));
121         }
122         inputMatcher.appendTail(decodedValue);
123 
124         return decodedValue.toString();
125     }

127     /***
128      * Replace the macros in given String
129      * 
130      * @param replaceValue
131      *            The String in which known macros should get replaced
132      * @return returnData The String with replaced macros
133      * @throws PermErrorException
134      *             Get thrown if an error in processing happen
135      */
136     private String replaceCell(String replaceValue) throws PermErrorException {
137 
138         String variable = "";
139         String domainNumber = "";
140         boolean isReversed = false;
141         String delimeters = ".";
142 
143         if (isExplanation) {
144             // Find command
145             cellPattern = Pattern.compile("[ctCT]");
146             cellMatcher = cellPattern.matcher(replaceValue);
147             while (cellMatcher.find()) {
148                 if (cellMatcher.group().toUpperCase().equals(
149                         cellMatcher.group())) {
150                     variable = encodeURL(matchMacro(cellMatcher.group()));
151                 } else {
152                     variable = matchMacro(cellMatcher.group());
153                 }
154             }
155         }
156         // Get only command character so that 'r' command and 'r' modifier don't
157         // clash
158         String commandCharacter = replaceValue.substring(0, 1);
159         // Find command
160         cellPattern = Pattern.compile("[lsodipvhrLSODIPVHR]");
161         cellMatcher = cellPattern.matcher(commandCharacter);
162         while (cellMatcher.find()) {
163             if (cellMatcher.group().toUpperCase().equals(cellMatcher.group())) {
164                 variable = encodeURL(matchMacro(cellMatcher.group()));
165             } else {
166                 variable = matchMacro(cellMatcher.group());
167             }
168         }
169         // Remove Macro code so that r macro code does not clash with r the
170         // reverse modifier
171         replaceValue = replaceValue.substring(1);
172 
173         // Find number of domains to use
174         cellPattern = Pattern.compile("//d+");
175         cellMatcher = cellPattern.matcher(replaceValue);
176         while (cellMatcher.find()) {
177             domainNumber = cellMatcher.group();
178             if (Integer.parseInt(domainNumber) == 0) {
179                 throw new PermErrorException(
180                         "Digit transformer must be non-zero");
181             }
182         }
183         // find if reversed
184         cellPattern = Pattern.compile("r");
185         cellMatcher = cellPattern.matcher(replaceValue);
186         while (cellMatcher.find()) {
187             isReversed = true;
188         }
189 
190         // find delimeters
191         cellPattern = Pattern.compile("[//.//-//+//,/////_//=]+");
192         cellMatcher = cellPattern.matcher(replaceValue);
193         while (cellMatcher.find()) {
194             delimeters = cellMatcher.group();
195         }
196 
197         // Reverse domains as necessary
198         ArrayList data = split(variable, delimeters);
199         if (isReversed) {
200             data = reverse(data);
201         }
202 
203         // Truncate domain name to number of sub sections
204         String returnData;
205         if (!domainNumber.equals("")) {
206             returnData = subset(data, Integer.parseInt(domainNumber));
207         } else {
208             returnData = subset(data);
209         }
210 
211         return returnData;
212 
213     }

215     /***
216      * Get the value for the given macro like descripted in the RFC
217      * 
218      * @param macro
219      *            The macro we want to get the value for
220      * @return rValue The value for the given macro
221      * @throws PermErrorException
222      *             Get thrown if the given variable is an unknown macro
223      * 
224      */
225     private String matchMacro(String macro) throws PermErrorException {
226 
227         String rValue = null;
228 
229         String variable = macro.toLowerCase();
230         if (variable.equalsIgnoreCase("i")) {
231             rValue = spfData.getMacroIpAddress();
232         } else if (variable.equalsIgnoreCase("s")) {
233             rValue = spfData.getMailFrom();
234         } else if (variable.equalsIgnoreCase("h")) {
235             rValue = spfData.getHostName();
236         } else if (variable.equalsIgnoreCase("l")) {
237             rValue = spfData.getCurrentSenderPart();
238         } else if (variable.equalsIgnoreCase("d")) {
239             rValue = spfData.getCurrentDomain();
240         } else if (variable.equalsIgnoreCase("v")) {
241             rValue = spfData.getInAddress();
242         } else if (variable.equalsIgnoreCase("t")) {
243             rValue = Long.toString(spfData.getTimeStamp());
244         } else if (variable.equalsIgnoreCase("c")) {
245             rValue = spfData.getReadableIP();
246         } else if (variable.equalsIgnoreCase("p")) {
247             rValue = spfData.getClientDomain();
248         } else if (variable.equalsIgnoreCase("o")) {
249             rValue = spfData.getSenderDomain();
250         } else if (variable.equalsIgnoreCase("r")) {
251             rValue = spfData.getReceivingDomain();
252         }
253 
254         if (rValue == null) {
255             throw new PermErrorException("Unknown command : " + variable);
256 
257         } else {
258 
259             log.debug("Used macro: " + macro + " replaced with: " + rValue);
260 
261             return rValue;
262         }
263     }

265     /***
266      * Create an ArrayList by the given String. The String get splitted by given
267      * delimeters and one entry in the Array will be made for each splited
268      * String
269      * 
270      * @param data
271      *            The String we want to put in the Array
272      * @param delimeters
273      *            The delimeter we want to use to split the String
274      * @return ArrayList which contains the String parts
275      */
276     private ArrayList split(String data, String delimeters) {
277 
278         String currentChar;
279         StringBuffer element = new StringBuffer();
280         ArrayList splitParts = new ArrayList();
281 
282         for (int i = 0; i < data.length(); i++) {
283             currentChar = data.substring(i, i + 1);
284             if (delimeters.indexOf(currentChar) > -1) {
285                 splitParts.add(element.toString());
286                 element.setLength(0);
287             } else {
288                 element.append(currentChar);
289             }
290         }
291         splitParts.add(element.toString());
292         return splitParts;
293     }

295     /***
296      * Reverse an ArrayList
297      * 
298      * @param data
299      *            The ArrayList we want to get reversed
300      * @return reversed The reversed given ArrayList
301      */
302     private ArrayList reverse(ArrayList data) {
303 
304         ArrayList reversed = new ArrayList();
305         for (int i = 0; i < data.size(); i++) {
306             reversed.add(0, data.get(i));
307         }
308         return reversed;
309     }

315         return subset(data, data.size());

318     /***
319      * Convert a ArrayList to a String which holds the entries seperated by dots
320      * 
321      * @param data The ArrayList which should be converted
322      * @param length The ArrayLength
323      * @return A String which holds all entries seperated by dots
324      */
325     private String subset(ArrayList data, int length) {
326 
327         StringBuffer buildString = new StringBuffer();
328         if (data.size() < length) {
329             length = data.size();
330         }
331         int start = data.size() - length;
332         for (int i = start; i < data.size(); i++) {
333             if (buildString.length() > 0) {
334                 buildString.append(".");
335             }
336             buildString.append(data.get(i));
337         }
338         return buildString.toString();
339 
340     }

342     /***
343      * Replace all literals in the given String
344      * 
345      * @param data
346      *            The String we want to replace the literals
347      * @return given The given String with all literales replaced
348      */
349     private String replaceLiterals(String data) {
350 
351         log.debug("Replace literals on String: " + data);
352 
353         data = data.replaceAll("%%", "%");
354         data = data.replaceAll("%_", " ");
355         data = data.replaceAll("%-", "%20");
356 
357         log.debug("Replaced String: " + data);
358 
359         return data;
360     }

362     /***
363      * Encode the given URL to UTF-8
364      * 
365      * @param data
366      *            url to encode
367      * @return encoded URL
368      */
369     private String encodeURL(String data) {
370 
371         try {
372             // TODO URLEncoder method is not RFC2396 compatible, known
373             // difference
374             // is Space character gets converted to "+" rather than "%20"
375             // Is there anything else which is not correct with URLEncoder?
376             // Couldn't find a RFC2396 encoder
377             data = URLEncoder.encode(data, "UTF-8");
378         } catch (UnsupportedEncodingException e) {
379             // This shouldn't happen ignore it!
380         }
381 
382         // workaround for the above descripted problem
383         return data.replaceAll("//+", "%20");
384 
385     }
