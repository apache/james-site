84      public String expand(String input, MacroData macroData, boolean isExplanation) throws PermErrorException {
85          if (isExplanation) {
86              return expandExplanation(input, macroData);

88              return expandDomain(input, macroData);

92      /***
93       * This method expand the given a explanation
94       * 
95       * @param input
96       *            The explanation which should be expanded
97       * @return expanded The expanded explanation
98       * @throws PermErrorException
99       *             Get thrown if invalid macros are used
100      */
101     private String expandExplanation(String input, MacroData macroData) throws PermErrorException {
102 
103         log.debug("Start do expand explanation: " + input);
104 
105         String[] parts = input.split(" ");
106         StringBuffer res = new StringBuffer();
107         for (int i = 0; i < parts.length; i++) {
108             if (i > 0) res.append(" ");
109             res.append(expandMacroString(parts[i], macroData, true));
110         }
111         log.debug("Done expand explanation: " + res);
112         
113         return res.toString();
114     }

116     /***
117      * This method expand the given domain. So all known macros get replaced
118      * 
119      * @param input
120      *            The domain which should be expand
121      * @return expanded The domain with replaced macros
122      * @throws PermErrorException
123      *             This get thrown if invalid macros are used
124      */
125     private String expandDomain(String input, MacroData macroData) throws PermErrorException {
126 
127         log.debug("Start expand domain: " + input);
128 
129         Matcher inputMatcher = domainSpecPattern.matcher(input);
130         if (!inputMatcher.matches() || inputMatcher.groupCount() != 2) {
131             throw new PermErrorException("Invalid DomainSpec: "+input);
132         }
133 
134         StringBuffer res = new StringBuffer();
135         if (inputMatcher.group(1) != null && inputMatcher.group(1).length() > 0) {
136             res.append(expandMacroString(inputMatcher.group(1), macroData, false));
137         }
138         if (inputMatcher.group(2) != null && inputMatcher.group(2).length() > 0) {
139             if (inputMatcher.group(2).startsWith(".")) {
140                 res.append(inputMatcher.group(2));
141             } else {
142                 res.append(expandMacroString(inputMatcher.group(2), macroData, false));
143             }
144         }
145         
146         String domainName = expandMacroString(input, macroData, false);
147         // reduce to less than 255 characters, deleting subdomains from left
148         int split = 0;
149         while (domainName.length() > 255 && split > -1) {
150             split = domainName.indexOf(".");
151             domainName = domainName.substring(split + 1);
152         }
153 
154         log.debug("Domain expanded: " + domainName);
155         
156         return domainName;
157     }

159     /***
160      * Expand the given String
161      * 
162      * @param input
163      *            The inputString which should get expanded
164      * @return expanded The expanded given String
165      * @throws PermErrorException
166      *             This get thrown if invalid macros are used
167      */
168     private String expandMacroString(String input, MacroData macroData, boolean isExplanation) throws PermErrorException {
169 
170         StringBuffer decodedValue = new StringBuffer();
171         Matcher inputMatcher = macroStringPattern.matcher(input);
172         String macroCell;
173         int pos = 0;
174 
175         while (inputMatcher.find()) {
176             String match2 = inputMatcher.group();
177             if (pos != inputMatcher.start()) {
178                 throw new PermErrorException("Middle part does not match: "+input.substring(0,pos)+">>"+input.substring(pos, inputMatcher.start())+"<<"+input.substring(inputMatcher.start())+" ["+input+"]");
179             }
180             if (match2.length() > 0) {
181                 if (match2.startsWith("%{")) {
182                     macroCell = input.substring(inputMatcher.start() + 2, inputMatcher
183                             .end() - 1);
184                     inputMatcher
185                             .appendReplacement(decodedValue, replaceCell(macroCell, macroData, isExplanation));
186                 } else if (match2.length() == 2 && match2.startsWith("%")) {
187                     // handle the % escaping
188                     inputMatcher.appendReplacement(decodedValue, match2.substring(1));
189                 }
190             }
191             
192             pos = inputMatcher.end();
193         }
194         
195         if (input.length() != pos) {
196             throw new PermErrorException("End part does not match: "+input.substring(pos));
197         }
198         
199         inputMatcher.appendTail(decodedValue);
200 
201         return decodedValue.toString();
202     }

204     /***
205      * Replace the macros in given String
206      * 
207      * @param replaceValue
208      *            The String in which known macros should get replaced
209      * @return returnData The String with replaced macros
210      * @throws PermErrorException
211      *             Get thrown if an error in processing happen
212      */
213     private String replaceCell(String replaceValue, MacroData macroData, boolean isExplanation) throws PermErrorException {
214 
215         String variable = "";
216         String domainNumber = "";
217         boolean isReversed = false;
218         String delimeters = ".";
219 
220         
221         // Get only command character so that 'r' command and 'r' modifier don't
222         // clash
223         String commandCharacter = replaceValue.substring(0, 1);
224         Matcher cellMatcher;
225         // Find command
226         if (isExplanation) {
227             cellMatcher = macroLettersExpPattern.matcher(commandCharacter);
228         } else {
229             cellMatcher = macroLettersPattern.matcher(commandCharacter);
230         }
231         if (cellMatcher.find()) {
232             if (cellMatcher.group().toUpperCase().equals(cellMatcher.group())) {
233                 variable = encodeURL(matchMacro(cellMatcher.group(), macroData));
234             } else {
235                 variable = matchMacro(cellMatcher.group(), macroData);
236             }
237             // Remove Macro code so that r macro code does not clash with r the
238             // reverse modifier
239             replaceValue = replaceValue.substring(1);
240         } else {
241             throw new PermErrorException("MacroLetter not found: "+replaceValue);
242         }
243 
244         // Find number of domains to use
245         cellPattern = Pattern.compile("//d+");
246         cellMatcher = cellPattern.matcher(replaceValue);
247         while (cellMatcher.find()) {
248             domainNumber = cellMatcher.group();
249             if (Integer.parseInt(domainNumber) == 0) {
250                 throw new PermErrorException(
251                         "Digit transformer must be non-zero");
252             }
253         }
254         // find if reversed
255         cellPattern = Pattern.compile("r");
256         cellMatcher = cellPattern.matcher(replaceValue);
257         while (cellMatcher.find()) {
258             isReversed = true;
259         }
260 
261         // find delimeters
262         cellPattern = Pattern.compile("[//.//-//+//,/////_//=]+");
263         cellMatcher = cellPattern.matcher(replaceValue);
264         while (cellMatcher.find()) {
265             delimeters = cellMatcher.group();
266         }
267 
268         // Reverse domains as necessary
269         ArrayList data = split(variable, delimeters);
270         if (isReversed) {
271             data = reverse(data);
272         }
273 
274         // Truncate domain name to number of sub sections
275         String returnData;
276         if (!domainNumber.equals("")) {
277             returnData = subset(data, Integer.parseInt(domainNumber));
278         } else {
279             returnData = subset(data);
280         }
281 
282         return returnData;
283 
284     }

286     /***
287      * Get the value for the given macro like descripted in the RFC
288      * 
289      * @param macro
290      *            The macro we want to get the value for
291      * @return rValue The value for the given macro
292      * @throws PermErrorException
293      *             Get thrown if the given variable is an unknown macro
294      * 
295      */
296     private String matchMacro(String macro, MacroData macroData) throws PermErrorException {
297 
298         String rValue = null;
299 
300         String variable = macro.toLowerCase();
301         if (variable.equalsIgnoreCase("i")) {
302             rValue = macroData.getMacroIpAddress();
303         } else if (variable.equalsIgnoreCase("s")) {
304             rValue = macroData.getMailFrom();
305         } else if (variable.equalsIgnoreCase("h")) {
306             rValue = macroData.getHostName();
307         } else if (variable.equalsIgnoreCase("l")) {
308             rValue = macroData.getCurrentSenderPart();
309         } else if (variable.equalsIgnoreCase("d")) {
310             rValue = macroData.getCurrentDomain();
311         } else if (variable.equalsIgnoreCase("v")) {
312             rValue = macroData.getInAddress();
313         } else if (variable.equalsIgnoreCase("t")) {
314             rValue = Long.toString(macroData.getTimeStamp());
315         } else if (variable.equalsIgnoreCase("c")) {
316             rValue = macroData.getReadableIP();
317         } else if (variable.equalsIgnoreCase("p")) {
318             rValue = macroData.getClientDomain();
319             if (rValue == null) {
320                 rValue = "unknown";
321                 if (macroData instanceof SPF1Data) {
322                     SPF1Data spf1data = (SPF1Data) macroData;
323                     try {
324                         boolean ip6 = IPAddr.isIPV6(spf1data.getIpAddress());
325                         List records = dnsProbe.getRecords(IPAddr.getAddress(spf1data.getIpAddress()).getReverseIP(), DNSService.PTR);
326     
327                         if (records != null && records.size() > 0) {
328                             String record = (String) records.get(0);
329                             records = dnsProbe.getRecords(record, ip6 ? DNSService.AAAA : DNSService.A);
330                             if (records != null && records.size() > 0) {
331                                 Iterator i = records.iterator();
332                                 while (i.hasNext()) {
333                                     String next = (String) i.next();
334                                     if (IPAddr.getAddress(spf1data.getIpAddress()).toString().equals(IPAddr.getAddress(next).toString())) {
335                                         rValue = record;
336                                         break;
337                                     }
338                                 }
339                             }
340                         }
341                     } catch (TimeoutException e) {
342                         // just return the default "unknown".
343                     } catch (PermErrorException e) {
344                         // just return the default "unknown".
345                     }
346                     spf1data.setClientDomain(rValue);
347                 }
348             }
349         } else if (variable.equalsIgnoreCase("o")) {
350             rValue = macroData.getSenderDomain();
351         } else if (variable.equalsIgnoreCase("r")) {
352             rValue = macroData.getReceivingDomain();
353             if (rValue == null) {
354                 rValue = "unknown";
355                 List dNames = dnsProbe.getLocalDomainNames();
356 
357                 for (int i = 0; i < dNames.size(); i++) {
358                     // check if the domainname is a FQDN
359                     if (SPF1Utils.checkFQDN(dNames.get(i).toString())) {
360                         rValue = dNames.get(i).toString();
361                         if (macroData instanceof SPF1Data) {
362                             ((SPF1Data) macroData).setReceivingDomain(rValue);
363                         }
364                         break;
365                     }
366                 }
367             }
368         }
369 
370         if (rValue == null) {
371             throw new PermErrorException("Unknown command : " + variable);
372 
373         } else {
374             log.debug("Used macro: " + macro + " replaced with: " + rValue);
375 
376             return rValue;
377         }
378     }

380     /***
381      * Create an ArrayList by the given String. The String get splitted by given
382      * delimeters and one entry in the Array will be made for each splited
383      * String
384      * 
385      * @param data
386      *            The String we want to put in the Array
387      * @param delimeters
388      *            The delimeter we want to use to split the String
389      * @return ArrayList which contains the String parts
390      */
391     private ArrayList split(String data, String delimeters) {
392 
393         String currentChar;
394         StringBuffer element = new StringBuffer();
395         ArrayList splitParts = new ArrayList();
396 
397         for (int i = 0; i < data.length(); i++) {
398             currentChar = data.substring(i, i + 1);
399             if (delimeters.indexOf(currentChar) > -1) {
400                 splitParts.add(element.toString());
401                 element.setLength(0);
402             } else {
403                 element.append(currentChar);
404             }
405         }
406         splitParts.add(element.toString());
407         return splitParts;
408     }

410     /***
411      * Reverse an ArrayList
412      * 
413      * @param data
414      *            The ArrayList we want to get reversed
415      * @return reversed The reversed given ArrayList
416      */
417     private ArrayList reverse(ArrayList data) {
418 
419         ArrayList reversed = new ArrayList();
420         for (int i = 0; i < data.size(); i++) {
421             reversed.add(0, data.get(i));
422         }
423         return reversed;
424     }

433     /***
434      * Convert a ArrayList to a String which holds the entries seperated by dots
435      * 
436      * @param data The ArrayList which should be converted
437      * @param length The ArrayLength
438      * @return A String which holds all entries seperated by dots
439      */
440     private String subset(ArrayList data, int length) {
441 
442         StringBuffer buildString = new StringBuffer();
443         if (data.size() < length) {
444             length = data.size();
445         }
446         int start = data.size() - length;
447         for (int i = start; i < data.size(); i++) {
448             if (buildString.length() > 0) {
449                 buildString.append(".");
450             }
451             buildString.append(data.get(i));
452         }
453         return buildString.toString();
454 
455     }

457     /***
458      * Encode the given URL to UTF-8
459      * 
460      * @param data
461      *            url to encode
462      * @return encoded URL
463      */
464     private String encodeURL(String data) {
465 
466         try {
467             // TODO URLEncoder method is not RFC2396 compatible, known
468             // difference
469             // is Space character gets converted to "+" rather than "%20"
470             // Is there anything else which is not correct with URLEncoder?
471             // Couldn't find a RFC2396 encoder
472             data = URLEncoder.encode(data, "UTF-8");
473         } catch (UnsupportedEncodingException e) {
474             // This shouldn't happen ignore it!
475         }
476 
477         // workaround for the above descripted problem
478         return data.replaceAll("//+", "%20");
479 
480     }
