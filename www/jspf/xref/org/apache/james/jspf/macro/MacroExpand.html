61      public MacroExpand(MacroData macroData, Logger logger) {
62          this.macroData = macroData;
63          // This matches 2 groups
64          domainSpecPattern = Pattern.compile(SPFTermsRegexps.DOMAIN_SPEC_REGEX_R);
65          // The real pattern replacer
66          macroStringPattern = Pattern.compile(SPFTermsRegexps.MACRO_STRING_REGEX_TOKEN);
67          // The macro letters pattern

71          log = logger;

74      /***
75       * This method expand the given a explanation
76       * 
77       * @param input
78       *            The explanation which should be expanded
79       * @return expanded The expanded explanation
80       * @throws PermErrorException
81       *             Get thrown if invalid macros are used
82       */
83      public String expandExplanation(String input) throws PermErrorException {
84  
85          log.debug("Start do expand explanation: " + input);
86  
87          String[] parts = input.split(" ");
88          StringBuffer res = new StringBuffer();
89          for (int i = 0; i < parts.length; i++) {
90              if (i > 0) res.append(" ");
91              res.append(expandMacroString(parts[i], true));
92          }
93          log.debug("Done expand explanation: " + res);
94          
95          return res.toString();
96      }

98      /***
99       * This method expand the given domain. So all known macros get replaced
100      * 
101      * @param input
102      *            The domain which should be expand
103      * @return expanded The domain with replaced macros
104      * @throws PermErrorException
105      *             This get thrown if invalid macros are used
106      */
107     public String expandDomain(String input) throws PermErrorException {
108 
109         log.debug("Start expand domain: " + input);
110 
111         Matcher inputMatcher = domainSpecPattern.matcher(input);
112         if (!inputMatcher.matches() || inputMatcher.groupCount() != 2) {
113             throw new PermErrorException("Invalid DomainSpec: "+input);
114         }
115 
116         StringBuffer res = new StringBuffer();
117         if (inputMatcher.group(1) != null && inputMatcher.group(1).length() > 0) {
118             res.append(expandMacroString(inputMatcher.group(1), false));
119         }
120         if (inputMatcher.group(2) != null && inputMatcher.group(2).length() > 0) {
121             if (inputMatcher.group(2).startsWith(".")) {
122                 res.append(inputMatcher.group(2));
123             } else {
124                 res.append(expandMacroString(inputMatcher.group(2), false));
125             }
126         }
127         
128         String domainName = expandMacroString(input, false);
129         // reduce to less than 255 characters, deleting subdomains from left
130         int split = 0;
131         while (domainName.length() > 255 && split > -1) {
132             split = domainName.indexOf(".");
133             domainName = domainName.substring(split + 1);
134         }
135 
136         log.debug("Domain expanded: " + domainName);
137         
138         return domainName;
139     }

141     /***
142      * Expand the given String
143      * 
144      * @param input
145      *            The inputString which should get expanded
146      * @return expanded The expanded given String
147      * @throws PermErrorException
148      *             This get thrown if invalid macros are used
149      */
150     private String expandMacroString(String input, boolean isExplanation) throws PermErrorException {
151 
152         StringBuffer decodedValue = new StringBuffer();
153         Matcher inputMatcher = macroStringPattern.matcher(input);
154         String macroCell;
155         int pos = 0;
156 
157         while (inputMatcher.find()) {
158             String match2 = inputMatcher.group();
159             if (pos != inputMatcher.start()) {
160                 throw new PermErrorException("Middle part does not match: "+input.substring(0,pos)+">>"+input.substring(pos, inputMatcher.start())+"<<"+input.substring(inputMatcher.start())+" ["+input+"]");
161             }
162             if (match2.length() > 0) {
163                 if (match2.startsWith("%{")) {
164                     macroCell = input.substring(inputMatcher.start() + 2, inputMatcher
165                             .end() - 1);
166                     inputMatcher
167                             .appendReplacement(decodedValue, replaceCell(macroCell, isExplanation));
168                 } else if (match2.length() == 2 && match2.startsWith("%")) {
169                     // handle the % escaping
170                     inputMatcher.appendReplacement(decodedValue, match2.substring(1));
171                 }
172             }
173             
174             pos = inputMatcher.end();
175         }
176         
177         if (input.length() != pos) {
178             throw new PermErrorException("End part does not match: "+input.substring(pos));
179         }
180         
181         inputMatcher.appendTail(decodedValue);
182 
183         return decodedValue.toString();
184     }

186     /***
187      * Replace the macros in given String
188      * 
189      * @param replaceValue
190      *            The String in which known macros should get replaced
191      * @return returnData The String with replaced macros
192      * @throws PermErrorException
193      *             Get thrown if an error in processing happen
194      */
195     private String replaceCell(String replaceValue, boolean isExplanation) throws PermErrorException {
196 
197         String variable = "";
198         String domainNumber = "";
199         boolean isReversed = false;
200         String delimeters = ".";
201 
202         
203         // Get only command character so that 'r' command and 'r' modifier don't
204         // clash
205         String commandCharacter = replaceValue.substring(0, 1);
206         Matcher cellMatcher;
207         // Find command
208         if (isExplanation) {
209             cellMatcher = macroLettersExpPattern.matcher(commandCharacter);
210         } else {
211             cellMatcher = macroLettersPattern.matcher(commandCharacter);
212         }
213         if (cellMatcher.find()) {
214             if (cellMatcher.group().toUpperCase().equals(cellMatcher.group())) {
215                 variable = encodeURL(matchMacro(cellMatcher.group()));
216             } else {
217                 variable = matchMacro(cellMatcher.group());
218             }
219             // Remove Macro code so that r macro code does not clash with r the
220             // reverse modifier
221             replaceValue = replaceValue.substring(1);
222         } else {
223             throw new PermErrorException("MacroLetter not found: "+replaceValue);
224         }
225 
226         // Find number of domains to use
227         cellPattern = Pattern.compile("//d+");
228         cellMatcher = cellPattern.matcher(replaceValue);
229         while (cellMatcher.find()) {
230             domainNumber = cellMatcher.group();
231             if (Integer.parseInt(domainNumber) == 0) {
232                 throw new PermErrorException(
233                         "Digit transformer must be non-zero");
234             }
235         }
236         // find if reversed
237         cellPattern = Pattern.compile("r");
238         cellMatcher = cellPattern.matcher(replaceValue);
239         while (cellMatcher.find()) {
240             isReversed = true;
241         }
242 
243         // find delimeters
244         cellPattern = Pattern.compile("[//.//-//+//,/////_//=]+");
245         cellMatcher = cellPattern.matcher(replaceValue);
246         while (cellMatcher.find()) {
247             delimeters = cellMatcher.group();
248         }
249 
250         // Reverse domains as necessary
251         ArrayList data = split(variable, delimeters);
252         if (isReversed) {
253             data = reverse(data);
254         }
255 
256         // Truncate domain name to number of sub sections
257         String returnData;
258         if (!domainNumber.equals("")) {
259             returnData = subset(data, Integer.parseInt(domainNumber));
260         } else {
261             returnData = subset(data);
262         }
263 
264         return returnData;
265 
266     }

268     /***
269      * Get the value for the given macro like descripted in the RFC
270      * 
271      * @param macro
272      *            The macro we want to get the value for
273      * @return rValue The value for the given macro
274      * @throws PermErrorException
275      *             Get thrown if the given variable is an unknown macro
276      * 
277      */
278     private String matchMacro(String macro) throws PermErrorException {
279 
280         String rValue = null;
281 
282         String variable = macro.toLowerCase();
283         if (variable.equalsIgnoreCase("i")) {
284             rValue = macroData.getMacroIpAddress();
285         } else if (variable.equalsIgnoreCase("s")) {
286             rValue = macroData.getMailFrom();
287         } else if (variable.equalsIgnoreCase("h")) {
288             rValue = macroData.getHostName();
289         } else if (variable.equalsIgnoreCase("l")) {
290             rValue = macroData.getCurrentSenderPart();
291         } else if (variable.equalsIgnoreCase("d")) {
292             rValue = macroData.getCurrentDomain();
293         } else if (variable.equalsIgnoreCase("v")) {
294             rValue = macroData.getInAddress();
295         } else if (variable.equalsIgnoreCase("t")) {
296             rValue = Long.toString(macroData.getTimeStamp());
297         } else if (variable.equalsIgnoreCase("c")) {
298             rValue = macroData.getReadableIP();
299         } else if (variable.equalsIgnoreCase("p")) {
300             rValue = macroData.getClientDomain();
301         } else if (variable.equalsIgnoreCase("o")) {
302             rValue = macroData.getSenderDomain();
303         } else if (variable.equalsIgnoreCase("r")) {
304             rValue = macroData.getReceivingDomain();
305         }
306 
307         if (rValue == null) {
308             throw new PermErrorException("Unknown command : " + variable);
309 
310         } else {
311             log.debug("Used macro: " + macro + " replaced with: " + rValue);
312 
313             return rValue;
314         }
315     }

317     /***
318      * Create an ArrayList by the given String. The String get splitted by given
319      * delimeters and one entry in the Array will be made for each splited
320      * String
321      * 
322      * @param data
323      *            The String we want to put in the Array
324      * @param delimeters
325      *            The delimeter we want to use to split the String
326      * @return ArrayList which contains the String parts
327      */
328     private ArrayList split(String data, String delimeters) {
329 
330         String currentChar;
331         StringBuffer element = new StringBuffer();
332         ArrayList splitParts = new ArrayList();
333 
334         for (int i = 0; i < data.length(); i++) {
335             currentChar = data.substring(i, i + 1);
336             if (delimeters.indexOf(currentChar) > -1) {
337                 splitParts.add(element.toString());
338                 element.setLength(0);
339             } else {
340                 element.append(currentChar);
341             }
342         }
343         splitParts.add(element.toString());
344         return splitParts;
345     }

347     /***
348      * Reverse an ArrayList
349      * 
350      * @param data
351      *            The ArrayList we want to get reversed
352      * @return reversed The reversed given ArrayList
353      */
354     private ArrayList reverse(ArrayList data) {
355 
356         ArrayList reversed = new ArrayList();
357         for (int i = 0; i < data.size(); i++) {
358             reversed.add(0, data.get(i));
359         }
360         return reversed;
361     }

370     /***
371      * Convert a ArrayList to a String which holds the entries seperated by dots
372      * 
373      * @param data The ArrayList which should be converted
374      * @param length The ArrayLength
375      * @return A String which holds all entries seperated by dots
376      */
377     private String subset(ArrayList data, int length) {
378 
379         StringBuffer buildString = new StringBuffer();
380         if (data.size() < length) {
381             length = data.size();
382         }
383         int start = data.size() - length;
384         for (int i = start; i < data.size(); i++) {
385             if (buildString.length() > 0) {
386                 buildString.append(".");
387             }
388             buildString.append(data.get(i));
389         }
390         return buildString.toString();
391 
392     }

394     /***
395      * Encode the given URL to UTF-8
396      * 
397      * @param data
398      *            url to encode
399      * @return encoded URL
400      */
401     private String encodeURL(String data) {
402 
403         try {
404             // TODO URLEncoder method is not RFC2396 compatible, known
405             // difference
406             // is Space character gets converted to "+" rather than "%20"
407             // Is there anything else which is not correct with URLEncoder?
408             // Couldn't find a RFC2396 encoder
409             data = URLEncoder.encode(data, "UTF-8");
410         } catch (UnsupportedEncodingException e) {
411             // This shouldn't happen ignore it!
412         }
413 
414         // workaround for the above descripted problem
415         return data.replaceAll("//+", "%20");
416 
417     }
