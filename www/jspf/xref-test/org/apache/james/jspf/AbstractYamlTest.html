70      protected AbstractYamlTest(String name) throws IOException {
71          super(name);
72          List tests = internalLoadTests(getFilename());
73          Iterator i = tests.iterator();
74          while (i.hasNext() && data == null) {
75              SPFYamlTestSuite def = (SPFYamlTestSuite) i.next();
76              Iterator j = def.getTests().keySet().iterator();
77              while (j.hasNext() && data == null) {
78                  String test = (String) j.next();
79                  if (name.equals(def.getComment()+ " #"+test)) {
80                      data = def;
81                      this.test = test;
82                  }
83              }
84          }
85          assertNotNull(data);
86          assertNotNull(test);
87      }

129         if (log == null) {

169         String sender = null;

171     
172         if (currentTest.get("helo") != null) {
173             helo = (String) currentTest.get("helo");
174         }

176             ip = (String) currentTest.get("host");
177         }

188             assertEquals("Test "+next+" ("+currentTest.get("description")+") failed. Returned: "+res.getResult()+" Expected: "+currentTest.get("result")+" [["+res.getResult()+"||"+res.getHeaderText()+"]]", currentTest.get("result"), res.getResult());
189         } else {
190             ArrayList results = (ArrayList) currentTest.get("result");
191             boolean match = false;
192             for (int i = 0; i < results.size(); i++) {
193                 if (results.get(i).equals(resultSPF)) match = true;

205                 // See http://java.sun.com/j2se/1.4.2/docs/api/java/net/Inet6Address.html    
206                 // For methods that return a textual representation as output value, the full form is used.

268         public List getRecords(String hostname, int recordType, int depth) throws TimeoutException {
269             String type = getRecordTypeDescription(recordType);
270 
271             List res;
272             
273             // remove trailing dot before running the search.
274             if (hostname.endsWith(".")) hostname = hostname.substring(0, hostname.length()-1);
275             
276             // dns search lowercases:
277             hostname = hostname.toLowerCase(Locale.US);
278             
279             if (zonedata.get(hostname) != null) {
280                 List l = (List) zonedata.get(hostname);
281                 Iterator i = l.iterator();
282                 res = new ArrayList();
283                 while (i.hasNext()) {
284                     Object o = i.next();
285                     if (o instanceof HashMap) {
286                         HashMap hm = (HashMap) o;
287                         if (hm.get(type) != null) {
288                             if (recordType == DNSService.MX) {
289                                 List mxList = (List) hm.get(type);
290     
291                                 // For MX records we overwrite the result ignoring the priority.
292                                 Iterator mxs = mxList.iterator();
293                                 while (mxs.hasNext()) {
294                                     // skip the MX priority
295                                     mxs.next();
296                                     String cname = (String) mxs.next();
297                                     res.add(cname);
298                                 }
299                             } else {
300                                 Object obj = hm.get(type);
301                                 
302                                 if (obj instanceof String) {
303                                     res.add((String) obj);
304                                 } else if (obj instanceof ArrayList) {
305                                     ArrayList a = (ArrayList) obj;
306                                     StringBuffer sb = new StringBuffer();
307                                     
308                                     for (int i2 = 0; i2 < a.size(); i2++) {
309                                         sb.append(a.get(i2));
310                                     }
311                                     res.add(sb.toString());
312                                 }
313                             }
314                         }
315                         if (hm.get("CNAME") != null && depth > 0) {
316                             return getRecords((String) hm.get("CNAME"), recordType, depth - 1);
317                         }
318                     } else if ("TIMEOUT".equals(o)) {
319                         throw new TimeoutException();
320                     } else {
321                         throw new IllegalStateException("getRecord found an unexpected data");
322                     }
323                 }
324                 return res.size() > 0 ? res : null;
325             }
326             return null;
327         }

334      * @param recordType the DNSService.CONSTANT type to convert

337     public static String getRecordTypeDescription(int recordType) {
338         switch (recordType) {
339             case DNSService.A: return "A";
340             case DNSService.AAAA: return "AAAA";
341             case DNSService.MX: return "MX";
342             case DNSService.PTR: return "PTR";
343             case DNSService.TXT: return "TXT";
344             case DNSService.SPF: return "SPF";
345             default: return null;

378         public void setZonedata(HashMap zonedata) {
379             this.zonedata = new HashMap();
380             Set keys = zonedata.keySet();
381             for (Iterator i = keys.iterator(); i.hasNext(); ) {
382                 String hostname = (String) i.next();
383                 String lowercase = hostname.toLowerCase(Locale.US);
384                 this.zonedata.put(lowercase, zonedata.get(hostname));
385             }
