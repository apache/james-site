57      protected AbstractYamlTest(String name) throws IOException {
58          super(name);
59          List tests = internalLoadTests(getFilename());
60          Iterator i = tests.iterator();
61          while (i.hasNext() && data == null) {
62              SPFYamlTestSuite def = (SPFYamlTestSuite) i.next();
63              Iterator j = def.getTests().keySet().iterator();
64              while (j.hasNext() && data == null) {
65                  String test = (String) j.next();
66                  if (name.equals(def.getComment()+ " #"+test)) {
67                      data = def;
68                      this.test = test;
69                  }
70              }
71          }
72          assertNotNull(data);
73          assertNotNull(test);
74      }

76      public static List loadTests(String filename) throws IOException {
77          List tests = new ArrayList();
78      
79          //InputStream is = SPFYamlTest.class.getResourceAsStream("pyspf-tests.yml");
80          InputStream is = SPFYamlTest.class.getResourceAsStream(filename);
81          
82          Reader br = new BufferedReader(new InputStreamReader(is)) {
83      
84              public int read(char[] arg0) throws IOException {
85                  int rl = super.read(arg0);
86                  // System.out.println("<<< "+new String(arg0));
87                  return rl;
88              }
89              
90          };
91          
92          YAMLFactory fact = new DefaultYAMLFactory();
93          
94          Constructor ctor = fact.createConstructor(fact.createComposer(fact.createParser(fact.createScanner(br)),fact.createResolver()));
95          int i = 1;
96          while(ctor.checkData()) {
97              Object o = ctor.getData();
98              if (o instanceof HashMap) {
99                HashMap m = (HashMap) o;
100               SPFYamlTestSuite ts = new SPFYamlTestSuite(m, i);
101               tests.add(ts);
102             }
103             i++;
104         }
105     
106         return tests;
107     }

109     protected void runTest() throws Throwable {
110         String next = test;
111         HashMap currentTest = (HashMap) data.getTests().get(next);
112         
113         System.out.println("testing "+next+": "+currentTest.get("description"));
114     
115         SPF spf = new SPF(new LoggingDNSService(getDNSService()), new ConsoleLogger());
116         
117         String ip = null;
118         String sender = null;
119         String helo = null;
120     
121         if (currentTest.get("helo") != null) {
122             helo = (String) currentTest.get("helo");
123         }
124         if (currentTest.get("host") != null) {
125             ip = (String) currentTest.get("host");
126         }
127         if (currentTest.get("mailfrom") != null) {
128             sender = (String) currentTest.get("mailfrom");
129         } else {
130             sender = "";
131         }
132     
133         SPFResult res = spf.checkSPF(ip, sender, helo);
134         String resultSPF = res.getResult();
135         
136         if (currentTest.get("result") instanceof String) {
137             assertEquals("Test "+next+" ("+currentTest.get("description")+") failed. Returned: "+res.getResult()+" Expected: "+currentTest.get("result")+" [["+res.getResultChar()+"||"+res.getHeaderText()+"]]", currentTest.get("result"), res.getResult());
138         } else {
139             ArrayList results = (ArrayList) currentTest.get("result");
140             boolean match = false;
141             for (int i = 0; i < results.size(); i++) {
142                 if (results.get(i).equals(resultSPF)) match = true;
143                 System.err.println("checking "+results.get(i)+" => "+resultSPF);
144             }
145             assertTrue(match);
146         }
147         
148         if (currentTest.get("explanation") != null) {
149             
150             // Check for our default explanation!
151             if (currentTest.get("explanation").equals("DEFAULT") || currentTest.get("explanation").equals("postmaster") ) {
152                 assertTrue(res.getExplanation().startsWith("http://www.openspf.org/why.html?sender="));
153             } else {
154                 assertEquals(currentTest.get("explanation"),res.getExplanation());
155             }
156     
157         }

206         public List getRecords(String hostname, int recordType) throws TimeoutException {
207             String type = getRecordTypeDescription(recordType);
208 
209             List res;
210             
211             // remove trailing dot before running the search.
212             if (hostname.endsWith(".")) hostname = hostname.substring(0, hostname.length()-1);
213             
214             if (zonedata.get(hostname) != null) {
215                 List l = (List) zonedata.get(hostname);
216                 Iterator i = l.iterator();
217                 res = new ArrayList();
218                 while (i.hasNext()) {
219                     Object o = i.next();
220                     if (o instanceof HashMap) {
221                         HashMap hm = (HashMap) o;
222                         if (hm.get(type) != null) {
223                             if (recordType == DNSService.MX) {
224                                 List mxList = (List) hm.get(type);
225     
226                                 // For MX records we overwrite the result ignoring the priority.
227                                 Iterator mxs = mxList.iterator();
228                                 while (mxs.hasNext()) {
229                                     // skip the MX priority
230                                     mxs.next();
231                                     String cname = (String) mxs.next();
232                                     res.add(cname);
233                                 }
234                             } else {
235                                 res.add((String) hm.get(type));
236                             }
237                         }
238                     } else if ("TIMEOUT".equals(o)) {
239                         throw new TimeoutException();
240                     } else {
241                         throw new IllegalStateException("getRecord found an unexpected data");
242                     }
243                 }
244                 return res.size() > 0 ? res : null;
245             }
246             return null;
247         }

252     /***
253      * Return a string representation of a DNSService record type.
254      * 
255      * @param recordType the DNSService.CONSTANT type to convert
256      * @return a string representation of the given record type
257      */
258     public static String getRecordTypeDescription(int recordType) {
259         switch (recordType) {
260             case DNSService.A: return "A";
261             case DNSService.AAAA: return "AAAA";
262             case DNSService.MX: return "MX";
263             case DNSService.PTR: return "PTR";
264             case DNSService.TXT: return "TXT";
265             case DNSService.SPF: return "SPF";
266             default: return null;
267         }
