136     public static List loadTests() throws IOException {
137         List tests = new ArrayList();
138 
139         BufferedReader br = new BufferedReader(new InputStreamReader(
140                 SPFTest.class.getResourceAsStream("test.txt")));
141 
142         String line;
143 
144         Pattern p = Pattern.compile("[ ]+");
145 
146         SPFTestDef def = null;
147 
148         String defaultCommands = "";
149 
150         while ((line = br.readLine()) != null) {
151             // skip comments and empty lines
152             if (line.length() != 0 && line.charAt(0) != '#') {
153 
154                 if (line.startsWith("default")) {
155                     defaultCommands = line.replaceFirst("default ", "");
156                 } else {
157 
158                     String[] tokens = p.split(line, 3);
159 
160                     if (tokens.length >= 2) {
161 
162                         if ("spfquery".equals(tokens[0])) {
163                             if (def != null) {
164                                 if (def.result != null) {
165                                     tests.add(def);
166                                 } else {
167                                     System.err
168                                             .println("Unexpected test sequence: "
169                                                     + def.command
170                                                     + "|"
171                                                     + def.result
172                                                     + "|"
173                                                     + def.smtpComment
174                                                     + "|"
175                                                     + def.headerComment
176                                                     + "|"
177                                                     + def.receivedSPF);
178                                 }
179                             }
180                             def = new SPFTestDef();
181                             def.name = tokens[1] + " " + tokens[2];
182                             def.command = tokens[1] + " " + tokens[2] + " "
183                                     + defaultCommands;
184                         } else if ("/.*/".equals(tokens[1])
185                                 || "jspf".equals(tokens[1])) {
186 
187                             if ("result".equals(tokens[0])) {
188                                 if (def.result == null)
189                                     def.result = tokens[2];
190                             } else if ("smtp-comment".equals(tokens[0])) {
191                                 if (def.smtpComment == null)
192                                     def.smtpComment = tokens[2];
193                             } else if ("received-spf".equals(tokens[0])) {
194                                 if (def.receivedSPF == null)
195                                     def.receivedSPF = tokens[2].replaceFirst(
196                                             "Received-SPF: ", "");
197                             } else if ("header-comment".equals(tokens[0])) {
198                                 if (def.headerComment == null)
199                                     def.headerComment = tokens[2];
200                             } else if ("explanation".equals(tokens[0])) {
201                                 if (def.explanation.equals(""))
202                                         def.explanation = tokens[2];
203                             } else {
204                                 System.err.println("Unknown token: "
205                                         + tokens[0]);
206                             }
207 
208                         } else {
209                             // System.out.println("Ignored line: " + line);
210                         }
211 
212                     } else {
213                         throw new IllegalStateException("Bad format: " + line);
214                     }
215                 }
216             }
217 
218         }
219 
220         if (def != null && def.command != null) {
221             if (def.result != null) {
222                 tests.add(def);
223             } else {
224                 System.err.println("Unexpected test sequence: " + def.command
225                         + "|" + def.result + "|" + def.smtpComment + "|"
226                         + def.headerComment + "|" + def.receivedSPF);
227             }
228         }
229 
230         br.close();
231 
232         return tests;
233     }
