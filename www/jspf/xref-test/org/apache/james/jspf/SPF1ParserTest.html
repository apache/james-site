41      public SPF1ParserTest(String name) throws IOException {
42          super(name);
43          List tests = loadTests();
44          Iterator i = tests.iterator();
45          while (i.hasNext()) {
46              SPF1RecordTestDef def = (SPF1RecordTestDef) i.next();
47              if (name.equals(def.recIn)) {
48                  data = def;
49                  break;
50              }
51          }
52          assertNotNull(data);
53          parser = new SPF1Parser(new ConsoleLogger());
54      }

64      public SPF1ParserTest(SPF1RecordTestDef def, SPF1Parser parser) {
65          super(def.recIn);

70      protected void runTest() throws Throwable {
71  
72          try {
73  
74              System.out.println("testing [" + data.recIn + "]");
75  
76              parser.parse(data.recIn);
77  
78              assertEquals("Expected <" + data.errMsg + "> but was <"
79                      + "no errors" + ">", data.errMsg, "no errors");
80          } catch (NoneException e) {
81              e.printStackTrace();
82              assertNotNull(data.errMsg);
83              assertTrue("Expected <" + data.errMsg + "> but was <"
84                      + e.getMessage() + ">", !"no errors".equals(data.errMsg));
85              // assertEquals("Expected <" + data.errMsg + "> but was <"
86              // + e.getMessage() + ">", data.errMsg, e.getMessage());
87          } catch (PermErrorException e) {
88              e.printStackTrace();
89              assertNotNull(data.errMsg);
90              assertTrue("Expected <" + data.errMsg + "> but was <"
91                      + e.getMessage() + ">\n" + data.recOut + "\n"
92                      + data.recOutAuto, !"no errors".equals(data.errMsg));
93              // assertEquals("Expected <" + data.errMsg + "> but was <"
94              // + e.getMessage() + ">", data.errMsg, e.getMessage());
95          }
96  
97      }

99      public static List loadTests() throws IOException {
100         List tests = new ArrayList();
101 
102         BufferedReader br = new BufferedReader(new InputStreamReader(Thread
103                 .currentThread().getContextClassLoader().getResourceAsStream(
104                         "org/apache/james/jspf/test_parser.txt")));
105 
106         String line;
107 
108         Pattern p = Pattern.compile("[ ]+");
109 
110         SPF1RecordTestDef def = null;
111 
112         while ((line = br.readLine()) != null) {
113             // skip comments and empty lines
114             if (line.length() != 0 && line.charAt(0) != '#') {
115                 String[] tokens = p.split(line, 3);
116 
117                 if (tokens.length >= 2) {
118 
119                     if ("spftest".equals(tokens[0])) {
120                         if (def != null && def.recIn != null) {
121                             tests.add(def);
122                         }
123                         def = new SPF1RecordTestDef();
124                         def.name = tokens[2];
125                     } else if ("/.*/".equals(tokens[1])
126                             || "jspf".equals(tokens[1])) {
127 
128                         if ("rec-in".equals(tokens[0])) {
129                             if (def.recIn == null)
130                                 def.recIn = tokens[2].replaceFirst(
131                                         "SPF record in:  ", "");
132                         } else if ("err-msg".equals(tokens[0])) {
133                             if (def.errMsg == null)
134                                 def.errMsg = tokens[2];
135                         } else if ("rec-out".equals(tokens[0])) {
136                             if (def.recOut == null)
137                                 def.recOut = tokens[2].replaceFirst(
138                                         "SPF record:  ", "");
139                         } else if ("rec-out-auto".equals(tokens[0])) {
140                             if (tokens.length == 3) {
141                                 if (def.recOutAuto == null)
142                                     def.recOutAuto = tokens[2];
143                             } else {
144                                 if (def.recOutAuto == null)
145                                     def.recOutAuto = "";
146                             }
147                         } else {
148                             System.err.println("Unknown token: " + tokens[0]);
149                         }
150 
151                     } else {
152                         System.err
153                                 .println("Ignored line for different implementation: "
154                                         + tokens[1]);
155                     }
156 
157                 } else {
158                     throw new IllegalStateException("Bad format: " + line);
159                 }
160             }
161 
162         }
163 
164         if (def != null && def.recIn != null) {
165             tests.add(def);
166         }
167 
168         br.close();
169 
170         return tests;
171     }
