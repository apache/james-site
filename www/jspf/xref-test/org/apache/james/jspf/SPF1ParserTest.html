38      public SPF1ParserTest(String name) throws IOException {
39          super(name);
40          List tests = loadTests();
41          Iterator i = tests.iterator();
42          while (i.hasNext()) {
43              SPF1RecordTestDef def = (SPF1RecordTestDef) i.next();
44              if (name.equals(def.recIn)) {
45                  data = def;
46                  break;
47              }
48          }
49          assertNotNull(data);
50          parser = new SPF1Parser();
51      }

67      protected void runTest() throws Throwable {

73              parser.parse(data.recIn);

83              // + e.getMessage() + ">", data.errMsg, e.getMessage());
84          } catch (PermErrorException e) {
85              e.printStackTrace();
86              assertNotNull(data.errMsg);
87              assertTrue("Expected <" + data.errMsg + "> but was <"
88                      + e.getMessage() + ">\n" + data.recOut + "\n"
89                      + data.recOutAuto, !"no errors".equals(data.errMsg));
90              // assertEquals("Expected <" + data.errMsg + "> but was <"
91              // + e.getMessage() + ">", data.errMsg, e.getMessage());
92          }

96      public static List loadTests() throws IOException {
97          List tests = new ArrayList();
98  
99          BufferedReader br = new BufferedReader(new InputStreamReader(Thread
100                 .currentThread().getContextClassLoader().getResourceAsStream(
101                         "org/apache/james/jspf/test_parser.txt")));
102 
103         String line;
104 
105         Pattern p = Pattern.compile("[ ]+");
106 
107         SPF1RecordTestDef def = null;
108 
109         while ((line = br.readLine()) != null) {
110             // skip comments and empty lines
111             if (line.length() != 0 && line.charAt(0) != '#') {
112                 String[] tokens = p.split(line, 3);
113 
114                 if (tokens.length >= 2) {
115 
116                     if ("spftest".equals(tokens[0])) {
117                         if (def != null && def.recIn != null) {
118                             tests.add(def);
119                         }
120                         def = new SPF1RecordTestDef();
121                         def.name = tokens[2];
122                     } else if ("/.*/".equals(tokens[1])
123                             || "jspf".equals(tokens[1])) {
124 
125                         if ("rec-in".equals(tokens[0])) {
126                             if (def.recIn == null)
127                                 def.recIn = tokens[2].replaceFirst(
128                                         "SPF record in:  ", "");
129                         } else if ("err-msg".equals(tokens[0])) {
130                             if (def.errMsg == null)
131                                 def.errMsg = tokens[2];
132                         } else if ("rec-out".equals(tokens[0])) {
133                             if (def.recOut == null)
134                                 def.recOut = tokens[2].replaceFirst(
135                                         "SPF record:  ", "");
136                         } else if ("rec-out-auto".equals(tokens[0])) {
137                             if (tokens.length == 3) {
138                                 if (def.recOutAuto == null)
139                                     def.recOutAuto = tokens[2];
140                             } else {
141                                 if (def.recOutAuto == null)
142                                     def.recOutAuto = "";
143                             }
144                         } else {
145                             System.err.println("Unknown token: " + tokens[0]);
146                         }
147 
148                     } else {
149                         System.err
150                                 .println("Ignored line for different implementation: "
151                                         + tokens[1]);
152                     }
153 
154                 } else {
155                     throw new IllegalStateException("Bad format: " + line);
156                 }
157             }
158 
159         }
160 
161         if (def != null && def.recIn != null) {
162             tests.add(def);
163         }
164 
165         br.close();
166 
167         return tests;
168     }
