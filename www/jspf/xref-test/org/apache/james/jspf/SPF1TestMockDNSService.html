132     public List getAddressList(String list, int mask) throws PermErrorException {
133         if (list == null || "".equals(list)) {
134             return new ArrayList();
135         }
136         String[] s = list.split(",");
137         IPAddr[] ips = new IPAddr[s.length];
138         for (int i = 0; i < s.length; i++) {
139             ips[i] = IPAddr.getAddress(s[i], mask);
140         }
141         return new ArrayList(Arrays.asList(ips));
142     }
