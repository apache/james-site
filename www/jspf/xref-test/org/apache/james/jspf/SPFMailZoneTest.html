91                      rcptTo = val;

134     public static List loadTests() throws IOException {
135         List tests = new ArrayList();
136 
137         BufferedReader br = new BufferedReader(new InputStreamReader(
138                 SPFMailZoneTest.class.getResourceAsStream("test_mailzone.txt")));
139 
140         String line;
141 
142         Pattern p = Pattern.compile("[ ]+");
143 
144         SPFTestDef def = null;
145 
146         String defaultCommands = "";
147 
148         while ((line = br.readLine()) != null) {
149             // skip comments and empty lines
150             if (line.length() != 0 && line.charAt(0) != '#') {
151 
152                 if (line.startsWith("default")) {
153                     defaultCommands = line.replaceFirst("default ", "");
154                 } else {
155 
156                     String[] tokens = p.split(line, 3);
157 
158                     if (tokens.length >= 2) {
159 
160                         if ("spfquery".equals(tokens[0])) {
161                             if (def != null) {
162                                 if (def.result != null) {
163                                     tests.add(def);
164                                 } else {
165                                     System.err
166                                             .println("Unexpected test sequence: "
167                                                     + def.command
168                                                     + "|"
169                                                     + def.result
170                                                     + "|"
171                                                     + def.smtpComment
172                                                     + "|"
173                                                     + def.headerComment
174                                                     + "|"
175                                                     + def.receivedSPF);
176                                 }
177                             }
178                             def = new SPFTestDef();
179                             def.name = tokens[1] + " " + tokens[2];
180                             def.command = tokens[1] + " " + tokens[2] + " "
181                                     + defaultCommands;
182                         } else if ("/.*/".equals(tokens[1])
183                                 || "jspf".equals(tokens[1])) {
184 
185                             if ("result".equals(tokens[0])) {
186                                 if (def.result == null)
187                                     def.result = tokens[2];
188                             } else if ("smtp-comment".equals(tokens[0])) {
189                                 if (def.smtpComment == null)
190                                     def.smtpComment = tokens[2];
191                             } else if ("received-spf".equals(tokens[0])) {
192                                 if (def.receivedSPF == null)
193                                     def.receivedSPF = tokens[2].replaceFirst(
194                                             "Received-SPF: ", "");
195                             } else if ("header-comment".equals(tokens[0])) {
196                                 if (def.headerComment == null)
197                                     def.headerComment = tokens[2];
198                             } else {
199                                 System.err.println("Unknown token: "
200                                         + tokens[0]);
201                             }
202 
203                         } else {
204                             // System.out.println("Ignored line: " + line);
205                         }
206 
207                     } else {
208                         throw new IllegalStateException("Bad format: " + line);
209                     }
210                 }
211             }
212 
213         }
214 
215         if (def != null && def.command != null) {
216             if (def.result != null) {
217                 tests.add(def);
218             } else {
219                 System.err.println("Unexpected test sequence: " + def.command
220                         + "|" + def.result + "|" + def.smtpComment + "|"
221                         + def.headerComment + "|" + def.receivedSPF);
222             }
223         }
224 
225         br.close();
226 
227         return tests;
228     }
