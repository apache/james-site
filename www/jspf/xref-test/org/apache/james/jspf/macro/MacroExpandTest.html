115     public void testPercD() throws PermErrorException {
116         assertEquals("email.example.com", defIp4me.expandDomain("%{d}"));
117         assertEquals("email.example.com", defIp4me.expandDomain("%{d4}"));
118         assertEquals("email.example.com", defIp4me.expandDomain("%{d3}"));
119         assertEquals("example.com", defIp4me.expandDomain("%{d2}"));
120         assertEquals("com", defIp4me.expandDomain("%{d1}"));
121         assertEquals("com.example.email", defIp4me.expandDomain("%{dr}"));
122         assertEquals("example.email", defIp4me.expandDomain("%{d2r}"));
123     }

125     public void testPercL() throws PermErrorException {

127         assertEquals("strong.bad", defIp4me.expandDomain("%{l-}"));
