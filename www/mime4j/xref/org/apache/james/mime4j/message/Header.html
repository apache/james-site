45  /**
46   * The header of an entity (see RFC 2045).
47   *
48   * 
49   * @version $Id$
50   */
51  public class Header {
52      private List fields = new LinkedList();
53      private Map/*<String, List<Field>>*/ fieldMap = new HashMap();
54      
55      /**
56       * Creates a new empty <code>Header</code>.
57       */
58      public Header() {
59      }
60  
61      /**
62       * Creates a new <code>Header</code> from the specified stream.
63       * 
64       * @param is the stream to read the header from.
65       */
66      public Header(InputStream is) throws MimeException, IOException {
67          final MimeStreamParser parser = new MimeStreamParser();
68          parser.setContentHandler(new AbstractContentHandler() {
69              public void endHeader() {
70                  parser.stop();
71              }
72              public void field(String fieldData) throws MimeException {
73                  addField(Field.parse(fieldData));
74              }
75          });
76          parser.parse(is);
77      }
78  
79      /**
80       * Adds a field to the end of the list of fields.
81       * 
82       * @param field the field to add.
83       */
84      public void addField(Field field) {
85          List values = (List) fieldMap.get(field.getName().toLowerCase());
86          if (values == null) {
87              values = new LinkedList();
88              fieldMap.put(field.getName().toLowerCase(), values);
89          }
90          values.add(field);
91          fields.add(field);
92      }
93      
94      /**
95       * Gets the fields of this header. The returned list will not be
96       * modifiable.
97       * 
98       * @return the list of <code>Field</code> objects.
99       */
100     public List getFields() {
101         return Collections.unmodifiableList(fields);
102     }
103 
104     /**
105      * Gets a <code>Field</code> given a field name. If there are multiple
106      * such fields defined in this header the first one will be returned.
107      * 
108      * @param name the field name (e.g. From, Subject).
109      * @return the field or <code>null</code> if none found.
110      */
111     public Field getField(String name) {
112         List l = (List) fieldMap.get(name.toLowerCase());
113         if (l != null && !l.isEmpty()) {
114             return (Field) l.get(0);
115         }
116         return null;
117     }
118     
119     /**
120      * Gets all <code>Field</code>s having the specified field name. 
121      * 
122      * @param name the field name (e.g. From, Subject).
123      * @return the list of fields.
124      */
125     public List getFields(final String name) {
126         final String lowerCaseName = name.toLowerCase();
127         final List l = (List) fieldMap.get(lowerCaseName);
128         final List results;
129         if (l == null || l.isEmpty()) {
130             results = Collections.EMPTY_LIST;
131         } else {
132             results = Collections.unmodifiableList(l);
133         }
134         return results;
135     }
136 
137     /**
138      * Removes all <code>Field</code>s having the specified field name.
139      * 
140      * @param name
141      *            the field name (e.g. From, Subject).
142      * @return number of fields removed.
143      */
144     public int removeFields(String name) {
145         final String lowerCaseName = name.toLowerCase();
146         List removed = (List) fieldMap.remove(lowerCaseName);
147         if (removed == null || removed.isEmpty())
148             return 0;
149 
150         for (Iterator iterator = fields.iterator(); iterator.hasNext();) {
151             Field field = (Field) iterator.next();
152             if (field.getName().equalsIgnoreCase(name))
153                 iterator.remove();
154         }
155 
156         return removed.size();
157     }
158 
159     /**
160      * Sets or replaces a field. This method is useful for header fields such as
161      * Subject or Message-ID that should not occur more than once in a message.
162      * 
163      * If this <code>Header</code> does not already contain a header field of
164      * the same name as the given field then it is added to the end of the list
165      * of fields (same behavior as {@link #addField(Field)}). Otherwise the
166      * first occurrence of a field with the same name is replaced by the given
167      * field and all further occurrences are removed.
168      * 
169      * @param field the field to set.
170      */
171     public void setField(Field field) {
172         final String lowerCaseName = field.getName().toLowerCase();
173         List l = (List) fieldMap.get(lowerCaseName);
174         if (l == null || l.isEmpty()) {
175             addField(field);
176             return;
177         }
178 
179         l.clear();
180         l.add(field);
181 
182         int firstOccurrence = -1;
183         int index = 0;
184         for (Iterator iterator = fields.iterator(); iterator.hasNext(); index++) {
185             Field f = (Field) iterator.next();
186             if (f.getName().equalsIgnoreCase(field.getName())) {
187                 iterator.remove();
188 
189                 if (firstOccurrence == -1)
190                     firstOccurrence = index;
191             }
192         }
193 
194         fields.add(firstOccurrence, field);
195     }

197     /**
198      * Return Header Object as String representation. Each headerline is
199      * seperated by "\r\n"
200      * 
201      * @return headers
202      */
203     public String toString() {
204         CharArrayBuffer str = new CharArrayBuffer(128);
205         for (Iterator it = fields.iterator(); it.hasNext();) {
206             str.append(it.next().toString());
207             str.append("\r\n");
208         }
209         return str.toString();
210     }

213     /**
214      * Write the Header to the given OutputStream. 
215      * <p>
216      * Compatibility mode:
217      * <ul>
218      *  <li>
219      *   {@link MessageUtils#LENIENT}: use charset of the Content-Type header
220      *  </li>
221      *  <li>
222      *   {@link MessageUtils#STRICT_ERROR}: use US-ASCII and throw {@link MimeException} 
223      *    if a non ASCII character is encountered
224      *   </li>
225      *  <li>
226      *   {@link MessageUtils#STRICT_ERROR}: ignore non ASCII characters if encountered
227      *   </li>
228      * </ul>
229      * @param out the OutputStream to write to
230      * @param mode compatibility mode:
231      *   {@link MessageUtils#LENIENT}, {@link MessageUtils#STRICT_ERROR}, {@link MessageUtils#STRICT_IGNORE}  
232      * 
233      * @throws IOException if case of an I/O error
234      * @throws MimeException if case of a MIME protocol violation
235      */
236     public void writeTo(final OutputStream out, int mode) throws IOException, MimeException {
237         Charset charset = null;
238         if (mode == MessageUtils.LENIENT) {
239             final ContentTypeField contentTypeField = ((ContentTypeField) getField(Field.CONTENT_TYPE));
240             if (contentTypeField == null) {
241                 charset = MessageUtils.DEFAULT_CHARSET;
242             } else {
243                 final String contentTypeFieldCharset = contentTypeField.getCharset();
244                 if (contentTypeField != null && contentTypeFieldCharset != null) {
245                     charset = CharsetUtil.getCharset(contentTypeFieldCharset);
246                 } else {
247                     charset = MessageUtils.ISO_8859_1;
248                 }
249             }
250         } else {
251             charset = MessageUtils.DEFAULT_CHARSET;
252         }
253         BufferedWriter writer = new BufferedWriter(
254                 new OutputStreamWriter(out, charset), 8192);
255         for (Iterator it = fields.iterator(); it.hasNext();) {
256             Field field = (Field) it.next();
257             String fs = field.toString();
258             if (mode == MessageUtils.STRICT_ERROR && !MessageUtils.isASCII(fs)) {
259                 throw new MimeException("Header '" + fs + "' violates RFC 822");
260             }
261             writer.write(fs);
262             writer.write(MessageUtils.CRLF);
263         }
264         writer.write(MessageUtils.CRLF);
265         writer.flush();
266     }
