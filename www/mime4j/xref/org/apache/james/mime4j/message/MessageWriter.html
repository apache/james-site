39  /**
40   * Writes a message (or a part of a message) to an output stream.
41   * <p>
42   * This class cannot be instantiated; instead the three static instances
43   * {@link #STRICT_ERROR}, {@link #STRICT_IGNORE} or {@link #LENIENT} implement
44   * different strategies for encoding header fields and preamble and epilogue of
45   * a multipart.
46   * <p>
47   * This class can also be subclassed to implement custom strategies for writing
48   * messages.
49   */
50  public class MessageWriter {
51  
52      private static final Charset LENIENT_FALLBACK_CHARSET = CharsetUtil.ISO_8859_1;
53      private static final String CRLF = CharsetUtil.CRLF;
54      private static final int WRITER_BUFFER_SIZE = 4096;
55  
56      /**
57       * A message writer that uses US-ASCII for encoding and throws
58       * {@link MimeIOException} if a non ASCII character is encountered.
59       */
60      public static final MessageWriter STRICT_ERROR = new MessageWriter();
61  
62      /**
63       * A message writer that uses US-ASCII for encoding but ignores non ASCII
64       * characters.
65       */
66      public static final MessageWriter STRICT_IGNORE = new MessageWriter() {
67          @Override
68          protected CharsetEncoder getCharsetEncoder(ContentTypeField contentType) {
69              return ignoreEncoder(CharsetUtil.DEFAULT_CHARSET);
70          }
71      };
72  
73      /**
74       * A message writer that uses the charset of the Content-Type header for
75       * encoding.
76       */
77      public static final MessageWriter LENIENT = new MessageWriter() {
78          @Override
79          protected CharsetEncoder getCharsetEncoder(ContentTypeField contentType) {
80              if (contentType == null) {
81                  return ignoreEncoder(LENIENT_FALLBACK_CHARSET);
82              } else {
83                  String charset = contentType.getCharset();
84                  if (charset != null) {
85                      return ignoreEncoder(CharsetUtil.getCharset(charset));
86                  } else {
87                      return ignoreEncoder(LENIENT_FALLBACK_CHARSET);
88                  }
89              }
90          }
91      };
92  
93      /**
94       * Protected constructor prevents direct instantiation.
95       */
96      protected MessageWriter() {
97      }

99      /**
100      * Write the specified <code>Body</code> to the specified
101      * <code>OutputStream</code>.
102      * 
103      * @param body
104      *            the <code>Body</code> to write.
105      * @param out
106      *            the OutputStream to write to.
107      * @throws IOException
108      *             if an I/O error occurs.
109      * @throws MimeIOException
110      *             in case of a MIME protocol violation
111      */
112     public void writeBody(Body body, OutputStream out) throws IOException,
113             MimeIOException {
114         if (body instanceof Message) {
115             writeEntity((Message) body, out);
116         } else if (body instanceof Multipart) {
117             writeMultipart((Multipart) body, out);
118         } else if (body instanceof SingleBody) {
119             ((SingleBody) body).writeTo(out);
120         } else
121             throw new IllegalArgumentException("Unsupported body class");
122 
123         out.flush();
124     }
125 
126     /**
127      * Write the specified <code>Entity</code> to the specified
128      * <code>OutputStream</code>.
129      * 
130      * @param entity
131      *            the <code>Entity</code> to write.
132      * @param out
133      *            the OutputStream to write to.
134      * @throws IOException
135      *             if an I/O error occurs.
136      * @throws MimeIOException
137      *             in case of a MIME protocol violation
138      */
139     public void writeEntity(Entity entity, OutputStream out)
140             throws IOException, MimeIOException {
141         final Header header = entity.getHeader();
142         if (header == null)
143             throw new IllegalArgumentException("Missing header");
144 
145         writeHeader(header, out);
146         out.flush();
147 
148         final Body body = entity.getBody();
149         if (body == null)
150             throw new IllegalArgumentException("Missing body");
151 
152         boolean binaryBody = body instanceof BinaryBody;
153         OutputStream encOut = encodeStream(out, entity
154                 .getContentTransferEncoding(), binaryBody);
155 
156         writeBody(body, encOut);
157 
158         // close if wrapped (base64 or quoted-printable)
159         if (encOut != out)
160             encOut.close();
161     }
162 
163     /**
164      * Write the specified <code>Multipart</code> to the specified
165      * <code>OutputStream</code>.
166      * 
167      * @param multipart
168      *            the <code>Multipart</code> to write.
169      * @param out
170      *            the OutputStream to write to.
171      * @throws IOException
172      *             if an I/O error occurs.
173      * @throws MimeIOException
174      *             in case of a MIME protocol violation
175      */
176     public void writeMultipart(Multipart multipart, OutputStream out)
177             throws IOException, MimeIOException {
178         ContentTypeField contentType = getContentType(multipart);
179 
180         String boundary = getBoundary(contentType);
181 
182         Writer writer = getWriter(contentType, out);
183 
184         try {
185             writer.write(multipart.getPreamble());
186             writer.write(CRLF);
187 
188             for (BodyPart bodyPart : multipart.getBodyParts()) {
189                 writer.write("--");
190                 writer.write(boundary);
191                 writer.write(CRLF);
192                 writer.flush();
193 
194                 writeEntity(bodyPart, out);
195                 writer.write(CRLF);
196             }
197 
198             writer.write("--");
199             writer.write(boundary);
200             writer.write("--");
201             writer.write(CRLF);
202 
203             writer.write(multipart.getEpilogue());
204 
205             writer.flush();
206         } catch (CharacterCodingException e) {
207             throw new MimeIOException("Multipart violates RFC 822");
208         }
209     }
210 
211     /**
212      * Write the specified <code>Header</code> to the specified
213      * <code>OutputStream</code>.
214      * 
215      * @param header
216      *            the <code>Header</code> to write.
217      * @param out
218      *            the OutputStream to write to.
219      * @throws IOException
220      *             if an I/O error occurs.
221      * @throws MimeIOException
222      *             in case of a MIME protocol violation
223      */
224     public void writeHeader(Header header, OutputStream out)
225             throws IOException, MimeIOException {
226         Writer writer = getWriter((ContentTypeField) header
227                 .getField(Field.CONTENT_TYPE), out);
228 
229         try {
230             for (Field field : header) {
231                 writer.write(field.getRaw());
232                 writer.write(CRLF);
233             }
234 
235             writer.write(CRLF);
236             writer.flush();
237         } catch (CharacterCodingException e) {
238             throw new MimeIOException("Header violates RFC 822");
239         }
240     }
241 
242     CharsetEncoder getCharsetEncoder(ContentTypeField contentType) {
243         return CharsetUtil.DEFAULT_CHARSET.newEncoder();
244     }
245 
246     CharsetEncoder ignoreEncoder(Charset charset) {
247         return charset.newEncoder().onMalformedInput(CodingErrorAction.REPLACE)
248                 .onUnmappableCharacter(CodingErrorAction.REPLACE);
249     }
250 
251     private OutputStream encodeStream(OutputStream out, String encoding,
252             boolean binaryBody) throws IOException {
253         if (MimeUtil.isBase64Encoding(encoding)) {
254             return CodecUtil.wrapBase64(out);
255         } else if (MimeUtil.isQuotedPrintableEncoded(encoding)) {
256             return CodecUtil.wrapQuotedPrintable(out, binaryBody);
257         } else {
258             return out;
259         }
260     }
261 
262     private ContentTypeField getContentType(Multipart multipart) {
263         Entity parent = multipart.getParent();
264         if (parent == null)
265             throw new IllegalArgumentException(
266                     "Missing parent entity in multipart");
267 
268         Header header = parent.getHeader();
269         if (header == null)
270             throw new IllegalArgumentException(
271                     "Missing header in parent entity");
272 
273         ContentTypeField contentType = (ContentTypeField) header
274                 .getField(Field.CONTENT_TYPE);
275         if (contentType == null)
276             throw new IllegalArgumentException(
277                     "Content-Type field not specified");
278 
279         return contentType;
280     }
281 
282     private String getBoundary(ContentTypeField contentType) {
283         String boundary = contentType.getBoundary();
284         if (boundary == null)
285             throw new IllegalArgumentException(
286                     "Multipart boundary not specified");
287 
288         return boundary;
289     }
290 
291     private Writer getWriter(ContentTypeField contentType, OutputStream out) {
292         CharsetEncoder encoder = getCharsetEncoder(contentType);
293 
294         return new BufferedWriter(new OutputStreamWriter(out, encoder),
295                 WRITER_BUFFER_SIZE);
296     }
