115     /**
116      * Parses the specified MIME message stream into a <code>Message</code>
117      * instance using given {@link MimeEntityConfig} and {@link StorageProvider}.
118      * 
119      * @param is the stream to parse.
120      * @param config {@link MimeEntityConfig} to use.
121      * @param storageProvider {@link StorageProvider} to use for storing text
122      *        and binary message bodies.
123      * @throws IOException on I/O errors.
124      * @throws MimeIOException on MIME protocol violations.
125      */
126     public Message(InputStream is, MimeEntityConfig config,
127             StorageProvider storageProvider) throws IOException,
128             MimeIOException {
129         try {
130             MimeStreamParser parser = new MimeStreamParser(config);
131             parser.setContentHandler(new MessageBuilder(this, storageProvider));
132             parser.parse(is);
133         } catch (MimeException e) {
134             throw new MimeIOException(e);
135         }
136     }
