38  /**
39   * Represents a MIME message. The following code parses a stream into a 
40   * <code>Message</code> object.
41   * 
42   * <pre>
43   *      Message msg = new Message(new BufferedInputStream(
44   *                                      new FileInputStream("mime.msg")));
45   * </pre>
46   * 
47   *
48   * 
49   * @version $Id: Message.java,v 1.3 2004/10/02 12:41:11 ntherning Exp $
50   */
51  public class Message extends Entity implements Body {
52      
53      /**
54       * Creates a new empty <code>Message</code>.
55       */
56      public Message() {
57      }
58      
59      /**
60       * Parses the specified MIME message stream into a <code>Message</code>
61       * instance.
62       * 
63       * @param is the stream to parse.
64       * @throws IOException on I/O errors.
65       */
66      public Message(InputStream is) throws IOException {
67          MimeStreamParser parser = new MimeStreamParser();
68          parser.setContentHandler(new MessageBuilder());
69          try {
70              parser.parse(is);
71          } catch (MimeException e) {
72              IllegalStateException ise = new IllegalStateException(
73                      "Unexpected message processing error");
74              ise.initCause(e);
75              throw ise;
76          }
77      }
78  
79      
80      /**
81       * Gets the <code>Subject</code> field.
82       * 
83       * @return the <code>Subject</code> field or <code>null</code> if it
84       *         doesn't exist.
85       */
86      public UnstructuredField getSubject() {
87          return (UnstructuredField) getHeader().getField(Field.SUBJECT);
88      }
89  
90      /**
91       * 
92       * @see org.apache.james.mime4j.message.Entity#writeTo(java.io.OutputStream)
93       */
94      public void writeTo(OutputStream out) throws IOException {
95          getHeader().writeTo(out);
96  
97          Body body = getBody();
98          if (body instanceof Multipart) {
99              Multipart mp = (Multipart) body;
100             mp.writeTo(out);
101         } else {
102             body.writeTo(out);
103         }
104     }

107     private class MessageBuilder implements ContentHandler {
108         private Stack stack = new Stack();
109         
110         public MessageBuilder() {
111         }
112         
113         private void expect(Class c) {
114             if (!c.isInstance(stack.peek())) {
115                 throw new IllegalStateException("Internal stack error: "
116                         + "Expected '" + c.getName() + "' found '"
117                         + stack.peek().getClass().getName() + "'");
118             }
119         }
120         
121         /**
122          * @see org.apache.james.mime4j.ContentHandler#startMessage()
123          */
124         public void startMessage() {
125             if (stack.isEmpty()) {
126                 stack.push(Message.this);
127             } else {
128                 expect(Entity.class);
129                 Message m = new Message();
130                 ((Entity) stack.peek()).setBody(m);
131                 stack.push(m);
132             }
133         }
134         
135         /**
136          * @see org.apache.james.mime4j.ContentHandler#endMessage()
137          */
138         public void endMessage() {
139             expect(Message.class);
140             stack.pop();
141         }
142         
143         /**
144          * @see org.apache.james.mime4j.ContentHandler#startHeader()
145          */
146         public void startHeader() {
147             stack.push(new Header());
148         }
149         
150         /**
151          * @see org.apache.james.mime4j.ContentHandler#field(java.lang.String)
152          */
153         public void field(String fieldData) {
154             expect(Header.class);
155             ((Header) stack.peek()).addField(Field.parse(fieldData));
156         }
157         
158         /**
159          * @see org.apache.james.mime4j.ContentHandler#endHeader()
160          */
161         public void endHeader() {
162             expect(Header.class);
163             Header h = (Header) stack.pop();
164             expect(Entity.class);
165             ((Entity) stack.peek()).setHeader(h);
166         }
167         
168         /**
169          * @see org.apache.james.mime4j.ContentHandler#startMultipart(org.apache.james.mime4j.BodyDescriptor)
170          */
171         public void startMultipart(BodyDescriptor bd) {
172             expect(Entity.class);
173             
174             Entity e = (Entity) stack.peek();
175             Multipart multiPart = new Multipart();
176             e.setBody(multiPart);
177             stack.push(multiPart);
178         }
179         
180         /**
181          * @see org.apache.james.mime4j.ContentHandler#body(org.apache.james.mime4j.BodyDescriptor, java.io.InputStream)
182          */
183         public void body(BodyDescriptor bd, InputStream is) throws IOException {
184             expect(Entity.class);
185             
186             String enc = bd.getTransferEncoding();
187             if (MimeUtil.ENC_BASE64.equals(enc)) {
188                 is = new Base64InputStream(is);
189             } else if (MimeUtil.ENC_QUOTED_PRINTABLE.equals(enc)) {
190                 is = new QuotedPrintableInputStream(is);
191             }
192             
193             Body body = null;
194             if (bd.getMimeType().startsWith("text/")) {
195                 body = new TempFileTextBody(is, bd.getCharset());
196             } else {
197                 body = new TempFileBinaryBody(is);
198             }
199             
200             ((Entity) stack.peek()).setBody(body);
201         }
202         
203         /**
204          * @see org.apache.james.mime4j.ContentHandler#endMultipart()
205          */
206         public void endMultipart() {
207             stack.pop();
208         }
209         
210         /**
211          * @see org.apache.james.mime4j.ContentHandler#startBodyPart()
212          */
213         public void startBodyPart() {
214             expect(Multipart.class);
215             
216             BodyPart bodyPart = new BodyPart();
217             ((Multipart) stack.peek()).addBodyPart(bodyPart);
218             stack.push(bodyPart);
219         }
220         
221         /**
222          * @see org.apache.james.mime4j.ContentHandler#endBodyPart()
223          */
224         public void endBodyPart() {
225             expect(BodyPart.class);
226             stack.pop();
227         }
228         
229         /**
230          * @see org.apache.james.mime4j.ContentHandler#epilogue(java.io.InputStream)
231          */
232         public void epilogue(InputStream is) throws IOException {
233             expect(Multipart.class);
234             StringBuffer sb = new StringBuffer();
235             int b;
236             while ((b = is.read()) != -1) {
237                 sb.append((char) b);
238             }
239             ((Multipart) stack.peek()).setEpilogue(sb.toString());
240         }
241         
242         /**
243          * @see org.apache.james.mime4j.ContentHandler#preamble(java.io.InputStream)
244          */
245         public void preamble(InputStream is) throws IOException {
246             expect(Multipart.class);
247             StringBuffer sb = new StringBuffer();
248             int b;
249             while ((b = is.read()) != -1) {
250                 sb.append((char) b);
251             }
252             ((Multipart) stack.peek()).setPreamble(sb.toString());
253         }
