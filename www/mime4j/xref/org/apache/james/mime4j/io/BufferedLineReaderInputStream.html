27  /**
28   * Input buffer that can be used to search for patterns using Quick Search 
29   * algorithm in data read from an {@link InputStream}. 
30   */
31  public class BufferedLineReaderInputStream extends LineReaderInputStream {

33      private byte[] buffer;
34      
35      private int bufpos;
36      private int buflen;

38      private final int maxLineLen;

40      public BufferedLineReaderInputStream(
41              final InputStream instream, 
42              int buffersize,
43              int maxLineLen) {
44          super(instream);
45          if (instream == null) {
46              throw new IllegalArgumentException("Input stream may not be null");
47          }
48          if (buffersize <= 0) {
49              throw new IllegalArgumentException("Buffer size may not be negative or zero");
50          }
51          this.buffer = new byte[buffersize];
52          this.bufpos = 0;
53          this.buflen = 0;
54          this.maxLineLen = maxLineLen;
55      }

57      public BufferedLineReaderInputStream(
58              final InputStream instream, 
59              int buffersize) {
60          this(instream, buffersize, -1);
61      }

63      private void expand(int newlen) {
64          byte newbuffer[] = new byte[newlen];
65          int len = this.buflen - this.bufpos;
66          if (len > 0) {
67              System.arraycopy(this.buffer, this.bufpos, newbuffer, this.bufpos, len);
68          }
69          this.buffer = newbuffer;
70      }
71      
72      public void ensureCapacity(int len) {
73          if (len > this.buffer.length) {
74              expand(len);
75          }
76      }
77      
78      public int fillBuffer() throws IOException {
79          // compact the buffer if necessary
80          if (this.bufpos > 0) {
81              int len = this.buflen - this.bufpos;
82              if (len > 0) {
83                  System.arraycopy(this.buffer, this.bufpos, this.buffer, 0, len);
84              }
85              this.bufpos = 0;
86              this.buflen = len;
87          }
88          int l;
89          int off = this.buflen;
90          int len = this.buffer.length - off;
91          l = in.read(this.buffer, off, len);
92          if (l == -1) {
93              return -1;
94          } else {
95              this.buflen = off + l;
96              return l;
97          }
98      }

100     public boolean hasBufferedData() {
101         return this.bufpos < this.buflen;
102     }

104     @Override
105     public int read() throws IOException {
106         int noRead = 0;
107         while (!hasBufferedData()) {
108             noRead = fillBuffer();
109             if (noRead == -1) {
110                 return -1;
111             }
112         }
113         return this.buffer[this.bufpos++] & 0xff;
114     }

116     @Override
117     public int read(final byte[] b, int off, int len) throws IOException {
118         if (b == null) {
119             return 0;
120         }
121         int noRead = 0;
122         while (!hasBufferedData()) {
123             noRead = fillBuffer();
124             if (noRead == -1) {
125                 return -1;
126             }
127         }
128         int chunk = this.buflen - this.bufpos;
129         if (chunk > len) {
130             chunk = len;
131         }
132         System.arraycopy(this.buffer, this.bufpos, b, off, chunk);
133         this.bufpos += chunk;
134         return chunk;
135     }

137     @Override
138     public int read(final byte[] b) throws IOException {
139         if (b == null) {
140             return 0;
141         }
142         return read(b, 0, b.length);
143     }

151     @Override
152     public int readLine(final ByteArrayBuffer dst) throws IOException {
153         if (dst == null) {
154             throw new IllegalArgumentException("Buffer may not be null");
155         }
156         int total = 0;
157         boolean found = false;
158         int bytesRead = 0;
159         while (!found) {
160             if (!hasBufferedData()) {
161                 bytesRead = fillBuffer();
162                 if (bytesRead == -1) {
163                     break;
164                 }
165             }
166             int i = indexOf((byte)'\n');
167             int chunk;
168             if (i != -1) {
169                 found = true;
170                 chunk = i + 1 - pos();
171             } else {
172                 chunk = length();
173             }
174             if (chunk > 0) {
175                 dst.append(buf(), pos(), chunk);
176                 skip(chunk);
177                 total += chunk;
178             }
179             if (this.maxLineLen > 0 && dst.length() >= this.maxLineLen) {
180                 throw new MaxLineLimitException("Maximum line length limit exceeded");
181             }
182         }
183         if (total == 0 && bytesRead == -1) {
184             return -1;
185         } else {
186             return total;
187         }
188     }

190     /**
191      * Implements quick search algorithm as published by
192      * <p> 
193      * SUNDAY D.M., 1990, 
194      * A very fast substring search algorithm, 
195      * Communications of the ACM . 33(8):132-142.
196      * </p>
197      */
198     public int indexOf(final byte[] pattern, int off, int len) {
199         if (pattern == null) {
200             throw new IllegalArgumentException("Pattern may not be null");
201         }
202         if (off < this.bufpos || len < 0 || off + len > this.buflen) {
203             throw new IndexOutOfBoundsException();
204         }
205         if (len < pattern.length) {
206             return -1;
207         }
208         
209         int[] shiftTable = new int[256];
210         for (int i = 0; i < shiftTable.length; i++) {
211             shiftTable[i] = pattern.length + 1;
212         }
213         for (int i = 0; i < pattern.length; i++) {
214             int x = pattern[i] & 0xff;
215             shiftTable[x] = pattern.length - i;
216         }
217         
218         int j = 0;
219         while (j <= len - pattern.length) {
220             int cur = off + j;
221             boolean match = true;
222             for (int i = 0; i < pattern.length; i++) {
223                 if (this.buffer[cur + i] != pattern[i]) {
224                     match = false;
225                     break;
226                 }
227             }
228             if (match) {
229                 return cur;
230             }
231             
232             int pos = cur + pattern.length; 
233             if (pos >= this.buffer.length) {
234                 break;
235             }
236             int x = this.buffer[pos] & 0xff;
237             j += shiftTable[x];
238         }
239         return -1;
240     }

243      * Implements quick search algorithm as published by
244      * <p>

246      * A very fast substring search algorithm, 
247      * Communications of the ACM . 33(8):132-142.
248      * </p>

251         return indexOf(pattern, this.bufpos, this.buflen - this.bufpos);

254     public int indexOf(byte b, int off, int len) {
255         if (off < this.bufpos || len < 0 || off + len > this.buflen) {
256             throw new IndexOutOfBoundsException();
257         }
258         for (int i = off; i < off + len; i++) {
259             if (this.buffer[i] == b) {
260                 return i;
261             }
262         }
263         return -1;
264     }

266     public int indexOf(byte b) {
267         return indexOf(b, this.bufpos, this.buflen - this.bufpos);
268     }

297     public int skip(int n) {
298         int chunk = Math.min(n, this.buflen - this.bufpos);
299         this.bufpos += chunk; 
300         return chunk;

309     public String toString() {
310         StringBuilder buffer = new StringBuilder();
311         buffer.append("[pos: ");
312         buffer.append(this.bufpos);
313         buffer.append("]");
314         buffer.append("[limit: ");
315         buffer.append(this.buflen);

317         buffer.append("[");
