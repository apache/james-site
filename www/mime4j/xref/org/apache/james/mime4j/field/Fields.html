44      public static ContentTypeField contentType(String mimeType,
45              Map<String, String> parameters) {
46          if (!isValidMimeType(mimeType))
47              throw new IllegalArgumentException();
48  
49          if (parameters == null || parameters.isEmpty()) {
50              return parse(ContentTypeField.class, Field.CONTENT_TYPE, mimeType);
51          } else {
52              StringBuilder sb = new StringBuilder(mimeType);
53              for (Map.Entry<String, String> entry : parameters.entrySet()) {
54                  sb.append("; ");
55                  sb.append(EncoderUtil.encodeHeaderParameter(entry.getKey(),
56                          entry.getValue()));
57              }
58              String contentType = sb.toString();
59              return contentType(contentType);
60          }
61      }

107     public static ContentDispositionField contentDisposition(
108             String dispositionType, String filename, long size,
109             Date creationDate, Date modificationDate, Date readDate) {
110         Map<String, String> parameters = new HashMap<String, String>();
111         if (filename != null) {
112             parameters.put(ContentDispositionField.PARAM_FILENAME, filename);
113         }
114         if (size >= 0) {
115             parameters.put(ContentDispositionField.PARAM_SIZE, Long
116                     .toString(size));
117         }
118         if (creationDate != null) {
119             parameters.put(ContentDispositionField.PARAM_CREATION_DATE,
120                     MimeUtil.formatDate(creationDate, null));
121         }
122         if (modificationDate != null) {
123             parameters.put(ContentDispositionField.PARAM_MODIFICATION_DATE,
124                     MimeUtil.formatDate(modificationDate, null));
125         }
126         if (readDate != null) {
127             parameters.put(ContentDispositionField.PARAM_READ_DATE, MimeUtil
128                     .formatDate(readDate, null));
129         }
130         return contentDisposition(dispositionType, parameters);
131     }

287     private static boolean isValidMimeType(String mimeType) {
288         if (mimeType == null)
289             return false;
290 
291         int idx = mimeType.indexOf('/');
292         if (idx == -1)
293             return false;
294 
295         String type = mimeType.substring(0, idx);
296         String subType = mimeType.substring(idx + 1);
297         return EncoderUtil.isToken(type) && EncoderUtil.isToken(subType);
298     }
299 
300     private static boolean isValidDispositionType(String dispositionType) {
301         if (dispositionType == null)
302             return false;
303 
304         return EncoderUtil.isToken(dispositionType);
305     }

307     private static <F extends Field> F parse(Class<F> fieldClass,
308             String fieldName, String fieldBody) {
309         try {
310             String raw = MimeUtil.fold(fieldName + ": " + fieldBody, 0);
311 
312             Field field = Field.parse(raw);
313             if (!fieldClass.isInstance(field)) {
314                 throw new IllegalArgumentException("Incompatible field name: "
315                         + fieldName);
316             }
317 
318             return fieldClass.cast(field);
319         } catch (MimeException e) {
320             throw new IllegalArgumentException("Illegal field name: "
321                     + fieldName);
322         }
323     }

325     private static String encodeAddresses(Iterable<? extends Address> addresses) {
326         StringBuilder sb = new StringBuilder();
327 
328         for (Address address : addresses) {
329             if (sb.length() > 0) {
330                 sb.append(", ");
331             }
332             sb.append(address.getEncodedString());
333         }
334 
335         return sb.toString();
336     }
