38  /**
39   * <p>
40   * Parses MIME (or RFC822) message streams of bytes or characters.
41   * The stream is converted into an event stream.
42   * <p>
43   * <p>
44   * Typical usage:
45   * </p>
46   * <pre>
47   *      MimeTokenStream stream = new MimeTokenStream();
48   *      stream.parse(new BufferedInputStream(new FileInputStream("mime.msg")));
49   *      for (int state = stream.getState();
50   *           state != MimeTokenStream.T_END_OF_STREAM;
51   *           state = stream.next()) {
52   *          switch (state) {
53   *            case MimeTokenStream.T_BODY:
54   *              System.out.println("Body detected, contents = "
55   *                + stream.getInputStream() + ", header data = "
56   *                + stream.getBodyDescriptor());
57   *              break;
58   *            case MimeTokenStream.T_FIELD:
59   *              System.out.println("Header field detected: "
60   *                + stream.getField());
61   *              break;
62   *            case MimeTokenStream.T_START_MULTIPART:
63   *              System.out.println("Multipart message detexted,"
64   *                + " header data = "
65   *                + stream.getBodyDescriptor());
66   *            ...
67   *          }
68   *      }
69   * </pre>
70   * <p>
71   * <strong>NOTE:</strong> All lines must end with CRLF 
72   * (<code>\r\n</code>). If you are unsure of the line endings in your stream 
73   * you should wrap it in a {@link org.apache.james.mime4j.EOLConvertingInputStream}
74   * instance.</p>
75   * <p>Instances of {@link MimeTokenStream} are reusable: Invoking the
76   * method {@link #parse(InputStream)} resets the token streams internal
77   * state. However, they are definitely <em>not</em> thread safe. If you
78   * have a multi threaded application, then the suggested use is to have
79   * one instance per thread.</p>
80   * 
81   * @version $Id: MimeStreamParser.java,v 1.8 2005/02/11 10:12:02 ntherning Exp $
82   */
83  public class MimeTokenStream {
84      private static final Log log = LogFactory.getLog(MimeTokenStream.class);

86      /**
87       * This token indicates, that the MIME stream has been completely
88       * and successfully parsed, and no more data is available.
89       */
90      public static final int T_END_OF_STREAM = -1;
91      /**
92       * This token indicates, that the MIME stream is currently
93       * at the beginning of a message.
94       */
95      public static final int T_START_MESSAGE = 0;
96      /**
97       * This token indicates, that the MIME stream is currently
98       * at the end of a message.
99       */
100     public static final int T_END_MESSAGE = 1;
101     /**
102      * This token indicates, that a raw entity is currently being processed.
103      * You may call {@link #getInputStream()} to obtain the raw entity
104      * data.
105      */
106     public static final int T_RAW_ENTITY = 2;
107     /**
108      * This token indicates, that a message parts headers are now
109      * being parsed.
110      */
111     public static final int T_START_HEADER = 3;
112     /**
113      * This token indicates, that a message parts field has now
114      * been parsed. You may call {@link #getField()} to obtain the
115      * raw field contents.
116      */
117     public static final int T_FIELD = 4;
118     /**
119      * This token indicates, that part headers have now been
120      * parsed.
121      */
122     public static final int T_END_HEADER = 5;
123     /**
124      * This token indicates, that a multipart body is being parsed.
125      */
126     public static final int T_START_MULTIPART = 6;
127     /**
128      * This token indicates, that a multipart body has been parsed.
129      */
130     public static final int T_END_MULTIPART = 7;
131     /**
132      * This token indicates, that a multiparts preamble is being
133      * parsed. You may call {@link #getInputStream()} to access the
134      * preamble contents.
135      */
136     public static final int T_PREAMBLE = 8;
137     /**
138      * This token indicates, that a multiparts epilogue is being
139      * parsed. You may call {@link #getInputStream()} to access the
140      * epilogue contents.
141      */
142     public static final int T_EPILOGUE = 9;
143     /**
144      * This token indicates, that the MIME stream is currently
145      * at the beginning of a body part.
146      */
147     public static final int T_START_BODYPART = 10;
148     /**
149      * This token indicates, that the MIME stream is currently
150      * at the end of a body part.
151      */
152     public static final int T_END_BODYPART = 11;
153     /**
154      * This token indicates, that an atomic entity is being parsed.
155      * Use {@link #getInputStream()} to access the entity contents.
156      */
157     public static final int T_BODY = 12;
158     /**
159      * Internal state, not exposed.
160      */
161     private static final int T_IN_BODYPART = -2;
162     /**
163      * Internal state, not exposed.
164      */
165     private static final int T_IN_MESSAGE = -3;

167     private static final BitSet fieldChars = new BitSet();
168     static {
169         for (int i = 0x21; i <= 0x39; i++) {
170             fieldChars.set(i);
171         }
172         for (int i = 0x3b; i <= 0x7e; i++) {
173             fieldChars.set(i);
174         }
175     }

222             case T_START_MULTIPART:
223                 result = "Start multipart";
224                 break;
225             case T_END_MULTIPART:
226                 result = "End multipart";
227                 break;
228             case T_PREAMBLE:

230                 break;
231             case T_EPILOGUE:

307     abstract static class StateMachine {
308         int state;
309         abstract int next() throws IOException, MimeException;
310         abstract InputStream read();
311     }

313     private static class RawEntity extends StateMachine {
314         private InputStream stream;
315         RawEntity(InputStream stream) {
316             this.stream = stream;
317             state = T_RAW_ENTITY;
318         }
319         int next() {
320             state = T_END_OF_STREAM;
321             return state;
322         }
323         InputStream read() {
324             return stream;
325         }

328     private abstract class Entity extends StateMachine {
329         private final BodyDescriptor parent;
330         private final Cursor cursor;
331         private final StringBuffer sb = new StringBuffer();
332         private BodyDescriptor body;
333         private int pos, start;
334         private int lineNumber, startLineNumber;
335         private final int endState;
336         
337         String field, fieldName, fieldValue;
338 
339         Entity(Cursor cursor, BodyDescriptor parent, int startState, int endState) {
340             this.parent = parent;
341             this.cursor = cursor;
342             state = startState;
343             this.endState = endState;
344         }
345 
346         private void setParsingFieldState() {
347             state = parseField() ? T_FIELD : T_END_HEADER;
348         }
349 
350         private int setParseBodyPartState() throws IOException, MimeException {
351             cursor.advanceToBoundary();
352             if (cursor.isEnded()) {
353                 monitor(Event.MIME_BODY_PREMATURE_END);
354             } else {
355                 if (cursor.moreMimeParts()) {
356                     final String boundary = body.getBoundary();
357                     cursor.boundary(boundary);
358                    
359                     if (isRaw()) {
360                         currentStateMachine = new RawEntity(cursor.nextSection());
361                     } else {
362                         currentStateMachine = new BodyPart(cursor.nextMimePartCursor(), body);
363                     }
364                     entities.add(currentStateMachine);
365                     state = T_IN_BODYPART;
366                     return currentStateMachine.state;
367                 }
368             }
369             state = T_EPILOGUE;
370             return T_EPILOGUE;
371         }
372 
373         int next() throws IOException, MimeException {
374             switch (state) {
375                 case T_START_MESSAGE:
376                 case T_START_BODYPART:
377                     state = T_START_HEADER;
378                     break;
379                 case T_START_HEADER:
380                     initHeaderParsing();
381                     setParsingFieldState();
382                     break;
383                 case T_FIELD:
384                     setParsingFieldState();
385                     break;
386                 case T_END_HEADER:
387                     final String mimeType = body.getMimeType();
388                     if (MimeUtil.isMultipart(mimeType)) {
389                         state = T_START_MULTIPART;
390                     } else if (recursionMode != M_NO_RECURSE && MimeUtil.isMessage(mimeType)) {
391                         Cursor nextCursor = cursor;
392                         final String transferEncoding = body.getTransferEncoding();
393                         if (MimeUtil.isBase64Encoding(transferEncoding)) {
394                             log.debug("base64 encoded message/rfc822 detected");
395                             nextCursor = cursor.decodeBase64();
396                         } else if (MimeUtil.isQuotedPrintableEncoded(transferEncoding)) {
397                             log.debug("quoted-printable encoded message/rfc822 detected");
398                             nextCursor = cursor.decodeQuotedPrintable();
399                         }
400                         state = T_IN_MESSAGE;
401                         return parseMessage(nextCursor, body);
402                     } else {
403                         state = T_BODY;
404                         break;
405                     }
406                     break;
407                 case T_START_MULTIPART:
408                     cursor.boundary(body.getBoundary());
409                     if (cursor.isEnded()) {
410                         state = T_END_MULTIPART;
411                     } else {
412                         state = T_PREAMBLE;
413                     }
414                     break;
415                 case T_PREAMBLE:
416                     return setParseBodyPartState();
417                 case T_IN_BODYPART:
418                     return setParseBodyPartState();
419                 case T_EPILOGUE:
420                     state = T_END_MULTIPART;
421                     break;
422                 case T_BODY:
423                 case T_END_MULTIPART:
424                 case T_IN_MESSAGE:
425                     state = endState;
426                     break;
427                 default:
428                     if (state == endState) {
429                         state = T_END_OF_STREAM;
430                         break;
431                     }
432                     throw new IllegalStateException("Invalid state: " + state);
433             }
434             return state;
435         }
436 
437         InputStream read() {
438             switch (getState()) {
439                 case T_PREAMBLE:
440                 case T_EPILOGUE:
441                 case T_BODY:
442                     return cursor.nextSection();
443                 case T_START_MULTIPART:
444                     return cursor.rest();
445                 default:
446                     throw new IllegalStateException("Expected state to be either of T_RAW_ENTITY, T_PREAMBLE, or T_EPILOGUE.");
447             }
448         }
449         
450         private void initHeaderParsing() throws IOException, MimeException {
451             body = newBodyDescriptor(parent);
452             startLineNumber = lineNumber = cursor.getLineNumber();
453 
454             int curr = 0;
455             int prev = 0;
456             while ((curr = cursor.advance()) != -1) {
457                 if (curr == '\n' && (prev == '\n' || prev == 0)) {
458                     /*
459                      * [\r]\n[\r]\n or an immediate \r\n have been seen.
460                      */
461                     sb.deleteCharAt(sb.length() - 1);
462                     break;
463                 }
464                 sb.append((char) curr);
465                 prev = curr == '\r' ? prev : curr;
466             }
467             
468             if (curr == -1) {
469                 monitor(Event.HEADERS_PREMATURE_END);
470             }
471         }
472 
473         private boolean parseField() {
474             while (pos < sb.length()) {
475                 while (pos < sb.length() && sb.charAt(pos) != '\r') {
476                     pos++;
477                 }
478                 if (pos < sb.length() - 1 && sb.charAt(pos + 1) != '\n') {
479                     pos++;
480                     continue;
481                 }
482                 if (pos >= sb.length() - 2 || fieldChars.get(sb.charAt(pos + 2))) {
483                     /*
484                      * field should be the complete field data excluding the 
485                      * trailing \r\n.
486                      */
487                     field = sb.substring(start, pos);
488                     start = pos + 2;
489                     
490                     /*
491                      * Check for a valid field.
492                      */
493                     int index = field.indexOf(':');
494                     boolean valid = false;
495                     if (index != -1 && fieldChars.get(field.charAt(0))) {
496                         valid = true;
497                         fieldName = field.substring(0, index).trim();
498                         for (int i = 0; i < fieldName.length(); i++) {
499                             if (!fieldChars.get(fieldName.charAt(i))) {
500                                 valid = false;
501                                 break;
502                             }
503                         }
504                         if (valid) {
505                             fieldValue = field.substring(index + 1);
506                             body.addField(fieldName, fieldValue);
507                             startLineNumber = lineNumber;
508                             pos += 2;
509                             lineNumber++;
510                             return true;
511                         }
512                     }
513                     if (log.isWarnEnabled()) {
514                         log.warn("Line " + startLineNumber 
515                                 + ": Ignoring invalid field: '" + field.trim() + "'");
516                     }
517                     startLineNumber = lineNumber;
518                 }
519                 pos += 2;
520                 lineNumber++;
521             }
522             return false;
523         }
524     }

560     /** Instructs the {@code MimeTokenStream} to parse the given streams contents.
561      * If the {@code MimeTokenStream} has already been in use, resets the streams
562      * internal state.
563      */
564     public void parse(InputStream stream) {
565         entities.clear();
566         cursor = new StreamCursor(stream);
567         state = parseMessage(cursor, null);
568     }

570     private int parseMessage(Cursor cursor, BodyDescriptor parent) {
571         switch (recursionMode) {
572             case M_RAW:
573                 currentStateMachine = new RawEntity(cursor.nextSection());
574                 break;
575             case M_NO_RECURSE:
576                 // expected to be called only at start of paring
577             case M_RECURSE:
578                 currentStateMachine = new Message(cursor, parent);
579                 break;
580         }
581         entities.add(currentStateMachine);
582         return currentStateMachine.state;
583     }

596     /**
597      * Enables or disables raw mode. In raw mode all future entities 
598      * (messages or body parts) in the stream will be reported to the
599      * {@link ContentHandler#raw(InputStream)} handler method only.
600      * The stream will contain the entire unparsed entity contents 
601      * including header fields and whatever is in the body.
602      * 
603      * @param raw <code>true</code> enables raw mode, <code>false</code>
604      *        disables it.
605      * @deprecated pass {@link #M_RAW} to {@link #setRecursionMode(int)} 
606      */
607     public void setRaw(boolean raw) {
608         if (raw) {
609             recursionMode = M_RAW;
610         } else {
611             recursionMode = M_RECURSE;
612         }
613     }

709     /**
710      * Monitors the given event.
711      * Subclasses may override to perform actions upon events.
712      * Base implementation logs at warn.
713      * @param event <code>Event</code>, not null
714      * @throws MimeException subclasses may elect to throw this exception upon
715      * invalid content
716      * @throws IOException subclasses may elect to throw this exception
717      */
718     protected void monitor(Event event) throws MimeException, IOException {
719         if (strictParsing) {
720             throw new MimeParseEventException(event);
721         } else {
722             warn(event);
723         }
724     }

726     /**
727      * Creates an indicative message suitable for display
728      * based on the given event and the current state of the system.
729      * @param event <code>Event</code>, not null
730      * @return message suitable for use as a message in an exception
731      * or for logging
732      */
733     protected String message(Event event) {
734         String preamble = "";
735         try {
736             preamble = "Line " + cursor.getLineNumber() + ": ";
737         } catch (IOException e) {

851      * @throws IllegalStateException The method has been called, although
852      *   {@link #getState()} was already {@link #T_END_OF_STREAM}.
853      */
854     public int next() throws IOException, MimeException {
855         if (state == T_END_OF_STREAM  ||  currentStateMachine == null) {
856             throw new IllegalStateException("No more tokens are available.");
857         }
858         while (currentStateMachine != null) {
859             state = currentStateMachine.next();
860             if (state != T_END_OF_STREAM) {
861                 return state;
862             }
863             entities.remove(entities.size()-1);
864             if (entities.size() == 0) {
865                 currentStateMachine = null;
866             } else {
867                 currentStateMachine = (StateMachine) entities.get(entities.size()-1);
868             }
869         }
870         state = T_END_OF_STREAM;
