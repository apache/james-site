92      public void skipHeader(String contentType) {
93          if (state != EntityStates.T_START_MESSAGE) {
94              throw new IllegalStateException("Invalid state: " + stateToString(state));

96          skipHeader = true;

108     public EntityStateMachine advance() throws IOException, MimeException {
109         switch (state) {
110         case EntityStates.T_START_MESSAGE:
111             if (skipHeader) {
112                 state = EntityStates.T_END_HEADER;
113             } else {
114                 state = EntityStates.T_START_HEADER;
115             }
116             break;
117         case EntityStates.T_START_BODYPART:
118             state = EntityStates.T_START_HEADER;
119             break;
120         case EntityStates.T_START_HEADER:
121         case EntityStates.T_FIELD:
122             state = parseField() ? EntityStates.T_FIELD : EntityStates.T_END_HEADER;
123             break;
124         case EntityStates.T_END_HEADER:
125             String mimeType = body.getMimeType();
126             if (recursionMode == RecursionMode.M_FLAT) {
127                 state = EntityStates.T_BODY;
128             } else if (MimeUtil.isMultipart(mimeType)) {
129                 state = EntityStates.T_START_MULTIPART;
130                 clearMimeStream();
131             } else if (recursionMode != RecursionMode.M_NO_RECURSE 
132                     && MimeUtil.isMessage(mimeType)) {
133                 state = T_IN_MESSAGE;
134                 return nextMessage();
135             } else {
136                 state = EntityStates.T_BODY;
137             }
138             break;
139         case EntityStates.T_START_MULTIPART:
140             if (dataStream.isUsed()) {
141                 advanceToBoundary();            
142                 state = EntityStates.T_END_MULTIPART;
143             } else {
144                 createMimeStream();
145                 state = EntityStates.T_PREAMBLE;
146             }
147             break;
148         case EntityStates.T_PREAMBLE:
149             advanceToBoundary();            
150             if (mimeStream.isLastPart()) {
151                 clearMimeStream();
152                 state = EntityStates.T_END_MULTIPART;
153             } else {
154                 clearMimeStream();
155                 createMimeStream();
156                 state = T_IN_BODYPART;
157                 return nextMimeEntity();
158             }
159             break;
160         case T_IN_BODYPART:
161             advanceToBoundary();
162             if (mimeStream.eof() && !mimeStream.isLastPart()) {
163                 monitor(Event.MIME_BODY_PREMATURE_END);
164             } else {
165                 if (!mimeStream.isLastPart()) {
166                     clearMimeStream();
167                     createMimeStream();
168                     state = T_IN_BODYPART;
169                     return nextMimeEntity();
170                 }
171             }
172             clearMimeStream();
173             state = EntityStates.T_EPILOGUE;
174             break;
175         case EntityStates.T_EPILOGUE:
176             state = EntityStates.T_END_MULTIPART;
177             break;
178         case EntityStates.T_BODY:
179         case EntityStates.T_END_MULTIPART:
180         case T_IN_MESSAGE:
181             state = endState;
182             break;
183         default:
184             if (state == endState) {
185                 state = EntityStates.T_END_OF_STREAM;
186                 break;
187             }
188             throw new IllegalStateException("Invalid state: " + stateToString(state));
189         }
190         return null;
191     }

193     private void createMimeStream() throws IOException {
194         String boundary = body.getBoundary();
195         int bufferSize = 2 * boundary.length();
196         if (bufferSize < 4096) {
197             bufferSize = 4096;
198         }
199         try {
200             if (mimeStream != null) {
201                 mimeStream = new MimeBoundaryInputStream(
202                         new BufferedLineReaderInputStream(
203                                 mimeStream, 
204                                 bufferSize, 
205                                 config.getMaxLineLen()), 
206                         boundary);
207             } else {
208                 inbuffer.ensureCapacity(bufferSize);
209                 mimeStream = new MimeBoundaryInputStream(inbuffer, boundary);
210             }
211         } catch (IllegalArgumentException e) {
212             // thrown when boundary is too long
213             throw new MimeException(e.getMessage(), e);
214         }
215         dataStream = new LineReaderInputStreamAdaptor(
216                 mimeStream,
217                 config.getMaxLineLen()); 
218     }
219     
220     private void clearMimeStream() {
221         mimeStream = null;
222         dataStream = new LineReaderInputStreamAdaptor(
223                 inbuffer,
224                 config.getMaxLineLen()); 
225     }

227     private void advanceToBoundary() throws IOException {
228         if (!dataStream.eof()) {
229             if (tmpbuf == null) {
230                 tmpbuf = new byte[2048];
231             }
232             InputStream instream = getLimitedContentStream();
233             while (instream.read(tmpbuf)!= -1) {
234             }
235         }
236     }

238     private EntityStateMachine nextMessage() {
239         String transferEncoding = body.getTransferEncoding();
240         InputStream instream;
241         if (MimeUtil.isBase64Encoding(transferEncoding)) {
242             log.debug("base64 encoded message/rfc822 detected");
243             instream = new Base64InputStream(dataStream);                    
244         } else if (MimeUtil.isQuotedPrintableEncoded(transferEncoding)) {
245             log.debug("quoted-printable encoded message/rfc822 detected");
246             instream = new QuotedPrintableInputStream(dataStream);                    
247         } else {
248             instream = dataStream;
249         }
250         
251         if (recursionMode == RecursionMode.M_RAW) {
252             RawEntity message = new RawEntity(instream);
253             return message;
254         } else {
255             MimeEntity message = new MimeEntity(
256                     rootStream, 
257                     new BufferedLineReaderInputStream(
258                             instream, 
259                             4 * 1024,
260                             config.getMaxLineLen()),
261                     body, 
262                     EntityStates.T_START_MESSAGE, 
263                     EntityStates.T_END_MESSAGE,
264                     config);
265             message.setRecursionMode(recursionMode);
266             return message;
267         }
268     }
269     
270     private EntityStateMachine nextMimeEntity() {
271         if (recursionMode == RecursionMode.M_RAW) {
272             RawEntity message = new RawEntity(mimeStream);
273             return message;
274         } else {
275             BufferedLineReaderInputStream stream = new BufferedLineReaderInputStream(
276                     mimeStream, 
277                     4 * 1024,
278                     config.getMaxLineLen());
279             MimeEntity mimeentity = new MimeEntity(
280                     rootStream, 
281                     stream,
282                     body, 
283                     EntityStates.T_START_BODYPART, 
284                     EntityStates.T_END_BODYPART,
285                     config);
286             mimeentity.setRecursionMode(recursionMode);
287             return mimeentity;
288         }
289     }

291     private InputStream getLimitedContentStream() {
292         long maxContentLimit = config.getMaxContentLen();
293         if (maxContentLimit >= 0) {
294             return new LimitedInputStream(dataStream, maxContentLimit);
295         } else {
296             return dataStream;
297         }
298     }

300     public InputStream getContentStream() {
301         switch (state) {
302         case EntityStates.T_START_MULTIPART:
303         case EntityStates.T_PREAMBLE:
304         case EntityStates.T_EPILOGUE:
305         case EntityStates.T_BODY:
306             return getLimitedContentStream();
307         default:
308             throw new IllegalStateException("Invalid state: " + stateToString(state));
309         }
310     }
