20  package org.apache.james.mime4j.parser;
21  
22  /**
23   * Enumeration of states an entity is expected to go through
24   * in the process of the MIME stream parsing.
25   */
26  public interface EntityStates {
27  
28      /**
29       * This token indicates, that the MIME stream has been completely
30       * and successfully parsed, and no more data is available.
31       */
32      int T_END_OF_STREAM = -1;
33      /**
34       * This token indicates, that the MIME stream is currently
35       * at the beginning of a message.
36       */
37      int T_START_MESSAGE = 0;
38      /**
39       * This token indicates, that the MIME stream is currently
40       * at the end of a message.
41       */
42      int T_END_MESSAGE = 1;
43      /**
44       * This token indicates, that a raw entity is currently being processed.
45       */
46      int T_RAW_ENTITY = 2;
47      /**
48       * This token indicates, that a message parts headers are now
49       * being parsed.
50       */
51      int T_START_HEADER = 3;
52      /**
53       * This token indicates, that a message parts field has now
54       * been parsed. 
55       */
56      int T_FIELD = 4;
57      /**
58       * This token indicates, that part headers have now been
59       * parsed.
60       */
61      int T_END_HEADER = 5;
62      /**
63       * This token indicates, that a multipart body is being parsed.
64       */
65      int T_START_MULTIPART = 6;
66      /**
67       * This token indicates, that a multipart body has been parsed.
68       */
69      int T_END_MULTIPART = 7;
70      /**
71       * This token indicates, that a multiparts preamble is being
72       * parsed. 
73       */
74      int T_PREAMBLE = 8;
75      /**
76       * This token indicates, that a multiparts epilogue is being
77       * parsed. 
78       */
79      int T_EPILOGUE = 9;
80      /**
81       * This token indicates, that the MIME stream is currently
82       * at the beginning of a body part.
83       */
84      int T_START_BODYPART = 10;
85      /**
86       * This token indicates, that the MIME stream is currently
87       * at the end of a body part.
88       */
89      int T_END_BODYPART = 11;
90      /**
91       * This token indicates, that an atomic entity is being parsed.
92       */
93      int T_BODY = 12;
94  
95  }
