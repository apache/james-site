20  package org.apache.james.mime4j.parser;
21  
22  import org.apache.james.mime4j.MimeException;
23  import org.apache.james.mime4j.util.CharArrayBuffer;
24  
25  /**
26   * MIME entity configuration
27   */
28  public final class MimeEntityConfig implements Cloneable {
29  
30      private boolean maximalBodyDescriptor;
31      private boolean strictParsing;
32      private int maxLineLen;
33      private int maxHeaderCount;
34      private long maxContentLen;
35      
36      public MimeEntityConfig() {
37          super();
38          this.maximalBodyDescriptor = false;
39          this.strictParsing = false;
40          this.maxLineLen = 1000;
41          this.maxHeaderCount = 1000;
42          this.maxContentLen = -1;
43      }
44      
45      public boolean isMaximalBodyDescriptor() {
46          return this.maximalBodyDescriptor;
47      }
48      
49      public void setMaximalBodyDescriptor(boolean maximalBodyDescriptor) {
50          this.maximalBodyDescriptor = maximalBodyDescriptor;
51      }
52      
53      /**
54       * Defines whether minor violations of the MIME specification should be 
55       * tolerated or should result in a {@link MimeException}. If this parameter
56       * is set to <code>true</code>, a strict interpretation of the MIME 
57       * specification will be enforced, If this parameter is set to <code>false</code>
58       * minor violations will result in a warning in the log.
59       * 
60       * @param strictParsing value of the strict parsing mode
61       */
62      public void setStrictParsing(boolean strictParsing) {
63          this.strictParsing = strictParsing;
64      }
65  
66      /**
67       * Returns the value of the strict parsing mode
68       * @see #setStrictParsing(boolean)
69       * 
70       * @return value of the strict parsing mode
71       */
72      public boolean isStrictParsing() {
73          return this.strictParsing;
74      }
75      
76      /**
77       * Sets the maximum line length limit. Parsing of a MIME entity will be terminated 
78       * with a {@link MimeException} if a line is encountered that exceeds the maximum
79       * length limit. If this parameter is set to a non positive value the line length
80       * check will be disabled.
81       * 
82       * @param maxLineLen maximum line length limit
83       */
84      public void setMaxLineLen(int maxLineLen) {
85          this.maxLineLen = maxLineLen;
86      }
87      
88      /** 
89       * Returns the maximum line length limit
90       * @see #setMaxLineLen(int)
91       * 
92       * @return value of the the maximum line length limit
93       */
94      public int getMaxLineLen() {
95          return this.maxLineLen;
96      }
97  
98      /**
99       * Sets the maximum header limit. Parsing of a MIME entity will be terminated 
100      * with a {@link MimeException} if the number of headers exceeds the maximum
101      * limit. If this parameter is set to a non positive value the header limit check 
102      * will be disabled.
103      * 
104      * @param maxHeaderCount maximum header limit
105      */
106     public void setMaxHeaderCount(int maxHeaderCount) {
107         this.maxHeaderCount = maxHeaderCount;
108     }
109 
110     /** 
111      * Returns the maximum header limit
112      * @see #setMaxHeaderCount(int)
113      * 
114      * @return value of the the maximum header limit
115      */
116     public int getMaxHeaderCount() {
117         return this.maxHeaderCount;
118     }
119 
120     /**
121      * Sets the maximum line length limit. Parsing of a MIME entity will be terminated 
122      * with a {@link MimeException} if a content body exceeds the maximum length limit. 
123      * If this parameter is set to a non positive value the content length
124      * check will be disabled.
125      * 
126      * @param maxLineLen maximum line length limit
127      */
128     public void setMaxContentLen(long maxContentLen) {
129         this.maxContentLen = maxContentLen;
130     }
131 
132     /** 
133      * Returns the maximum content length limit
134      * @see #setMaxContentLen(long)
135      * 
136      * @return value of the the maximum content length limit
137      */
138     public long getMaxContentLen() {
139         return maxContentLen;
140     }
141 
142     public Object clone() throws CloneNotSupportedException {
143         return super.clone();
144     }
145     
146     public String toString() {
147         CharArrayBuffer buffer = new CharArrayBuffer(128);
148         buffer.append("[max body descriptor: ");
149         buffer.append(Boolean.toString(this.maximalBodyDescriptor));
150         buffer.append("][strict parsing: ");
151         buffer.append(Boolean.toString(this.strictParsing));
152         buffer.append("][max header length: ");
153         buffer.append(Integer.toString(this.maxLineLen));
154         buffer.append("]");
155         return buffer.toString();
156     }
