161     /**
162      * <p>Parses a complex field value into a map of key/value pairs. You may
163      * use this, for example, to parse a definition like
164      * <pre>
165      *   text/plain; charset=UTF-8; boundary=foobar
166      * </pre>
167      * The above example would return a map with the keys "", "charset",
168      * and "boundary", and the values "text/plain", "UTF-8", and "foobar".
169      * </p><p>
170      * Header value will be unfolded and excess white space trimmed.
171      * </p>
172      * @param pValue The field value to parse.
173      * @return The result map; use the key "" to retrieve the first value.
174      */
175     public static Map<String, String> getHeaderParams(String pValue) {
176         pValue = pValue.trim();
177         
178         /*
179          * Unfold Content-Type value
180          */
181         StringBuilder sb = new StringBuilder(128);
182         for (int i = 0; i < pValue.length(); i++) {
183             char c = pValue.charAt(i);
184             if (c == '\r' || c == '\n') {
185                 continue;
186             }
187             sb.append(c);
188         }
189         pValue = sb.toString();
190         
191         Map<String, String> result = new HashMap<String, String>();
192 
193         // split main value and parameters
194         String main;
195         String rest;
196         if (pValue.indexOf(";") == -1) {
197             main = pValue;
198             rest = null;
199         } else {
200             main = pValue.substring(0, pValue.indexOf(";"));
201             rest = pValue.substring(main.length() + 1);
202         }
203 
204         result.put("", main);
205         if (rest != null) {
206             char[] chars = rest.toCharArray();
207             StringBuilder paramName = new StringBuilder(64);
208             StringBuilder paramValue = new StringBuilder(64);
209 
210             final byte READY_FOR_NAME = 0;
211             final byte IN_NAME = 1;
212             final byte READY_FOR_VALUE = 2;
213             final byte IN_VALUE = 3;
214             final byte IN_QUOTED_VALUE = 4;
215             final byte VALUE_DONE = 5;
216             final byte ERROR = 99;
217 
218             byte state = READY_FOR_NAME;
219             boolean escaped = false;
220             for (char c : chars) {
221                 switch (state) {
222                     case ERROR:
223                         if (c == ';')
224                             state = READY_FOR_NAME;
225                         break;
226 
227                     case READY_FOR_NAME:
228                         if (c == '=') {
229                             log.error("Expected header param name, got '='");
230                             state = ERROR;
231                             break;
232                         }
233 
234                         paramName.setLength(0);
235                         paramValue.setLength(0);
236 
237                         state = IN_NAME;
238                         // fall-through
239 
240                     case IN_NAME:
241                         if (c == '=') {
242                             if (paramName.length() == 0)
243                                 state = ERROR;
244                             else
245                                 state = READY_FOR_VALUE;
246                             break;
247                         }
248 
249                         // not '='... just add to name
250                         paramName.append(c);
251                         break;
252 
253                     case READY_FOR_VALUE:
254                         boolean fallThrough = false;
255                         switch (c) {
256                             case ' ':
257                             case '\t':
258                                 break;  // ignore spaces, especially before '"'
259 
260                             case '"':
261                                 state = IN_QUOTED_VALUE;
262                                 break;
263 
264                             default:
265                                 state = IN_VALUE;
266                                 fallThrough = true;
267                                 break;
268                         }
269                         if (!fallThrough)
270                             break;
271 
272                         // fall-through
273 
274                     case IN_VALUE:
275                         fallThrough = false;
276                         switch (c) {
277                             case ';':
278                             case ' ':
279                             case '\t':
280                                 result.put(
281                                    paramName.toString().trim().toLowerCase(),
282                                    paramValue.toString().trim());
283                                 state = VALUE_DONE;
284                                 fallThrough = true;
285                                 break;
286                             default:
287                                 paramValue.append(c);
288                                 break;
289                         }
290                         if (!fallThrough)
291                             break;
292 
293                     case VALUE_DONE:
294                         switch (c) {
295                             case ';':
296                                 state = READY_FOR_NAME;
297                                 break;
298 
299                             case ' ':
300                             case '\t':
301                                 break;
302 
303                             default:
304                                 state = ERROR;
305                                 break;
306                         }
307                         break;
308                         
309                     case IN_QUOTED_VALUE:
310                         switch (c) {
311                             case '"':
312                                 if (!escaped) {
313                                     // don't trim quoted strings; the spaces could be intentional.
314                                     result.put(
315                                             paramName.toString().trim().toLowerCase(),
316                                             paramValue.toString());
317                                     state = VALUE_DONE;
318                                 } else {
319                                     escaped = false;
320                                     paramValue.append(c);                                    
321                                 }
322                                 break;
323                                 
324                             case '\\':
325                                 if (escaped) {
326                                     paramValue.append('\\');
327                                 }
328                                 escaped = !escaped;
329                                 break;
330 
331                             default:
332                                 if (escaped) {
333                                     paramValue.append('\\');
334                                 }
335                                 escaped = false;
336                                 paramValue.append(c);
337                                 break;
338                         }
339                         break;
340 
341                 }
342             }
343 
344             // done looping.  check if anything is left over.
345             if (state == IN_VALUE) {
346                 result.put(
347                         paramName.toString().trim().toLowerCase(),
348                         paramValue.toString().trim());
349             }
350         }
351 
352         return result;
353     }

375     /**
376      * Creates a new unique message identifier that can be used in message
377      * header field such as Message-ID or In-Reply-To. If the given host name is
378      * not <code>null</code> it will be used as suffix for the message ID
379      * (following an at sign).
380      * 
381      * The resulting string is enclosed in angle brackets (&lt; and &gt;);
382      * 
383      * @param hostName host name to be included in the message ID or
384      *            <code>null</code> if no host name should be included.
385      * @return a new unique message identifier.
386      */
387     public static String createUniqueMessageId(String hostName) {
388         StringBuilder sb = new StringBuilder("<Mime4j.");
389         sb.append(Integer.toHexString(nextCounterValue()));
390         sb.append('.');
391         sb.append(Long.toHexString(random.nextLong()));
392         sb.append('.');
393         sb.append(Long.toHexString(System.currentTimeMillis()));
394         if (hostName != null) {
395             sb.append('@');
396             sb.append(hostName);
397         }
398         sb.append('>');
399         return sb.toString();
400     }

424     /**
425      * Splits the specified string into a multiple-line representation with
426      * lines no longer than 76 characters (because the line might contain
427      * encoded words; see <a href='http://www.faqs.org/rfcs/rfc2047.html'>RFC
428      * 2047</a> section 2). If the string contains non-whitespace sequences
429      * longer than 76 characters a line break is inserted at the whitespace
430      * character following the sequence resulting in a line longer than 76
431      * characters.
432      * 
433      * @param s
434      *            string to split.
435      * @param usedCharacters
436      *            number of characters already used up. Usually the number of
437      *            characters for header field name plus colon and one space.
438      * @return a multiple-line representation of the given string.
439      */
440     public static String fold(String s, int usedCharacters) {
441         final int maxCharacters = 76;
442 
443         final int length = s.length();
444         if (usedCharacters + length <= maxCharacters)
445             return s;
446 
447         StringBuilder sb = new StringBuilder();
448 
449         int lastLineBreak = -usedCharacters;
450         int wspIdx = indexOfWsp(s, 0);
451         while (true) {
452             if (wspIdx == length) {
453                 sb.append(s.substring(Math.max(0, lastLineBreak)));
454                 return sb.toString();
455             }
456 
457             int nextWspIdx = indexOfWsp(s, wspIdx + 1);
458 
459             if (nextWspIdx - lastLineBreak > maxCharacters) {
460                 sb.append(s.substring(Math.max(0, lastLineBreak), wspIdx));
461                 sb.append("\r\n");
462                 lastLineBreak = wspIdx;
463             }
464 
465             wspIdx = nextWspIdx;
466         }
467     }

469     private static int indexOfWsp(String s, int fromIndex) {
470         final int len = s.length();
471         for (int index = fromIndex; index < len; index++) {
472             char c = s.charAt(index);
473             if (c == ' ' || c == '\t')
474                 return index;
475         }
476         return len;
477     }

490     private static final class Rfc822DateFormat extends SimpleDateFormat {
491         private static final long serialVersionUID = 1L;
492 
493         public Rfc822DateFormat() {
494             super("EEE, d MMM yyyy HH:mm:ss ", Locale.US);
495         }
496 
497         @Override
498         public StringBuffer format(Date date, StringBuffer toAppendTo,
499                 FieldPosition pos) {
500             StringBuffer sb = super.format(date, toAppendTo, pos);
501 
502             int zoneMillis = calendar.get(GregorianCalendar.ZONE_OFFSET);
503             int dstMillis = calendar.get(GregorianCalendar.DST_OFFSET);
504             int minutes = (zoneMillis + dstMillis) / 1000 / 60;
505 
506             if (minutes < 0) {
507                 sb.append('-');
508                 minutes = -minutes;
509             } else {
510                 sb.append('+');
511             }
512 
513             sb.append(String.format("%02d%02d", minutes / 60, minutes % 60));
514 
515             return sb;
516         }
