155     /**
156      * <p>Parses a complex field value into a map of key/value pairs. You may
157      * use this, for example, to parse a definition like
158      * <pre>
159      *   text/plain; charset=UTF-8; boundary=foobar
160      * </pre>
161      * The above example would return a map with the keys "", "charset",
162      * and "boundary", and the values "text/plain", "UTF-8", and "foobar".
163      * </p><p>
164      * Header value will be unfolded and excess white space trimmed.
165      * </p>
166      * @param pValue The field value to parse.
167      * @return The result map; use the key "" to retrieve the first value.
168      */
169     public static Map getHeaderParams(String pValue) {
170         pValue = pValue.trim();
171         
172         /*
173          * Unfold Content-Type value
174          */
175         CharArrayBuffer sb = new CharArrayBuffer(128);
176         for (int i = 0; i < pValue.length(); i++) {
177             char c = pValue.charAt(i);
178             if (c == '\r' || c == '\n') {
179                 continue;
180             }
181             sb.append(c);
182         }
183         pValue = sb.toString();
184         
185         Map result = new HashMap();
186 
187         // split main value and parameters
188         String main;
189         String rest;
190         if (pValue.indexOf(";") == -1) {
191             main = pValue;
192             rest = null;
193         } else {
194             main = pValue.substring(0, pValue.indexOf(";"));
195             rest = pValue.substring(main.length() + 1);
196         }
197 
198         result.put("", main);
199         if (rest != null) {
200             char[] chars = rest.toCharArray();
201             CharArrayBuffer paramName = new CharArrayBuffer(64);
202             CharArrayBuffer paramValue = new CharArrayBuffer(64);
203 
204             final byte READY_FOR_NAME = 0;
205             final byte IN_NAME = 1;
206             final byte READY_FOR_VALUE = 2;
207             final byte IN_VALUE = 3;
208             final byte IN_QUOTED_VALUE = 4;
209             final byte VALUE_DONE = 5;
210             final byte ERROR = 99;
211 
212             byte state = READY_FOR_NAME;
213             boolean escaped = false;
214             for (int i = 0; i < chars.length; i++) {
215                 char c = chars[i];
216 
217                 switch (state) {
218                     case ERROR:
219                         if (c == ';')
220                             state = READY_FOR_NAME;
221                         break;
222 
223                     case READY_FOR_NAME:
224                         if (c == '=') {
225                             log.error("Expected header param name, got '='");
226                             state = ERROR;
227                             break;
228                         }
229 
230                         paramName = new CharArrayBuffer(64);
231                         paramValue = new CharArrayBuffer(64);
232 
233                         state = IN_NAME;
234                         // fall-through
235 
236                     case IN_NAME:
237                         if (c == '=') {
238                             if (paramName.length() == 0)
239                                 state = ERROR;
240                             else
241                                 state = READY_FOR_VALUE;
242                             break;
243                         }
244 
245                         // not '='... just add to name
246                         paramName.append(c);
247                         break;
248 
249                     case READY_FOR_VALUE:
250                         boolean fallThrough = false;
251                         switch (c) {
252                             case ' ':
253                             case '\t':
254                                 break;  // ignore spaces, especially before '"'
255 
256                             case '"':
257                                 state = IN_QUOTED_VALUE;
258                                 break;
259 
260                             default:
261                                 state = IN_VALUE;
262                                 fallThrough = true;
263                                 break;
264                         }
265                         if (!fallThrough)
266                             break;
267 
268                         // fall-through
269 
270                     case IN_VALUE:
271                         fallThrough = false;
272                         switch (c) {
273                             case ';':
274                             case ' ':
275                             case '\t':
276                                 result.put(
277                                    paramName.toString().trim().toLowerCase(),
278                                    paramValue.toString().trim());
279                                 state = VALUE_DONE;
280                                 fallThrough = true;
281                                 break;
282                             default:
283                                 paramValue.append(c);
284                                 break;
285                         }
286                         if (!fallThrough)
287                             break;
288 
289                     case VALUE_DONE:
290                         switch (c) {
291                             case ';':
292                                 state = READY_FOR_NAME;
293                                 break;
294 
295                             case ' ':
296                             case '\t':
297                                 break;
298 
299                             default:
300                                 state = ERROR;
301                                 break;
302                         }
303                         break;
304                         
305                     case IN_QUOTED_VALUE:
306                         switch (c) {
307                             case '"':
308                                 if (!escaped) {
309                                     // don't trim quoted strings; the spaces could be intentional.
310                                     result.put(
311                                             paramName.toString().trim().toLowerCase(),
312                                             paramValue.toString());
313                                     state = VALUE_DONE;
314                                 } else {
315                                     escaped = false;
316                                     paramValue.append(c);                                    
317                                 }
318                                 break;
319                                 
320                             case '\\':
321                                 if (escaped) {
322                                     paramValue.append('\\');
323                                 }
324                                 escaped = !escaped;
325                                 break;
326 
327                             default:
328                                 if (escaped) {
329                                     paramValue.append('\\');
330                                 }
331                                 escaped = false;
332                                 paramValue.append(c);
333                                 break;
334                         }
335                         break;
336 
337                 }
338             }
339 
340             // done looping.  check if anything is left over.
341             if (state == IN_VALUE) {
342                 result.put(
343                         paramName.toString().trim().toLowerCase(),
344                         paramValue.toString().trim());
345             }
346         }
347 
348         return result;
349     }

351     /**
352      * Creates a new unique message boundary string that can be used as boundary
353      * parameter for the Content-Type header field of a message.
354      * 
355      * @return a new unique message boundary string.
356      */
357     public static String createUniqueBoundary() {
358         StringBuffer sb = new StringBuffer();
359         sb.append("-=Part.");
360         sb.append(Integer.toHexString(nextCounterValue()));
361         sb.append('.');
362         sb.append(Long.toHexString(random.nextLong()));
363         sb.append('.');
364         sb.append(Long.toHexString(System.currentTimeMillis()));
365         sb.append('.');
366         sb.append(Long.toHexString(random.nextLong()));
367         sb.append("=-");
368         return sb.toString();
369     }

371     /**
372      * Creates a new unique message identifier that can be used in message
373      * header field such as Message-ID or In-Reply-To. If the given host name is
374      * not <code>null</code> it will be used as suffix for the message ID
375      * (following an at sign).
376      * 
377      * The resulting string is enclosed in angle brackets (&lt; and &gt;);
378      * 
379      * @param hostName host name to be included in the message ID or
380      *            <code>null</code> if no host name should be included.
381      * @return a new unique message identifier.
382      */
383     public static String createUniqueMessageId(String hostName) {
384         StringBuffer sb = new StringBuffer("<Mime4j.");
385         sb.append(Integer.toHexString(nextCounterValue()));
386         sb.append('.');
387         sb.append(Long.toHexString(random.nextLong()));
388         sb.append('.');
389         sb.append(Long.toHexString(System.currentTimeMillis()));
390         if (hostName != null) {
391             sb.append('@');
392             sb.append(hostName);
393         }
394         sb.append('>');
395         return sb.toString();
396     }
