42      public static final String CRLF = "\r\n";
43      
44      public static final int CR = 13; // <US-ASCII CR, carriage return (13)>
45      public static final int LF = 10; // <US-ASCII LF, linefeed (10)>
46      public static final int SP = 32; // <US-ASCII SP, space (32)>
47      public static final int HT = 9;  // <US-ASCII HT, horizontal-tab (9)>
48  
49      public static boolean isASCII(char ch) {
50          return (0xFF80 & ch) == 0;
51      }
52      
53      public static boolean isASCII(final String s) {
54          if (s == null) {
55              throw new IllegalArgumentException("String may not be null");
56          }
57          int len = s.length();
58          for (int i = 0; i < len; i++) {
59              if (!isASCII(s.charAt(i))) {
60                  return false;
61              }
62          }
63          return true;
64      }
65      
66      public static boolean isWhitespace(char ch) {
67          return ch == SP || ch == HT || ch == CR || ch == LF; 
68      }
