32  /**
33   * Static methods for decoding strings, byte arrays and encoded words.
34   *
35   * 
36   * @version $Id$
37   */
38  public class DecoderUtil {
39      private static Log log = LogFactory.getLog(DecoderUtil.class);
40      
41      /**
42       * Decodes a string containing quoted-printable encoded data. 
43       * 
44       * @param s the string to decode.
45       * @return the decoded bytes.
46       */
47      public static byte[] decodeBaseQuotedPrintable(String s) {
48          ByteArrayOutputStream baos = new ByteArrayOutputStream();
49          
50          try {
51              byte[] bytes = s.getBytes("US-ASCII");
52              
53              QuotedPrintableInputStream is = new QuotedPrintableInputStream(
54                                                 new ByteArrayInputStream(bytes));
55              
56              int b = 0;
57              while ((b = is.read()) != -1) {
58                  baos.write(b);
59              }
60          } catch (IOException e) {
61              /*
62               * This should never happen!
63               */
64              log.error(e);
65          }
66          
67          return baos.toByteArray();
68      }

70      /**
71       * Decodes a string containing base64 encoded data. 
72       * 
73       * @param s the string to decode.
74       * @return the decoded bytes.
75       */
76      public static byte[] decodeBase64(String s) {
77          ByteArrayOutputStream baos = new ByteArrayOutputStream();
78          
79          try {
80              byte[] bytes = s.getBytes("US-ASCII");
81              
82              Base64InputStream is = new Base64InputStream(
83                                          new ByteArrayInputStream(bytes));
84              
85              int b = 0;
86              while ((b = is.read()) != -1) {
87                  baos.write(b);
88              }
89          } catch (IOException e) {
90              /*
91               * This should never happen!
92               */
93              log.error(e);
94          }
95          
96          return baos.toByteArray();
97      }

99      /**
100      * Decodes an encoded word encoded with the 'B' encoding (described in 
101      * RFC 2047) found in a header field body.
102      * 
103      * @param encodedWord the encoded word to decode.
104      * @param charset the Java charset to use.
105      * @return the decoded string.
106      * @throws UnsupportedEncodingException if the given Java charset isn't 
107      *         supported.
108      */
109     public static String decodeB(String encodedWord, String charset) 
110             throws UnsupportedEncodingException {
111         
112         return new String(decodeBase64(encodedWord), charset);
113     }
114     
115     /**
116      * Decodes an encoded word encoded with the 'Q' encoding (described in 
117      * RFC 2047) found in a header field body.
118      * 
119      * @param encodedWord the encoded word to decode.
120      * @param charset the Java charset to use.
121      * @return the decoded string.
122      * @throws UnsupportedEncodingException if the given Java charset isn't 
123      *         supported.
124      */
125     public static String decodeQ(String encodedWord, String charset)
126             throws UnsupportedEncodingException {
127            
128         /*
129          * Replace _ with =20
130          */
131         CharArrayBuffer sb = new CharArrayBuffer(128);
132         for (int i = 0; i < encodedWord.length(); i++) {
133             char c = encodedWord.charAt(i);
134             if (c == '_') {
135                 sb.append("=20");
136             } else {
137                 sb.append(c);
138             }
139         }
140         
141         return new String(decodeBaseQuotedPrintable(sb.toString()), charset);
142     }

144     /**
145      * Decodes a string containing encoded words as defined by RFC 2047.
146      * Encoded words in have the form 
147      * =?charset?enc?Encoded word?= where enc is either 'Q' or 'q' for 
148      * quoted-printable and 'B' or 'b' for Base64.
149      * 
150      * @param body the string to decode.
151      * @return the decoded string.
152      */
153     public static String decodeEncodedWords(String body) {
154         CharArrayBuffer sb = new CharArrayBuffer(128);
155         
156         int p1 = 0;
157         int p2 = 0;
158         
159         try {
160             
161             /*
162              * Encoded words in headers have the form 
163              * =?charset?enc?Encoded word?= where enc is either 'Q' or 'q' for 
164              * quoted printable and 'B' and 'b' for Base64
165              */
166             
167             while (p2 < body.length()) {
168                 /*
169                  * Find beginning of first encoded word
170                  */
171                 p1 = body.indexOf("=?", p2);
172                 if (p1 == -1) {
173                     /*
174                      * None found. Emit the rest of the header and exit.
175                      */
176                     sb.append(body.substring(p2));
177                     break;
178                 }
179                 
180                 /*
181                  * p2 points to the previously found end marker or the start
182                  * of the entire header text. Append the text between that
183                  * marker and the one pointed to by p1.
184                  */
185                 if (p1 - p2 > 0) {
186                     sb.append(body.substring(p2, p1));
187                 }
188 
189                 /*
190                  * Find the first and second '?':s after the marker pointed to
191                  * by p1.
192                  */
193                 int t1 = body.indexOf('?', p1 + 2);
194                 int t2 = t1 != -1 ? body.indexOf('?', t1 + 1) : -1;
195 
196                 /*
197                  * Find this words end marker.
198                  */
199                 p2 = t2 != -1 ? body.indexOf("?=", t2 + 1) : -1;
200                 if (p2 == -1) {
201                     if (t2 != -1 && (body.length() - 1 == t2 || body.charAt(t2 + 1) == '=')) {
202                         /*
203                          * Treat "=?charset?enc?" and "=?charset?enc?=" as
204                          * empty strings.
205                          */
206                         p2 = t2;
207                     } else {
208                         /*
209                          * No end marker was found. Append the rest of the 
210                          * header and exit.
211                          */
212                         sb.append(body.substring(p1));
213                         break;
214                     }
215                 }
216 
217                 /*
218                  * [p1+2, t1] -> charset
219                  * [t1+1, t2] -> encoding
220                  * [t2+1, p2] -> encoded word
221                  */
222                 
223                 String decodedWord = null;
224                 if (t2 == p2) {
225                     /*
226                      * The text is empty
227                      */
228                     decodedWord = "";
229                 } else {
230 
231                     String mimeCharset = body.substring(p1 + 2, t1);
232                     String enc = body.substring(t1 + 1, t2);
233                     String encodedWord = body.substring(t2 + 1, p2);
234 
235                     /*
236                      * Convert the MIME charset to a corresponding Java one.
237                      */
238                     String charset = CharsetUtil.toJavaCharset(mimeCharset);
239                     if (charset == null) {
240                         decodedWord = body.substring(p1, p2 + 2);
241                         if (log.isWarnEnabled()) {
242                             log.warn("MIME charset '" + mimeCharset 
243                                     + "' in header field doesn't have a "
244                                     +"corresponding Java charset");
245                         }
246                     } else if (!CharsetUtil.isDecodingSupported(charset)) {
247                         decodedWord = body.substring(p1, p2 + 2);
248                         if (log.isWarnEnabled()) {
249                             log.warn("Current JDK doesn't support decoding "
250                                    + "of charset '" + charset 
251                                    + "' (MIME charset '" 
252                                    + mimeCharset + "')");
253                         }
254                     } else {
255                         if (enc.equalsIgnoreCase("Q")) {
256                             decodedWord = DecoderUtil.decodeQ(encodedWord, charset);
257                         } else if (enc.equalsIgnoreCase("B")) {
258                             decodedWord = DecoderUtil.decodeB(encodedWord, charset);
259                         } else {
260                             decodedWord = encodedWord;
261                             if (log.isWarnEnabled()) {
262                                 log.warn("Warning: Unknown encoding in "
263                                         + "header field '" + enc + "'");
264                             }
265                         }
266                     }
267                 }
268                 p2 += 2;
269                 sb.append(decodedWord);
270             }
271         } catch (Throwable t) {
272             log.error("Decoding header field body '" + body + "'", t);
273         }
274         
275         return sb.toString();
276     }
