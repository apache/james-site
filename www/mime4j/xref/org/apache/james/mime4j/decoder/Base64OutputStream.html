27  /**
28   * This class is based on Base64 and Base64OutputStream code from Commons-Codec 1.4
29   * 
30   * Provides Base64 encoding and decoding as defined by RFC 2045.

32   * <p>
33   * This class implements section <cite>6.8. Base64 Content-Transfer-Encoding</cite> from RFC 2045 <cite>Multipurpose
34   * Internet Mail Extensions (MIME) Part One: Format of Internet Message Bodies</cite> by Freed and Borenstein.
35   * </p>
36   * 
37   * @see <a href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045</a>

104     /** Mask used to extract 6 bits, used when encoding */
105     private static final int MASK_6BITS = 0x3f;

112     /**
113      * Line length for encoding.  Not used when decoding.  A value of zero or less implies
114      * no chunking of the base64 encoded data.
115      */
116     private final int lineLength;

118     /**
119      * Line separator for encoding.  Not used when decoding.  Only used if lineLength > 0.

121     private final byte[] lineSeparator;

123     /**

127     private final int encodeSize;

129     /**

132     private byte[] buf = new byte[1024];

139     /**
140      * Variable tracks how many characters have been written to the current line.

144     private int currentLinePos;

146     /**
147      * Writes to the buffer only occur after every 3 reads when encoding, an

166      * Default constructor:  lineLength is 76, and the lineSeparator is CRLF

169     public Base64OutputStream(OutputStream os) {

173     /**
174      * <p>
175      * Consumer can use this constructor to choose a different lineLength
176      * when encoding (lineSeparator is still CRLF).  All forms of data can
177      * be decoded.
178      * </p><p>
179      * Note:  lineLengths that aren't multiples of 4 will still essentially
180      * end up being multiples of 4 in the encoded data.
181      * </p>
182      *
183      * @param lineLength each line of encoded data will be at most this long
184      * (rounded up to nearest multiple of 4). 
185      * If lineLength <= 0, then the output will not be divided into lines (chunks).  
186      * Ignored when decoding.
187      */
188     public Base64OutputStream(OutputStream os, int lineLength) {
189         this(os, lineLength, CHUNK_SEPARATOR);
190     }

192     /**
193      * <p>
194      * Consumer can use this constructor to choose a different lineLength
195      * and lineSeparator when encoding.  All forms of data can
196      * be decoded.
197      * </p><p>
198      * Note:  lineLengths that aren't multiples of 4 will still essentially
199      * end up being multiples of 4 in the encoded data.
200      * </p>
201      * @param lineLength    Each line of encoded data will be at most this long
202      *                      (rounded up to nearest multiple of 4).  Ignored when decoding.
203      *                      If <= 0, then output will not be divided into lines (chunks).
204      * @param lineSeparator Each line of encoded data will end with this
205      *                      sequence of bytes.
206      *                      If lineLength <= 0, then the lineSeparator is not used.
207      * @throws IllegalArgumentException The provided lineSeparator included
208      *                                  some base64 characters.  That's not going to work!
209      */
210     public Base64OutputStream(OutputStream os, int lineLength, byte[] lineSeparator) {
211         super(os);
212         this.lineLength = lineLength;
213         this.lineSeparator = new byte[lineSeparator.length];
214         System.arraycopy(lineSeparator, 0, this.lineSeparator, 0, lineSeparator.length);
215         if (lineLength > 0) {
216             this.encodeSize = 4 + lineSeparator.length;
217         } else {
218             this.encodeSize = 4;
219         }
220         if (containsBase64Byte(lineSeparator)) {
221             String sep;
222             try {
223                 sep = new String(lineSeparator, "UTF-8");
224             } catch (UnsupportedEncodingException uee) {
225                 sep = new String(lineSeparator);
226             }
227             throw new IllegalArgumentException("lineSeperator must not contain base64 characters: [" + sep + "]");
228         }
229     }

231     /** Doubles our buffer. */
232     private void resizeBuf() {
233         byte[] b = new byte[buf.length * 2];
234         System.arraycopy(buf, 0, b, 0, buf.length);
235         buf = b;
236     }

238     /**
239      * Returns whether or not the <code>octet</code> is in the base 64 alphabet.
240      * 
241      * @param octet
242      *            The value to test
243      * @return <code>true</code> if the value is defined in the the base 64 alphabet, <code>false</code> otherwise.
244      */
245     public static boolean isBase64(byte octet) {
246         return octet == PAD || (octet >= 0 && octet < base64ToInt.length && base64ToInt[octet] != -1);
247     }
248 
249     /*
250      * Tests a given byte array to see if it contains only valid characters within the Base64 alphabet.
251      * 
252      * @param arrayOctet
253      *            byte array to test
254      * @return <code>true</code> if any byte is a valid character in the Base64 alphabet; false herwise
255      */
256     private static boolean containsBase64Byte(byte[] arrayOctet) {
257         for (int i = 0; i < arrayOctet.length; i++) {
258             if (isBase64(arrayOctet[i])) {
259                 return true;
260             }
261         }
262         return false;
263     }

278     /**
279      * Writes <code>len</code> bytes from the specified
280      * <code>b</code> array starting at <code>offset</code> to
281      * this output stream.
282      *
283      * @param b source byte array
284      * @param offset where to start reading the bytes
285      * @param len maximum number of bytes to write
286      * 
287      * @throws IOException if an I/O error occurs.
288      * @throws NullPointerException if the byte array parameter is null
289      * @throws IndexOutOfBoundsException if offset, len or buffer size are invalid
290      */
291     public void write(byte b[], int offset, int len) throws IOException {
292         if (closed) {
293         	throw new IOException("Base64OutputStream has been closed");
294         }
295         if (b == null) {
296             throw new NullPointerException();
297         } else if (len < 0) {
298             // len < 0 is how we're informed of EOF in the underlying data we're
299             // encoding.
300             closed = true;
301             if (buf.length - pos < encodeSize) {
302                 resizeBuf();
303             }
304             switch (modulus) {
305                 case 1:
306                     buf[pos++] = intToBase64[(x >> 2) & MASK_6BITS];
307                     buf[pos++] = intToBase64[(x << 4) & MASK_6BITS];
308                     buf[pos++] = PAD;
309                     buf[pos++] = PAD;
310                     break;
311 
312                 case 2:
313                     buf[pos++] = intToBase64[(x >> 10) & MASK_6BITS];
314                     buf[pos++] = intToBase64[(x >> 4) & MASK_6BITS];
315                     buf[pos++] = intToBase64[(x << 2) & MASK_6BITS];
316                     buf[pos++] = PAD;
317                     break;
318             }
319             if (lineLength > 0) {
320                 System.arraycopy(lineSeparator, 0, buf, pos, lineSeparator.length);
321                 pos += lineSeparator.length;
322                 // TODO I had to add this to make it work as the quoted printable encoder.
323                 // not sure this is generally speaking ok.
324                 System.arraycopy(lineSeparator, 0, buf, pos, lineSeparator.length);
325                 pos += lineSeparator.length;
326             }
327         } else if (offset < 0 || len < 0 || offset + len < 0) {
328             throw new IndexOutOfBoundsException();
329         } else if (offset > b.length || offset + len > b.length) {
330             throw new IndexOutOfBoundsException();
331         } else if (len > 0) {
332             for (int i = 0; i < len; i++) {
333                 if (buf.length - pos < encodeSize) {
334                     resizeBuf();
335                 }
336                 modulus = (++modulus) % 3;
337                 int bc = b[offset++];
338                 if (bc < 0) { bc += 256; }
339                 x = (x << 8) + bc;
340                 if (0 == modulus) {
341                     buf[pos++] = intToBase64[(x >> 18) & MASK_6BITS];
342                     buf[pos++] = intToBase64[(x >> 12) & MASK_6BITS];
343                     buf[pos++] = intToBase64[(x >> 6) & MASK_6BITS];
344                     buf[pos++] = intToBase64[x & MASK_6BITS];
345                     currentLinePos += 4;
346                     if (lineLength > 0 && lineLength <= currentLinePos) {
347                         System.arraycopy(lineSeparator, 0, buf, pos, lineSeparator.length);
348                         pos += lineSeparator.length;
349                         currentLinePos = 0;
350                     }
351                 }
352             }
353         }
354         flushBuffer();
355     }

357     /**
358      * Flushes this output stream and forces any buffered output bytes
359      * to be written out to the stream.  If propogate is true, the wrapped
360      * stream will also be flushed.
361      *
362      * @param propogate boolean flag to indicate whether the wrapped
363      *                  OutputStream should also be flushed.
364      * @throws IOException if an I/O error occurs.
365      */
366     private void flushBuffer() throws IOException {
367         if (pos > 0) {
368             out.write(buf, 0, pos);
369             // buf = null;
370             pos = 0;
371         }
372     }
373 
374     /**
375      * Flushes this output stream and forces any buffered output bytes
376      * to be written out to the stream.
377      *
378      * @throws IOException if an I/O error occurs.
379      */
380     public void flush() throws IOException {
381         flushBuffer();
382         out.flush();
383     }
384     
385     /**
386      * Terminates the BASE64 coded content and flushes the internal buffers. This method does
387      * NOT close the underlying output stream.
388      */
389     public void close() throws IOException {
390         if (closed) {
391             return;
392         }
393 
394         try {
395             // Notify encoder of EOF (-1).
396             write(singleByte, 0, -1);
397             flush();
398         } finally {
399             closed = true;
400         }
401     }
