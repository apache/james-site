16  package org.mime4j.util;
17  
18  import org.apache.commons.logging.Log;
19  import org.apache.commons.logging.LogFactory;
20  
21  /***
22   * 
23   *
24   * @author Niklas Therning
25   * @version $Id: TempStorage.java,v 1.2 2004/10/02 12:41:11 ntherning Exp $
26   */
27  public abstract class TempStorage {
28      private static Log log = LogFactory.getLog(TempStorage.class);
29      private static TempStorage inst = null;
30      
31      static {
32          
33          String clazz = System.getProperty("org.mime4j.tempStorage");
34          try {
35              
36              if (inst != null) {
37                  inst = (TempStorage) Class.forName(clazz).newInstance();
38              }
39              
40          } catch (Throwable t) {
41              log.error("Unable to create or instantiate TempStorage class '" 
42                        + clazz + "' using SimpleTempStorage instead", t);
43          }
44  
45          if (inst == null) {
46              inst = new SimpleTempStorage();            
47          }
48      }
49      
50      /***
51       * Gets the root temporary path which should be used to 
52       * create new temporary paths or files.
53       * 
54       * @return the root temporary path.
55       */
56      public abstract TempPath getRootTempPath();
57      
58      public static TempStorage getInstance() {
59          return inst;
60      }
61      
62      public static void setInstance(TempStorage inst) {
63          if (inst == null) {
64              throw new NullPointerException("inst");
65          }
66          TempStorage.inst = inst;
67      }
68  }
