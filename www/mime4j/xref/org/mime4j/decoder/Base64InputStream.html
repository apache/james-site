16  package org.mime4j.decoder;
17  
18  import java.io.IOException;
19  import java.io.InputStream;
20  
21  import org.apache.commons.logging.Log;
22  import org.apache.commons.logging.LogFactory;
23  
24  /***
25   * Performs Base-64 decoding on an underlying stream.
26   * 
27   * @author Joe Cheng
28   * @version $Id: Base64InputStream.java,v 1.3 2004/11/29 13:15:47 ntherning Exp $
29   */
30  public class Base64InputStream extends InputStream {
31      private static Log log = LogFactory.getLog(Base64InputStream.class);
32  
33      private final InputStream s;
34      private final ByteQueue byteq = new ByteQueue(3);
35      private boolean done = false;
36  
37      public Base64InputStream(InputStream s) {
38          this.s = s;
39      }
40  
41      /***
42       * Closes the underlying stream.
43       * 
44       * @throws IOException on I/O errors.
45       */
46      public void close() throws IOException {
47          s.close();
48      }
49      
50      public int read() throws IOException {
51          if (byteq.count() == 0) {
52              fillBuffer();
53              if (byteq.count() == 0) {
54                  return -1;
55              }
56          }
57  
58          byte val = byteq.dequeue();
59          if (val >= 0)
60              return val;
61          else
62              return val & 0xFF;
63      }
64  
65      /***
66       * Retrieve data from the underlying stream, decode it,
67       * and put the results in the byteq.
68       * @throws IOException
69       */
70      private void fillBuffer() throws IOException {
71          byte[] data = new byte[4];
72          int pos = 0;
73  
74          int i;
75          while (!done) {
76              switch (i = s.read()) {
77                  case -1:
78                      if (pos > 0) {
79                          log.error("Unexpected EOF in MIME parser, dropping " 
80                                  + pos + " sextets");
81                      }
82                      return;
83                  case '=':
84                      decodeAndEnqueue(data, pos);
85                      done = true;
86                      break;
87                  default:
88                      byte sX = TRANSLATION[i];
89                      if (sX < 0)
90                          continue;
91                      data[pos++] = sX;
92                      if (pos == data.length) {
93                          decodeAndEnqueue(data, pos);
94                          return;
95                      }
96                      break;
97              }
98          }
99      }
100 
101     private void decodeAndEnqueue(byte[] data, int len) {
102         int accum = 0;
103         accum |= data[0] << 18;
104         accum |= data[1] << 12;
105         accum |= data[2] << 6;
106         accum |= data[3];
107 
108         byte b1 = (byte)(accum >>> 16);
109         byteq.enqueue(b1);
110 
111         if (len > 2) {
112             byte b2 = (byte)((accum >>> 8) & 0xFF);
113             byteq.enqueue(b2);
114 
115             if (len > 3) {
116                 byte b3 = (byte)(accum & 0xFF);
117                 byteq.enqueue(b3);
118             }
119         }
120     }
