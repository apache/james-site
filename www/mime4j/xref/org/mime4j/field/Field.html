16  package org.mime4j.field;
17  
18  import java.util.regex.Matcher;
19  import java.util.regex.Pattern;
20  
21  import org.apache.commons.logging.Log;
22  import org.apache.commons.logging.LogFactory;
23  
24  
25  /***
26   * The base class of all field classes.
27   *
28   * @author Niklas Therning
29   * @version $Id: Field.java,v 1.6 2004/10/25 07:26:46 ntherning Exp $
30   */
31  public abstract class Field {
32      private static Log log = LogFactory.getLog(Field.class);
33      
34      public static final String SENDER = "Sender";
35      public static final String FROM = "From";
36      public static final String TO = "To";
37      public static final String CC = "Cc";
38      public static final String BCC = "Bcc";
39      public static final String REPLY_TO = "Reply-To";
40      public static final String RESENT_SENDER = "Resent-Sender";
41      public static final String RESENT_FROM = "Resent-From";
42      public static final String RESENT_TO = "Resent-To";
43      public static final String RESENT_CC = "Resent-Cc";
44      public static final String RESENT_BCC = "Resent-Bcc";
45  
46      public static final String DATE = "Date";
47      public static final String RESENT_DATE = "Resent-Date";
48  
49      public static final String SUBJECT = "Subject";
50      public static final String CONTENT_TYPE = "Content-Type";
51      public static final String CONTENT_TRANSFER_ENCODING = 
52                                          "Content-Transfer-Encoding";
53      
54      private static final String FIELD_NAME_PATTERN = 
55          "^([//x21-//x39//x3b-//x7e]+)[ \t]*:";
56      private static Pattern fieldNamePattern = 
57          Pattern.compile(FIELD_NAME_PATTERN);
58      
59      private String name;
60      private String body;
61      private String raw;
62      
63      protected Field() {
64      }
65      
66      /***
67       * Parsers the given string and returns an instance of the 
68       * <code>Field</code> class. The type of the class returned depends on
69       * the field name:
70       * <table>
71       *      <tr>
72       *          <td><em>Field name</em></td><td><em>Class returned</em></td>
73       *          <td>Content-Type</td><td>org.mime4j.field.ContentTypeField</td>
74       *          <td>other</td><td>org.mime4j.field.UnstructuredField</td>
75       *      </tr>
76       * </table>
77       * 
78       * @param s the string to parse.
79       * @return a <code>Field</code> instance.
80       * @throws IllegalArgumentException on parse errors.
81       */
82      public static Field parse(String s) {
83          String raw = s;
84          
85          /*
86           * Unfold the field.
87           */
88          s = s.replaceAll("\r|\n", "");
89          
90          /*
91           * Split into name and value.
92           */
93          Matcher fieldMatcher = fieldNamePattern.matcher(s);
94          if (!fieldMatcher.find()) {
95              throw new IllegalArgumentException("Invalid field in string");
96          }
97          String name = fieldMatcher.group(1);
98          
99          String body = s.substring(fieldMatcher.end());
100         if (body.length() > 0 && body.charAt(0) == ' ') {
101             body = body.substring(1);
102         }
103         
104         Field f = null;
105 
106         // TODO jcheng 2004-09-27: Clean this up
107 
108         if (name.equalsIgnoreCase(CONTENT_TRANSFER_ENCODING)) {
109             f = new ContentTransferEncodingField();
110         } else if (name.equalsIgnoreCase(CONTENT_TYPE)) {
111             f = new ContentTypeField();
112         } else if (name.equalsIgnoreCase(DATE) || name.equalsIgnoreCase(RESENT_DATE)) {
113             f = new DateTimeField();
114         } else if (name.equalsIgnoreCase(FROM) || name.equalsIgnoreCase(RESENT_FROM)) {
115             f = new MailboxListField();
116         } else if (name.equalsIgnoreCase(SENDER) || name.equalsIgnoreCase(RESENT_SENDER)) {
117             f = new MailboxField();
118         } else if (name.equalsIgnoreCase(TO) || name.equalsIgnoreCase(CC) || name.equalsIgnoreCase(BCC)
119                 || name.equalsIgnoreCase(RESENT_TO) || name.equalsIgnoreCase(RESENT_CC)
120                 || name.equalsIgnoreCase(RESENT_BCC) || name.equalsIgnoreCase(REPLY_TO)) {
121             f = new AddressListField();
122         } else {
123             f = new UnstructuredField();
124         }
125         
126         f.name = name;
127         f.raw = raw;
128         f.body = body;
129         f.parseBody(body);
130         
131         return f;
132     }
133     
134     /***
135      * Parses the field body.
136      * 
137      * @param body the field unfolded body.
138      */
139     protected abstract void parseBody(String body);
140     
141     /***
142      * Gets the name of the field (<code>Subject</code>, 
143      * <code>From</code>, etc).
144      * 
145      * @return the field name.
146      */
147     public String getName() {
148         return name;
149     }
150     
151     /***
152      * Gets the original raw field string.
153      * 
154      * @return the original raw field string.
155      */
156     public String getRaw() {
157         return raw;
158     }
159     
160     /***
161      * Gets the unfolded, unparsed and possibly encoded (see RFC 2047) field 
162      * body string.
163      * 
164      * @return the unfolded unparsed field body string.
165      */
166     public String getBody() {
167         return body;
168     }
169     
170     /***
171      * Determines if this is a <code>Content-Type</code> field.
172      * 
173      * @return <code>true</code> if this is a <code>Content-Type</code> field,
174      *         <code>false</code> otherwise.
175      */
176     public boolean isContentType() {
177         return CONTENT_TYPE.equalsIgnoreCase(name);
178     }
179     
180     /***
181      * Determines if this is a <code>Subject</code> field.
182      * 
183      * @return <code>true</code> if this is a <code>Subject</code> field,
184      *         <code>false</code> otherwise.
185      */
186     public boolean isSubject() {
187         return SUBJECT.equalsIgnoreCase(name);
188     }
189     
190     /***
191      * Determines if this is a <code>From</code> field.
192      * 
193      * @return <code>true</code> if this is a <code>From</code> field,
194      *         <code>false</code> otherwise.
195      */
196     public boolean isFrom() {
197         return FROM.equalsIgnoreCase(name);
198     }
199     
200     /***
201      * Determines if this is a <code>To</code> field.
202      * 
203      * @return <code>true</code> if this is a <code>To</code> field,
204      *         <code>false</code> otherwise.
205      */
206     public boolean isTo() {
207         return TO.equalsIgnoreCase(name);
208     }
209     
210     public String toString() {
211         return raw;
212     }
