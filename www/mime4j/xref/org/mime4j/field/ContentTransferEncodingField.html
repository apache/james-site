16  package org.mime4j.field;
17  
18  
19  /***
20   * Represents a <code>Content-Transfer-Encoding</code> field.
21   *
22   * @author Niklas Therning
23   * @version $Id: ContentTransferEncodingField.java,v 1.2 2004/10/02 12:41:11 ntherning Exp $
24   */
25  public class ContentTransferEncodingField extends Field {
26      /***
27       * The <code>7bit</code> encoding.
28       */
29      public static final String ENC_7BIT = "7bit";
30      /***
31       * The <code>8bit</code> encoding.
32       */
33      public static final String ENC_8BIT = "8bit";
34      /***
35       * The <code>binary</code> encoding.
36       */
37      public static final String ENC_BINARY = "binary";
38      /***
39       * The <code>quoted-printable</code> encoding.
40       */
41      public static final String ENC_QUOTED_PRINTABLE = "quoted-printable";
42      /***
43       * The <code>base64</code> encoding.
44       */
45      public static final String ENC_BASE64 = "base64";
46      
47      private String encoding;
48      
49      protected ContentTransferEncodingField() {
50      }
51      
52      protected void parseBody(String body) {
53          encoding = body.trim().toLowerCase();
54      }
55      
56      /***
57       * Gets the encoding defined in this field.
58       * 
59       * @return the encoding or an empty string if not set.
60       */
61      public String getEncoding() {
62          return encoding;
63      }
64      
65      /***
66       * Gets the encoding of the given field if. Returns the default 
67       * <code>7bit</code> if not set or if
68       * <code>f</code> is <code>null</code>.
69       * 
70       * @return the encoding.
71       */
72      public static String getEncoding(ContentTransferEncodingField f) {
73          if (f != null && f.getEncoding().length() != 0) {
74              return f.getEncoding();
75          }
76          return ENC_7BIT;
77      }
78  }
