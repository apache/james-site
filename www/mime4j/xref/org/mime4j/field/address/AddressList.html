16  package org.mime4j.field.address;

18  import org.mime4j.field.address.parser.AddressListParser;
19  import org.mime4j.field.address.parser.ParseException;

21  import java.io.StringReader;
22  import java.util.ArrayList;
23  import java.util.Iterator;
24  
25  /***
26   * An immutable, random-access list of Address objects.
27   *
28   * @author Joe Cheng &lt;code@joecheng.com&gt;
29   */
30  public class AddressList {
31      
32      private ArrayList addresses;
33  
34      /***
35       * @param addresses An ArrayList that contains only Address objects. 
36       * @param dontCopy true iff it is not possible for the addresses ArrayList to be modified by someone else.
37       */
38      public AddressList(ArrayList addresses, boolean dontCopy) {
39          if (addresses != null)
40              this.addresses = (dontCopy ? addresses : (ArrayList) addresses.clone());
41          else
42              this.addresses = new ArrayList(0);
43      }
44  
45      public Iterator iterator() {
46          return new Iterator() {
47              private int pos = 0;
48              public void remove() { throw new UnsupportedOperationException(); }
49              public boolean hasNext() { return pos < size(); }
50              public Object next() { return get(pos++); }
51          };
52      }
53  
54      /***
55       * The number of elements in this list.
56       */
57      public int size() {
58          return addresses.size();
59      }
60  
61      /***
62       * Gets an address. 
63       */
64      public Address get(int index) {
65          if (0 > index || size() <= index)
66              throw new IndexOutOfBoundsException();
67          return (Address) addresses.get(index);
68      }
69  
70      /***
71       * Returns a flat list of all mailboxes represented
72       * in this address list. Use this if you don't care
73       * about grouping. 
74       */
75      public MailboxList flatten() {
76          // in the common case, all addresses are mailboxes
77          boolean groupDetected = false;
78          for (int i = 0; i < size(); i++) {
79              if (!(get(i) instanceof Mailbox)) {
80                  groupDetected = true;
81                  break;
82              }
83          }
84          
85          if (!groupDetected)
86              return new MailboxList(addresses, true);
87          
88          ArrayList results = new ArrayList();
89          for (int i = 0; i < size(); i++) {
90              Address addr = get(i);
91              addr.addMailboxesTo(results);
92          }
93          
94          // copy-on-construct this time, because subclasses
95          // could have held onto a reference to the results
96          return new MailboxList(results, false);
97      }
98      
99      /***
100    * Dumps a representation of this address list to
101    * stdout, for debugging purposes.
102    */
103   public void print() {
104       for (int i = 0; i < size(); i++) {
105           Address addr = get(i);
106           System.out.println(addr.toString());
107       }
108   }

112   /***
113    * Parse the address list string, such as the value 
114    * of a From, To, Cc, Bcc, Sender, or Reply-To
115    * header.
116    * 
117    * The string MUST be unfolded already.
118    */
119   public static AddressList parse(String rawAddressList) throws ParseException {
120       AddressListParser parser = new AddressListParser(new StringReader(rawAddressList));
121       return Builder.getInstance().buildAddressList(parser.parse());
122   }
