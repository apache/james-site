16  package org.mime4j.field.address;
17  
18  import java.util.ArrayList;
19  
20  /***
21   * An immutable, random-access list of Mailbox objects.
22   *
23   * @author Joe Cheng &lt;code@joecheng.com&gt;
24   */
25  public class MailboxList {
26  
27      private ArrayList mailboxes;
28      
29      /***
30       * @param mailboxes An ArrayList that contains only Mailbox objects. 
31       * @param dontCopy true iff it is not possible for the mailboxes ArrayList to be modified by someone else.
32       */
33      public MailboxList(ArrayList mailboxes, boolean dontCopy) {
34          if (mailboxes != null)
35              this.mailboxes = (dontCopy ? mailboxes : (ArrayList) mailboxes.clone());
36          else
37              mailboxes = new ArrayList(0);
38      }
39      
40      /***
41       * The number of elements in this list.
42       */
43      public int size() {
44          return mailboxes.size();
45      }
46      
47      /***
48       * Gets an address. 
49       */
50      public Mailbox get(int index) {
51          if (0 > index || size() <= index)
52              throw new IndexOutOfBoundsException();
53          return (Mailbox) mailboxes.get(index);
54      }
55      
56      /***
57       * Dumps a representation of this mailbox list to
58       * stdout, for debugging purposes.
59       */
60      public void print() {
61          for (int i = 0; i < size(); i++) {
62              Mailbox mailbox = get(i);
63              System.out.println(mailbox.toString());
64          }
65      }
66  
67  }
