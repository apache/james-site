40      public AddressList buildAddressList(ASTaddress_list node) {
41          ArrayList list = new ArrayList();
42          for (int i = 0; i < node.jjtGetNumChildren(); i++) {
43              ASTaddress childNode = (ASTaddress) node.jjtGetChild(i);
44              Address address = buildAddress(childNode);
45              list.add(address);
46          }
47          return new AddressList(list, true);
48      }

50      private Address buildAddress(ASTaddress node) {
51          ChildNodeIterator it = new ChildNodeIterator(node);
52          Node n = it.nextNode();
53          if (n instanceof ASTaddr_spec) {
54              return buildAddrSpec((ASTaddr_spec)n);
55          }
56          else if (n instanceof ASTangle_addr) {
57              return buildAngleAddr((ASTangle_addr)n);
58          }
59          else if (n instanceof ASTphrase) {
60              String name = buildString((ASTphrase)n, false);
61              Node n2 = it.nextNode();
62              if (n2 instanceof ASTgroup_body) {
63                  return new Group(name, buildGroupBody((ASTgroup_body)n2));
64              }
65              else if (n2 instanceof ASTangle_addr) {
66                  name = DecoderUtil.decodeEncodedWords(name);
67                  return new NamedMailbox(name, buildAngleAddr((ASTangle_addr)n2));
68              }
69              else {
70                  throw new IllegalStateException();
71              }
72          }
73          else {
74              throw new IllegalStateException();
75          }
76      }

80      private MailboxList buildGroupBody(ASTgroup_body node) {
81          ArrayList results = new ArrayList();
82          ChildNodeIterator it = new ChildNodeIterator(node);
83          while (it.hasNext()) {
84              Node n = it.nextNode();
85              if (n instanceof ASTmailbox)
86                  results.add(buildMailbox((ASTmailbox)n));
87              else
88                  throw new IllegalStateException();
89          }
90          return new MailboxList(results, true);
91      }

93      private Mailbox buildMailbox(ASTmailbox node) {
94          ChildNodeIterator it = new ChildNodeIterator(node);
95          Node n = it.nextNode();
96          if (n instanceof ASTaddr_spec) {
97              return buildAddrSpec((ASTaddr_spec)n);
98          }
99          else if (n instanceof ASTangle_addr) {
100           return buildAngleAddr((ASTangle_addr)n);
101       }
102       else if (n instanceof ASTname_addr) {
103           return buildNameAddr((ASTname_addr)n);
104       }
105       else {
106           throw new IllegalStateException();
107       }
108   }

110   private NamedMailbox buildNameAddr(ASTname_addr node) {
111       ChildNodeIterator it = new ChildNodeIterator(node);
112       Node n = it.nextNode();
113       String name;
114       if (n instanceof ASTphrase) {
115           name = buildString((ASTphrase)n, false);
116       }
117       else {
118           throw new IllegalStateException();
119       }
120       
121       n = it.nextNode();
122       if (n instanceof ASTangle_addr) {
123             name = DecoderUtil.decodeEncodedWords(name);
124           return new NamedMailbox(name, buildAngleAddr((ASTangle_addr) n));
125       }
126       else {
127           throw new IllegalStateException();
128       }
129   }
130   
131   private Mailbox buildAngleAddr(ASTangle_addr node) {
132       ChildNodeIterator it = new ChildNodeIterator(node);
133       DomainList route = null;
134       Node n = it.nextNode();
135       if (n instanceof ASTroute) {
136           route = buildRoute((ASTroute)n);
137           n = it.nextNode();
138       }
139       else if (n instanceof ASTaddr_spec)
140           ; // do nothing
141       else
142           throw new IllegalStateException();
143       
144       if (n instanceof ASTaddr_spec)
145           return buildAddrSpec(route, (ASTaddr_spec)n);
146       else
147           throw new IllegalStateException();
148   }
149 
150   private DomainList buildRoute(ASTroute node) {
151       ArrayList results = new ArrayList(node.jjtGetNumChildren());
152       ChildNodeIterator it = new ChildNodeIterator(node);
153       while (it.hasNext()) {
154           Node n = it.nextNode();
155           if (n instanceof ASTdomain)
156               results.add(buildString((ASTdomain)n, true));
157           else
158               throw new IllegalStateException();
159       }
160       return new DomainList(results, true);
161   }
162 
163   private Mailbox buildAddrSpec(ASTaddr_spec node) {
164       return buildAddrSpec(null, node);
165   }
166   private Mailbox buildAddrSpec(DomainList route, ASTaddr_spec node) {
167       ChildNodeIterator it = new ChildNodeIterator(node);
168       String localPart = buildString((ASTlocal_part)it.nextNode(), true);
169       String domain = buildString((ASTdomain)it.nextNode(), true);
170       return new Mailbox(route, localPart, domain);     
171   }

174   private String buildString(SimpleNode node, boolean stripSpaces) {
175       Token head = node.firstToken;
176       Token tail = node.lastToken;
177       StringBuffer out = new StringBuffer();
178       
179       while (head != tail) {
180           out.append(head.image);
181           head = head.next;
182           if (!stripSpaces)
183               addSpecials(out, head.specialToken);
184       }
185       out.append(tail.image);         
186       
187       return out.toString();
188   }
189 
190   private void addSpecials(StringBuffer out, Token specialToken) {
191       if (specialToken != null) {
192           addSpecials(out, specialToken.specialToken);
193           out.append(specialToken.image);
194       }
195   }

197   private static class ChildNodeIterator implements Iterator {
198 
199       private SimpleNode simpleNode;
200       private int index;
201       private int len;
202       
203       public ChildNodeIterator(SimpleNode simpleNode) {
204           this.simpleNode = simpleNode;
205           this.len = simpleNode.jjtGetNumChildren();
206           this.index = 0;
207       }
208       
209       public void remove() {
210           throw new UnsupportedOperationException();
211       }
212 
213       public boolean hasNext() {
214           return index < len;
215       }
216 
217       public Object next() {
218           return nextNode();
219       }
220       
221       public Node nextNode() {
222           return simpleNode.jjtGetChild(index++);
223       }
224       
225   }
