16  package org.mime4j.field.address;
17  
18  import java.util.ArrayList;
19  
20  /***
21   * An immutable, random-access list of Strings (that 
22   * are supposedly domain names or domain literals).
23   *
24   * @author Joe Cheng &lt;code@joecheng.com&gt;
25   */
26  public class DomainList {
27      private ArrayList domains;
28      
29      /***
30       * @param domains An ArrayList that contains only String objects. 
31       * @param dontCopy true iff it is not possible for the domains ArrayList to be modified by someone else.
32       */
33      public DomainList(ArrayList domains, boolean dontCopy) {
34          if (domains != null)
35              this.domains = (dontCopy ? domains : (ArrayList) domains.clone());
36          else
37              this.domains = new ArrayList(0);
38      }
39      
40      /***
41       * The number of elements in this list.
42       */
43      public int size() {
44          return domains.size();
45      }
46  
47      /***
48       * Gets the domain name or domain literal at the
49       * specified index.
50       * @throws IndexOutOfBoundsException If index is &lt; 0 or &gt;= size().
51       */
52      public String get(int index) {
53          if (0 > index || size() <= index)
54              throw new IndexOutOfBoundsException();
55          return (String) domains.get(index);
56      }
57  
58      /***
59       * Returns the list of domains formatted as a route
60       * string (not including the trailing ':'). 
61       */
62      public String toRouteString() {
63          StringBuffer out = new StringBuffer();
64          for (int i = 0; i < domains.size(); i++) {
65              out.append("@");
66              out.append(get(i));
67              if (i + 1 < domains.size())
68                  out.append(",");
69          }
70          return out.toString();
71      }
72  }
