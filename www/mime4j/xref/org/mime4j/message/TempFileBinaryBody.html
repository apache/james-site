30  /***
31   * Binary body backed by a {@link org.mime4j.util.TempFile}.
32   *
33   * @author Niklas Therning
34   * @version $Id: TempFileBinaryBody.java,v 1.2 2004/10/02 12:41:11 ntherning Exp $
35   */
36  class TempFileBinaryBody extends AbstractBody implements BinaryBody {
37      private static Log log = LogFactory.getLog(TempFileBinaryBody.class);
38      
39      private Entity parent = null;
40      private TempFile tempFile = null;
41  
42      public TempFileBinaryBody(InputStream is) throws IOException {
43          
44          TempPath tempPath = TempStorage.getInstance().getRootTempPath();
45          tempFile = tempPath.createTempFile("attachment", ".bin");
46          
47          OutputStream out = tempFile.getOutputStream();
48          CopyUtils.copy(is, out);
49          out.close();
50      }
51      
52      public Entity getParent() {
53          return parent;
54      }
55      
56      public void setParent(Entity parent) {
57          this.parent = parent;
58      }
59      
60      public InputStream getInputStream() throws IOException {
61          return tempFile.getInputStream();
62      }
63      
64      public void writeTo(OutputStream out) throws IOException {
65      }
66  }
