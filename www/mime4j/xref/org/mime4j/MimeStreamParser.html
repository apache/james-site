18  import java.io.IOException;
19  import java.io.InputStream;
20  import java.util.BitSet;
21  import java.util.LinkedList;

23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  import org.mime4j.decoder.Base64InputStream;
26  import org.mime4j.decoder.QuotedPrintableInputStream;

28  /***
29   * <p>
30   * Parses MIME (or RFC822) message streams of bytes or characters and reports 
31   * parsing events to a <code>ContentHandler</code> instance.
32   * </p>
33   * <p>
34   * Typical usage:<br/>
35   * <pre>
36   *      ContentHandler handler = new MyHandler();
37   *      MimeStreamParser parser = new MimeStreamParser();
38   *      parser.setContentHandler(handler);
39   *      parser.parse(new BufferedInputStream(new FileInputStream("mime.msg")));
40   * </pre>
41   * <strong>NOTE:</strong> All lines must end with CRLF 
42   * (<code>\r\n</code>). If you are unsure of the line endings in your stream 
43   * you should wrap it in a {@link org.mime4j.EOLConvertingInputStream} instance.
44   *
45   * @author Niklas Therning
46   * @version $Id: MimeStreamParser.java,v 1.8 2005/02/11 10:12:02 ntherning Exp $
47   */
48  public class MimeStreamParser {
49      private static final Log log = LogFactory.getLog(MimeStreamParser.class);
50      
51      private static BitSet fieldChars = null;
52      
53      private RootInputStream rootStream = null;
54      private LinkedList bodyDescriptors = new LinkedList();
55      private ContentHandler handler = null;
56      private boolean raw = false;
57      
58      static {
59          fieldChars = new BitSet();
60          for (int i = 0x21; i <= 0x39; i++) {
61              fieldChars.set(i);
62          }
63          for (int i = 0x3b; i <= 0x7e; i++) {
64              fieldChars.set(i);
65          }
66      }
67      
68      /***
69       * Creates a new <code>MimeStreamParser</code> instance.
70       */
71      public MimeStreamParser() {
72      }
73  
74      /***
75       * Parses a stream of bytes containing a MIME message.
76       * 
77       * @param is the stream to parse.
78       * @throws IOException on I/O errors.
79       */
80      public void parse(InputStream is) throws IOException {
81          rootStream = new RootInputStream(is);
82          parseMessage(rootStream);
83      }
84      
85      /***
86       * Determines if this parser is currently in raw mode.
87       * 
88       * @return <code>true</code> if in raw mode, <code>false</code>
89       *         otherwise.
90       * @see #setRaw(boolean)
91       */
92      public boolean isRaw() {
93          return raw;
94      }
95      
96      /***
97       * Enables or disables raw mode. In raw mode all future entities 
98       * (messages or body parts) in the stream will be reported to the
99       * {@link ContentHandler#raw(InputStream)} handler method only.
100      * The stream will contain the entire unparsed entity contents 
101      * including header fields and whatever is in the body.
102      * 
103      * @param raw <code>true</code> enables raw mode, <code>false</code>
104      *        disables it.
105      */
106     public void setRaw(boolean raw) {
107         this.raw = raw;
108     }
109     
110     /***
111      * Finishes the parsing and stops reading lines.
112      * NOTE: No more lines will be parsed but the parser
113      * will still call 
114      * {@link ContentHandler#endMultipart()},
115      * {@link ContentHandler#endBodyPart()},
116      * {@link ContentHandler#endMessage()}, etc to match previous calls
117      * to 
118      * {@link ContentHandler#startMultipart(BodyDescriptor)},
119      * {@link ContentHandler#startBodyPart()},
120      * {@link ContentHandler#startMessage()}, etc.
121      */
122     public void stop() {
123         rootStream.truncate();
124     }
125     
126     /***
127      * Parses an entity which consists of a header followed by a body containing
128      * arbitrary data, body parts or an embedded message.
129      * 
130      * @param is the stream to parse.
131      * @throws IOException on I/O errors.
132      */
133     private void parseEntity(InputStream is) throws IOException {
134         BodyDescriptor bd = parseHeader(is);
135         
136         if (bd.isMultipart()) {
137             bodyDescriptors.addFirst(bd);
138             
139             handler.startMultipart(bd);
140             
141             MimeBoundaryInputStream tempIs = 
142                 new MimeBoundaryInputStream(is, bd.getBoundary());
143             handler.preamble(new CloseShieldInputStream(tempIs));
144             tempIs.consume();
145 
146             while (tempIs.hasMoreParts()) {
147                 tempIs = new MimeBoundaryInputStream(is, bd.getBoundary());
148                 parseBodyPart(tempIs);
149                 tempIs.consume();
150                 if (tempIs.parentEOF()) {
151                     if (log.isWarnEnabled()) {
152                         log.warn("Line " + rootStream.getLineNumber() 
153                                 + ": Body part ended prematurely. "
154                                 + "Higher level boundary detected or "
155                                 + "EOF reached.");
156                     }
157                     break;
158                 }
159             }
160             
161             handler.epilogue(new CloseShieldInputStream(is));
162             
163             handler.endMultipart();
164             
165             bodyDescriptors.removeFirst();
166             
167         } else if (bd.isMessage()) {
168             if (bd.isBase64Encoded()) {
169                 log.warn("base64 encoded message/rfc822 detected");
170                 is = new EOLConvertingInputStream(
171                         new Base64InputStream(is));
172             } else if (bd.isQuotedPrintableEncoded()) {
173                 log.warn("quoted-printable encoded message/rfc822 detected");
174                 is = new EOLConvertingInputStream(
175                         new QuotedPrintableInputStream(is));
176             }
177             bodyDescriptors.addFirst(bd);
178             parseMessage(is);
179             bodyDescriptors.removeFirst();
180         } else {
181             handler.body(bd, new CloseShieldInputStream(is));
182         }
183         
184         /*
185          * Make sure the stream has been consumed.
186          */
187         while (is.read() != -1) {
188         }
189     }
190     
191     private void parseMessage(InputStream is) throws IOException {
192         if (raw) {
193             handler.raw(new CloseShieldInputStream(is));
194         } else {
195             handler.startMessage();
196             parseEntity(is);
197             handler.endMessage();
198         }
199     }
200     
201     private void parseBodyPart(InputStream is) throws IOException {
202         if (raw) {
203             handler.raw(new CloseShieldInputStream(is));
204         } else {
205             handler.startBodyPart();
206             parseEntity(is);
207             handler.endBodyPart();
208         }
209     }

211     /***
212      * Parses a header.
213      * 
214      * @param is the stream to parse.
215      * @return a <code>BodyDescriptor</code> describing the body following 
216      *         the header.
217      */
218     private BodyDescriptor parseHeader(InputStream is) throws IOException {
219         BodyDescriptor bd = new BodyDescriptor(bodyDescriptors.isEmpty() 
220                         ? null : (BodyDescriptor) bodyDescriptors.getFirst());
221         
222         handler.startHeader();
223         
224         int lineNumber = rootStream.getLineNumber();
225         
226         StringBuffer sb = new StringBuffer();
227         int curr = 0;
228         int prev = 0;
229         while ((curr = is.read()) != -1) {
230             if (curr == '\n' && (prev == '\n' || prev == 0)) {
231                 /*
232                  * [\r]\n[\r]\n or an immediate \r\n have been seen.
233                  */
234                 sb.deleteCharAt(sb.length() - 1);
235                 break;
236             }
237             sb.append((char) curr);
238             prev = curr == '\r' ? prev : curr;
239         }
240         
241         if (curr == -1 && log.isWarnEnabled()) {
242             log.warn("Line " + rootStream.getLineNumber()  
243                     + ": Unexpected end of headers detected. "
244                     + "Boundary detected in header or EOF reached.");
245         }
246 
247         int start = 0;
248         int pos = 0;
249         int startLineNumber = lineNumber;
250         while (pos < sb.length()) {
251             while (pos < sb.length() && sb.charAt(pos) != '\r') {
252                 pos++;
253             }
254             if (pos < sb.length() - 1 && sb.charAt(pos + 1) != '\n') {
255                 pos++;
256                 continue;
257             }
258             
259             if (pos >= sb.length() - 2 || fieldChars.get(sb.charAt(pos + 2))) {
260                 
261                 /*
262                  * field should be the complete field data excluding the 
263                  * trailing \r\n.
264                  */
265                 String field = sb.substring(start, pos);
266                 start = pos + 2;
267                 
268                 /*
269                  * Check for a valid field.
270                  */
271                 int index = field.indexOf(':');
272                 boolean valid = false;
273                 if (index != -1 && fieldChars.get(field.charAt(0))) {
274                     valid = true;
275                     String fieldName = field.substring(0, index).trim();
276                     for (int i = 0; i < fieldName.length(); i++) {
277                         if (!fieldChars.get(fieldName.charAt(i))) {
278                             valid = false;
279                             break;
280                         }
281                     }
282                     
283                     if (valid) {
284                         handler.field(field);
285                         bd.addField(fieldName, field.substring(index + 1));
286                     }                        
287                 }
288                 
289                 if (!valid && log.isWarnEnabled()) {
290                     log.warn("Line " + startLineNumber 
291                             + ": Ignoring invalid field: '" + field.trim() + "'");
292                 }          
293                 
294                 startLineNumber = lineNumber;
295             }
296             
297             pos += 2;
298             lineNumber++;
299         }
300         
301         handler.endHeader();
302         
303         return bd;
304     }

313         this.handler = h;
314     }
