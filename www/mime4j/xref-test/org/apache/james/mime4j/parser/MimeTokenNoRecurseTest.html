80      public void testWhenRecurseShouldRecurseInnerMail() throws Exception {

82          nextIs(MimeTokenStream.T_START_HEADER);

145     public void testWhenNoRecurseInputStreamShouldContainInnerMail() throws Exception {
146         stream.setRecursionMode(MimeTokenStream.M_NO_RECURSE);
147         nextIs(MimeTokenStream.T_START_HEADER);
148         nextIs(MimeTokenStream.T_FIELD);
149         nextIs(MimeTokenStream.T_FIELD);
150         nextIs(MimeTokenStream.T_FIELD);
151         nextIs(MimeTokenStream.T_FIELD);
152         nextIs(MimeTokenStream.T_FIELD);
153         nextIs(MimeTokenStream.T_FIELD);
154         nextIs(MimeTokenStream.T_END_HEADER);
155         
156         nextIs(MimeTokenStream.T_START_MULTIPART);
157         nextIs(MimeTokenStream.T_PREAMBLE);
158         nextShouldBeStandardPart(false);
159         
160         nextShouldBeStandardPart(true);
161         nextIs(MimeTokenStream.T_START_BODYPART);
162         nextIs(MimeTokenStream.T_START_HEADER);
163         nextIs(MimeTokenStream.T_FIELD);
164         nextIs(MimeTokenStream.T_END_HEADER);
165         nextIs(MimeTokenStream.T_BODY);
166         InputStream inputStream = stream.getInputStream();
167         int next = inputStream.read();
168         int i=0;
169         while (next != -1) {
170             assertEquals("@" + i, INNER_MAIL.charAt(i++), (char) next);
171             next = inputStream.read();
172         }
173         assertEquals(INNER_MAIL.length()-2, i);
174     }

207     private void nextShouldBeStandardPart(boolean withHeader) throws Exception {
208         nextIs(MimeTokenStream.T_START_BODYPART);
209         nextIs(MimeTokenStream.T_START_HEADER);
210         if (withHeader) {
211             nextIs(MimeTokenStream.T_FIELD);
212         }
213         nextIs(MimeTokenStream.T_END_HEADER);
214         nextIs(MimeTokenStream.T_BODY);
215         nextIs(MimeTokenStream.T_END_BODYPART);
216     }
217     
218     private void nextIs(int state) throws Exception {
219         assertEquals(MimeTokenStream.stateToString(state), MimeTokenStream.stateToString(stream.next()));
220     }
