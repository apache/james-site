39      public void testSimpleEntity() throws Exception {
40          String message = 
41              "To: Road Runner <runner@example.org>\r\n" +
42              "From: Wile E. Cayote <wile@example.org>\r\n" +
43              "Date: Tue, 12 Feb 2008 17:34:09 +0000 (GMT)\r\n" +
44              "Subject: Mail\r\n" +
45              "Content-Type: text/plain\r\n" +
46              "\r\n" +
47              "a very important message";
48          byte[] raw = message.getBytes("US-ASCII");
49          ByteArrayInputStream instream = new ByteArrayInputStream(raw);
50          RootInputStream rootStream = new RootInputStream(instream); 
51          BufferedLineReaderInputStream rawstream = new BufferedLineReaderInputStream(rootStream, 12); 
52          
53          MimeEntity entity = new MimeEntity(
54                  rootStream,
55                  rawstream,
56                  null,
57                  EntityStates.T_START_MESSAGE,
58                  EntityStates.T_END_MESSAGE);
59          
60          
61          assertEquals(EntityStates.T_START_MESSAGE, entity.getState());
62          entity.advance();
63          assertEquals(EntityStates.T_START_HEADER, entity.getState());
64          entity.advance();
65          assertEquals(EntityStates.T_FIELD, entity.getState());
66          assertEquals("To", entity.getFieldName());
67          assertEquals(" Road Runner <runner@example.org>", entity.getFieldValue());
68          entity.advance();
69          assertEquals(EntityStates.T_FIELD, entity.getState());
70          assertEquals("From", entity.getFieldName());
71          assertEquals(" Wile E. Cayote <wile@example.org>", entity.getFieldValue());
72          entity.advance();
73          assertEquals(EntityStates.T_FIELD, entity.getState());
74          assertEquals("Date", entity.getFieldName());
75          assertEquals(" Tue, 12 Feb 2008 17:34:09 +0000 (GMT)", entity.getFieldValue());
76          entity.advance();
77          assertEquals(EntityStates.T_FIELD, entity.getState());
78          assertEquals("Subject", entity.getFieldName());
79          assertEquals(" Mail", entity.getFieldValue());
80          entity.advance();
81          assertEquals(EntityStates.T_FIELD, entity.getState());
82          assertEquals("Content-Type", entity.getFieldName());
83          assertEquals(" text/plain", entity.getFieldValue());
84          entity.advance();
85          assertEquals(EntityStates.T_END_HEADER, entity.getState());
86          try {
87              entity.getFieldName();
88              fail("IllegalStateException should have been thrown");
89          } catch (IllegalStateException expected) {
90          }
91          try {
92              entity.getFieldValue();
93              fail("IllegalStateException should have been thrown");
94          } catch (IllegalStateException expected) {
95          }
96          
97          entity.advance();
98          assertEquals(EntityStates.T_BODY, entity.getState());
99          assertEquals("a very important message", IOUtils.toString(entity.getContentStream()));
100         entity.advance();
101         assertEquals(EntityStates.T_END_MESSAGE, entity.getState());
102         try {
103             entity.getContentStream();
104             fail("IllegalStateException should have been thrown");
105         } catch (IllegalStateException expected) {
106         }
107         entity.advance();
108         assertEquals(EntityStates.T_END_OF_STREAM, entity.getState());
109         try {
110             entity.advance();
111             fail("IllegalStateException should have been thrown");
112         } catch (IllegalStateException expected) {
113         }
114     }

137         RootInputStream rootStream = new RootInputStream(instream);

150         entity.advance();
151         assertEquals(EntityStates.T_FIELD, entity.getState());
152         assertEquals("To", entity.getFieldName());
153         assertEquals(" Road Runner <runner@example.org>", entity.getFieldValue());
154         entity.advance();

156         assertEquals("From", entity.getFieldName());

170         entity.advance();
171         assertEquals(EntityStates.T_END_HEADER, entity.getState());
172         entity.advance();
173         assertEquals(EntityStates.T_START_MULTIPART, entity.getState());
174         entity.advance();

184         p1.advance();

186         assertEquals("Content-Type", p1.getFieldName());
187         assertEquals(" text/plain; charset=US-ASCII", p1.getFieldValue());
188         p1.advance();
189         assertEquals(EntityStates.T_END_HEADER, p1.getState());
190         p1.advance();

200         
201         assertEquals(EntityStates.T_START_BODYPART, p2.getState());
202         p2.advance();
203         assertEquals(EntityStates.T_START_HEADER, p2.getState());

205         assertEquals(EntityStates.T_FIELD, p2.getState());

208         p2.advance();
209         assertEquals(EntityStates.T_END_HEADER, p2.getState());
210         p2.advance();
211         assertEquals(EntityStates.T_BODY, p2.getState());
212         assertEquals("yada yada yada", IOUtils.toString(p2.getContentStream()));
213         p2.advance();
214         assertEquals(EntityStates.T_END_BODYPART, p2.getState());
215         p2.advance();
216         assertEquals(EntityStates.T_END_OF_STREAM, p2.getState());
217 
218         entity.advance();
219         assertEquals(EntityStates.T_EPILOGUE, entity.getState());
220         assertEquals("Goodbye!", IOUtils.toString(entity.getContentStream()));
221         entity.advance();
222         assertEquals(EntityStates.T_END_MULTIPART, entity.getState());
223         entity.advance();
224         assertEquals(EntityStates.T_END_MESSAGE, entity.getState());
225         entity.advance();
226         assertEquals(EntityStates.T_END_OF_STREAM, entity.getState());
227     }
