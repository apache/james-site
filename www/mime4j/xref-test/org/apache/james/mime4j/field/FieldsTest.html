233     public void testSubject() throws Exception {
234         assertEquals("Subject: ", Fields.subject("").getRaw());
235         assertEquals("Subject: test", Fields.subject("test").getRaw());
236         assertEquals("Subject: =?ISO-8859-1?Q?Sm=F8rebr=F8d?=", Fields.subject(
237                 "Sm\370rebr\370d").getRaw());
238 
239         String seventyEight = "12345678901234567890123456789012345678901234567890123456789012345678";
240         assertEquals("Subject:\r\n " + seventyEight, Fields.subject(
241                 seventyEight).getRaw());
242 
243         String seventyNine = seventyEight + "9";
244         String expected = "Subject: =?US-ASCII?Q?1234567890123456789012345678901234?="
245                 + "\r\n =?US-ASCII?Q?56789012345678901234567890123456789?=";
