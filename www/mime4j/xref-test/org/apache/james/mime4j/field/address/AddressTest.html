117     public void testTortureTest() throws ParseException {
118 
119         // Source: http://mailformat.dan.info/headers/from.html
120         // (Commented out pending confirmation of legality--I think the local-part is illegal.) 
121         // AddressList.parse("\"Guy Macon\" <guymacon+\" http://www.guymacon.com/ \"00@spamcop.net>");
122 
123         // Taken mostly from RFC822.
124 
125         // Just make sure these are recognized as legal address lists;
126         // there shouldn't be any aspect of the RFC that is tested here
127         // but not in the other unit tests.
128 
129         AddressList.parse("Alfred Neuman <Neuman@BBN-TENEXA>");
130         AddressList.parse("Neuman@BBN-TENEXA");
131         AddressList.parse("\"George, Ted\" <Shared@Group.Arpanet>");
132         AddressList.parse("Wilt . (the Stilt) Chamberlain@NBA.US");
133 
134         // NOTE: In RFC822 8.1.5, the following example did not have "Galloping Gourmet"
135         // in double-quotes.  I can only assume this was a typo, since 6.2.4 specifically
136         // disallows spaces in unquoted local-part.
137         AddressList.parse("     Gourmets:  Pompous Person <WhoZiWhatZit@Cordon-Bleu>," +
138                 "                Childs@WGBH.Boston, \"Galloping Gourmet\"@" +
139                 "                ANT.Down-Under (Australian National Television)," +
140                 "                Cheapie@Discount-Liquors;," +
141                 "       Cruisers:  Port@Portugal, Jones@SEA;," +
142                 "         Another@Somewhere.SomeOrg");
143 
144         // NOTE: In RFC822 8.3.3, the following example ended with a lone ">" after
145         // Tops-20-Host.  I can only assume this was a typo, since 6.1 clearly shows
146         // ">" requires a matching "<".
147         AddressList.parse("Important folk:" +
148                 "                   Tom Softwood <Balsa@Tree.Root>," +
149                 "                   \"Sam Irving\"@Other-Host;," +
150                 "                 Standard Distribution:" +
151                 "                   /main/davis/people/standard@Other-Host," +
152                 "                   \"<Jones>standard.dist.3\"@Tops-20-Host;");
153 
154         // The following are from a Usenet post by Dan J. Bernstein:
155         // http://groups.google.com/groups?selm=1996Aug1418.21.01.28081%40koobera.math.uic.edu
156         AddressList.parse("\":sysmail\"@  Some-Group.\t         Some-Org, Muhammed.(I am  the greatest) Ali @(the)Vegas.WBA");
157         AddressList.parse("me@home.com (comment (nested (deeply\\))))");
158         AddressList.parse("mailing list: me@home.com, route two <you@work.com>, them@play.com ;");
159 
160     }

212     public void testInteractiveMain() throws Exception {
213         PrintStream out_orig = System.out;
214         InputStream in_orig = System.in;
215         PrintStream err_orig = System.err;
216         try {
217             PipedOutputStream piped = new PipedOutputStream();
218             PipedInputStream newInput = new PipedInputStream(piped);
219             
220             PipedInputStream inOut = new PipedInputStream();
221             PrintStream outPs = new PrintStream(new PipedOutputStream(inOut));
222             BufferedReader outReader = new BufferedReader(new InputStreamReader(inOut));
223             PipedInputStream inErr = new PipedInputStream();
224             PrintStream errPs = new PrintStream(new PipedOutputStream(inErr));
225             BufferedReader errReader = new BufferedReader(new InputStreamReader(inErr));
226             
227             
228             System.setIn(newInput);
229             System.setOut(outPs);
230             System.setErr(errPs);
231             Thread t = new Thread() {
232                 public void run() {
233                     try {
234                         AddressList.main(null);
235                     } catch (Exception e) {
236                         fail("Catched an exception in main: "+e);
237                     }
238                 }
239             };
240             t.start();
241             
242             PrintWriter input = new PrintWriter(piped);
243             
244             input.write("Test <test@example.com>\r\n");
245             input.flush();
246 
247             String out = outReader.readLine();
248             assertEquals("> Test <test@example.com>", out);
249 
250             input.write("A <foo@example.com>\r\n");
251             input.flush();
252             
253             String out2 = outReader.readLine();
254             assertEquals("> A <foo@example.com>", out2);
255 
256             input.write("\"Foo Bar\" <foo>\r\n");
257             input.flush();
258             
259             String out3 = errReader.readLine();
260             assertNotNull(out3);
261 
262             input.write("quit\r\n");
263             input.flush();
264             
265             // we read 2 angular brackets because one was from the previous exception error.
266             String out4 = outReader.readLine();
267             assertEquals("> > Goodbye.", out4);
268 
269             t.join();
270         } finally {
271             System.setIn(in_orig);
272             System.setOut(out_orig);
273             System.setErr(err_orig);
274         }
275     }

351     public void testGroupSerialization() {
352         ArrayList al = new ArrayList();
353         al.add(new Mailbox("test", "example.com"));
354         al.add(new NamedMailbox("Foo!", "foo", "example.com"));
355         DomainList dl = new DomainList(new ArrayList(Arrays.asList(new String[] {"foo.example.com"})), true);
