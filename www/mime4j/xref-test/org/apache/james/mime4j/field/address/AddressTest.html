111     public void testTortureTest() throws ParseException {
112 
113         // Source: http://mailformat.dan.info/headers/from.html
114         // (Commented out pending confirmation of legality--I think the local-part is illegal.) 
115         // AddressList.parse("\"Guy Macon\" <guymacon+\" http://www.guymacon.com/ \"00@spamcop.net>");
116 
117         // Taken mostly from RFC822.
118 
119         // Just make sure these are recognized as legal address lists;
120         // there shouldn't be any aspect of the RFC that is tested here
121         // but not in the other unit tests.
122 
123         AddressList.parse("Alfred Neuman <Neuman@BBN-TENEXA>");
124         AddressList.parse("Neuman@BBN-TENEXA");
125         AddressList.parse("\"George, Ted\" <Shared@Group.Arpanet>");
126         AddressList.parse("Wilt . (the Stilt) Chamberlain@NBA.US");
127 
128         // NOTE: In RFC822 8.1.5, the following example did not have "Galloping Gourmet"
129         // in double-quotes.  I can only assume this was a typo, since 6.2.4 specifically
130         // disallows spaces in unquoted local-part.
131         AddressList.parse("     Gourmets:  Pompous Person <WhoZiWhatZit@Cordon-Bleu>," +
132                 "                Childs@WGBH.Boston, \"Galloping Gourmet\"@" +
133                 "                ANT.Down-Under (Australian National Television)," +
134                 "                Cheapie@Discount-Liquors;," +
135                 "       Cruisers:  Port@Portugal, Jones@SEA;," +
136                 "         Another@Somewhere.SomeOrg");
137 
138         // NOTE: In RFC822 8.3.3, the following example ended with a lone ">" after
139         // Tops-20-Host.  I can only assume this was a typo, since 6.1 clearly shows
140         // ">" requires a matching "<".
141         AddressList.parse("Important folk:" +
142                 "                   Tom Softwood <Balsa@Tree.Root>," +
143                 "                   \"Sam Irving\"@Other-Host;," +
144                 "                 Standard Distribution:" +
145                 "                   /main/davis/people/standard@Other-Host," +
146                 "                   \"<Jones>standard.dist.3\"@Tops-20-Host;");
147 
148         // The following are from a Usenet post by Dan J. Bernstein:
149         // http://groups.google.com/groups?selm=1996Aug1418.21.01.28081%40koobera.math.uic.edu
150         AddressList.parse("\":sysmail\"@  Some-Group.\t         Some-Org, Muhammed.(I am  the greatest) Ali @(the)Vegas.WBA");
151         AddressList.parse("me@home.com (comment (nested (deeply//))))");
152         AddressList.parse("mailing list: me@home.com, route two <you@work.com>, them@play.com ;");
153

206     public void testInteractiveMain() throws Exception {
207         PrintStream out_orig = System.out;
208         InputStream in_orig = System.in;
209         PrintStream err_orig = System.err;
210         try {
211             PipedOutputStream piped = new PipedOutputStream();
212             PipedInputStream newInput = new PipedInputStream(piped);
213             
214             PipedInputStream inOut = new PipedInputStream();
215             PrintStream outPs = new PrintStream(new PipedOutputStream(inOut));
216             BufferedReader outReader = new BufferedReader(new InputStreamReader(inOut));
217             PipedInputStream inErr = new PipedInputStream();
218             PrintStream errPs = new PrintStream(new PipedOutputStream(inErr));
219             BufferedReader errReader = new BufferedReader(new InputStreamReader(inErr));
220             
221             
222             System.setIn(newInput);
223             System.setOut(outPs);
224             System.setErr(errPs);
225             Thread t = new Thread() {
226                 public void run() {
227                     try {
228                         AddressList.main(null);
229                     } catch (Exception e) {
230                         fail("Catched an exception in main: "+e);
231                     }
232                 }
233             };
234             t.start();
235             
236             PrintWriter input = new PrintWriter(piped);
237             
238             input.write("Test <test@example.com>\r\n");
239             input.flush();
240 
241             String out = outReader.readLine();
242             assertEquals("> Test <test@example.com>", out);
243 
244             input.write("A <foo@example.com>\r\n");
245             input.flush();
246             
247             String out2 = outReader.readLine();
248             assertEquals("> A <foo@example.com>", out2);
249 
250             input.write("\"Foo Bar\" <foo>\r\n");
251             input.flush();
252             
253             String out3 = errReader.readLine();
254             assertNotNull(out3);
255 
256             input.write("quit\r\n");
257             input.flush();
258             
259             // we read 2 angular brackets because one was from the previous exception error.
260             String out4 = outReader.readLine();
261             assertEquals("> > Goodbye.", out4);
262 
263             t.join();
264         } finally {
265             System.setIn(in_orig);
266             System.setOut(out_orig);
267             System.setErr(err_orig);
268         }
269     }

345     public void testGroupSerialization() {
346         ArrayList al = new ArrayList();
347         al.add(new Mailbox("test", "example.com"));
348         al.add(new NamedMailbox("Foo!", "foo", "example.com"));
349         DomainList dl = new DomainList(new ArrayList(Arrays.asList(new String[] {"foo.example.com"})), true);
350         NamedMailbox namedMailbox = new NamedMailbox("Foo Bar", dl, "foo2", "example.com");
351         assertSame(dl, namedMailbox.getRoute());
352         al.add(namedMailbox);
353         Group g = new Group("group", new MailboxList(al, false));
354         assertEquals("group:<test@example.com>,Foo! <foo@example.com>,Foo Bar <foo2@example.com>;", g.toString());
