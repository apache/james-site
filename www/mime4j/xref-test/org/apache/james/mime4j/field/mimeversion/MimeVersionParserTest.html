20  package org.apache.james.mime4j.field.mimeversion;
21  
22  import java.io.StringReader;
23  
24  import junit.framework.TestCase;
25  
26  public class MimeVersionParserTest extends TestCase {
27  
28      @Override
29      protected void setUp() throws Exception {
30          super.setUp();
31      }
32  
33      @Override
34      protected void tearDown() throws Exception {
35          super.tearDown();
36      }
37  
38      public void testPlainLine() throws Exception {
39          check("2.4", 2, 4);
40          check("25.344", 25, 344);
41          check("0.1", 0, 1);
42          check("123234234.0", 123234234, 0);
43      }
44      
45      public void testLineWithComments() throws Exception {
46          check("2(A comment).4", 2, 4);
47          check("2(.8).4", 2, 4);
48          check("(A comment)2.4", 2, 4);
49          check("2.4(A comment)", 2, 4);
50          check("2.(A comment)4", 2, 4);
51      }
52      
53      public void testLineWithNestedComments() throws Exception {
54          check("2(4.45 ( Another ()comment () blah (Wobble(mix)))Whatever).4", 2, 4);
55      }
56      
57      public void testEmptyLine() throws Exception {
58          try {
59              parse("(This is just a comment)");
60              fail("Expected exception to be thrown");
61          } catch (ParseException e) {
62              //expected
63          }
64      }
65      
66      private void check(String input, int expectedMajorVersion, int expectedMinorVersion) throws Exception {
67          MimeVersionParser parser = parse(input);
68          assertEquals("Major version number", expectedMajorVersion, parser.getMajorVersion());
69          assertEquals("Minor version number", expectedMinorVersion, parser.getMinorVersion());
70      }
71  
72      private MimeVersionParser parse(String input) throws ParseException {
73          MimeVersionParser parser = new MimeVersionParser(new StringReader(input));
74          parser.parseAll();
75          return parser;
76      }
77  }
