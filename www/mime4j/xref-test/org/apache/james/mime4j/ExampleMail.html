260     public static final byte[] join(byte[] one, byte[] two) {
261         byte[] results = new byte[one.length + two.length];
262         System.arraycopy(one, 0, results, 0, one.length);
263         System.arraycopy(two, 0, results, one.length, two.length);
264         return results;
265     }

267     public static final byte[] join(byte[][] byteArrays) {
268         int length = 0;
269         for (int i = 0; i < byteArrays.length; i++) {
270             byte[] bytes = byteArrays[i];
271             length += bytes.length;
272         }
273         byte[] results = new byte[length];
274         int count = 0;
275         for (int i = 0; i < byteArrays.length; i++) {
276             byte[] bytes = byteArrays[i];
277             System.arraycopy(bytes, 0, results, count, bytes.length);
278             count += bytes.length;
279         }
280         return results;
281     }

283     public static String breakLines(String original) {
284         StringBuffer buffer = new StringBuffer(original);
285         int count = 76;
286         while(count < buffer.length()) {
287             if (buffer.charAt(count) == '=') {
288                 count = count - 1;
289             } else if (buffer.charAt(count-1) == '=') {
290                 count = count - 4;                
291             } else if (buffer.charAt(count-2) == '=') {
292                 count = count - 3;
293             }    
294             buffer.insert(count, '\n');
295             buffer.insert(count, '\r');
296             buffer.insert(count, '=');
297             count += 79;
298         }
299         final String result = buffer.toString();
300         return result;
301     }
