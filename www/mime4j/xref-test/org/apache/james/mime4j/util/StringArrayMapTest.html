33  public class StringArrayMapTest extends TestCase {
34      private StringArrayMap getSampleMap() {
35          final StringArrayMap map = new StringArrayMap();
36          map.addValue("xYz", "a");
37          map.addValue("Xyz", "B");
38          map.addValue("xyz", "c");
39          map.addValue("xs", "1");
40          map.addValue("XS", "2");
41          map.addValue("foo", "bAr");
42          return map;
43      }

45      public void testGetMap() {
46          final Map<String, String[]> map = getSampleMap().getMap();
47          final List<String> keys = new ArrayList<String>(map.keySet());
48          assertEquals(keys.size(), 3);
49          Collections.sort(keys);
50          assertEquals("foo", keys.get(0));
51          assertEquals("xs", keys.get(1));
52          assertEquals("xyz", keys.get(2));
53          final String[] foo = map.get("foo");
54          assertEquals("bAr", foo[0]);
55          final String[] xs = map.get("xs");
56          assertEquals("1", xs[0]);
57          assertEquals("2", xs[1]);
58          final String[] xyz = map.get("xyz");
59          assertEquals(3, xyz.length);
60          assertEquals("a", xyz[0]);
61          assertEquals("B", xyz[1]);
62          assertEquals("c", xyz[2]);
63      }

65      public void testGetNameArray() {
66          final String[] names = getSampleMap().getNameArray();

72      }

79      public void testGetValue() {
80          final StringArrayMap map = getSampleMap();
81          assertEquals("bAr", map.getValue("foo"));
82          assertEquals("bAr", map.getValue("FOO"));
83          assertEquals("1", map.getValue("xs"));
84          assertEquals("a", map.getValue("xyz"));
85          assertEquals("a", map.getValue("xyZ"));
86          assertNull(map.getValue("xz"));

89      public void testGetValues() {
90          final StringArrayMap map = getSampleMap();
91          final String[] foo = map.getValues("foo");
92          assertEquals(1, foo.length);
93          assertEquals("bAr", foo[0]);
94          final String[] FOO = map.getValues("FOO");
95          assertEquals(1, FOO.length);
96          assertEquals("bAr", FOO[0]);
97          final String[] xs = map.getValues("xs");
98          assertEquals(2, xs.length);
99          assertEquals("1", xs[0]);
100         assertEquals("2", xs[1]);
101         final String[] XS = map.getValues("XS");
102         assertEquals(2, XS.length);
103         assertEquals("1", XS[0]);
104         assertEquals("2", XS[1]);
105         final String[] xyz = map.getValues("xyz");
106         assertEquals("a", xyz[0]);
107         assertEquals("B", xyz[1]);
108         assertEquals("c", xyz[2]);
109         final String[] XYZ = map.getValues("XYZ");
110         assertEquals("a", XYZ[0]);
111         assertEquals("B", XYZ[1]);
112         assertEquals("c", XYZ[2]);
113         assertNull(map.getValues("xz"));
114     }

116     private void assertEquals(String[] pArray, Enumeration<String> pEnum) {
117         final List<String> list = new ArrayList<String>();
118         while (pEnum.hasMoreElements()) {
119             list.add(pEnum.nextElement());
120         }
121         Collections.sort(list, Collator.getInstance(Locale.US));
122         assertEquals(pArray.length, list.size());
123         for (int i = 0;  i < pArray.length;  i++) {
124             assertEquals(pArray[i], list.get(i));
125         }
126     }

131         assertEquals(new String[]{"bAr"}, map.getValueEnum("FOO"));
132         assertEquals(new String[]{"1", "2"}, map.getValueEnum("xs"));
133         assertEquals(new String[]{"1", "2"}, map.getValueEnum("Xs"));
134         assertEquals(new String[]{"a", "B", "c"}, map.getValueEnum("xyz"));
135         assertEquals(new String[]{"a", "B", "c"}, map.getValueEnum("XYZ"));
136         assertNull(map.getValues("xz"));
137     }
