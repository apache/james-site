46      /**

48       */
49      public void testGetMap() {
50          final Map map = getSampleMap().getMap();
51          final List keys = new ArrayList(map.keySet());
52          assertEquals(keys.size(), 3);
53          Collections.sort(keys);
54          assertEquals("foo", keys.get(0));
55          assertEquals("xs", keys.get(1));
56          assertEquals("xyz", keys.get(2));
57          final String[] foo = (String[]) map.get("foo");
58          assertEquals("bAr", foo[0]);
59          final String[] xs = (String[]) map.get("xs");
60          assertEquals("1", xs[0]);
61          assertEquals("2", xs[1]);
62          final String[] xyz = (String[]) map.get("xyz");
63          assertEquals(3, xyz.length);
64          assertEquals("a", xyz[0]);
65          assertEquals("B", xyz[1]);
66          assertEquals("c", xyz[2]);
67      }

103      * Test case for {@link StringArrayMap#getValue(String)}.
104      */
105     public void testGetValues() {
106         final StringArrayMap map = getSampleMap();
107         final String[] foo = map.getValues("foo");
108         assertEquals(1, foo.length);
109         assertEquals("bAr", foo[0]);
110         final String[] FOO = map.getValues("FOO");
111         assertEquals(1, FOO.length);
112         assertEquals("bAr", FOO[0]);
113         final String[] xs = map.getValues("xs");
114         assertEquals(2, xs.length);
115         assertEquals("1", xs[0]);
116         assertEquals("2", xs[1]);
117         final String[] XS = map.getValues("XS");
118         assertEquals(2, XS.length);
119         assertEquals("1", XS[0]);
120         assertEquals("2", XS[1]);
121         final String[] xyz = map.getValues("xyz");
122         assertEquals("a", xyz[0]);
123         assertEquals("B", xyz[1]);
124         assertEquals("c", xyz[2]);
125         final String[] XYZ = map.getValues("XYZ");
126         assertEquals("a", XYZ[0]);
127         assertEquals("B", XYZ[1]);
128         assertEquals("c", XYZ[2]);
129         assertNull(map.getValues("xz"));

132     private void assertEquals(String[] pArray, Enumeration pEnum) {
133         final List list = new ArrayList();
134         while (pEnum.hasMoreElements()) {
135             list.add(pEnum.nextElement());
136         }
137         Collections.sort(list, Collator.getInstance(Locale.US));
138         assertEquals(pArray.length, list.size());
139         for (int i = 0;  i < pArray.length;  i++) {
140             assertEquals(pArray[i], list.get(i));
141         }
142     }
