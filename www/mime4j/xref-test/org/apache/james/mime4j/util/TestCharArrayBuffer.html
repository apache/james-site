61      public void testSimpleAppend() throws Exception {
62          CharArrayBuffer buffer = new CharArrayBuffer(16);
63          assertEquals(16, buffer.capacity()); 
64          assertEquals(0, buffer.length());
65          char[] b1 = buffer.toCharArray();
66          assertNotNull(b1);
67          assertEquals(0, b1.length);
68          assertTrue(buffer.isEmpty());
69          assertFalse(buffer.isFull());
70          
71          char[] tmp = new char[] { '1', '2', '3', '4'};
72          buffer.append(tmp, 0, tmp.length);
73          assertEquals(16, buffer.capacity()); 
74          assertEquals(4, buffer.length());
75          assertFalse(buffer.isEmpty());
76          assertFalse(buffer.isFull());
77          
78          char[] b2 = buffer.toCharArray();
79          assertNotNull(b2);
80          assertEquals(4, b2.length);
81          for (int i = 0; i < tmp.length; i++) {
82              assertEquals(tmp[i], b2[i]);
83              assertEquals(tmp[i], buffer.charAt(i));
84          }
85          assertEquals("1234", buffer.toString());
86          
87          buffer.clear();
88          assertEquals(16, buffer.capacity()); 
89          assertEquals(0, buffer.length());
90          assertTrue(buffer.isEmpty());
91          assertFalse(buffer.isFull());
92      }
