94      public void testBoundaryInEpilogue() throws Exception {
95          StringBuffer sb = new StringBuffer();
96          sb.append("From: foo@bar.com\r\n");
97          sb.append("To: someone@else.com\r\n");
98          sb.append("Content-type: multipart/something; boundary=myboundary\r\n");
99          sb.append("\r\n");
100         sb.append("This is the preamble.\r\n");
101         sb.append("--myboundary\r\n");
102         sb.append("Content-type: text/plain\r\n");
103         sb.append("\r\n");
104         sb.append("This is the first body.\r\n");
105         sb.append("It's completely meaningless.\r\n");
106         sb.append("After this line the body ends.\r\n");
107         sb.append("\r\n");
108         sb.append("--myboundary--\r\n");
109         
110         StringBuffer epilogue = new StringBuffer();
111         epilogue.append("Content-type: text/plain\r\n");
112         epilogue.append("\r\n");
113         epilogue.append("This is actually the epilogue but it looks like a second body.\r\n");
114         epilogue.append("Yada yada yada.\r\n");
115         epilogue.append("\r\n");
116         epilogue.append("--myboundary--\r\n");
117         epilogue.append("This is still the epilogue.\r\n");
118         
119         sb.append(epilogue.toString());
120         
121         ByteArrayInputStream bais = new ByteArrayInputStream(sb.toString().getBytes("US-ASCII"));
122         
123         final StringBuffer actual = new StringBuffer();
124         
125         MimeStreamParser parser = new MimeStreamParser();
126         parser.setContentHandler(new AbstractContentHandler() {
127             public void epilogue(InputStream is) throws IOException {
128                 int b;
129                 while ((b = is.read()) != -1) {
130                     actual.append((char) b);
131                 }
132             }
133         });
134         parser.parse(bais);
135         
136         assertEquals(epilogue.toString(), actual.toString());
137     }

418     public void testParse() throws Exception {
419         File dir = new File("src/test/resources/testmsgs");
420         File[] files = dir.listFiles();
421         
422         for (int i = 0; i < files.length; i++) {
423             File f = files[i];
424             
425             if (f.getName().toLowerCase().endsWith(".msg")) {
426                 MimeStreamParser parser = null;
427                 TestHandler handler = null;
428                 parser = new MimeStreamParser();
429                 handler = new TestHandler();
430                 
431                 System.out.println("Parsing " + f.getName());
432                 parser.setContentHandler(handler);
433                 parser.parse(new FileInputStream(f));
434                 
435                 String result = handler.sb.toString();
436                 String xmlFile = f.getAbsolutePath().substring(0, f.getAbsolutePath().lastIndexOf('.')) + ".xml";
437                 String xmlFileMime4j = f.getAbsolutePath().substring(0, f.getAbsolutePath().lastIndexOf('.')) + ".mime4j.xml";
438                 
439                 try {
440                     String expected = IOUtils.toString(new FileInputStream(xmlFile), "ISO8859-1");
441                     assertEquals("Error parsing " + f.getName(), expected, result);
442                 } catch (FileNotFoundException e) {
443                     FileOutputStream fos = new FileOutputStream(xmlFileMime4j);
444                     fos.write(result.getBytes());
445                     fos.flush();
446                     fos.close();
447                     fail("XML file not found: generated a file with the expected result!");
448                 }
449             }
450         }

453     private static class TestHandler implements ContentHandler {
454         private StringBuffer sb = new StringBuffer();
455 
456         private String escape(char c) {
457             if (c == '&') {
458                 return "&amp;";
459             }
460             if (c == '>') {
461                 return "&gt;";
462             }
463             if (c == '<') {
464                 return "&lt;";
465             }
466             return "" + c;
467         }
468         
469         private String escape(String s) {
470             s = s.replaceAll("&", "&amp;");
471             s = s.replaceAll(">", "&gt;");
472             s = s.replaceAll("<", "&lt;");
473             return s;
474         }
475         
476         public void epilogue(InputStream is) throws IOException {
477             sb.append("<epilogue>\r\n");
478             int b = 0;
479             while ((b = is.read()) != -1) {
480                 sb.append(escape((char) b));
481             }
482             sb.append("</epilogue>\r\n");
483         }
484         public void preamble(InputStream is) throws IOException {
485             sb.append("<preamble>\r\n");
486             int b = 0;
487             while ((b = is.read()) != -1) {
488                 sb.append(escape((char) b));
489             }
490             sb.append("</preamble>\r\n");
491         }
492         public void startMultipart(BodyDescriptor bd) {
493             sb.append("<multipart>\r\n");
494         }
495         public void body(BodyDescriptor bd, InputStream is) throws IOException {
496             sb.append("<body>\r\n");
497             int b = 0;
498             while ((b = is.read()) != -1) {
499                 sb.append(escape((char) b));
500             }
501             sb.append("</body>\r\n");
502         }
503         public void endMultipart() {
504             sb.append("</multipart>\r\n");
505         }
506         public void startBodyPart() {
507             sb.append("<body-part>\r\n");
508         }
509         public void endBodyPart() {
510             sb.append("</body-part>\r\n");
511         }
512         public void startHeader() {
513             sb.append("<header>\r\n");
514         }
515         public void field(String fieldData) {
516             sb.append("<field>\r\n" + escape(fieldData) + "</field>\r\n");
517         }
518         public void endHeader() {
519             sb.append("</header>\r\n");
520         }
521         public void startMessage() {
522             sb.append("<message>\r\n");
523         }
524         public void endMessage() {
525             sb.append("</message>\r\n");
526         }
527 
528         public void raw(InputStream is) throws IOException {
529             fail("raw should never be called");
530         }
