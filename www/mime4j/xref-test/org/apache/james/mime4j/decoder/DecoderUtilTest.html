53      public void testDecodeB() throws UnsupportedEncodingException {
54          String s = DecoderUtil.decodeB("VGhpcyBpcyB0aGUgcGxhaW4gd"
55                      + "GV4dCBtZXNzYWdlIQ==", "ISO8859-1");
56          assertEquals("This is the plain text message!", s);
57      }
58      
59  
60      public void testDecodeQ() throws UnsupportedEncodingException {
61          String s = DecoderUtil.decodeQ("=e1_=e2=09=E3_=E4_", 
62                                                           "ISO8859-1");
63          assertEquals("с т\tу ф ", s);
64      }
65      
66      public void testDecodeEncodedWords() {
67          assertEquals("", DecoderUtil.decodeEncodedWords(""));
68          assertEquals("Yada yada", DecoderUtil.decodeEncodedWords("Yada yada"));
69          assertEquals("  сту\tф", 
70                  DecoderUtil.decodeEncodedWords("=?iso-8859-1?Q?_=20=e1=e2=E3=09=E4?="));
71          assertEquals("Word 1 '  ту\tф'. Word 2 '  ту\tф'", 
72                  DecoderUtil.decodeEncodedWords("Word 1 '=?iso-8859-1?Q?_=20=e2=E3=09=E4?="
73                          + "'. Word 2 '=?iso-8859-1?q?_=20=e2=E3=09=E4?='"));
74          assertEquals("=?iso-8859-YADA?Q?_=20=t1=e2=E3=09=E4?=", 
75                  DecoderUtil.decodeEncodedWords("=?iso-8859-YADA?Q?_=20=t1=e2=E3=09=E4?="));
76          assertEquals("A short text", 
77                  DecoderUtil.decodeEncodedWords("=?US-ASCII?B?QSBzaG9ydCB0ZXh0?="));
78          assertEquals("A short text again!", 
79                  DecoderUtil.decodeEncodedWords("=?US-ASCII?b?QSBzaG9ydCB0ZXh0IGFnYWluIQ==?="));
80          assertEquals("", DecoderUtil.decodeEncodedWords("=?iso8859-1?Q?="));
81          assertEquals("", DecoderUtil.decodeEncodedWords("=?iso8859-1?b?="));
82          assertEquals("", DecoderUtil.decodeEncodedWords("=?ISO-8859-1?Q?"));
83          
84          /*

89      }
