181   /**
182    *
183    *   Writes each individual char from str to a Base64OutputStream and compares
184    *   the resulting output stream contents with the expected encoded string.  
185    *
186    *   @param  str         the decoded string
187    *   @param  str_encoded the encoded string
188    *
189    **/    
190   
191   private void _testWriteChar(String str, String str_encoded) throws IOException
192   {
193     // create a Base64OutputStream 
194     StringWriter strwriter = new StringWriter(); 
195     BufferedWriter buffwriter = new BufferedWriter(strwriter);
196     Base64OutputStream b64_out = new Base64OutputStream(buffwriter);
197     
198     // write out each char in str to the stream
199     for (int i = 0; i<str.length(); i++) 
200     {
201       b64_out.write(str.charAt(i));
202     }
203     // remember to add padding characters (if necessary)
204     b64_out.close();
205         
206     // compare the contents of the outputstream with the expected encoded string
207     assertEquals(strwriter.toString(), str_encoded);  
208   }
