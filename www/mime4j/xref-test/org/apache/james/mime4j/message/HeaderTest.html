20  package org.apache.james.mime4j.message;

22  import org.apache.commons.io.output.ByteArrayOutputStream;
23  import org.apache.james.mime4j.MimeException;
24  import org.apache.james.mime4j.field.Field;
25  import org.apache.james.mime4j.message.Header;
26  import org.apache.james.mime4j.util.MessageUtils;
27  
28  import junit.framework.TestCase;
29  
30  public class HeaderTest extends TestCase {
31  
32      public static final String SUBJECT = "Subject: test";
33  
34      public static final String TO = "To: anyuser <any@user>";
35  
36      public void testHeader() {
37          Header header = new Header();
38          header.addField(Field.parse(SUBJECT));
39          header.addField(Field.parse(TO));
40  
41          assertNotNull("Subject found", header.getField("Subject"));
42          assertNotNull("To found", header.getField("To"));
43  
44          assertEquals("Headers equals", SUBJECT + "\r\n" + TO + "\r\n", header
45                  .toString());
46      }
47      
48      static final int SWISS_GERMAN_HELLO [] = {
49          0x47, 0x72, 0xFC, 0x65, 0x7A, 0x69, 0x5F, 0x7A, 0xE4, 0x6D, 0xE4
50      };
51          
52      private static String constructString(int [] unicodeChars) {
53          StringBuffer buffer = new StringBuffer();
54          if (unicodeChars != null) {
55              for (int i = 0; i < unicodeChars.length; i++) {
56                  buffer.append((char)unicodeChars[i]); 
57              }
58          }
59          return buffer.toString();
60      }
61  
62      public void testWriteInStrictMode() throws Exception {
63          String hello = constructString(SWISS_GERMAN_HELLO);
64          Header header = new Header();
65          header.addField(Field.parse("Hello: " + hello));
66          
67          Field field = header.getField("Hello");
68          assertNotNull(field);
69          assertEquals(hello, field.getBody());
70          
71          ByteArrayOutputStream buffer = new ByteArrayOutputStream();
72          
73          header.writeTo(buffer, MessageUtils.STRICT_IGNORE);
74          String s = buffer.toString(MessageUtils.ASCII.name());
75          
76          assertEquals("Hello: Gr?ezi_z?m?\r\n\r\n", s);
77  
78          buffer.reset();
79          
80          try {
81              header.writeTo(buffer, MessageUtils.STRICT_ERROR);
82              fail("MimeException should have been thrown");
83          } catch (MimeException expected) {
84          }
85      }
86      
87      public void testWriteInLenientMode() throws Exception {
88          String hello = constructString(SWISS_GERMAN_HELLO);
89          Header header = new Header();
90          header.addField(Field.parse("Hello: " + hello));
91          header.addField(Field.parse("Content-type: text/plain; charset=" + 
92                  MessageUtils.ISO_8859_1.name()));
93          
94          Field field = header.getField("Hello");
95          assertNotNull(field);
96          assertEquals(hello, field.getBody());
97          
98          ByteArrayOutputStream buffer = new ByteArrayOutputStream();
99          
100         header.writeTo(buffer, MessageUtils.LENIENT);
101         String s = buffer.toString(MessageUtils.ISO_8859_1.name());
102         
103         assertEquals("Hello: " + hello + "\r\n" +
104               "Content-type: text/plain; charset=ISO-8859-1\r\n\r\n", s);
105     }
