129     private String escape(String s) {
130         s = s.replaceAll("&", "&amp;");
131         s = s.replaceAll("<", "&lt;");
132         return s.replaceAll(">", "&gt;");
133     }

135     private String getStructure(Entity e, String prefix, String id) 
136             throws IOException {
137         
138         StringBuffer sb = new StringBuffer();
139         
140         if (e instanceof Message) {
141             sb.append("<message>\r\n");
142         } else {
143             sb.append("<body-part>\r\n");
144         }            
145         
146         sb.append("<header>\r\n");
147         for (Iterator it = e.getHeader().getFields().iterator(); it.hasNext();) {
148             sb.append("<field>\r\n" + escape(((Field) it.next()).getRaw()) 
149                     + "</field>\r\n");
150         }
151         sb.append("</header>\r\n");
152         
153         if (e.getBody() instanceof Multipart) {
154             sb.append("<multipart>\r\n");
155             
156             Multipart multipart =(Multipart) e.getBody(); 
157             List parts =multipart.getBodyParts();
158 
159             sb.append("<preamble>\r\n");
160             sb.append(escape(multipart.getPreamble()));
161             sb.append("</preamble>\r\n");
162             
163             int i = 1;
164             for (Iterator it = parts.iterator(); it.hasNext();) {
165                 sb.append(getStructure((Entity) it.next(), prefix, id + "_" + (i++)));
166             }
167 
168             sb.append("<epilogue>\r\n");
169             sb.append(escape(multipart.getEpilogue()));
170             sb.append("</epilogue>\r\n");
171             
172             sb.append("</multipart>\r\n");
173             
174         } else if (e.getBody() instanceof Message) {
175             sb.append(getStructure((Message) e.getBody(), prefix, id + "_1"));
176         } else {
177             Body b = e.getBody();
178             String name = prefix + "_decoded_" + id 
179                             + (b instanceof TextBody ? ".txt" : ".bin");
180             String tag = b instanceof TextBody ? "text-body" : "binary-body";
181             sb.append("<" + tag + " name=\"" + name + "\"/>\r\n");
182                 
183             File expectedFile = new File(new File(fileName).getParent(), name);
184             File mime4jFile = new File(new File(fileName).getParent(), 
185                               name.substring(0, name.length() - 4) + ".mime4j"
186                                + (b instanceof TextBody ? ".txt" : ".bin"));
187                 
188             InputStream expected = null;
189             try {
190                 expected = new BufferedInputStream(new FileInputStream(expectedFile));
191             } catch (FileNotFoundException ex) {
192                 writeToFile(b, mime4jFile);
193                 fail("Test file not found. Generated the expected result with mime4j prefix: "+ex.getMessage());
194             }
195             
196             try {
197                 if (b instanceof TextBody) {
198                     String charset = CharsetUtil.toJavaCharset(e.getCharset());
199                     if (charset == null) {
200                         charset = "ISO8859-1";
201                     }
202 
203                     String s1 = IOUtils.toString(expected, charset).replaceAll(
204                             "\r", "");
205                     String s2 = IOUtils.toString(((TextBody) b).getReader())
206                             .replaceAll("\r", "");
207                     assertEquals(expectedFile.getName(), s1, s2);
208                 } else {
209                     assertEqualsBinary(expectedFile.getName(), expected,
210                             ((BinaryBody) b).getInputStream());
211                 }
212             } catch (AssertionError er) {
213                 writeToFile(b, mime4jFile);
214                 throw er;
215             }
216         }
217         
218         
219         if (e instanceof Message) {
220             sb.append("</message>\r\n");
221         } else {
222             sb.append("</body-part>\r\n");
223         }            
224         
225         return sb.toString();
226     }

228     private void writeToFile(Body b, File mime4jFile)
229             throws FileNotFoundException, IOException {
230         if (b instanceof TextBody) {
231             OutputStream out = new FileOutputStream(mime4jFile);
232             IOUtils.copy(((TextBody) b).getReader(), out);
233         } else {
234             OutputStream out = new FileOutputStream(mime4jFile);
235             IOUtils.copy(((BinaryBody) b).getInputStream(), out);
236         }
237     }

239     private void assertEqualsBinary(String msg, InputStream a, InputStream b) 
240             throws IOException {
241         
242         int pos = 0;
243         while (true) {
244             int b1 = a.read();
245             int b2 = b.read();
246             assertEquals(msg + " (Position " + (++pos) + ")", b1, b2);
247             
248             if (b1 == -1 || b2 == -1) {
249                 break;
250             }
251         }
252     }
