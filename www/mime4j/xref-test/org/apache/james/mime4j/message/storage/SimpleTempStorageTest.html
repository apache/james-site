50          TempPath path = man.getRootTempPath().createTempPath();
51          File tmpdir = new File(System.getProperty("java.io.tmpdir"));
52          assertTrue(path.getAbsolutePath().startsWith(tmpdir.getAbsolutePath()));

54          String fileName = path.getAbsolutePath().substring(
55                  path.getAbsolutePath().lastIndexOf(File.separatorChar) + 1);

77      public void testCreateTempFile() throws IOException {
78          SimpleTempStorage man = new SimpleTempStorage();
79          TempPath path = man.getRootTempPath().createTempPath();
80          TempFile file = path.createTempFile();
81          assertTrue(file.getAbsolutePath().startsWith(path.getAbsolutePath()));
82          
83          String fileName = file.getAbsolutePath().substring(
84                  file.getAbsolutePath().lastIndexOf(File.separatorChar) + 1);
85          assertTrue("Unexpected chars in file name " + fileName, 
86                      fileName.matches("^[0-9]+\\.tmp$"));
87          assertTrue("Temp file doesn't exist", 
88                     new File(file.getAbsolutePath()).exists());
89          
90          String s = "A short string";
91          OutputStream out = file.getOutputStream();
92          out.write(s.getBytes());
93          out.close();
94  
95          byte[] buffer = new byte[s.length() * 2];
96          InputStream in = file.getInputStream();
97          int i = 0;
98          for (int data; (data = in.read()) != -1; i++) {
99              buffer[i] = (byte) data;
100         }
101         in.close();
102         assertTrue(s.length() == i);
103         assertEquals(s, new String(buffer, 0, i));
104     }
