30  public class MessageTest extends TestCase {
31      private Header headerTextPlain = null;
32      private Header headerMessageRFC822 = null;
33      private Header headerEmpty = null;
34      private Header headerMultipartMixed = null;
35      private Header headerMultipartDigest = null;

37      public void setUp() {
38          headerTextPlain = new Header();
39          headerMessageRFC822 = new Header();
40          headerEmpty = new Header();
41          headerMultipartMixed = new Header();
42          headerMultipartDigest = new Header();
43          
44          headerTextPlain.addField(
45                  Field.parse("Content-Type: text/plain"));
46          headerMessageRFC822.addField(
47                  Field.parse("Content-Type: message/RFC822"));
48          headerMultipartMixed.addField(
49                  Field.parse("Content-Type: multipart/mixed; boundary=foo"));
50          headerMultipartDigest.addField(
51                  Field.parse("Content-Type: multipart/digest; boundary=foo"));
52      }

55      }

57      public void testGetMimeType() {
58          Message parent = null;
59          Message child = null;
60          
61          parent = new Message();
62          child = new Message();
63          child.setParent(parent);
64          parent.setHeader(headerMultipartDigest);
65          child.setHeader(headerEmpty);
66          assertEquals("multipart/digest, empty", "message/rfc822", 
67                  child.getMimeType());
68          child.setHeader(headerTextPlain);
69          assertEquals("multipart/digest, text/plain", "text/plain", 
70                  child.getMimeType());
71          child.setHeader(headerMessageRFC822);
72          assertEquals("multipart/digest, message/rfc822", "message/rfc822", 
73                  child.getMimeType());
74          
75          parent = new Message();
76          child = new Message();
77          child.setParent(parent);
78          parent.setHeader(headerMultipartMixed);
79          child.setHeader(headerEmpty);
80          assertEquals("multipart/mixed, empty", "text/plain", 
81                  child.getMimeType());
82          child.setHeader(headerTextPlain);
83          assertEquals("multipart/mixed, text/plain", "text/plain", 
84                  child.getMimeType());
85          child.setHeader(headerMessageRFC822);
86          assertEquals("multipart/mixed, message/rfc822", "message/rfc822", 
87                  child.getMimeType());
88          
89          child = new Message();
90          child.setHeader(headerEmpty);
91          assertEquals("null, empty", "text/plain", child.getMimeType());
92          child.setHeader(headerTextPlain);
93          assertEquals("null, text/plain", "text/plain", child.getMimeType());
94          child.setHeader(headerMessageRFC822);
95          assertEquals("null, message/rfc822", "message/rfc822", 
96                  child.getMimeType());
97      }
98  
99      public void testIsMultipart() {
100         Message m = new Message();
101         
102         m.setHeader(headerEmpty);
103         assertTrue("empty", !m.isMultipart());
104         
105         m.setHeader(headerTextPlain);
106         assertTrue("text/plain", !m.isMultipart());
107         
108         m.setHeader(headerMultipartDigest);
109         assertTrue("multipart/digest", m.isMultipart());
110         
111         m.setHeader(headerMultipartMixed);
112         assertTrue("multipart/mixed", m.isMultipart());
113     }
