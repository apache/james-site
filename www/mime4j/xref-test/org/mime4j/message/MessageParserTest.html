103     private String escape(String s) {
104         s = s.replaceAll("&", "&amp;");
105         s = s.replaceAll("<", "&lt;");
106         return s.replaceAll(">", "&gt;");
107     }

109     private String getStructure(Entity e, String prefix, String id) 
110             throws IOException {
111         
112         StringBuffer sb = new StringBuffer();
113         
114         if (e instanceof Message) {
115             sb.append("<message>\r\n");
116         } else {
117             sb.append("<body-part>\r\n");
118         }            
119         
120         sb.append("<header>\r\n");
121         for (Iterator it = e.getHeader().getFields().iterator(); it.hasNext();) {
122             sb.append("<field>\r\n" + escape(((Field) it.next()).getRaw()) 
123                     + "</field>\r\n");
124         }
125         sb.append("</header>\r\n");
126         
127         if (e.getBody() instanceof Multipart) {
128             sb.append("<multipart>\r\n");
129             
130             Multipart multipart =(Multipart) e.getBody(); 
131             List parts =multipart.getBodyParts();
132 
133             sb.append("<preamble>\r\n");
134             sb.append(escape(multipart.getPreamble()));
135             sb.append("</preamble>\r\n");
136             
137             int i = 1;
138             for (Iterator it = parts.iterator(); it.hasNext();) {
139                 sb.append(getStructure((Entity) it.next(), prefix, id + "_" + (i++)));
140             }
141 
142             sb.append("<epilogue>\r\n");
143             sb.append(escape(multipart.getEpilogue()));
144             sb.append("</epilogue>\r\n");
145             
146             sb.append("</multipart>\r\n");
147             
148         } else if (e.getBody() instanceof Message) {
149             sb.append(getStructure((Message) e.getBody(), prefix, id + "_1"));
150         } else {
151             Body b = e.getBody();
152             String name = prefix + "_decoded_" + id 
153                             + (b instanceof TextBody ? ".txt" : ".bin");
154             String tag = b instanceof TextBody ? "text-body" : "binary-body";
155             sb.append("<" + tag + " name=\"" + name + "\"/>\r\n");
156                 
157             File perlFile = new File(new File(fileName).getParent(), name);
158             File mime4jFile = new File(new File(fileName).getParent(), 
159                               name.substring(0, name.length() - 4) + ".mime4j"
160                                + (b instanceof TextBody ? ".txt" : ".bin"));
161                 
162             InputStream expected = 
163                 new BufferedInputStream(new FileInputStream(perlFile));
164             
165             if (b instanceof TextBody) {
166                 String charset = CharsetUtil.toJavaCharset(e.getCharset());
167                 if (charset == null) {
168                     charset = "ISO8859-1";
169                 }
170                 
171                 String s1 = IOUtils.toString(expected, charset).replaceAll("\r", "");
172                 String s2 = IOUtils.toString(((TextBody) b).getReader()).replaceAll("\r", "");
173                 assertEquals(perlFile.getName(), s1, s2);
174             } else {
175                 OutputStream out = new FileOutputStream(mime4jFile);
176                 CopyUtils.copy(((BinaryBody) b).getInputStream(), out);
177             
178                 assertEqualsBinary(perlFile.getName(), expected, ((BinaryBody) b).getInputStream());
179             }
180         }
181         
182         
183         if (e instanceof Message) {
184             sb.append("</message>\r\n");
185         } else {
186             sb.append("</body-part>\r\n");
187         }            
188         
189         return sb.toString();
190     }

192     private void assertEqualsBinary(String msg, InputStream a, InputStream b) 
193             throws IOException {
194         
195         int pos = 0;
196         while (true) {
197             int b1 = a.read();
198             int b2 = b.read();
199             assertEquals(msg + " (Position " + (++pos) + ")", b1, b2);
200             
201             if (b1 == -1 || b2 == -1) {
202                 break;
203             }
204         }
205     }
