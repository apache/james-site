48      public void testDecodeB() throws UnsupportedEncodingException {
49          String s = DecoderUtil.decodeB("VGhpcyBpcyB0aGUgcGxhaW4gd"
50                      + "GV4dCBtZXNzYWdlIQ==", "ISO8859-1");
51          assertEquals("This is the plain text message!", s);
52      }
53      
54  
55      public void testDecodeQ() throws UnsupportedEncodingException {
56          String s = DecoderUtil.decodeQ("=e1_=e2=09=E3_=E4_", 
57                                                           "ISO8859-1");
58          assertEquals("с т\tу ф ", s);
59      }
60      
61      public void testDecodeEncodedWords() {
62          assertEquals("", DecoderUtil.decodeEncodedWords(""));
63          assertEquals("Yada yada", DecoderUtil.decodeEncodedWords("Yada yada"));
64          assertEquals("  сту\tф", 
65                  DecoderUtil.decodeEncodedWords("=?iso-8859-1?Q?_=20=e1=e2=E3=09=E4?="));
66          assertEquals("Word 1 '  ту\tф'. Word 2 '  ту\tф'", 
67                  DecoderUtil.decodeEncodedWords("Word 1 '=?iso-8859-1?Q?_=20=e2=E3=09=E4?="
68                          + "'. Word 2 '=?iso-8859-1?q?_=20=e2=E3=09=E4?='"));
69          assertEquals("=?iso-8859-YADA?Q?_=20=t1=e2=E3=09=E4?=", 
70                  DecoderUtil.decodeEncodedWords("=?iso-8859-YADA?Q?_=20=t1=e2=E3=09=E4?="));
71          assertEquals("A short text", 
72                  DecoderUtil.decodeEncodedWords("=?US-ASCII?B?QSBzaG9ydCB0ZXh0?="));
73          assertEquals("A short text again!", 
74                  DecoderUtil.decodeEncodedWords("=?US-ASCII?b?QSBzaG9ydCB0ZXh0IGFnYWluIQ==?="));
75          assertEquals("", DecoderUtil.decodeEncodedWords("=?iso8859-1?Q?="));
76          assertEquals("", DecoderUtil.decodeEncodedWords("=?iso8859-1?b?="));
77          
78          /*

83      }
