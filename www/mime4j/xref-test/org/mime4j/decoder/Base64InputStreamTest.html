42          ByteArrayInputStream bis = null;

49          bis = new ByteArrayInputStream(

51          decoder = new Base64InputStream(bis);
52          assertEquals("This is the plain text message!", toString(read(decoder)));
53          
54          /*
55           * Test encoded text padded once, twice and not at all.
56           */
57          bis = new ByteArrayInputStream(
58                  fromString("VGhpcyBpcyBhIHRleHQgd2hpY2ggaGFzIHRvIGJl"
59                          + "IHBhZGRlZCBvbmNlLi4="));
60          decoder = new Base64InputStream(bis);
61          assertEquals("This is a text which has to be padded once..", toString(read(decoder)));
62          bis = new ByteArrayInputStream(
63                  fromString("VGhpcyBpcyBhIHRleHQgd2hpY2ggaGFzIHRvIGJl"
64                          + "IHBhZGRlZCB0d2ljZQ=="));
65          decoder = new Base64InputStream(bis);
66          assertEquals("This is a text which has to be padded twice", toString(read(decoder)));
67          bis = new ByteArrayInputStream(
68                  fromString("VGhpcyBpcyBhIHRleHQgd2hpY2ggd2lsbCBub3Qg"
69                          + "YmUgcGFkZGVk"));
70          decoder = new Base64InputStream(bis);
71          assertEquals("This is a text which will not be padded", toString(read(decoder)));

73          /*
74           * Test that non base64 characters are ignored.
75           */
76          bis = new ByteArrayInputStream(

135     private byte[] read(InputStream is) throws IOException {
136         ByteArrayOutputStream bos = new ByteArrayOutputStream();
137         int b;
138         while ((b = is.read()) != -1) {
139             bos.write(b);
140         }
141         return bos.toByteArray();
142     }
143     
144     private byte[] fromString(String s) {
145         try {
146             return s.getBytes("US-ASCII");
147         } catch (UnsupportedEncodingException e) {
148             fail(e.getMessage());
149             return null;
150         }
151     }
152     
153     private String toString(byte[] b) {
154         try {
155             return new String(b, "US-ASCII");
156         } catch (UnsupportedEncodingException e) {
157             fail(e.getMessage());
158             return null;
159         }
160     }
