37      public void testGetRootTempPath() {
38          SimpleTempStorage man = new SimpleTempStorage();
39          TempPath path = man.getRootTempPath();
40          File tmpdir = new File(System.getProperty("java.io.tmpdir"));
41          assertEquals(tmpdir.getAbsolutePath(), path.getAbsolutePath());
42      }

44      public void testCreateTempPath() throws IOException {
45          SimpleTempStorage man = new SimpleTempStorage();
46          TempPath path = man.getRootTempPath().createTempPath();
47          File tmpdir = new File(System.getProperty("java.io.tmpdir"));
48          assertTrue(path.getAbsolutePath().startsWith(tmpdir.getAbsolutePath()));
49          
50          String fileName = path.getAbsolutePath().substring(
51                  path.getAbsolutePath().lastIndexOf(File.separatorChar) + 1);
52          assertTrue("Unexpected chars in file name " + fileName, 
53                      fileName.matches("^[0-9]+$"));
54          assertTrue("Temp dir doesn't exist", 
55                     new File(path.getAbsolutePath()).exists());
56      }

73      public void testCreateTempFile() throws IOException {
74          SimpleTempStorage man = new SimpleTempStorage();
75          TempPath path = man.getRootTempPath().createTempPath();
76          TempFile file = path.createTempFile();
77          assertTrue(file.getAbsolutePath().startsWith(path.getAbsolutePath()));
78          
79          String fileName = file.getAbsolutePath().substring(
80                  file.getAbsolutePath().lastIndexOf(File.separatorChar) + 1);
81          assertTrue("Unexpected chars in file name " + fileName, 
82                      fileName.matches("^[0-9]+//.tmp$"));
83          assertTrue("Temp file doesn't exist", 
84                     new File(file.getAbsolutePath()).exists());
85          
86          String s = "A short string";
87          OutputStream out = file.getOutputStream();
88          out.write(s.getBytes());
89          out.close();
90  
91          byte[] buffer = new byte[s.length() * 2];
92          InputStream in = file.getInputStream();
93          int i = 0;
94          for (int data; (data = in.read()) != -1; i++) {
95              buffer[i] = (byte) data;
96          }
97          in.close();
98          assertTrue(s.length() == i);
99          assertEquals(s, new String(buffer, 0, i));
100     }
