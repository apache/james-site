88      public void testBoundaryInEpilogue() throws IOException {
89          StringBuffer sb = new StringBuffer();
90          sb.append("From: foo@bar.com\r\n");
91          sb.append("To: someone@else.com\r\n");
92          sb.append("Content-type: multipart/something; boundary=myboundary\r\n");
93          sb.append("\r\n");
94          sb.append("This is the preamble.\r\n");
95          sb.append("--myboundary\r\n");
96          sb.append("Content-type: text/plain\r\n");
97          sb.append("\r\n");
98          sb.append("This is the first body.\r\n");
99          sb.append("It's completely meaningless.\r\n");
100         sb.append("After this line the body ends.\r\n");
101         sb.append("\r\n");
102         sb.append("--myboundary--\r\n");
103         
104         StringBuffer epilogue = new StringBuffer();
105         epilogue.append("Content-type: text/plain\r\n");
106         epilogue.append("\r\n");
107         epilogue.append("This is actually the epilogue but it looks like a second body.\r\n");
108         epilogue.append("Yada yada yada.\r\n");
109         epilogue.append("\r\n");
110         epilogue.append("--myboundary--\r\n");
111         epilogue.append("This is still the epilogue.\r\n");
112         
113         sb.append(epilogue.toString());
114         
115         ByteArrayInputStream bais = new ByteArrayInputStream(sb.toString().getBytes("US-ASCII"));
116         
117         final StringBuffer actual = new StringBuffer();
118         
119         MimeStreamParser parser = new MimeStreamParser();
120         parser.setContentHandler(new AbstractContentHandler() {
121             public void epilogue(InputStream is) throws IOException {
122                 int b;
123                 while ((b = is.read()) != -1) {
124                     actual.append((char) b);
125                 }
126             }
127         });
128         parser.parse(bais);
129         
130         assertEquals(epilogue.toString(), actual.toString());
131     }

438     private static class TestHandler implements ContentHandler {
439         private StringBuffer sb = new StringBuffer();
440 
441         private String escape(char c) {
442             if (c == '&') {
443                 return "&amp;";
444             }
445             if (c == '>') {
446                 return "&gt;";
447             }
448             if (c == '<') {
449                 return "&lt;";
450             }
451             return "" + c;
452         }
453         
454         private String escape(String s) {
455             s = s.replaceAll("&", "&amp;");
456             s = s.replaceAll(">", "&gt;");
457             s = s.replaceAll("<", "&lt;");
458             return s;
459         }
460         
461         public void epilogue(InputStream is) throws IOException {
462             sb.append("<epilogue>\r\n");
463             int b = 0;
464             while ((b = is.read()) != -1) {
465                 sb.append(escape((char) b));
466             }
467             sb.append("</epilogue>\r\n");
468         }
469         public void preamble(InputStream is) throws IOException {
470             sb.append("<preamble>\r\n");
471             int b = 0;
472             while ((b = is.read()) != -1) {
473                 sb.append(escape((char) b));
474             }
475             sb.append("</preamble>\r\n");
476         }
477         public void startMultipart(BodyDescriptor bd) {
478             sb.append("<multipart>\r\n");
479         }
480         public void body(BodyDescriptor bd, InputStream is) throws IOException {
481             sb.append("<body>\r\n");
482             int b = 0;
483             while ((b = is.read()) != -1) {
484                 sb.append(escape((char) b));
485             }
486             sb.append("</body>\r\n");
487         }
488         public void endMultipart() {
489             sb.append("</multipart>\r\n");
490         }
491         public void startBodyPart() {
492             sb.append("<body-part>\r\n");
493         }
494         public void endBodyPart() {
495             sb.append("</body-part>\r\n");
496         }
497         public void startHeader() {
498             sb.append("<header>\r\n");
499         }
500         public void field(String fieldData) {
501             sb.append("<field>\r\n" + escape(fieldData) + "</field>\r\n");
502         }
503         public void endHeader() {
504             sb.append("</header>\r\n");
505         }
506         public void startMessage() {
507             sb.append("<message>\r\n");
508         }
509         public void endMessage() {
510             sb.append("</message>\r\n");
511         }
512 
513         public void raw(InputStream is) throws IOException {
514             fail("raw should never be called");
515         }
516     }
