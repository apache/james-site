27      public void testParse1() throws ParseException {
28          AddressList addrList = AddressList.parse("John Doe <jdoe@machine(comment).  example>");

53      public void testMessyGroupAndMailbox() throws ParseException {
54          AddressList addrList = AddressList.parse("Marketing  folks :  Jane Smith < jane @ example . net >, \" Jack //\"Jackie//\" Jones \" < jjones@example.com > (comment(comment)); ,, (comment)  , <@example . net,@example(ignore//)).com:(ignore)john@(ignore)example.net>");
55          assertEquals(2, addrList.size());
56  
57          Group group = (Group)addrList.get(0);
58          assertEquals("Marketing  folks", group.getName());
59          assertEquals(2, group.getMailboxes().size());
60  
61          NamedMailbox namedMailbox1 = (NamedMailbox)group.getMailboxes().get(0);
62          NamedMailbox namedMailbox2 = (NamedMailbox)group.getMailboxes().get(1);
63  
64          assertEquals("Jane Smith", namedMailbox1.getName());
65          assertEquals("jane", namedMailbox1.getLocalPart());
66          assertEquals("example.net", namedMailbox1.getDomain());
67  
68          assertEquals(" Jack \"Jackie\" Jones ", namedMailbox2.getName());
69          assertEquals("jjones", namedMailbox2.getLocalPart());
70          assertEquals("example.com", namedMailbox2.getDomain());
71  
72          Mailbox mailbox = (Mailbox)addrList.get(1);
73          assertFalse(mailbox instanceof NamedMailbox);
74          assertEquals("john", mailbox.getLocalPart());
75          assertEquals("example.net", mailbox.getDomain());
76          assertEquals(2, mailbox.getRoute().size());
77          assertEquals("example.net", mailbox.getRoute().get(0));
78          assertEquals("example.com", mailbox.getRoute().get(1));
79      }

100     public void testTortureTest() throws ParseException {
101 
102         // Source: http://mailformat.dan.info/headers/from.html
103         // (Commented out pending confirmation of legality--I think the local-part is illegal.) 
104         // AddressList.parse("\"Guy Macon\" <guymacon+\" http://www.guymacon.com/ \"00@spamcop.net>");
105 
106         // Taken mostly from RFC822.
107 
108         // Just make sure these are recognized as legal address lists;
109         // there shouldn't be any aspect of the RFC that is tested here
110         // but not in the other unit tests.
111 
112         AddressList.parse("Alfred Neuman <Neuman@BBN-TENEXA>");
113         AddressList.parse("Neuman@BBN-TENEXA");
114         AddressList.parse("\"George, Ted\" <Shared@Group.Arpanet>");
115         AddressList.parse("Wilt . (the Stilt) Chamberlain@NBA.US");
116 
117         // NOTE: In RFC822 8.1.5, the following example did not have "Galloping Gourmet"
118         // in double-quotes.  I can only assume this was a typo, since 6.2.4 specifically
119         // disallows spaces in unquoted local-part.
120         AddressList.parse("     Gourmets:  Pompous Person <WhoZiWhatZit@Cordon-Bleu>," +
121                 "                Childs@WGBH.Boston, \"Galloping Gourmet\"@" +
122                 "                ANT.Down-Under (Australian National Television)," +
123                 "                Cheapie@Discount-Liquors;," +
124                 "       Cruisers:  Port@Portugal, Jones@SEA;," +
125                 "         Another@Somewhere.SomeOrg");
126 
127         // NOTE: In RFC822 8.3.3, the following example ended with a lone ">" after
128         // Tops-20-Host.  I can only assume this was a typo, since 6.1 clearly shows
129         // ">" requires a matching "<".
130         AddressList.parse("Important folk:" +
131                 "                   Tom Softwood <Balsa@Tree.Root>," +
132                 "                   \"Sam Irving\"@Other-Host;," +
133                 "                 Standard Distribution:" +
134                 "                   /main/davis/people/standard@Other-Host," +
135                 "                   \"<Jones>standard.dist.3\"@Tops-20-Host;");
136 
137         // The following are from a Usenet post by Dan J. Bernstein:
138         // http://groups.google.com/groups?selm=1996Aug1418.21.01.28081%40koobera.math.uic.edu
139         AddressList.parse("\":sysmail\"@  Some-Group.\t         Some-Org, Muhammed.(I am  the greatest) Ali @(the)Vegas.WBA");
140         AddressList.parse("me@home.com (comment (nested (deeply//))))");
141         AddressList.parse("mailing list: me@home.com, route two <you@work.com>, them@play.com ;");
