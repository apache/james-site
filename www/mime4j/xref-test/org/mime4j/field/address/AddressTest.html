49      public void testMessyGroupAndMailbox() throws ParseException {
50          AddressList addrList = AddressList.parse("Marketing  folks :  Jane Smith < jane @ example . net >, \" Jack //\"Jackie//\" Jones \" < jjones@example.com > (comment(comment)); ,, (comment)  , <@example . net,@example(ignore//)).com:(ignore)john@(ignore)example.net>");
51          assertEquals(2, addrList.size());
52  
53          Group group = (Group)addrList.get(0);
54          assertEquals("Marketing  folks", group.getName());
55          assertEquals(2, group.getMailboxes().size());
56  
57          NamedMailbox namedMailbox1 = (NamedMailbox)group.getMailboxes().get(0);
58          NamedMailbox namedMailbox2 = (NamedMailbox)group.getMailboxes().get(1);
59  
60          assertEquals("Jane Smith", namedMailbox1.getName());
61          assertEquals("jane", namedMailbox1.getLocalPart());
62          assertEquals("example.net", namedMailbox1.getDomain());
63  
64          assertEquals(" Jack \"Jackie\" Jones ", namedMailbox2.getName());
65          assertEquals("jjones", namedMailbox2.getLocalPart());
66          assertEquals("example.com", namedMailbox2.getDomain());
67  
68          Mailbox mailbox = (Mailbox)addrList.get(1);
69          assertFalse(mailbox instanceof NamedMailbox);
70          assertEquals("john", mailbox.getLocalPart());
71          assertEquals("example.net", mailbox.getDomain());
72          assertEquals(2, mailbox.getRoute().size());
73          assertEquals("example.net", mailbox.getRoute().get(0));
74          assertEquals("example.com", mailbox.getRoute().get(1));
75      }

96      public void testTortureTest() throws ParseException {
97  
98          // Source: http://mailformat.dan.info/headers/from.html
99          // (Commented out pending confirmation of legality--I think the local-part is illegal.) 
100         // AddressList.parse("\"Guy Macon\" <guymacon+\" http://www.guymacon.com/ \"00@spamcop.net>");
101 
102         // Taken mostly from RFC822.
103 
104         // Just make sure these are recognized as legal address lists;
105         // there shouldn't be any aspect of the RFC that is tested here
106         // but not in the other unit tests.
107 
108         AddressList.parse("Alfred Neuman <Neuman@BBN-TENEXA>");
109         AddressList.parse("Neuman@BBN-TENEXA");
110         AddressList.parse("\"George, Ted\" <Shared@Group.Arpanet>");
111         AddressList.parse("Wilt . (the Stilt) Chamberlain@NBA.US");
112 
113         // NOTE: In RFC822 8.1.5, the following example did not have "Galloping Gourmet"
114         // in double-quotes.  I can only assume this was a typo, since 6.2.4 specifically
115         // disallows spaces in unquoted local-part.
116         AddressList.parse("     Gourmets:  Pompous Person <WhoZiWhatZit@Cordon-Bleu>," +
117                 "                Childs@WGBH.Boston, \"Galloping Gourmet\"@" +
118                 "                ANT.Down-Under (Australian National Television)," +
119                 "                Cheapie@Discount-Liquors;," +
120                 "       Cruisers:  Port@Portugal, Jones@SEA;," +
121                 "         Another@Somewhere.SomeOrg");
122 
123         // NOTE: In RFC822 8.3.3, the following example ended with a lone ">" after
124         // Tops-20-Host.  I can only assume this was a typo, since 6.1 clearly shows
125         // ">" requires a matching "<".
126         AddressList.parse("Important folk:" +
127                 "                   Tom Softwood <Balsa@Tree.Root>," +
128                 "                   \"Sam Irving\"@Other-Host;," +
129                 "                 Standard Distribution:" +
130                 "                   /main/davis/people/standard@Other-Host," +
131                 "                   \"<Jones>standard.dist.3\"@Tops-20-Host;");
132 
133         // The following are from a Usenet post by Dan J. Bernstein:
134         // http://groups.google.com/groups?selm=1996Aug1418.21.01.28081%40koobera.math.uic.edu
135         AddressList.parse("\":sysmail\"@  Some-Group.\t         Some-Org, Muhammed.(I am  the greatest) Ali @(the)Vegas.WBA");
136         AddressList.parse("me@home.com (comment (nested (deeply//))))");
137         AddressList.parse("mailing list: me@home.com, route two <you@work.com>, them@play.com ;");
