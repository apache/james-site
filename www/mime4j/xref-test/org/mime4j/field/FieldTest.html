31  public class FieldTest extends TestCase {
32  
33      public void testGetName() {
34          Field f = null;
35          
36          f = Field.parse("Subject: Yada yada yada");
37          assertEquals("Testing simple field", "Subject", f.getName());
38          
39          f = Field.parse("X-yada-yada: Yada yada yada");
40          assertEquals("Testing an X- field", "X-yada-yada", f.getName());
41          
42          try {
43              f = Field.parse("Yada yada yada");
44              fail("IllegalArgumentException not thrown when using an invalid "
45                      + "field");
46          } catch (IllegalArgumentException e) {
47          }
48      }
49  
50      public void testParse() {
51          Field f = null;
52          
53          f = Field.parse("Subject: Yada yada yada");
54          assertTrue("Field should be UnstructuredField", 
55                          f instanceof UnstructuredField);
56          f = Field.parse("Content-Type: text/plain");
57          assertTrue("Field should be ContentTypeField", 
58                          f instanceof ContentTypeField);
59          f = Field.parse("Content-Transfer-Encoding: 7bit");
60          assertTrue("Field should be ContentTransferEncodingField", 
61                          f instanceof ContentTransferEncodingField);
62      }
