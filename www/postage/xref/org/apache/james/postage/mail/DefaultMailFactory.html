42      private final static char[] CHARSET = new char[]
43                                  {'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j',
44                                   'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't',

51      public Message createMail(Session mailSession, MailSender mailSender, MailProcessingRecord mailProcessingRecord) {
52          MimeMessage message = new MimeMessage(mailSession);
53  
54          try {
55              message.addHeader("Mime-Version", "1.0");
56              message.addHeader(HeaderConstants.JAMES_POSTAGE_HEADER, "This is a test mail sent by James Postage");
57              message.addHeader(HeaderConstants.JAMES_POSTAGE_VALIDATORCLASSNAME_HEADER, "org.apache.james.postage.mail.DefaultMailValidator");
58              message.setSubject(mailSender.getSubject());
59              message.addHeader("Message-ID", "Postage-" + System.currentTimeMillis());
60              mailProcessingRecord.setSubject(mailSender.getSubject());
61              message.addHeader("Content-Type", "multipart/mixed");
62  
63              Multipart multipart = new MimeMultipart("mixed");
64  
65              if (mailProcessingRecord.getMailId() != null) {
66                  message.addHeader(HeaderConstants.MAIL_ID_HEADER, mailProcessingRecord.getMailId());
67              } else {
68                  // TODO resolve situations when ID is null
69                  log.warn("ID header is NULL!");
70                  throw new RuntimeException("could not create mail with ID = NULL");
71              }
72  
73              if (mailSender.sendTextPart()) {
74                  int sizeMinText = mailSender.getSizeMinText();
75                  int sizeMaxText = mailSender.getSizeMaxText();
76                  MimeBodyPart part = new MimeBodyPart();
77  
78                  int mailSize = generateRandomPartSize(sizeMinText, sizeMaxText);
79                  mailProcessingRecord.setByteSendText(mailSize);
80  
81                  StringBuffer textBody = new StringBuffer(mailSize);
82                  for (int i = 0; i < mailSize; i++) textBody.append(getRandomChar());
83  
84                  part.setText(textBody.toString());
85  
86  //                part.setDataHandler(new DataHandler(textBody.toString(), "text/plain"));
87                  
88                  multipart.addBodyPart(part);
89              }
90  
91              if (mailSender.sendBinaryPart()) {
92                  int sizeMinBinary = mailSender.getSizeMinBinary();
93                  int sizeMaxBinary = mailSender.getSizeMaxBinary();
94                  MimeBodyPart part = new MimeBodyPart();
95  
96                  int mailSize = generateRandomPartSize(sizeMinBinary, sizeMaxBinary);
97                  mailProcessingRecord.setByteSendBinary(mailSize);
98  
99                  byte[] bytes = new byte[mailSize];
100                 for (int i = 0; i < mailSize; i++) bytes[i] = getRandomByte();
101 
102                 part.setDataHandler(new DataHandler(new ByteArrayDataSource(bytes, "application/octet-stream")));
103                 multipart.addBodyPart(part);
104             }
105             message.setContent(multipart);
106         } catch (MessagingException e) {
107             mailProcessingRecord.setErrorTextSending(e.toString());
108             log.error("mail could not be created", e);
109             return null;
110         }
111         return message;
112     }
113 
114     private int generateRandomPartSize(int sizeMin, int sizeMax) {
115         return (int)(Math.random() * (double)(sizeMax - sizeMin)) + sizeMin;
116     }
117 
118     public static char getRandomChar() {
119         return CHARSET[getRandomInt()];
120     }
121 
122     private static int getRandomInt() {
123         return (int)(Math.random() * (double)(CHARSET.length - 1));
124     }
125 
126     public static byte getRandomByte() {
127         return (byte)CHARSET[getRandomInt()];
128     }
