17  package org.apache.james.postage;
18  
19  import org.apache.commons.configuration.ConfigurationException;
20  import org.apache.commons.configuration.XMLConfiguration;
21  import org.apache.james.postage.configuration.ConfigurationLoader;
22  import org.apache.james.postage.configuration.PostageConfiguration;
23  
24  import java.util.Iterator;
25  import java.util.Map;
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  /***
30   * bootstrapping the application
31   */
32  public class Main {
33      private static PostageRunner m_currentPostageRunner = null;
34  
35      public static void main(String[] args) {
36          if (args == null || args.length == 0) {
37              System.out.println("Please provide the configuration file");
38              return;
39          }
40  
41          String filename = args[0];
42  
43          List scenariosToRun = new ArrayList();
44          for (int i = 1; i < args.length; i++)
45          {
46              scenariosToRun.add(args[i]);
47          }
48  
49          // load all scenarios from configuration file
50          ConfigurationLoader configurationLoader = new ConfigurationLoader();
51          Map configurations;
52          try {
53              // TODO allow different (non-xml) configs - as Common-Configuration supports it
54              XMLConfiguration xmlConfiguration = new XMLConfiguration(filename);
55              //xmlConfiguration.setThrowExceptionOnMissing(false);
56              configurations = configurationLoader.create(xmlConfiguration);
57          } catch (ConfigurationException e) {
58              e.printStackTrace();
59              return;
60          }
61  
62          // register shutdown hook if this app is terminated from outside
63          Runtime.getRuntime().addShutdownHook(new Thread(){public void run() {shutdown();}});
64  
65          // run all scenarios
66          runScenarios(configurations, scenariosToRun);
67      }
68  
69      private static void runScenarios(Map configurations, List scenariosToRun) {
70          // run all scenarios which are requested to be run.
71          int playedScenarioCounter = 0;
72          Iterator iterator = configurations.keySet().iterator();
73          while (iterator.hasNext()) {
74              String id = (String)iterator.next();
75              // if no scenario is given on the command line, all get executed
76              // if one or more is given, all others are skipped
77              if (!scenariosToRun.isEmpty() && !scenariosToRun.contains(id)) continue;
78  
79              playedScenarioCounter++;
80              PostageConfiguration postageConfiguration = (PostageConfiguration) configurations.get(id);
81              m_currentPostageRunner = new PostageRunner(postageConfiguration);
82              m_currentPostageRunner.run(); // TODO maybe we want to run this Runnable in its own thread, but maybe not
83          }
84          m_currentPostageRunner = null;
85          if (playedScenarioCounter == 0) {
86              System.out.println("No scenario has been executed. ");
87              System.out.println("Either those on the command line where not matching those in the file.");
88              System.out.println("Or the configuration file is empty");
89          }
90      }
91  
92      private static void shutdown() {
93          if (m_currentPostageRunner != null) m_currentPostageRunner.terminate();
94      }
95  
96  }
