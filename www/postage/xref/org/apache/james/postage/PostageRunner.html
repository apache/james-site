74      /***
75       * sends messages to James in two ways:
76       * 1. internal users relay to internal or external users using (inbound) SMTP
77       * 2. external users send mail to internal users using (inbound) SMTP
78       * 
79       * the correct mail delivery is checked in two ways:
80       * 1. by checking internal users mails using POP3
81       * 2. by checking mail to external users by receiving all mail forwarded by James to outbound/forwarded SMTP
82       *  
83       * @param postageConfiguration
84       */
85      public PostageRunner(PostageConfiguration postageConfiguration) {
86          m_postageConfiguration = postageConfiguration;
87          
88          int totalMailsPerMin = m_postageConfiguration.getTotalMailsPerMin();
89          int durationMinutes = m_postageConfiguration.getDurationMinutes();
90          
91          m_postageConfiguration.addDescriptionItem("mails_per_min", "" + totalMailsPerMin);
92          m_postageConfiguration.addDescriptionItem("totally_running_min", "" + durationMinutes);
93          m_postageConfiguration.addDescriptionItem("totally_mails_target", "" + totalMailsPerMin * durationMinutes);
94          
95          m_results.setEnvironmentDescription(m_postageConfiguration.getDescriptionItems());
96      }

98      private void execute() {
99          if (m_postageConfiguration != null) m_currentPhase = PHASE_STARTING;
100 
101         // do initialisation, check if all services can be connected
102         try {
103             setupInternalUserAccounts();
104             setupExternalUserAccounts();
105             setupInboundMailing();
106             setupInboundMailingChecker();
107             setupForwardedMailInterceptor();
108             setupJMXRemoting();
109             prepareResultFile(getCanonicalMailResultFileName());
110             prepareResultFile(getCanonicalJVMStatisticsFileName());
111             prepareResultFile(getCanonicalErrorsFileName());
112         } catch (StartupException e) {
113             log.fatal("could not even start the runner successfully", e);
114             return;
115         }
116 
117 
118         m_currentPhase = PHASE_RUNNING;
119 
120         log.info("starting scenario " + m_postageConfiguration.getId());
121 
122         // fork the timeout controller thread. it issues the oneMinute checkpoint event, too.
123         startTimer();
124 
125         // start all threads
126         try {
127             recordData();
128         } catch (Exception e) {
129             log.error("recording data was aborted!", e);
130         }
131 
132         // has to be set by method stopRecording() 
133         // m_currentPhase = PHASE_COMPLETED;
134 
135         // writeMatchedMailResults (remaining) collected data
136         log.info("completing by writing data for scenario " + m_postageConfiguration.getId());
137         writeData(false);
138     }

140     private void prepareResultFile(String canonicalMailResultFileName) {
141         File writeCandidate = new File(canonicalMailResultFileName);
142         if (writeCandidate.exists()) {
143             // rename existing result file from previous run 
144             // to something like "result___.cvs.64906993" to make place for new results
145             writeCandidate.renameTo(new File(canonicalMailResultFileName + "." + writeCandidate.lastModified()));
146         }

165     /***
166      * set up a thread issueing one-minute events and finally shutting down data recording when time has run out.
167      */
168     private void startTimer() {
169         new Thread(
170                 new Runnable() {
171                     public void run() {
172                         try {
173                             int durationMinutes = m_postageConfiguration.getDurationMinutes();
174                             log.info("running for " + durationMinutes + " minute(s)");
175                             for (int i = 0; i < durationMinutes; i++) {
176                                 Thread.sleep(60*1000);
177                                 oneMinuteCheckpoint();
178                             }
179                             stopRecording();
180                         } catch (InterruptedException e) {
181                             ; // exit
182                         }
183                     }
184 
185                 }
186         ).start();
187     }

189     /***
190      * called after each fully completed minute in the running phase
191      */
192     private void oneMinuteCheckpoint() {
193         m_minutesRunning++;
194         log.info("reached checkpoint after " + m_minutesRunning + " of " 
195                   + m_postageConfiguration.getDurationMinutes() + " minute(s) running.");
196 
197         //TODO do this in a separate thread?
198         writeData(true);
199     }

201     private void stopRecording() {
202         log.info("stopping");
203         if (m_sendControllers != null) {
204             Iterator iterator = m_sendControllers.iterator();
205             while (iterator.hasNext()) {
206                 SampleController sendController = (SampleController)iterator.next();
207                 sendController.stop();
208             }
209         }
210         if (m_inboundMailingController != null) m_inboundMailingController.stop();
211         
212         if (m_outboundMailingInterceptorController != null) m_outboundMailingInterceptorController.stop();
213         m_currentPhase = PHASE_COMPLETED;
214     }
215 
216     /***
217      * interrupt the runner from outside
218      */
219     public void terminate() {
220         stopRecording();
221         m_currentPhase = PHASE_ABORTED;
222         writeData(false);
223     }

252     private void recordData() {
253 
254         Iterator iterator = m_sendControllers.iterator();
255         while (iterator.hasNext()) {
256             SampleController sendController = (SampleController)iterator.next();
257             sendController.runThreaded();
258         }
259 
260         m_inboundMailingController = new SampleController(m_inboundMailingChecker, m_postageConfiguration.getTestserverPOP3FetchesPerMinute());
261         m_inboundMailingController.runThreaded();
262 
263         m_outboundMailingInterceptorController = new SampleController(m_smtpMailSink, 10, m_postageConfiguration.getTestserverSMTPForwardingWaitSeconds());
264         m_outboundMailingInterceptorController.runThreaded();
265 
266         if (m_jvmResourceSampler != null) {
267             m_jvmResourceController = new SampleController(m_jvmResourceSampler, 4);
268             m_jvmResourceController.runThreaded();
269         }
270 
271         while(m_currentPhase == PHASE_RUNNING) {
272             try {
273                 Thread.sleep(50);
274             } catch (InterruptedException e) {
275                 ; // leave
276             }
277         }
278 
279         if (m_currentPhase == PHASE_COMPLETED) {
280             // walk through all internal users and check for un-matched mails  
281             log.info("checking all internal accounts for unmatched mail...");
282             m_inboundMailingChecker.doMatchMailForAllUsers();
283             log.info("...done checking internal accounts");
284         } else {
285             // if we didn't COMPLETE, we'd better skip this
286             log.info("skip checking internal accounts for unmatched mail.");
287         }
288     }

302     /***
303      * sets up the profile where mail is send from external users to internal users
304      * @throws StartupException
305      */
306     private void setupInboundMailing() throws StartupException {
307         if (m_postageConfiguration.getTestserverPortSMTPInbound() <= 0) return;
308 
309         Iterator profileIterator = m_postageConfiguration.getProfiles().iterator();
310         while (profileIterator.hasNext()) {
311             SendProfile sendProfile = (SendProfile)profileIterator.next();
312             Iterator mailSenderIterator = sendProfile.mailSenderIterator();
313             while (mailSenderIterator.hasNext()) {
314                 MailSender mailSender = (MailSender)mailSenderIterator.next();
315                 int sendPerMinute = mailSender.getSendPerMinute();
316 
317                 if (sendPerMinute < 1) continue;
318 
319                 SMTPClient smtpClient = new SMTPClient(m_postageConfiguration.getTestserverHost(),
320                         m_postageConfiguration.getTestserverPortSMTPInbound(),
321                         m_postageConfiguration.getInternalUsers(),
322                         m_postageConfiguration.getExternalUsers(),
323                         m_results,
324                         mailSender
325                 );
326 
327                 boolean available = smtpClient.checkAvailability();
328                 log.info("availability of inbound mailing " + (available ? "": "NOT ") + "verified");
329                 if (!available) continue;
330 
331                 SampleController sendController = new SampleController(smtpClient, sendPerMinute);
332                 m_sendControllers.add(sendController);
333             }
334         }
335 
336     }

358     private void setupInternalUserAccounts() throws StartupException {
359         try {
360             String host = m_postageConfiguration.getTestserverHost();
361             int remoteManagerPort = m_postageConfiguration.getTestserverRemoteManagerPort();
362             String remoteManagerUsername = m_postageConfiguration.getTestserverRemoteManagerUsername();
363             String remoteManagerPassword = m_postageConfiguration.getTestserverRemoteManagerPassword();
364             int internalUserCount = m_postageConfiguration.getInternalUsers().getCount();
365             String internalUsernamePrefix = m_postageConfiguration.getInternalUsers().getNamePrefix();
366             String internalPassword = m_postageConfiguration.getInternalUsers().getPassword();
367 
368             Set existingUsers = getExistingUsers(host, remoteManagerPort, remoteManagerUsername, remoteManagerPassword);
369 
370             RemoteManagerClient remoteManagerClient = new RemoteManagerClient(host, remoteManagerPort, remoteManagerUsername, remoteManagerPassword);
371             boolean loginSuccess = remoteManagerClient.login();
372             ArrayList internalUsers = new ArrayList();
373             for (int i = 1; i <= internalUserCount; i++) {
374                 String username = internalUsernamePrefix + i;
375                 if (existingUsers.contains(username)) {
376                     log.info("user already exists: " + username);
377                     if (!m_postageConfiguration.isInternalReuseExisting()) {
378                         remoteManagerClient.executeCommand("deluser " + username);
379                         List answers = remoteManagerClient.readAnswer();
380                         addUser(remoteManagerClient, username, internalPassword);
381                         answers = remoteManagerClient.readAnswer();
382                         log.info("user deleted and re-created: " + username);
383                     }
384                     remoteManagerClient.executeCommand("setpassword " + username + " " + internalPassword);
385                     List answers = remoteManagerClient.readAnswer();
386                 } else {
387                     addUser(remoteManagerClient, username, internalPassword);
388                 }
389                 internalUsers.add(username);
390             }
391             m_postageConfiguration.getInternalUsers().setExistingUsers(internalUsers);
392             remoteManagerClient.disconnect();
393         } catch (Exception e) {
394             throw new StartupException("error setting up internal user accounts", e);
395         }
396     }

435     /***
436      * aquire a list of all existing internal James accounts 
437      * @return Set<String>, each String a username
438      */
439     private Set getExistingUsers(String host, int remoteManagerPort, String remoteManagerUsername, String remoteManagerPassword) {
440         RemoteManagerClient remoteManagerClient = new RemoteManagerClient(host, remoteManagerPort, remoteManagerUsername, remoteManagerPassword);
441         boolean loginSuccess = remoteManagerClient.login();
442         if (!loginSuccess) throw new Error("failed to login to remote manager");
443         List rawUserList = remoteManagerClient.executeCommand("listusers");
444         remoteManagerClient.disconnect();
445 
446         Set existingUsers = new LinkedHashSet();
447         Iterator iterator = rawUserList.iterator();
448         while (iterator.hasNext()) {
449             String line = (String)iterator.next();
450             if (!line.startsWith("user: ")) continue;
451 
452             existingUsers.add(line.substring(6));
453         }
454         return existingUsers;
455     }
