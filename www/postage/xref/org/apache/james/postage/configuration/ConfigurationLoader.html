31  /***
32   * generates named PostageConfigurations
33   */
34  public class ConfigurationLoader {
35  
36      private static Log log = LogFactory.getLog(ConfigurationLoader.class);
37  
38      public Map create(Configuration configuration) {
39          log.debug("reading configuration.");
40  
41          Map postageConfigurations = new LinkedHashMap();
42  
43          List scenariosIds = configuration.getList("scenario[@id]");
44          log.debug("scenarios contained in configuration: " + scenariosIds.size());
45  
46          Iterator scenarioIter = scenariosIds.iterator();
47          int scenarioCount = 0;
48          while (scenarioIter.hasNext()) {
49              String scenarioId = (String) scenarioIter.next();
50  
51              if (postageConfigurations.containsKey(scenarioId)) {
52                  log.error("found in configuration more than one scenario which is named: " + scenarioId);
53                  continue;
54              }
55  
56              PostageConfiguration postageConfiguration = new PostageConfiguration(scenarioId);
57  
58              String scenario = getIndexedPropertyName("scenario", scenarioCount);
59  
60              postageConfiguration.setDurationMinutes(configuration.getInt(getAttributedPropertyName(scenario, "runtimeMinutes")));
61  
62              addDescription(postageConfiguration, configuration.subset(scenario + ".description"));
63  
64              String scenarioInternalUsers = scenario + ".users.internal";
65              UserList internals = new UserList(configuration.getInt(getAttributedPropertyName(scenarioInternalUsers, "count")),
66                                                configuration.getString(getAttributedPropertyName(scenarioInternalUsers, "username-prefix")),
67                                                configuration.getString(getAttributedPropertyName(scenarioInternalUsers, "domain")),
68                                                configuration.getString(getAttributedPropertyName(scenarioInternalUsers, "password")));
69              postageConfiguration.setInternalUsers(internals);
70              postageConfiguration.setInternalReuseExisting(configuration.getBoolean(getAttributedPropertyName(scenarioInternalUsers, "reuseExisting")));
71  
72              String scenarioExternalUsers = scenario + ".users.external";
73              UserList externals = new UserList(configuration.getInt(getAttributedPropertyName(scenarioExternalUsers, "count")),
74                                                configuration.getString(getAttributedPropertyName(scenarioExternalUsers, "username-prefix")),
75                                                configuration.getString(getAttributedPropertyName(scenarioExternalUsers, "domain")));
76              postageConfiguration.setExternalUsers(externals);
77  
78              String scenarioTestserver = scenario + ".testserver";
79              postageConfiguration.setTestserverHost(configuration.getString(getAttributedPropertyName(scenarioTestserver, "host")));
80              postageConfiguration.setTestserverPortPOP3(configuration.getInt(getAttributedPropertyName(scenarioTestserver + ".pop3", "port")));
81              postageConfiguration.setTestserverPOP3FetchesPerMinute(configuration.getInt(getAttributedPropertyName(scenarioTestserver + ".pop3", "count-per-min")));
82              postageConfiguration.setTestserverPortSMTPInbound(configuration.getInt(getAttributedPropertyName(scenarioTestserver + ".smtp-inbound", "port")));
83              postageConfiguration.setTestserverPortSMTPForwarding(configuration.getInt(getAttributedPropertyName(scenarioTestserver + ".smtp-forwarding", "port")));
84              postageConfiguration.setTestserverSMTPForwardingWaitSeconds(configuration.getInt(getAttributedPropertyName(scenarioTestserver + ".smtp-forwarding", "latecomer-wait-seconds")));
85              postageConfiguration.setTestserverRemoteManagerPort(configuration.getInt(getAttributedPropertyName(scenarioTestserver + ".remotemanager", "port")));
86              postageConfiguration.setTestserverRemoteManagerUsername(configuration.getString(getAttributedPropertyName(scenarioTestserver + ".remotemanager", "name")));
87              postageConfiguration.setTestserverRemoteManagerPassword(configuration.getString(getAttributedPropertyName(scenarioTestserver + ".remotemanager", "password")));
88              postageConfiguration.setTestserverSpamAccountUsername(configuration.getString(getAttributedPropertyName(scenarioTestserver + ".spam-account", "name")));
89              postageConfiguration.setTestserverSpamAccountPassword(configuration.getString(getAttributedPropertyName(scenarioTestserver + ".spam-account", "password")));
90              postageConfiguration.setTestserverPortJMXRemoting(configuration.getInt(getAttributedPropertyName(scenarioTestserver + ".jvm-resources", "jmx-remoting-port")));
91  
92              addSendProfiles(postageConfiguration, configuration, scenario);
93  
94              postageConfigurations.put(postageConfiguration.getId(), postageConfiguration);
95  
96              scenarioCount++;
97          }
98  
99          return postageConfigurations;
100     }

102     private void addDescription(PostageConfiguration postageConfiguration, Configuration configuration) {
103         Iterator keys = configuration.getKeys();
104 
105 
106         while (keys.hasNext()) {
107             String itemName = (String) keys.next();
108             String itemContent = configuration.getString(itemName);
109 
110             postageConfiguration.addDescriptionItem(itemName, itemContent);
111         }
112     }
113 
114     private void addSendProfiles(PostageConfiguration postageConfiguration, Configuration configuration, String scenario) {
115         List profileNames = configuration.getList(scenario + ".profiles.profile[@name]");
116         log.debug("profiles contained in scenario " + postageConfiguration.getId() + ": " + profileNames.size());
117 
118         Iterator profileIter = profileNames.iterator();
119         int profileCount = 0;
120         while (profileIter.hasNext()) {
121             String profileName = (String) profileIter.next();
122 
123             SendProfile profile = new SendProfile(profileName);
124 
125             String profilePath = getIndexedPropertyName(scenario + ".profiles.profile", profileCount);
126 
127             profile.setSourceInternal(convertToInternalExternalFlag(configuration.getString(getAttributedPropertyName(profilePath, "source"))));
128             profile.setTargetInternal(convertToInternalExternalFlag(configuration.getString(getAttributedPropertyName(profilePath, "target"))));
129 
130             addMailSender(profile, configuration, profilePath);
131 
132             postageConfiguration.addProfile(profile);
133 
134             profileCount++;
135         }
136     }
137 
138     private void addMailSender(SendProfile profile, Configuration configuration, String profilePath) {
139         List mailSenders = configuration.getList(profilePath + ".send[@count-per-min]");
140 
141         Iterator mailSenderIter = mailSenders.iterator();
142         int mailSenderCount = 0;
143         while (mailSenderIter.hasNext()) {
144             mailSenderIter.next(); // ignore
145 
146             String mailSenderPath = getIndexedPropertyName(profilePath + ".send", mailSenderCount);
147 
148             MailSender mailSender = new MailSender(profile);
149             mailSender.setSubject(configuration.getString(getAttributedPropertyName(mailSenderPath, "subject"), "Apache JAMES Postage test mail"));
150             mailSender.setSendPerMinute(configuration.getInt(getAttributedPropertyName(mailSenderPath, "count-per-min")));
151             mailSender.setSizeMinText(configuration.getInt(getAttributedPropertyName(mailSenderPath, "text-size-min"), 0));
152             mailSender.setSizeMaxText(configuration.getInt(getAttributedPropertyName(mailSenderPath, "text-size-max"), 0));
153             mailSender.setSizeMinBinary(configuration.getInt(getAttributedPropertyName(mailSenderPath, "binary-size-min"), 0));
154             mailSender.setSizeMaxBinary(configuration.getInt(getAttributedPropertyName(mailSenderPath, "binary-size-max"), 0));
155             mailSender.setMailFactoryClassname(configuration.getString(getAttributedPropertyName(mailSenderPath, "mail-factory-class"), null));
156 
157             profile.addMailSender(mailSender);
158 
159             mailSenderCount++;
160         }
161     }

163     private boolean convertToInternalExternalFlag(String flagCleartext) {
164         return flagCleartext == null || !"extern".equals(flagCleartext.toLowerCase().trim());
165     }
166 
167     private String getIndexedPropertyName(String name, int scenarioCount) {
168         return name + "(" + scenarioCount + ")";
169     }
170 
171     private String getAttributedPropertyName(String name, String attr) {
172         return name + "[@" + attr + "]";
173     }
