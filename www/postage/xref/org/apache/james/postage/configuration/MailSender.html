76      /***
77       * how much emails are to be sent in the specified minute
78       * (taking into account the increase and max values)
79       * @param minute
80       * @return mails to be sent
81       */
82      public int getSendPerMinute(int minute) {
83          int increased = sendPerMinute + (int)(increaseSendPerMinute * minute);
84          if (maxSendPerMinute > 0) {
85              if (increased > maxSendPerMinute) increased = maxSendPerMinute;
86          }
87          return increased;
88      }

155     public MailFactory getMailFactory() {
156         MailFactory mailFactory = null;
157 
158         // class is configured, but not yet loaded
159         if (m_mailFactoryClassname != null && m_mailFactoryClass == null) {
160             try {
161                 m_mailFactoryClass = Class.forName(m_mailFactoryClassname);
162             } catch (ClassNotFoundException e) {
163                 log.error("failed to load MailFactory class " + m_mailFactoryClassname, e);
164             }
165         }
166 
167         // create instance, if custom class is given
168         if (m_mailFactoryClass != null) {
169             try {
170                 mailFactory = (MailFactory)m_mailFactoryClass.newInstance();
171             } catch (Exception e) {
172                 log.error("failed to create instance if MailFactory class " + m_mailFactoryClassname, e);
173             }
174         }
175 
176         if (mailFactory == null) mailFactory = new DefaultMailFactory();
177         return mailFactory;
178     }
