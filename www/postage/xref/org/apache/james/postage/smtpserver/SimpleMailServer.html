51      public void sendMail(MailAddress sender, Collection recipients, MimeMessage msg) throws MessagingException {
52          //log.info("start processing incoming mail having id = " + msg.getMessageID());
53          MailProcessingRecord mailProcessingRecord = new MailProcessingRecord();
54          mailProcessingRecord.setReceivingQueue("smtpOutbound");
55          mailProcessingRecord.setTimeFetchStart(System.currentTimeMillis());
56          mailProcessingRecord.setByteReceivedTotal(msg.getSize());
57  
58          try {
59              String[] id = msg.getHeader(HeaderConstants.MAIL_ID_HEADER);
60              if (id == null || id.length == 0 ) {
61                  log.info("skipping non-postage mail. message is lost. subject was: " + msg.getSubject());
62                  return;
63              }
64              mailProcessingRecord.setMailId(id[0]);
65  
66              String[] subjectHeader = msg.getHeader("Subject");
67              if (subjectHeader != null && subjectHeader.length > 0) {
68                  mailProcessingRecord.setSubject(subjectHeader[0]);
69              }
70  
71              // TODO mailProcessingRecord.setByteReceivedText();
72              // TODO mailProcessingRecord.setByteReceivedBinary();
73              
74              mailProcessingRecord.setTimeFetchEnd(System.currentTimeMillis());
75          } catch(MessagingException e) {
76              log.error("error processing incoming mail: " + e.getMessage());
77              throw e; // rethrow after logging 
78          } finally{
79              boolean matched = m_results.matchMailRecord(mailProcessingRecord);
80              if (!matched) {
81                  if (mailProcessingRecord.getMailId() == null) mailProcessingRecord.setMailId(MailProcessingRecord.getNextId());
82                  m_results.addNewMailRecord(mailProcessingRecord);
83              }
84          }
85      }

96      public void sendMail(MimeMessage message) throws MessagingException {
97          // taken from class org.apache.james.James 
98          MailAddress sender = new MailAddress((InternetAddress)message.getFrom()[0]);
99          Collection recipients = new HashSet();
100         Address addresses[] = message.getAllRecipients();
101         if (addresses != null) {
102             for (int i = 0; i < addresses.length; i++) {
103                 // Javamail treats the "newsgroups:" header field as a
104                 // recipient, so we want to filter those out.
105                 if ( addresses[i] instanceof InternetAddress ) {
106                     recipients.add(new MailAddress((InternetAddress)addresses[i]));
107                 }
108             }
109         }
110         sendMail(sender, recipients, message);
111     }
