61      public boolean checkAvailability() throws StartupException {
62          try {
63              org.apache.commons.net.pop3.POP3Client pop3Client = openConnection(m_internalUsers.getRandomUsername());
64              closeSession(pop3Client);
65          } catch (PostageException e) {
66              throw new StartupException("error checking availability");
67          }
68          return true;

71      private void closeSession(org.apache.commons.net.pop3.POP3Client pop3Client) throws PostageException {
72          try {
73              pop3Client.sendCommand("QUIT");
74              pop3Client.disconnect();
75          } catch (IOException e) {
76              throw new PostageException("error closing pop3 session", e);
77          }
78      }

80      private org.apache.commons.net.pop3.POP3Client openConnection(String username) throws PostageException {
81          org.apache.commons.net.pop3.POP3Client pop3Client = new org.apache.commons.net.pop3.POP3Client();
82          try {
83              pop3Client.connect(m_host, m_port);
84              pop3Client.login(username, m_internalUsers.getPassword());
85          } catch (IOException e) {
86              throw new PostageException("POP3 service not available", e);
87          }
88          return pop3Client;
89      }

114     private void findAllMatchingTestMail(String username) throws SamplingException {
115         try {
116             org.apache.commons.net.pop3.POP3Client pop3Client = openConnection(username);
117 
118 
119 
120             // retrieve all messages
121             POP3MessageInfo[] entries = null;
122             try {
123                 entries = pop3Client.listMessages();
124             } catch (Exception e) {
125                 String errorMessage = "failed to read pop3 account mail list for " + username;
126                 m_results.addError(500, errorMessage);
127                 log.info(errorMessage);
128                 return;
129             }
130 
131             for (int i = 0; i < entries.length; i++) {
132 // TODO do we need to check the state?                assertEquals(1, pop3Client.getState());
133                 POP3MessageInfo entry = entries[i];
134                 int size = entry.size;
135 
136                 MailProcessingRecord mailProcessingRecord = new MailProcessingRecord();
137                 mailProcessingRecord.setReceivingQueue("pop3");
138                 mailProcessingRecord.setTimeFetchStart(System.currentTimeMillis());
139 
140                 String mailText;
141                 BufferedReader mailReader = null;
142                 Reader reader = null;
143                 try {
144                     reader = pop3Client.retrieveMessage(entry.number);
145                     mailReader = new BufferedReader(reader);
146 
147                     analyzeAndMatch(mailReader, mailProcessingRecord, pop3Client, i);
148                 } catch (Exception e) {
149                     log.info("failed to read pop3 mail #" + i + " for " + username);
150                     continue; // mail processing record is discarded
151                 } finally {
152                     if (reader != null) {
153                         try {
154                             reader.close();
155                         } catch (IOException e) {
156                             log.warn("error closing mail reader");
157                         }
158                     }
159                 }
160 
161                 continue;
162             }
163 
164             closeSession(pop3Client);
165         } catch (PostageException e) {
166             throw new SamplingException("sample failed", e);
167         }
168     }

170     private void analyzeAndMatch(Reader mailReader, MailProcessingRecord mailProcessingRecord, org.apache.commons.net.pop3.POP3Client pop3Client, int i) {
171 
172         InputStream in = new ReaderInputStream(mailReader);
173         MimeMessage message;
174         String id = null;
175         try {
176             message = new MimeMessage(null, in);
177             in.close();
178 
179             String[] idHeaders = message.getHeader(HeaderConstants.MAIL_ID_HEADER);
180             if (idHeaders != null && idHeaders.length > 0) {
181                 id = idHeaders[0]; // there should be exactly one.
182             }
183             if (id == null) {
184                 log.info("skipping non-postage mail. remains on server. subject was: " + message.getSubject());
185                 return;
186             }
187         } catch (IOException e) {
188             log.info("failed to close mail reader.");
189             return;
190         } catch (MessagingException e) {
191             log.info("failed to process mail. remains on server");
192             return;
193         }
194 
195         try {
196             mailProcessingRecord.setByteReceivedTotal(message.getSize());
197 
198             mailProcessingRecord.setMailId(id);
199             mailProcessingRecord.setSubject(message.getSubject());
200 
201             MimeMultipart mimeMultipart = new MimeMultipart(message.getDataHandler().getDataSource());
202 
203             // this assumes, there is zero or one part each, either of type plaintext or binary  
204             mailProcessingRecord.setByteReceivedText(getPartSize(mimeMultipart, "text/plain"));
205             mailProcessingRecord.setByteReceivedBinary(getPartSize(mimeMultipart, "application/octet-stream"));
206 
207             mailProcessingRecord.setTimeFetchEnd(System.currentTimeMillis());
208 
209             try {
210                 pop3Client.deleteMessage(i + 1); // don't retrieve again next time
211             } catch (Exception e) {
212                 log.info("failed to delete mail.");
213                 return;
214             }
215         } catch (MessagingException e) {
216             log.info("failed to process mail. remains on server");
217             return;
218         } finally {
219             matchMails(mailProcessingRecord);
220         }
221     }

223     private void matchMails(MailProcessingRecord mailProcessingRecord) {
224         boolean matched = m_results.matchMailRecord(mailProcessingRecord);
225         if (!matched) {
226             String oldMailId = mailProcessingRecord.getMailId();
227             String newMailId = MailProcessingRecord.getNextId();
228             mailProcessingRecord.setMailId(newMailId);
229             log.info("changed mail id from " + oldMailId + " to " + newMailId);
230             m_results.addNewMailRecord(mailProcessingRecord);
231         }
232     }

234     private int getPartSize(MimeMultipart parts, String mimeType) {
235         if (parts != null) {
236             try {
237                 for (int i = 0; i < parts.getCount(); i++) {
238                     BodyPart bodyPart = parts.getBodyPart(i);
239                     if (mimeType.equals(bodyPart.getContentType())) {
240                         return bodyPart.getSize();
241                     }
242                 }
243             } catch (MessagingException e) {
244                 log.info("failed to process body parts.", e);
245             }
246         }
247         return 0;
248     }
