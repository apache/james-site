230     public void merge(MailProcessingRecord anotherRecord) {
231         if (matchedSentAndReceived) throw new IllegalStateException("already merged");
232         matchedSentAndReceived = true;
233         
234         if (timeConnectStart == 0) timeConnectStart = anotherRecord.timeConnectStart;
235         if (mailId == null) mailId = anotherRecord.mailId;
236         if (subject == null) subject = anotherRecord.subject;
237         if (sender == null) sender = anotherRecord.sender;
238         if (senderMailAddress == null) senderMailAddress = anotherRecord.senderMailAddress;
239         if (receiver == null) receiver = anotherRecord.receiver;
240         if (receiverMailAddress == null) receiverMailAddress = anotherRecord.receiverMailAddress;
241         if (timeSendStart == 0) timeSendStart = anotherRecord.timeSendStart;
242         if (timeSendEnd == 0) timeSendEnd = anotherRecord.timeSendEnd;
243         if (byteSendText == 0) byteSendText = anotherRecord.byteSendText;
244         if (byteSendBinary == 0) byteSendBinary = anotherRecord.byteSendBinary;
245         if (byteReceivedText == 0) byteReceivedText = anotherRecord.byteReceivedText;
246         if (byteReceivedBinary == 0) byteReceivedBinary = anotherRecord.byteReceivedBinary;
247         if (byteReceivedTotal == 0) byteReceivedTotal = anotherRecord.byteReceivedTotal;
248         if (errorNumberSending == 0) errorNumberSending = anotherRecord.errorNumberSending;
249         if (errorTextSending == null) errorTextSending = anotherRecord.errorTextSending;
250         if (timeReceived == 0) timeReceived = anotherRecord.timeReceived;
251         if (timeFetchStart == 0) timeFetchStart = anotherRecord.timeFetchStart;
252         if (timeFetchEnd == 0) timeFetchEnd = anotherRecord.timeFetchEnd;
253         if (timeServerReceived == 0) timeServerReceived = anotherRecord.timeServerReceived;
254         if (receivingQueue == null) receivingQueue = anotherRecord.receivingQueue;
255     }
