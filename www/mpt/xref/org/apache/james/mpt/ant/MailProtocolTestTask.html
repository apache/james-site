166     @Override
167     public void execute() throws BuildException {
168         if (port <= 0) {
169             throw new BuildException("Port must be set to a positive integer");
170         }
171         
172         if (scripts == null && script == null) {
173             throw new BuildException("Scripts must be specified as an embedded resource collection"); 
174         }
175         
176         if (scripts != null && script != null) {
177             throw new BuildException("Scripts can be specified either by the script attribute or as resource collections but not both."); 
178         }
179         
180         for(final Iterator it=users.iterator();it.hasNext();) {
181             final AddUser user = (AddUser) it.next();
182             user.validate();
183         }
184         
185         if(skip) {
186             log("Skipping excution");
187         } else {
188             doExecute();
189         }
190     }

192     public void add(ResourceCollection resources) {
193         if (scripts == null) {
194             scripts = new Union();
195         }
196         scripts.add(resources);

199     private void doExecute() throws BuildException {
200         for (final Iterator it=users.iterator();it.hasNext();) {
201             final AddUser userAdder = (AddUser) it.next();
202             userAdder.execute();
203         }
204         
205         final ExternalHostSystem host = new ExternalHostSystem(getHost(), getPort(), this, getShabang(), null);
206         final ProtocolSessionBuilder builder = new ProtocolSessionBuilder();
207         
208         if (scripts == null) {
209             scripts = new Union();
210             scripts.add(new FileResource(script));
211         }
212     
213         for (final Iterator it=scripts.iterator();it.hasNext();) {
214             final Resource resource = (Resource) it.next();
215             try {
216                 final Runner runner = new Runner();
217                 
218                 try {
219                     
220                     final InputStream inputStream = resource.getInputStream();
221                     final String name = resource.getName();
222                     builder.addProtocolLines(name == null ? "[Unknown]" : name, inputStream, runner.getTestElements());
223                     runner.runSessions(host);
224                     
225                 } catch (UnsupportedOperationException e) {
226                     log("Resource cannot be read: " + resource.getName(), Project.MSG_WARN);
227                 }
228             } catch (IOException e) {
229                 throw new BuildException("Cannot load script " + resource.getName(), e);
230             } catch (Exception e) {
231                 log(e.getMessage(), Project.MSG_ERR);
232                 throw new BuildException("[FAILURE] in script " + resource.getName() + "\n" + e.getMessage(), e);
233             }
234             
235         }
236     
237     }

331         /**
332          * Validates mandatory fields have been filled.
333          */
334         void validate() throws BuildException {
335             if (script == null && scriptText == null) {
336                 throw new BuildException("Either the 'script' attribute must be set, or the body must contain the text of the script");
337             }
338             
339             if (script != null && scriptText != null) {
340                 throw new BuildException("Choose either script text or script attribute but not both.");
341             }
342             
343             if (port <= 0) {
344                 throw new BuildException("'port' attribute must be set on AddUser to the port against which the script should run.");
345             }
346         }
347         
348         /**
349          * Creates a user.
350          * @throws BuildException
351          */
352         void execute() throws BuildException {
353             validate();
354             try {
355                 final File scriptFile = getScript();
356                 final Reader reader;
357                 if (scriptFile == null) {
358                     reader = new StringReader(scriptText);
359                 } else {
360                     reader = new FileReader(scriptFile);
361                 }
362                 final ScriptedUserAdder adder = new ScriptedUserAdder(getHost(), getPort(), MailProtocolTestTask.this);
363                 adder.addUser(getUser(), getPasswd(), reader);
364             } catch (Exception e) {
365                 log(e.getMessage(), Project.MSG_ERR);
366                 throw new BuildException("User addition failed: \n" + e.getMessage(), e);
367             }
368         } 
369     }
