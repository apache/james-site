21  package org.apache.jsieve;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.jsieve.mail.MailAdapter;
25  
26  /***
27   * <p>A parsed representation of the RFC3028 BNF...</p>
28   * 
29   * <code>command = identifier arguments ( ";" / block )</code>
30   */
31  public class Command implements Executable
32  {
33      /***
34       * @see org.apache.jsieve.Executable#execute(MailAdapter)
35       */
36      public Object execute(MailAdapter mail) throws SieveException
37      {
38          Log log = Logger.getLog();
39          if (log.isDebugEnabled())
40              log.debug(toString());
41          return CommandManager.getInstance().newInstance(getName()).execute(
42              mail,
43              getArguments(),
44              getBlock(), context);
45      }
46  
47  
48      // The name of this Command
49      private String fieldName;
50      
51      // The Arguments for this Command
52      private Arguments fieldArguments;
53   
54      // The Block for this Command   
55      private Block fieldBlock;  
56      
57      private SieveContext context;
58      
59      /***
60       * Constructor for Test.
61       */
62      private Command()
63      {
64          super();
65      }
66      
67      /***
68       * Constructor for Command.
69       * @param name
70       * @param arguments
71       * @param block
72       */
73      public Command(String name, Arguments arguments, Block block, SieveContext context)
74      {
75          this();
76          this.context = context;
77          setName(name);
78          setArguments(arguments);
79          setBlock(block);        
80      }        
81  
82      /***
83       * Returns the name.
84       * @return String
85       */
86      public String getName()
87      {
88          return fieldName;
89      }
90  
91      /***
92       * Sets the name.
93       * @param name The name to set
94       */
95      protected void setName(String name)
96      {
97          fieldName = name;
98      }
99  
100     /***
101      * @see java.lang.Object#toString()
102      */
103     public String toString()
104     {
105         return "Command name: "
106             + getName()
107             + " "
108             + ((getArguments() == null) ? "null" : getArguments().toString())
109             + " Block: "
110             + ((getBlock() == null) ? "null" : getBlock().toString());
111     }
112 
113     /***
114      * Returns the arguments.
115      * @return Arguments
116      */
117     public Arguments getArguments()
118     {
119         return fieldArguments;
120     }
121 
122     /***
123      * Returns the block.
124      * @return Block
125      */
126     public Block getBlock()
127     {
128         return fieldBlock;
129     }
130 
131     /***
132      * Sets the arguments.
133      * @param arguments The arguments to set
134      */
135     protected void setArguments(Arguments arguments)
136     {
137         fieldArguments = arguments;
138     }
139 
140     /***
141      * Sets the block.
142      * @param block The block to set
143      */
144     protected void setBlock(Block block)
145     {
146         fieldBlock = block;
147     }
148 
149 }
