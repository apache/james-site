20  package org.apache.jsieve;
21  
22  import java.util.Map;
23  
24  import org.apache.jsieve.exception.LookupException;
25  
26  /***
27   * Singleton class <code>CommandManager</code> maps Command names to
28   * configured Command implementation classes.
29   */
30  public class CommandManager {
31      /***
32       * The sole instance of the receiver.
33       */
34      static private CommandManager fieldInstance;
35  
36      /***
37       * Constructor for CommandManager.
38       */
39      private CommandManager() {
40          super();
41      }
42  
43      /***
44       * Returns the sole instance of the receiver, lazily initialised if
45       * required.
46       * 
47       * @return CommandManager
48       */
49      public static synchronized CommandManager getInstance() {
50          CommandManager instance = null;
51          if (null == (instance = getInstanceBasic())) {
52              updateInstance();
53              return getInstance();
54          }
55          return instance;
56      }
57  
58      /***
59       * Returns the sole instance of the receiver.
60       * 
61       * @return CommandManager
62       */
63      private static CommandManager getInstanceBasic() {
64          return fieldInstance;
65      }
66  
67      /***
68       * Computes a new instance of the receiver.
69       * 
70       * @return CommandManager
71       */
72      protected static CommandManager computeInstance() {
73          return new CommandManager();
74      }
75  
76      /***
77       * Sets the sole instance of the receiver.
78       * 
79       * @param instance
80       *                The current instance to set
81       */
82      protected static void setInstance(CommandManager instance) {
83          fieldInstance = instance;
84      }
85  
86      /***
87       * Resets the sole instance.
88       */
89      public static void resetInstance() {
90          setInstance(null);
91      }
92  
93      /***
94       * Updates the sole instance.
95       */
96      protected static void updateInstance() {
97          setInstance(computeInstance());
98      }
99  
100     /***
101      * <p>
102      * Method lookup answers the class to which a Command name is mapped.
103      * </p>
104      * 
105      * @param name -
106      *                The name of the Command
107      * @return Class - The class of the Command
108      * @throws LookupException
109      */
110     public Class lookup(String name) throws LookupException {
111         Class cmdClass = null;
112         try {
113             cmdClass = getClass().getClassLoader()
114                     .loadClass(getClassName(name));
115         } catch (ClassNotFoundException e) {
116             throw new LookupException("Command named '" + name + "' not found.");
117         }
118         if (!ExecutableCommand.class.isAssignableFrom(cmdClass))
119             throw new LookupException("Class " + cmdClass.getName()
120                     + " must implement " + ExecutableCommand.class.getName());
121         return cmdClass;
122     }
123 
124     /***
125      * <p>
126      * Method newInstance answers an instance of the class to which a Command
127      * name is mapped.
128      * </p>
129      * 
130      * @param name -
131      *                The name of the Command
132      * @return Class - The class of the Command
133      * @throws LookupException
134      */
135     public ExecutableCommand newInstance(String name) throws LookupException {
136         try {
137             return (ExecutableCommand) lookup(name).newInstance();
138         } catch (InstantiationException e) {
139             throw new LookupException(e.getMessage());
140         } catch (IllegalAccessException e) {
141             throw new LookupException(e.getMessage());
142         }
143     }
144 
145     /***
146      * Method isSupported answers a boolean indicating if a Command name is
147      * configured.
148      * 
149      * @param name -
150      *                The Command name
151      * @return boolean - True if the Command name is configured.
152      */
153     public boolean isSupported(String name) {
154         boolean isSupported = false;
155         try {
156             lookup(name);
157             isSupported = true;
158         } catch (LookupException e) {
159         }
160         return isSupported;
161     }
162 
163     /***
164      * <p>
165      * Method getClassName answers the name of the class to which a Command name
166      * is mapped.
167      * </p>
168      * 
169      * @param name -
170      *                The name of the Command
171      * @return String - The name of the class
172      * @throws LookupException
173      */
174     protected String getClassName(String name) throws LookupException {
175         String className;
176         try {
177             className = (String) getClassNameMap().get(name.toLowerCase());
178         } catch (SieveConfigurationException e) {
179             throw new LookupException(
180                     "Lookup failed due to a Configuration Exception: "
181                             + e.getMessage());
182         }
183         if (null == className)
184             throw new LookupException("Command named '" + name
185                     + "' not mapped.");
186         return className;
187     }
188 
189     /***
190      * Method getClassNameMap answers a Map of Command names and their class
191      * names.
192      * 
193      * @return Map
194      * @throws SieveConfigurationException
195      */
196     protected Map getClassNameMap() throws SieveConfigurationException {
197         return ConfigurationManager.getInstance().getCommandMap();
198     }
