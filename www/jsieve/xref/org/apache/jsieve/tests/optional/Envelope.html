38  /***
39   * Class Envelope implements the optional Envelope Test as defined in RFC 3028, 
40   * section 5.4.
41   */
42  public class Envelope extends AbstractCompatatorTest

46       * Constructor for EnvelopeAccessors.
47       */
48      public Envelope()
49      {
50          super();

53      /***
54       * @see org.apache.jsieve.tests.Address#getMatchingValues(MailAdapter, String)
55       */
56      protected List getMatchingValues(MailAdapter mail, String valueName)
57          throws SieveMailException
58      {
59          return ((EnvelopeAccessors) mail).getMatchingEnvelope(valueName);
60      }

62      /***
63       * Method match.
64       * @param addressPart
65       * @param comparator
66       * @param matchType
67       * @param headerValue
68       * @param key
69       * @return boolean
70       * @throws SieveMailException
71       */
72      protected boolean match(String addressPart, String comparator, String matchType, String headerValue, String key) throws SieveException {
73          // Attempt to create a new InternetAddress object from the headerValue
74          // If this fails, the header either is not intended to contain a valid
75          // Internet Address or is corrupt. Either way, its an Exception!
76          String address = null;
77          try
78          {
79              // address is a simple address; user@domain or user
80              address = new InternetAddress(headerValue).getAddress();
81          }
82          catch (AddressException e)
83          {
84              throw new InternetAddressException(e.getMessage());
85          }
86      
87          // Extract the part of the address we are matching on       
88          String matchAddress = null;
89          if (addressPart.equals(":all"))
90              matchAddress = address;
91          else
92          {
93              int localStart = 0;
94              int localEnd = 0;
95              int domainStart = 0;
96              int domainEnd = address.length();
97              int splitIndex = address.indexOf('@');
98              // If there is no domain part (-1), treat it as an empty String
99              if (splitIndex == -1)
100             {
101                 localEnd = domainEnd;
102                 domainStart = domainEnd;
103             }
104             else
105             {
106                 localEnd = splitIndex;
107                 domainStart = splitIndex + 1;
108             }
109             matchAddress =
110                 (addressPart.equals(LOCALPART_TAG)
111                     ? address.substring(localStart, localEnd)
112                     : address.substring(domainStart, domainEnd));
113         }
114     
115         // domain matches MUST ignore case, others should not
116         String matchKey = null;
117         if (addressPart.equals(DOMAIN_TAG))
118         {
119             matchKey = key.toLowerCase();
120             matchAddress = matchAddress.toLowerCase();
121         }
122         else
123             matchKey = key;
124     
125         // Match using the specified comparator          
126         return ComparatorUtils.match(
127             comparator,
128             matchType,
129             matchAddress,
130             matchKey);
131     }
132 
133     protected boolean match(MailAdapter mail, String addressPart, String comparator, String matchType, String headerName, String key) throws SieveException {
134         Iterator headerValuesIter =
135             getMatchingValues(mail, headerName).iterator();
136         boolean isMatched = false;
137         while (!isMatched && headerValuesIter.hasNext())
138         {
139             isMatched =
140                 match(
141                     addressPart,
142                     comparator,
143                     matchType,
144                     ((String) headerValuesIter.next()),
145                     key);
146         }
147         return isMatched;
148     }
