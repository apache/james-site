27  import org.apache.jsieve.mail.MailAdapter;

32  /***
33   * Class Envelope implements the optional Envelope Test as defined in RFC 3028,
34   * section 5.4.
35   */
36  public class Envelope extends AbstractCompatatorTest {
37  
38      /***
39       * Constructor for EnvelopeAccessors.
40       */
41      public Envelope() {
42          super();
43      }
44  
45      /***
46       * @see org.apache.jsieve.tests.Address#getMatchingValues(MailAdapter,
47       *      String)
48       */
49      protected List getMatchingValues(MailAdapter mail, String valueName)
50              throws SieveMailException {
51          return ((EnvelopeAccessors) mail).getMatchingEnvelope(valueName);
52      }
53  
54      /***
55       * Method match.
56       * 
57       * @param addressPart
58       * @param comparator
59       * @param matchType
60       * @param headerValue
61       * @param key
62       * @return boolean
63       * @throws SieveMailException
64       */
65      protected boolean match(String addressPart, String comparator,
66              String matchType, String headerValue, String key)
67              throws SieveException {
68  
69          // Extract the part of the address we are matching on
70          String matchAddress = null;
71          if (addressPart.equals(":all"))
72              matchAddress = headerValue;
73          else {
74              int localStart = 0;
75              int localEnd = 0;
76              int domainStart = 0;
77              int domainEnd = headerValue.length();
78              int splitIndex = headerValue.indexOf('@');
79              // If there is no domain part (-1), treat it as an empty String
80              if (splitIndex == -1) {
81                  localEnd = domainEnd;
82                  domainStart = domainEnd;
83              } else {
84                  localEnd = splitIndex;
85                  domainStart = splitIndex + 1;
86              }
87              matchAddress = (addressPart.equals(LOCALPART_TAG) ? headerValue
88                      .substring(localStart, localEnd) : headerValue.substring(
89                      domainStart, domainEnd));
90          }
91  
92          // domain matches MUST ignore case, others should not
93          String matchKey = null;
94          if (addressPart.equals(DOMAIN_TAG)) {
95              matchKey = key.toLowerCase();
96              matchAddress = matchAddress.toLowerCase();
97          } else
98              matchKey = key;
99  
100         // Match using the specified comparator
101         return ComparatorUtils.match(comparator, matchType, matchAddress,
102                 matchKey);
103     }
104 
105     protected boolean match(MailAdapter mail, String addressPart,
106             String comparator, String matchType, String headerName, String key)
107             throws SieveException {
108         Iterator headerValuesIter = getMatchingValues(mail, headerName)
109                 .iterator();
110         boolean isMatched = false;
111         while (!isMatched && headerValuesIter.hasNext()) {
112             isMatched = match(addressPart, comparator, matchType,
113                     ((String) headerValuesIter.next()), key);
114         }
115         return isMatched;
116     }
