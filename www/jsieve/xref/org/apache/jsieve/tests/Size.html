33  /***
34   * Class Size implements the Size Test as defined in RFC 3028, section 5.9.
35   */
36  public class Size extends AbstractTest
37  {
38  
39      /***
40       * Constructor for Size.
41       */
42      public Size()
43      {
44          super();
45      }
46  
47      /***
48       * @see org.apache.jsieve.tests.AbstractTest#executeBasic(MailAdapter, Arguments)
49       * <p>From RFC 3028, Section 5.9... </p>
50       * <code>  
51       *    Syntax: size &lt;&quote;:over"&quote; / &quote;:under&quote;&gt; &lt;limit: number&gt;
52       * </code>
53       */
54      protected boolean executeBasic(MailAdapter mail, Arguments arguments)
55          throws SyntaxException, SieveMailException
56      {
57          String comparator = null;
58          Integer size = null;
59          ListIterator argumentsIter = arguments.getArgumentList().listIterator();
60  
61          // First argument MUST be a tag of ":under" or ":over"
62          if (argumentsIter.hasNext())
63          {
64              Object argument = argumentsIter.next();
65              if (argument instanceof TagArgument)
66              {
67                  String tag = ((TagArgument) argument).getTag();
68                  if (tag.equals(":under") || tag.equals(":over"))
69                      comparator = tag;
70                  else
71                      throw new SyntaxException(
72                          "Found unexpected TagArgument: \"" + tag + "\"");
73              }
74          }
75          if (null == comparator)
76              throw new SyntaxException("Expecting a Tag");
77  
78          // Second argument MUST be a number
79          if (argumentsIter.hasNext())
80          {
81              Object argument = argumentsIter.next();
82              if (argument instanceof NumberArgument)
83                  size = ((NumberArgument) argument).getInteger();
84          }
85          if (null == size)
86              throw new SyntaxException("Expecting a Number");
87  
88          // There MUST NOT be any further arguments
89          if (argumentsIter.hasNext())
90              throw new SyntaxException("Found unexpected argument(s)");               
91  
92          return test(mail, comparator, size.intValue());
93      }
94  
95      /***
96       * Method test.
97       * @param mail
98       * @param comparator
99       * @param size
100      * @return boolean
101      * @throws SieveMailException
102      */
103     protected boolean test(MailAdapter mail, String comparator, int size)
104         throws SieveMailException
105     {
106         boolean isTestPassed = false;
107         if (comparator.equals(":over"))
108             isTestPassed = testOver(mail, size);
109         else if (comparator.equals(":under"))
110             isTestPassed = testUnder(mail, size);
111         return isTestPassed;
112     }
113 
114     /***
115      * Method testUnder.
116      * @param mail
117      * @param size
118      * @return boolean
119      * @throws SieveMailException
120      */
121     protected boolean testUnder(MailAdapter mail, int size)
122         throws SieveMailException
123     {
124         return mail.getSize() < size;
125     }
126 
127 
128     /***
129      * Method testOver.
130      * @param mail
131      * @param size
132      * @return boolean
133      * @throws SieveMailException
134      */
135     protected boolean testOver(MailAdapter mail, int size)
136         throws SieveMailException
137     {
138         return mail.getSize() > size;
139     }

146     protected void validateArguments(Arguments arguments) throws SieveException

149     }
