57      /***
58       * <p>From RFC 3028, Section 5.1... </p>
59       * <code>  
60       * Syntax: address [ADDRESS-PART] [COMPARATOR] [MATCH-TYPE]
61       *         &lt;header-list: string-list&gt; &lt;key-list: string-list&gt;
62       * </code>
63       * <p>Note that the spec. then goes on to give an example where the
64       * order of the optional parts is different, so I am assuming that the order of
65       * the optional parts is optional too!</p>
66       * 
67       * @see org.apache.jsieve.tests.AbstractTest#executeBasic(MailAdapter, Arguments, SieveContext)
68       */
69      protected boolean executeBasic(MailAdapter mail, Arguments arguments, SieveContext context)
70          throws SieveException
71      {
72          String addressPart = null;
73          String comparator = null;
74          String matchType = null;
75          List headerNames = null;
76          List keys = null;
77  
78          ListIterator argumentsIter = arguments.getArgumentList().listIterator();
79          boolean stop = false;
80  
81          // Tag processing
82          while (!stop && argumentsIter.hasNext())
83          {
84              Object argument = argumentsIter.next();
85              if (argument instanceof TagArgument)
86              {
87                  String tag = ((TagArgument) argument).getTag();
88  
89                  // [ADDRESS-PART]?
90                  if (null == addressPart
91                      && (tag.equals(LOCALPART_TAG)
92                          || tag.equals(DOMAIN_TAG)
93                          || tag.equals(ALL_TAG)))
94                      addressPart = tag;
95                  // [COMPARATOR]?    
96                  else if (null == comparator && tag.equals(COMPARATOR_TAG))
97                  {
98                      // The next argument must be a stringlist
99                      if (argumentsIter.hasNext())
100                     {
101                         argument = argumentsIter.next();
102                         if (argument instanceof StringListArgument)
103                         {
104                             List stringList =
105                                 ((StringListArgument) argument).getList();
106                             if (stringList.size() != 1)
107                                 throw new SyntaxException("Expecting exactly one String");
108                             comparator = (String) stringList.get(0);
109                         }
110                         else
111                             throw new SyntaxException("Expecting a StringList");
112                     }
113                 }
114                 // [MATCH-TYPE]?
115                 else if (
116                     null == matchType
117                         && (tag.equals(IS_TAG)
118                             || tag.equals(CONTAINS_TAG)
119                             || tag.equals(MATCHES_TAG)))
120                     matchType = tag;
121                 else
122                     throw context.getCoordinate().syntaxException("Found unexpected TagArgument");
123             }
124             else
125             {
126                 // Stop when a non-tag argument is encountered
127                 argumentsIter.previous();
128                 stop = true;
129             }
130         }
131 
132         // The next argument MUST be a string-list of header names
133         if (argumentsIter.hasNext())
134         {
135             Object argument = argumentsIter.next();
136             if (argument instanceof StringListArgument)
137                 headerNames = ((StringListArgument) argument).getList();
138         }
139         if (null == headerNames)
140             throw context.getCoordinate().syntaxException("Expecting a StringList of header names");
141 
142         // The next argument MUST be a string-list of keys
143         if (argumentsIter.hasNext())
144         {
145             Object argument = argumentsIter.next();
146             if (argument instanceof StringListArgument)
147                 keys = ((StringListArgument) argument).getList();
148         }
149         else if (null == keys)
150             throw context.getCoordinate().syntaxException("Expecting a StringList of keys");
151 
152         if (argumentsIter.hasNext())
153             throw context.getCoordinate().syntaxException("Found unexpected arguments");
154 
155         return match(
156             mail,
157             (addressPart == null ? ALL_TAG : addressPart),
158             (comparator == null ? ASCII_CASEMAP_COMPARATOR : comparator),
159             (matchType == null ? IS_TAG : matchType),
160             headerNames,
161             keys);
162     }

164     /***
165      * Method match.
166      * @param mail
167      * @param addressPart
168      * @param comparator
169      * @param matchType
170      * @param headerNames
171      * @param keys
172      * @return boolean
173      * @throws SieveMailException
174      */
175     protected boolean match(
176         MailAdapter mail,
177         String addressPart,
178         String comparator,
179         String matchType,
180         List headerNames,
181         List keys)
182         throws SieveException
183     {
184         // Iterate over the header names looking for a match
185         boolean isMatched = false;
186         Iterator headerNamesIter = headerNames.iterator();
187         while (!isMatched && headerNamesIter.hasNext())
188         {
189             isMatched =
190                 match(
191                     mail,
192                     addressPart,
193                     comparator,
194                     matchType,
195                     (String) headerNamesIter.next(),
196                     keys);
197         }
198         return isMatched;
199     }

287     /***
288      * Method match.
289      * @param addressPart
290      * @param comparator
291      * @param matchType
292      * @param headerValue
293      * @param key
294      * @return boolean
295      * @throws SieveMailException
296      */
297     protected boolean match(
298         String addressPart,
299         String comparator,
300         String matchType,
301         String headerValue,
302         String key)
303         throws SieveException
304     {
305         // Attempt to create a new InternetAddress object from the headerValue
306         // If this fails, the header either is not intended to contain a valid
307         // Internet Address or is corrupt. Either way, its an Exception!
308         String address = null;
309         try
310         {
311             // address is a simple address; user@domain or user
312             address = new InternetAddress(headerValue).getAddress();
313         }
314         catch (AddressException e)
315         {
316             throw new InternetAddressException(e.getMessage());
317         }
318 
319         // Extract the part of the address we are matching on       
320         String matchAddress = null;
321         if (addressPart.equals(":all"))
322             matchAddress = address;
323         else
324         {
325             int localStart = 0;
326             int localEnd = 0;
327             int domainStart = 0;
328             int domainEnd = address.length();
329             int splitIndex = address.indexOf('@');
330             // If there is no domain part (-1), treat it as an empty String
331             if (splitIndex == -1)
332             {
333                 localEnd = domainEnd;
334                 domainStart = domainEnd;
335             }
336             else
337             {
338                 localEnd = splitIndex;
339                 domainStart = splitIndex + 1;
340             }
341             matchAddress =
342                 (addressPart.equals(LOCALPART_TAG)
343                     ? address.substring(localStart, localEnd)
344                     : address.substring(domainStart, domainEnd));
345         }
346 
347         // domain matches MUST ignore case, others should not
348         String matchKey = null;
349         if (addressPart.equals(DOMAIN_TAG))
350         {
351             matchKey = key.toLowerCase();
352             matchAddress = matchAddress.toLowerCase();
353         }
354         else
355             matchKey = key;
356 
357         // Match using the specified comparator          
358         return ComparatorUtils.match(
359             comparator,
360             matchType,
361             matchAddress,
362             matchKey);
363     }
