20  package org.apache.jsieve.tests;
21  
22  import org.apache.jsieve.comparators.ComparatorUtils;
23  import org.apache.jsieve.exception.InternetAddressException;
24  import org.apache.jsieve.exception.SieveException;
25  import org.apache.jsieve.mail.MailAdapter;
26  import org.apache.jsieve.mail.SieveMailException;
27  
28  /**
29   * Class Address implements the Addresss Test as defined in RFC 3028, section
30   * 5.1.
31   */
32  public class Address extends AbstractCompatatorTest {
33      /**
34       * Constructor for Address.
35       */
36      public Address() {
37          super();
38      }
39  
40      protected boolean match(MailAdapter mail, String addressPart,
41              String comparator, String matchType, String headerName, String key)
42              throws SieveException {
43          final MailAdapter.Address[] addresses = getMatchingValues(mail,
44                  headerName);
45          final int length = addresses.length;
46          int i = 0;
47          boolean isMatched = false;
48          while (!isMatched && i < length) {
49              isMatched = match(addressPart, comparator, matchType,
50                      addresses[i++], key);
51          }
52          return isMatched;
53      }
54  
55      private MailAdapter.Address[] getMatchingValues(MailAdapter mail,
56              String valueName) throws SieveMailException,
57              InternetAddressException {
58          return mail.parseAddresses(valueName);
59      }
60  
61      protected boolean match(String addressPart, String comparator,
62              String matchType, MailAdapter.Address address, String key)
63              throws SieveException {
64          final String localPart = address.getLocalPart();
65          final String domain = address.getDomain();
66  
67          // Extract the part of the address we are matching on
68          final String matchAddress;
69          if (addressPart.equals(":all"))
70              matchAddress = localPart + "@" + domain;
71          else {
72              matchAddress = (addressPart.equals(LOCALPART_TAG) ? localPart
73                      : domain.toLowerCase());
74          }
75  
76          // domain matches MUST ignore case, others should not
77          String matchKey = null;
78          if (addressPart.equals(DOMAIN_TAG)) {
79              matchKey = key.toLowerCase();
80          } else {
81              matchKey = key;
82          }
83  
84          // Match using the specified comparator
85          return ComparatorUtils.match(comparator, matchType, matchAddress,
86                  matchKey);
87      }
88  }
