32  /**
33   * <p>
34   * Singleton class <code>ConfigurationManager</code> parses the XML statements
35   * in the Sieve configuration file and translates them to Java objects.
36   * </p>
37   * 
38   * <p>
39   * The Sieve configuration is read from 3 properties file. They are
40   * located by searching the classpath of the current ClassLoader.
41   * org/apache/jsieve/commandsmap.properties
42   * org/apache/jsieve/testsmap.properties
43   * org/apache/jsieve/comparatorsmap.properties
44   * </p>
45   */

74      /**
75       * Constructor for ConfigurationManager.
76       * 
77       * @throws SieveConfigurationException
78       */
79      private ConfigurationManager() throws SieveConfigurationException {
80          super();
81          Log log = Logger.getLog();
82          try {
83              parse();
84          } catch (SAXException e) {
85              if (log.isErrorEnabled())
86                  log.error("Exception processing Configuration: ", e);
87              throw new SieveConfigurationException(e);
88          } catch (IOException e) {
89              if (log.isErrorEnabled())
90                  log.error("Exception processing Configuration: ", e);
91              throw new SieveConfigurationException(e);
92          }
93      }

147     /**
148      * <p>
149      * Method getConfigStream answers an InputStream over the Sieve
150      * configuration file. It is located by searching the classpath of the
151      * current ClassLoader.
152      * </p>
153      * <p>
154      * The context classloader is searched first. If a suitably named resource
155      * is found then this is returned. Otherwise, the classloader used to load
156      * this class is searched for the resource.
157      * </p>
158      * 
159      * @return InputStream
160      * @throws IOException
161      */
162     static protected InputStream getConfigStream(String configName) throws IOException {
163         InputStream stream = null;
164         // Context classloader is usually right in a JEE evironment
165         final ClassLoader contextClassLoader = Thread.currentThread()
166                 .getContextClassLoader();
167         if (contextClassLoader != null) {
168             stream = contextClassLoader.getResourceAsStream(configName);
169         }
170 
171         // Sometimes context classloader will not be set conventionally
172         // So, try class classloader
173         if (null == stream) {
174             stream = ConfigurationManager.class.getClassLoader()
175                     .getResourceAsStream(configName);
176         }
177 
178         if (null == stream)
179             throw new IOException("Resource \"" + configName + "\" not found");
180         return stream;
181     }

222     /**
223      * Method parse uses the Digester to parse the XML statements in the Sieve
224      * configuration file into Java objects.
225      * 
226      * @throws SAXException
227      * @throws IOException
228      */
229     protected void parse() throws SAXException, IOException {
230         InputStream is;
231         Properties p;
232         is = getConfigStream(COMMANDSMAP_PROPERTIES);
233         p = new Properties();
234         p.load(is);
235         setCommandMap(p);
236         is = getConfigStream(TESTSMAP_PROPERTIES);
237         p = new Properties();
238         p.load(is);
239         setTestMap(p);
240         is = getConfigStream(COMPARATORSMAP_PROPERTIES);
241         p = new Properties();
242         p.load(is);
243         setComparatorMap(p);
244     }
