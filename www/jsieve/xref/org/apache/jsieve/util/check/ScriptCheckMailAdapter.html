42  /***
43   * Checks script execution for an email.
44   * The wrapped email is set by called {@link #setMail}.
45   * Actions are recorded on {@link #executedActions}
46   * and can be retrieved by {@link #getExecutedActions()}.
47   */
48  public class ScriptCheckMailAdapter implements MailAdapter {
49  
50      private final List actions;
51      private final List executedActions;
52      
53      private Message mail = null;
54      
55      public ScriptCheckMailAdapter() {
56          actions = new ArrayList();
57          executedActions = new ArrayList();
58      }
59      
60      /***
61       * Gets the wrapped email.
62       * @return <code>Message</code>, possibly null
63       */
64      public Message getMail() {
65          return mail;
66      }
67  
68      /***
69       * Sets the wrapped email and {@link #reset}s the adapter
70       * ready for another execution.
71       * @param mail <code>Message</code>, possibly null
72       */
73      public void setMail(Message mail) {
74          this.mail = mail;
75          reset();
76      }
77  
78  
79  
80      /***
81       * Method addAction adds an Action to the List of Actions to be performed by the
82       * receiver.
83       * @param action
84       */
85      public void addAction(final Action action) {
86          actions.add(action);
87      }
88  
89      /***
90       * Method executeActions. Applies the Actions accumulated by the receiver.
91       */
92      public void executeActions() throws SieveException {
93          executedActions.clear();
94          executedActions.addAll(actions);
95      }
96  
97      /***
98       * Gets the actions accumulated when {@link #executedActions}
99       * was last called.
100      * @return <code>List</code> of {@link Action}s, not null.
101      * This list is a modifiable copy
102      */
103     public List getExecutedActions() {
104         final ArrayList result = new ArrayList(executedActions);
105         return result;
106     }
107 
108     /***
109      * Method getActions answers the List of Actions accumulated by the receiver.
110      * Implementations may elect to supply an unmodifiable collection.
111      * @return <code>List</code> of {@link Action}'s, not null, possibly unmodifiable
112      */
113     public List getActions() {
114         final List result = Collections.unmodifiableList(actions);
115         return result;
116     }
117 
118     /***
119      * Method getActionIteraror answers an Iterator over the List of Actions
120      * accumulated by the receiver. Implementations may elect to supply
121      * an unmodifiable iterator.
122      * @return <code>ListIterator</code>, not null, possibly unmodifiable
123      */
124     public ListIterator getActionsIterator() {
125         final List unmodifiableActions = getActions();
126         final ListIterator result = unmodifiableActions.listIterator();
127         return result;
128     }
129     
130     /***
131      * Resets executed and accumlated actions.
132      * An instance may be safely reused to check a script
133      * once this method has been called.
134      */
135     public void reset() {
136         executedActions.clear();
137         actions.clear();
138     }

140     /***
141      * Method getHeader answers a List of all of the headers in the receiver whose 
142      * name is equal to the passed name. If no headers are found an empty List is 
143      * returned.
144      * 
145      * @param name
146      * @return <code>List</code> not null, possibly empty
147      * @throws SieveMailException
148      */
149     public List getHeader(String name) throws SieveMailException {
150         List result = Collections.EMPTY_LIST;
151         if (mail != null) {
152             try {
153                 String[] values = mail.getHeader(name);
154                 if (values != null)
155                 {
156                     result = Arrays.asList(values);
157                 }
158             } catch (MessagingException e) {
159                 throw new SieveMailException(e);
160             }
161         }
162         return result;
163     }
164 
165     /***
166      * Method getHeaderNames answers a List of all of the headers in the receiver.
167      * No duplicates are allowed.
168      * @return <code>List</code>, not null possible empty, possible unmodifiable
169      * @throws SieveMailException
170      */
171     public List getHeaderNames() throws SieveMailException {
172         List results = Collections.EMPTY_LIST;
173         if (mail != null)
174         {
175             try {
176                 results = new ArrayList();
177                 for (final Enumeration en=mail.getAllHeaders();en.hasMoreElements();) {
178                     final Header header = (Header) en.nextElement();
179                     final String name = header.getName();
180                     if (!results.contains(name)) {
181                         results.add(name);
182                     }
183                 }
184             } catch (MessagingException e) {
185                 throw new SieveMailException(e);
186             }
187         }
188         return results;
189     }

191     /***
192      * <p>Method getMatchingHeader answers a List of all of the headers in the 
193      * receiver with the passed name. If no headers are found an empty List is 
194      * returned.
195      * </p>
196      * 
197      * <p>This method differs from getHeader(String) in that it ignores case and the 
198      * whitespace prefixes and suffixes of a header name when performing the
199      * match, as required by RFC 3028. Thus "From", "from ", " From" and " from "
200      * are considered equal.
201      * </p>
202      * 
203      * @param name
204      * @return <code>List</code>, not null possibly empty
205      * @throws SieveMailException
206      */
207     public List getMatchingHeader(String name) throws SieveMailException {
208         List result = Collections.EMPTY_LIST;
209         if (mail != null)
210         {
211             result = MailUtils.getMatchingHeader(this, name);
212         }
213         return result;
214     }
