21  package org.apache.jsieve;
22  
23  import org.apache.jsieve.parser.generated.Token;
24  
25  /***
26   * <p>A parsed representation of the RFC3028 BNF...</p>
27   * 
28   * <code>1*DIGIT [QUANTIFIER]</code>
29   * 
30   * <p>Note that the stored value is the absolute value after applying the quantifier.
31   * </p>
32   * 
33   */
34  public class NumberArgument implements Argument
35  {
36      
37      /***
38       * The absolute value of the number after applying the quentifier. 
39       */ 
40      private Integer fieldValue;
41  
42      /***
43       * Constructor for NumberArgument.
44       */
45      private NumberArgument()
46      {
47          super();
48      }
49      
50      /***
51       * Constructor for NumberArgument.
52       * @param token
53       */
54      public NumberArgument(Token token)
55      {
56          this();
57          setValue(token);
58      }
59  
60      /***
61       * Sets the value of the reciver to an Integer.
62       * @param number The value to set
63       */
64      protected void setValue(Integer number)
65      {
66          fieldValue = number;
67      }
68  
69      /***
70       * @see org.apache.jsieve.Argument#getValue()
71       */
72      public Object getValue()
73      {
74          return fieldValue;
75      }
76      
77      /***
78       * Method getInteger answers the value of the receiver as an Integer.
79       * @return Integer
80       */
81      public Integer getInteger()
82      {
83          return fieldValue;
84      }    
85  
86      /***
87       * Sets the value of the receiver from a Token.
88       * @param aToken The Token from which to extract the value to set
89       */
90      protected void setValue(Token aToken)
91      {
92          int endIndex = aToken.image.length();
93          int magnitude = 1;
94          if (aToken.image.endsWith("K"))
95          {
96              magnitude = 1024;
97              endIndex--;
98          }
99          else if (aToken.image.endsWith("M"))
100         {
101             magnitude = 1048576;
102             endIndex--;
103         }
104         else if (aToken.image.endsWith("G"))
105         {
106             magnitude = 1073741824;
107             endIndex--;
108         }
109 
110         setValue(
111             new Integer(
112                 Integer.parseInt(aToken.image.substring(0, endIndex))
113                     * magnitude));
114     }
115 
116     /***
117      * @see java.lang.Object#toString()
118      */
119     public String toString()
120     {
121         return (getValue() == null) ? "null" : getValue().toString();
122     }
