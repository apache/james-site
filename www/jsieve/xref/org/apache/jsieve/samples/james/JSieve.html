44  /***
45   * <p>
46   * Class JSieve is a DEMONSTRATION Mailet that invokes JSieve to 
47   * perform mail processing. There is a single configuration parameter,
48   * <code>scriptURL</code>. This a URL pointing to the resource containing
49   * the Sieve script to run.
50   * </p>
51   * 
52   * <p>
53   * This Mailet is intended to replace the <code>LocalDelivery</code>
54   * Mailet in James. Sieve's Keep Action is functionally equivalent to the
55   * <code>LocalDelivery</code> Mailet's processing. The Sieve script may
56   * explicitily or implicitly invoke this Action, and/or any other 
57   * configured Action. Actions are configured in class
58   * <code>ActionDispatcher</code>. 
59   * </p>
60   * 
61   * <p>IMPORTANT NOTES</p>
62   * 
63   * <p>
64   * This is NOT production quality code! It is a test harness for exercising
65   * jSieve. At least the feutures listed below would be required to consider
66   * this of production quality.
67   * </p>
68   * 
69   * <p>REQUIRED FEATURES</p>
70   * 
71   * <p>
72   * To be truly useful, this mailet needs to be configurable to invoke
73   * user specific Sieve scripts so that indivual users have control of their
74   * mail processing.
75   * </p>
76   * 
77   * <p>
78   * In a Mailet environment, a generic Sieve command to invoke a Mailet
79   * would be extremely powerful as jSieve could then leverage the abilities
80   * of all available Mailets. Currently, Sieve must wastefully duplicate 
81   * the same behaviour as Sieve commands.
82   * </p> 
83   * 
84   * <p>The converse also applies. Provision should be made for a Mailet to 
85   * reuse jSieve commands. As the primary difference between the two is 
86   * that Sieve deals with a single recipient while Mailets deal with
87   * multiple recipients, a Mailet could simply iterate over all of its
88   * recipients invoking the Sieve command for each recipient.  
89   * </p>
90   */

105     /***
106      * @see org.apache.mailet.Mailet#service(Mail)
107      */
108     public void service(Mail mail) throws MessagingException
109     {
110         // If the mail has no recipients, do nothing
111         if (mail.getRecipients().isEmpty())
112             return;
113         // Sieve expects a single recipient. If the mail has more we need to
114         // clone the mail, with each mail having a single recipient and
115         // resend them.
116         if (mail.getRecipients().size() == 1)
117         {
118             // Evaluate the mail against the script.
119             // The default state for the mail is GHOST.
120             // Actions executed as a result of evaluating the script may
121             // change this.
122             mail.setState(Mail.GHOST);
123             evaluate(mail);
124         }
125         else
126         {
127             Iterator recipientsIter = mail.getRecipients().iterator();
128             List recipients = new ArrayList(1);
129             while (recipientsIter.hasNext())
130             {
131                 //                MailImpl mailClone = duplicate((MailImpl) mail);
132                 recipients.clear();
133                 recipients.add(recipientsIter.next());
134                 //               mailClone.setRecipients(recipients);
135                 //               getMailetContext().sendMail(mailClone);
136                 getMailetContext().sendMail(mail.getSender(), recipients,
137                         mail.getMessage(), mail.getState());
138             }
139             // Kill the original message
140             mail.setState(Mail.GHOST);
141         }
142     }

144     protected MailImpl duplicate(MailImpl aMail) throws MessagingException
145     {
146         // duplicates the Mail object, to be able to modify the new mail
147         // keeping
148         // the original untouched
149         MailImpl newMail = (MailImpl) aMail.duplicate(newName(aMail));
150         // We don't need to use the original Remote Address and Host,
151         // and doing so would likely cause a loop with spam detecting
152         // matchers.
153         try
154         {
155             ((MailImpl) newMail).setRemoteAddr(java.net.InetAddress
156                     .getLocalHost().getHostAddress());
157             ((MailImpl) newMail).setRemoteHost(java.net.InetAddress
158                     .getLocalHost().getHostName());
159         }
160         catch (java.net.UnknownHostException _)
161         {
162             ((MailImpl) newMail).setRemoteAddr("127.0.0.1");
163             ((MailImpl) newMail).setRemoteHost("localhost");
164         }
165         return newMail;
166     }

168     /***
169      * Create a unique new primary key name.
170      * 
171      * Borrowed from org.apache.james.transport.mailets.AbstractRedirect
172      * 
173      * @param mail the mail to use as the basis for the new mail name
174      * @return a new name
175      */
176     private String newName(MailImpl mail) throws MessagingException
177     {
178         String oldName = mail.getName();
179         // Checking if the original mail name is too long, perhaps because of a
180         // loop caused by a configuration error.
181         // it could cause a "null pointer exception" in AvalonMailRepository
182         // much
183         // harder to understand.
184         if (oldName.length() > 76)
185         {
186             int count = 0;
187             int index = 0;
188             while ((index = oldName.indexOf('!', index + 1)) >= 0)
189             {
190                 count++;
191             }
192             // It looks like a configuration loop. It's better to stop.
193             if (count > 7)
194             {
195                 throw new MessagingException(
196                         "Unable to create a new message name: too long."
197                                 + " Possible loop in config.xml.");
198             }
199             else
200             {
201                 oldName = oldName.substring(0, 76);
202             }
203         }
204         StringBuffer nameBuffer = new StringBuffer(64).append(oldName).append(
205                 "-!").append(random.nextInt(1048576));
206         return nameBuffer.toString();
207     }

209     /***
210      * Method evaluate evaluates the receivers script against aMail.
211      * 
212      * @param aMail
213      * @throws MessagingException
214      */
215     protected void evaluate(Mail aMail) throws MessagingException
216     {
217         // Evaluate the script against the mail
218         try
219         {
220             MailAdapter aMailAdapter = new SieveMailAdapter(aMail,
221                     getMailetContext());
222             log("Evaluating " + aMailAdapter.toString() + "against \""
223                     + getScriptURL().toExternalForm() + "\"");
224             SieveFactory.getInstance().evaluate(aMailAdapter, getStartNode());
225         }
226         catch (SieveException ex)
227         {
228             log("Exception evaluating Sieve script", ex);
229             // If there were errors, we redirect the email to the ERROR
230             // processor.
231             // In order for this server to meet the requirements of the SMTP
232             // specification,
233             // mails on the ERROR processor must be returned to the sender.
234             // Note that this
235             // email doesn't include any details regarding the details of the
236             // failure(s).
237             // In the future we may wish to address this.
238             getMailetContext().sendMail(aMail.getSender(),
239                     aMail.getRecipients(), aMail.getMessage(), Mail.ERROR);
240             throw new MessagingException("Exception evaluating Sieve script",
241                     ex);
242         }
243     }

292     /***
293      * Parses the script and returns its startNode.
294      * 
295      * @return Node
296      * @throws MessagingException
297      */
298     protected Node computeStartNode() throws MessagingException
299     {
300         Object content = null;
301         try
302         {
303             content = getScriptURL().getContent();
304             if (!(content instanceof InputStream))
305             {
306                 String msg = "Cannot get an InputStream for "
307                         + getScriptURL().toExternalForm();
308                 log(msg);
309                 throw new MessagingException(msg);
310             }
311             return SieveFactory.getInstance().parse(
312                     new BufferedInputStream((InputStream) content));
313         }
314         catch (IOException ex)
315         {
316             String msg = "Exception getting contents of script URL: "
317                     + getScriptURL().toExternalForm();
318             log(msg, ex);
319             throw new MessagingException(msg, ex);
320         }
321         catch (ParseException ex)
322         {
323             String msg = "Exception parsing Sieve script: "
324                     + getScriptURL().toExternalForm();
325             log(msg, ex);
326             throw new MessagingException(msg, ex);
327         }
328     }
