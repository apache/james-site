64      /***
65       * <p>
66       * Executes the passed ActionFileInto.
67       * </p>
68       * 
69       * <p>
70       * This implementation accepts any destination with the root of <code>INBOX</code>.
71       * </p>
72       * 
73       * <p>
74       * As the current POP3 server does not support sub-folders, the mail is
75       * stored in the INBOX for the recipient of the mail and the full intended
76       * destination added as a prefix to the message's subject.
77       * </p>
78       * 
79       * <p>
80       * When IMAP support is added to James, it will be possible to support
81       * sub-folders of <code>INBOX</code> fully.
82       * </p>
83       * 
84       * @param anAction
85       * @param aMail
86       * @param aMailetContext
87       * @throws MessagingException
88       */
89      static public void execute(ActionFileInto anAction, Mail aMail,
90              MailetContext aMailetContext) throws MessagingException
91      {
92          StringBuffer repositoryDestinationBuffer = null;
93          MailAddress recipient = null;
94          boolean delivered = false;
95          try
96          {
97              recipient = getSoleRecipient(aMail);
98              // Validate and translate the destination
99              // Note that we do not check that the repository implementation
100             // supports the destination or that it exists. That is left
101             // to the repository implementation
102             int endOfRootDestination = anAction.getDestination().indexOf('/');
103             String rootDestination = (endOfRootDestination > -1
104                     ? anAction.getDestination().substring(0,
105                             endOfRootDestination)
106                     : anAction.getDestination());
107             if (!rootDestination.equalsIgnoreCase("INBOX"))
108                 throw new DestinationException("Folder: \""
109                         + anAction.getDestination() + "\"");
110             repositoryDestinationBuffer = new StringBuffer(recipient.toString());
111             if (endOfRootDestination > -1)
112                 repositoryDestinationBuffer.append(anAction.getDestination()
113                         .substring(endOfRootDestination).replace('/', '.'));
114             // Adapted from LocalDelivery Mailet
115             // Add qmail's de facto standard Delivered-To header
116             MimeMessage localMessage = new MimeMessage(aMail.getMessage())
117             {
118                 protected void updateHeaders() throws MessagingException
119                 {
120                     if (getMessageID() == null)
121                         super.updateHeaders();
122                     else
123                         modified = false;
124                 }
125             };
126             localMessage.addHeader("Delivered-To", recipient.toString());
127             // For now, a prefix "[destination]" is added to the
128             // header so that we can see what the true destination
129             // would have been with support for sub-folders
130             localMessage.setSubject("["
131                     + repositoryDestinationBuffer.toString() + "] "
132                     + localMessage.getSubject());
133             localMessage.saveChanges();
134             // NOTE: recipient will have to change to a MailAddress built
135             // from the repositoryDestinationBuffer
136             aMailetContext
137                     .storeMail(aMail.getSender(), recipient, localMessage);
138             delivered = true;
139         }
140         catch (MessagingException ex)
141         {
142             aMailetContext.log("Error while storing mail.", ex);
143             throw ex;
144         }
145         finally
146         {
147             // Ensure the mail is always ghosted
148             aMail.setState(Mail.GHOST);
149         }
150         if (delivered)
151         {
152             aMailetContext.log("Filed Message ID: "
153                     + aMail.getMessage().getMessageID()
154                     + " into destination: \""
155                     + repositoryDestinationBuffer.toString() + "\"");
156         }
157     }

159     /***
160      * <p>
161      * Executes the passed ActionKeep.
162      * </p>
163      * 
164      * <p>
165      * In this implementation, "keep" is equivalent to "fileinto" with a
166      * destination of "INBOX".
167      * </p>
168      * 
169      * @param anAction
170      * @param aMail
171      * @param aMailetContext
172      * @throws MessagingException
173      */
174     public static void execute(ActionKeep anAction, Mail aMail,
175             MailetContext aMailetContext) throws MessagingException
176     {
177         ActionFileInto action = new ActionFileInto("INBOX");
178         execute(action, aMail, aMailetContext);
179     }

181     /***
182      * Method execute executes the passed ActionRedirect.
183      * 
184      * @param anAction
185      * @param aMail
186      * @param aMailetContext
187      * @throws MessagingException
188      */
189     public static void execute(ActionRedirect anAction, Mail aMail,
190             MailetContext aMailetContext) throws MessagingException
191     {
192         detectAndHandleLocalLooping(aMail, aMailetContext, "redirect");
193         Collection recipients = new ArrayList(1);
194         recipients.add(new InternetAddress(anAction.getAddress()));
195         aMailetContext.sendMail(aMail.getSender(), recipients, aMail
196                 .getMessage());
197         aMail.setState(Mail.GHOST);
198         aMailetContext.log("Redirected Message ID: "
199                 + aMail.getMessage().getMessageID() + " to \""
200                 + anAction.getAddress() + "\"");
201     }

203     /***
204      * <p>
205      * Method execute executes the passed ActionReject. It sends an RFC 2098
206      * compliant reject MDN back to the sender.
207      * </p>
208      * <p>
209      * NOTE: The Mimecontent type should be 'report', but as we do not yet have
210      * a DataHandler for this yet, its currently 'text'!
211      * 
212      * @param anAction
213      * @param aMail
214      * @param aMailetContext
215      * @throws MessagingException
216      */
217     public static void execute(ActionReject anAction, Mail aMail,
218             MailetContext aMailetContext) throws MessagingException
219     {
220         detectAndHandleLocalLooping(aMail, aMailetContext, "reject");
221 
222         // Create the MDN part
223         StringBuffer humanText = new StringBuffer(128);
224         humanText
225                 .append("This message was refused by the recipient's mail filtering program.");
226         humanText.append("\r\n");
227         humanText.append("The reason given was:");
228         humanText.append("\r\n");
229         humanText.append("\r\n");
230         humanText.append(anAction.getMessage());
231 
232         String reporting_UA_name = null;
233         try
234         {
235             reporting_UA_name = InetAddress.getLocalHost()
236                     .getCanonicalHostName();
237         }
238         catch (UnknownHostException ex)
239         {
240             reporting_UA_name = "localhost";
241         }
242 
243         String reporting_UA_product = aMailetContext.getServerInfo();
244 
245         String[] originalRecipients = aMail.getMessage().getHeader(
246                 "Original-Recipient");
247         String original_recipient = null;
248         if (null != originalRecipients && originalRecipients.length > 0)
249         {
250             original_recipient = originalRecipients[0];
251         }
252 
253         MailAddress soleRecipient = getSoleRecipient(aMail);
254         String final_recipient = soleRecipient.toString();
255 
256         String original_message_id = aMail.getMessage().getMessageID();
257 
258         DispositionModifier modifiers[] = {new ModifierError()};
259         Disposition disposition = new Disposition(new ActionModeAutomatic(),
260                 new SendingModeAutomatic(), new TypeDeleted(), modifiers);
261 
262         MimeMultipart multiPart = MDNFactory.create(humanText.toString(),
263                 reporting_UA_name, reporting_UA_product, original_recipient,
264                 final_recipient, original_message_id, disposition);
265 
266         // Send the message
267         MimeMessage reply = (MimeMessage) aMail.getMessage().reply(false);
268         reply.setFrom(soleRecipient.toInternetAddress());
269         reply.setContent(multiPart);
270         reply.saveChanges();
271         Address[] recipientAddresses = reply.getAllRecipients();
272         if (null != recipientAddresses)
273         {
274             Collection recipients = new ArrayList(recipientAddresses.length);
275             for (int i = 0; i < recipientAddresses.length; i++)
276             {
277                 recipients.add(new MailAddress(
278                         (InternetAddress) recipientAddresses[i]));
279             }
280             aMailetContext.sendMail(null, recipients, reply);
281         }
282         else
283         {
284             aMailetContext
285                     .log("Unable to send reject MDN. Could not determine the recipient.");
286         }
287         // Ghost the original mail
288         aMail.setState(Mail.GHOST);
289     }

291     /***
292      * Answers the sole intended recipient for aMail.
293      * 
294      * @param aMail
295      * @return String
296      * @throws MessagingException
297      */
298     protected static MailAddress getSoleRecipient(Mail aMail)
299             throws MessagingException
300     {
301         if (aMail.getRecipients() == null) {
302           throw new MessagingException("Invalid number of recipients - 0"
303               + ". Exactly 1 recipient is expected.");
304         } else if (1 != aMail.getRecipients().size())
305             throw new MessagingException("Invalid number of recipients - "
306                     + new Integer(aMail.getRecipients().size()).toString()
307                     + ". Exactly 1 recipient is expected.");
308         return (MailAddress) aMail.getRecipients().iterator().next();
309     }

311     /***
312      * Returns a lazy initialised attributePrefix.
313      * 
314      * @return String
315      */
316     protected static String getAttributePrefix()
317     {
318         String value = null;
319         if (null == (value = getAttributePrefixBasic()))
320         {
321             updateAttributePrefix();
322             return getAttributePrefix();
323         }
324         return value;

365     /***
366      * Detect and handle locally looping mail. External loop detection is left
367      * to the MTA.
368      * 
369      * @param aMail
370      * @param aMailetContext
371      * @param anAttributeSuffix
372      * @throws MessagingException
373      */
374     protected static void detectAndHandleLocalLooping(Mail aMail,
375             MailetContext aMailetContext, String anAttributeSuffix)
376             throws MessagingException
377     {
378         MailAddress thisRecipient = getSoleRecipient(aMail);
379         MailAddress lastRecipient = (MailAddress) aMail
380                 .getAttribute(getAttributePrefix() + anAttributeSuffix);
381         if (null != lastRecipient && lastRecipient.equals(thisRecipient))
382         {
383             MessagingException ex = new MessagingException(
384                     "This message is looping! Message ID: "
385                             + aMail.getMessage().getMessageID());
386             aMailetContext.log(ex.getMessage(), ex);
387             throw ex;
388         }
389         aMail.setAttribute(getAttributePrefix() + anAttributeSuffix,
390                 thisRecipient);
391     }
