21  package org.apache.jsieve;
22  
23  import org.apache.commons.logging.Log;
24  
25  import org.apache.jsieve.mail.*;
26  
27  /***
28   * <p>A parsed representation of an RFC3028 test argument...</p>
29   * 
30   * <code>test = identifier arguments</code>
31   */
32  public class Test implements Executable
33  {
34      // The name of this Test
35      private String fieldName;
36      
37      // The arguments for this Test
38      private Arguments fieldArguments;
39          
40      /***
41       * @see org.apache.jsieve.Executable#execute(MailAdapter)
42       */
43      public Object execute(MailAdapter mail) throws SieveException
44      {
45          Log log = Logger.getLog();
46          if (log.isDebugEnabled())
47              log.debug(toString());
48          return new Boolean(
49              TestManager.getInstance().newInstance(getName()).execute(
50                  mail,
51                  getArguments()));
52      }
53  
54      /***
55       * Constructor for Test.
56       */
57      private Test()
58      {
59          super();
60      }
61      
62      /***
63       * Constructor for Test.
64       * @param name
65       * @param arguments
66       */
67      public Test(String name, Arguments arguments)
68      {
69          this();
70          setName(name);
71          setArguments(arguments);
72      }    
73  
74      /***
75       * Returns the arguments.
76       * @return Arguments
77       */
78      public Arguments getArguments()
79      {
80          return fieldArguments;
81      }
82  
83      /***
84       * Returns the name.
85       * @return String
86       */
87      public String getName()
88      {
89          return fieldName;
90      }
91  
92      /***
93       * Sets the arguments.
94       * @param children The arguments to set
95       */
96      protected void setArguments(Arguments arguments)
97      {
98          fieldArguments = arguments;
99      }
100 
101     /***
102      * Sets the name.
103      * @param name The name to set
104      */
105     protected void setName(String name)
106     {
107         fieldName = name;
108     }
109 
110     /***
111      * @see java.lang.Object#toString()
112      */
113     public String toString()
114     {
115         return "Test name: "
116             + getName()
117             + " "
118             + (getArguments() == null ? "null" : getArguments().toString());
119     }
