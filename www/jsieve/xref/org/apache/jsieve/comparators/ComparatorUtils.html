58      /***
59       * Method <code>match</code> answers a boolean indicating if the parameter 
60       * <code>matchTarget</code> compares to parameter <code>matchArgument</code> 
61       * is a match of <code>matchType</code> using the comparator 
62       * <code>comparatorName</code>.
63       * 
64       * @param comparatorName
65       * @param matchType
66       * @param matchArgument
67       * @param matchTarget
68       * @return boolean
69       */
70      public static boolean match(
71          String comparatorName,
72          String matchType,
73          String matchTarget,
74          String matchArgument) throws SieveException
75      {
76          boolean isMatched = false;
77          if (matchType.equals(IS_TAG))
78              isMatched = is(comparatorName, matchTarget, matchArgument);
79          else if (matchType.equals(CONTAINS_TAG))
80              isMatched = contains(comparatorName, matchTarget, matchArgument);
81          else if (matchType.equals(MATCHES_TAG))
82              isMatched = matches(comparatorName, matchTarget, matchArgument);
83          return isMatched;
84      }

86      /***   
87       * <p>Method <code>matches</code> answers a boolean indicating if the parameter 
88       * <code>string</code> matches the glob pattern described by parameter 
89       * <code>glob</code>.
90       * @param string
91       * @param glob
92       * @return boolean
93       * @throws SievePatternException
94       */
95      static public boolean matches(String string, String glob)
96          throws SievePatternException
97      {
98          // This requires optimization
99          // 1) DONE - Keep one compiler and one matcher in thread variables
100         // 2) Is there a way to re-use the compiled pattern?
101         Pattern pattern = null;
102         try
103         {
104             pattern = getGlobCompiler().compile(glob);
105         }
106         catch (MalformedPatternException e)
107         {
108             throw new SievePatternException(e.getMessage());
109         }
110         PatternMatcher matcher = getPerl5Matcher();
111         return matcher.matches(string, pattern);
112     }

142     /***
143      * <p>Returns the globCompiler, intialises if required.</p>
144      * 
145      * <p>Note that this must be synchronized to prevent another thread 
146      * detecting the null state while this thread is initialising.</p>
147      * 
148      * @return GlobCompiler
149      */
150     static synchronized protected GlobCompiler getGlobCompiler()
151     {
152         GlobCompiler globCompiler = null;
153         if (null == (globCompiler = getGlobCompilerBasic()))
154         {
155             updateGlobCompiler();
156             return getGlobCompiler();
157         }
158         return globCompiler;
159     }
