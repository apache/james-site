21  package org.apache.jsieve.commands;
22  
23  import java.util.List;
24  
25  import org.apache.jsieve.Arguments;
26  import org.apache.jsieve.Block;
27  import org.apache.jsieve.SieveException;
28  import org.apache.jsieve.StringListArgument;
29  import org.apache.jsieve.SyntaxException;
30  import org.apache.jsieve.mail.ActionRedirect;
31  import org.apache.jsieve.mail.MailAdapter;
32  
33  /***
34   * Class Redirect implements the Redirect Command as defined in RFC 3028, section 4.3.
35   */
36  public class Redirect extends AbstractActionCommand
37  {
38  
39      /***
40       * Constructor for Redirect.
41       */
42      public Redirect()
43      {
44          super();
45      }
46  
47      /***
48       * <p>Add an ActionRedirect to the List of Actions to be performed passing the
49       * sole StringList argument as the recipient.</p>
50       * <p>Also,
51       * @see org.apache.jsieve.commands.AbstractCommand#executeBasic(MailAdapter, Arguments, Block)
52       * </p>
53       */  
54      protected Object executeBasic(
55          MailAdapter mail,
56          Arguments arguments,
57          Block block)
58          throws SieveException
59      {
60          String recipient =
61              (String) ((StringListArgument) arguments.getArgumentList().get(0))
62                  .getList()
63                  .get(
64                  0);
65  
66          mail.addAction(new ActionRedirect(recipient));
67  
68          return null;
69      }
70  
71      /***
72       * @see org.apache.jsieve.commands.AbstractCommand#validateArguments(Arguments)
73       */
74      protected void validateArguments(Arguments arguments) throws SieveException
75      {
76          List args = arguments.getArgumentList();
77          if (args.size() != 1)
78              throw new SyntaxException(
79                  "Exactly 1 argument permitted. Found " + args.size());
80  
81          Object argument = args.get(0);
82          if (!(argument instanceof StringListArgument))
83              throw new SyntaxException("Expecting a string-list");
84  
85          if (1 != ((StringListArgument) argument).getList().size())
86              throw new SyntaxException("Expecting exactly one argument");
87      }
88  
89  }
