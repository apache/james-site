20  package org.apache.jsieve;
21  
22  import java.util.Map;
23  
24  import org.apache.jsieve.comparators.Comparator;
25  import org.apache.jsieve.exception.LookupException;
26  
27  /**
28   * Singleton class <code>ComparatorManager</code> maps Comparator names to
29   * configured Comparator implementation classes.
30   */
31  public class ComparatorManagerImpl implements ComparatorManager {
32  
33      private final Map classNameMap;
34  
35      /**
36       * Constructor for ComparatorManager.
37       */
38      public ComparatorManagerImpl(final Map classNameMap) {
39          super();
40          this.classNameMap = classNameMap;
41      }
42  
43      /**
44       * <p>
45       * Method lookup answers the class to which a Comparator name is mapped.
46       * </p>
47       * 
48       * @param name -
49       *            The name of the Comparator
50       * @return Class - The class of the Comparator
51       * @throws LookupException
52       */
53      public Class lookup(String name) throws LookupException {
54          Class comparatorClass = null;
55          try {
56              comparatorClass = getClass().getClassLoader().loadClass(
57                      getClassName(name));
58          } catch (ClassNotFoundException e) {
59              throw new LookupException("Comparator named '" + name
60                      + "' not found.");
61          }
62          if (!Comparator.class.isAssignableFrom(comparatorClass))
63              throw new LookupException("Class " + comparatorClass.getName()
64                      + " must implement " + Comparator.class.getName());
65          return comparatorClass;
66      }
67  
68      /**
69       * <p>
70       * Method newInstance answers an instance of the class to which a Comparator
71       * name is mapped.
72       * </p>
73       * 
74       * @param name -
75       *            The name of the Comparator
76       * @return Class - The class of the Comparator
77       * @throws LookupException
78       */
79      public Comparator getComparator(String name) throws LookupException {
80          try {
81              return (Comparator) lookup(name).newInstance();
82          } catch (InstantiationException e) {
83              throw new LookupException(e.getMessage());
84          } catch (IllegalAccessException e) {
85              throw new LookupException(e.getMessage());
86          }
87      }
88  
89      /**
90       * <p>
91       * Method getClassName answers the name of the class to which a Comparator
92       * name is mapped.
93       * </p>
94       * 
95       * @param name -
96       *            The name of the Comparator
97       * @return String - The name of the class
98       * @throws LookupException
99       */
100     private String getClassName(String name) throws LookupException {
101         String className = (String) classNameMap.get(name.toLowerCase());
102         if (null == className)
103             throw new LookupException("Command named '" + name
104                     + "' not mapped.");
105         return className;
106     }
107 }
