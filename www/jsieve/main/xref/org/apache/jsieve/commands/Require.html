20  package org.apache.jsieve.commands;
21  
22  import java.util.List;
23  
24  import org.apache.jsieve.Argument;
25  import org.apache.jsieve.Arguments;
26  import org.apache.jsieve.Block;
27  import org.apache.jsieve.SieveContext;
28  import org.apache.jsieve.StringListArgument;
29  import org.apache.jsieve.exception.FeatureException;
30  import org.apache.jsieve.exception.LookupException;
31  import org.apache.jsieve.exception.SieveException;
32  import org.apache.jsieve.mail.MailAdapter;
33  
34  /**
35   * Class Require implements the Require Command as defined in RFC 3028, section
36   * 3.2.
37   */
38  public class Require extends AbstractPrologCommand {
39  
40      /**
41       * Constructor for Require.
42       */
43      public Require() {
44          super();
45      }
46  
47      /**
48       * <p>
49       * Ensure the required feature is configured.
50       * </p>
51       * <p>
52       * Also,
53       * 
54       * @see org.apache.jsieve.commands.AbstractCommand#executeBasic(MailAdapter,
55       *      Arguments, Block, SieveContext)
56       *      </p>
57       */
58      protected Object executeBasic(MailAdapter mail, Arguments arguments,
59              Block block, SieveContext context) throws SieveException {
60          final List<String> stringArgumentList = ((StringListArgument) arguments
61                                  .getArgumentList().get(0)).getList();
62          for (String stringArgument: stringArgumentList) {
63              validateFeature(stringArgument, mail, context);
64          }
65          return null;
66      }
67  
68      /**
69       * Method validateFeature validates the required feature is configured as
70       * either a Command or a Test.
71       * 
72       * @param name
73       * @param mail
74       * @param context not nul
75       * @throws FeatureException
76       */
77      protected void validateFeature(String name, MailAdapter mail,
78              SieveContext context) throws FeatureException {
79          // Validate as a Command
80          try {
81              validateCommand(name, context);
82              return;
83          } catch (LookupException e) {
84              // Not a command
85          }
86  
87          // Validate as a Test
88          try {
89              validateTest(name, context);
90          } catch (LookupException e) {
91              throw new FeatureException("Feature \"" + name
92                      + "\" is not supported.");
93          }
94      }
95  
96      /**
97       * Method validateCommand.
98       * 
99       * @param name
100      * @throws LookupException
101      */
102     protected void validateCommand(String name, SieveContext context)
103             throws LookupException {
104         context.getCommand(name);
105     }
106 
107     /**
108      * Method validateTest.
109      * 
110      * @param name not null
111      * @param context not null
112      * @throws LookupException
113      */
114     protected void validateTest(String name, SieveContext context)
115             throws LookupException {
116         context.getTest(name);
117     }
118 
119     /**
120      * @see org.apache.jsieve.commands.AbstractCommand#validateArguments(Arguments,
121      *      SieveContext)
122      */
123     protected void validateArguments(Arguments arguments, SieveContext context)
124             throws SieveException {
125         List<Argument> args = arguments.getArgumentList();
126         if (args.size() != 1)
127             throw context.getCoordinate().syntaxException(
128                     new StringBuilder("Exactly 1 argument permitted. Found ").append((int)args.size()));
129 
130         Argument argument = args.get(0);
131         if (!(argument instanceof StringListArgument))
132             throw context.getCoordinate().syntaxException(
133                     "Expecting a string-list");
134     }
135 
136 }
