45  /**
46   * Receives a Mail from JamesSpoolManager and takes care of delivery of the
47   * message to local inboxes or a specific repository.
48   * 
49   * Differently from LocalDelivery this does not lookup the UserRepository This
50   * simply store the message in a repository named like the local part of the
51   * recipient address.
52   * 
53   * If no repository is specified then this fallback to MailServer.getUserInbox.
54   * Otherwise you can add your own configuration for the repository
55   * 
56   * e.g: <repositoryUrl>file://var/spool/userspools/</repositoryUrl>
57   * <repositoryType>SPOOL</repositoryType>
58   * 
59   * <repositoryUrl>file://var/mail/inboxes/</repositoryUrl> <repositoryType>MAIL</repositoryType>
60   * 
61   * Header "Delivered-To" can be added to every message adding the
62   * <addDeliveryHeader>Delivered-To</addDeliveryHeader>
63   * 
64   */

119     /**
120      * Delivers a mail to a local mailbox.
121      * 
122      * @param mail
123      *            the mail being processed
124      * 
125      * @throws MessagingException
126      *             if an error occurs while storing the mail
127      */
128     public void service(Mail mail) throws MessagingException {
129         Collection recipients = mail.getRecipients();
130         Collection errors = new Vector();
131 
132         MimeMessage message = null;
133         if (deliveryHeader != null || resetReturnPath) {
134             message = mail.getMessage();
135         }
136 
137         if (resetReturnPath) {
138             // Set Return-Path and remove all other Return-Path headers from the
139             // message
140             // This only works because there is a placeholder inserted by
141             // MimeMessageWrapper
142             message.setHeader(RFC2822Headers.RETURN_PATH,
143                     (mail.getSender() == null ? "<>" : "<" + mail.getSender()
144                             + ">"));
145         }
146 
147         Enumeration headers;
148         InternetHeaders deliveredTo = new InternetHeaders();
149         if (deliveryHeader != null) {
150             // Copy any Delivered-To headers from the message
151             headers = message
152                     .getMatchingHeaders(new String[] { deliveryHeader });
153             while (headers.hasMoreElements()) {
154                 Header header = (Header) headers.nextElement();
155                 deliveredTo.addHeader(header.getName(), header.getValue());
156             }
157         }
158 
159         for (Iterator i = recipients.iterator(); i.hasNext();) {
160             MailAddress recipient = (MailAddress) i.next();
161             try {
162                 if (deliveryHeader != null) {
163                     // Add qmail's de facto standard Delivered-To header
164                     message.addHeader(deliveryHeader, recipient.toString());
165                 }
166 
167                 storeMail(mail.getSender(), recipient, mail);
168 
169                 if (deliveryHeader != null) {
170                     if (i.hasNext()) {
171                         // Remove headers but leave all placeholders
172                         message.removeHeader(deliveryHeader);
173                         headers = deliveredTo.getAllHeaders();
174                         // And restore any original Delivered-To headers
175                         while (headers.hasMoreElements()) {
176                             Header header = (Header) headers.nextElement();
177                             message.addHeader(header.getName(), header
178                                     .getValue());
179                         }
180                     }
181                 }
182             } catch (Exception ex) {
183                 getMailetContext().log("Error while storing mail.", ex);
184                 errors.add(recipient);
185             }
186         }
187 
188         if (!errors.isEmpty()) {
189             // If there were errors, we redirect the email to the ERROR
190             // processor.
191             // In order for this server to meet the requirements of the SMTP
192             // specification, mails on the ERROR processor must be returned to
193             // the sender. Note that this email doesn't include any details
194             // regarding the details of the failure(s).
195             // In the future we may wish to address this.
196             getMailetContext().sendMail(mail.getSender(), errors,
197                     mail.getMessage(), Mail.ERROR);
198         }
199         // We always consume this message
200         mail.setState(Mail.GHOST);
201     }

215      * @param recipient
216      * @param mail
217      * @throws MessagingException
218      */
219     public void storeMail(MailAddress sender, MailAddress recipient,
220             Mail mail) throws MessagingException {
221         String username;
222         if (recipient == null) {
223             throw new IllegalArgumentException(
224                     "Recipient for mail to be spooled cannot be null.");
225         }
226         if (mail.getMessage() == null) {
227             throw new IllegalArgumentException(
228                     "Mail message to be spooled cannot be null.");
229         }
230         // recipient.toString was used here (JD)
231         username = recipient.getUser();
232         
233         storeMessage(username, mail);
234  
235     }

237     void storeMessage(String username, Mail mail) throws MessagingException {
238         String sieveFileName="../apps/james/var/sieve/"+username+".sieve";
239         if (new File(sieveFileName).canRead()) {
240             getMailetContext().log("Passing over to sieve delivery for "+username);
241             storeMessageSieve(sieveFileName, username, mail);
242         } else {
243             getMailetContext().log(new File(sieveFileName).getAbsolutePath()+ " does not exists");
244             storeMessageInbox(username, mail);
245         }
246     }
247     
248     void storeMessageSieve(String sieveFileName, String username, Mail aMail) throws MessagingException {
249         // Evaluate the script against the mail
250         try
251         {
252             MailAdapter aMailAdapter = new SieveMailAdapter(aMail,
253                     getMailetContext());
254             log("Evaluating " + aMailAdapter.toString() + "against \""
255                     + sieveFileName + "\"");
256             factory.evaluate(aMailAdapter,
257                     factory.parse(new FileInputStream(sieveFileName)));
258         }
259         catch (Exception ex)
260         {
261             //
262             // SLIEVE is a mail filtering protocol.
263             // Rejecting the mail because it cannot be filtered
264             // seems very unfriendly.
265             // So just log and store in INBOX.
266             //
267             log("Cannot evaluate Sieve script. Storing mail in user INBOX.", ex);
268             storeMessageInbox(username, aMail);
269         }
270     }
271     
272     void storeMessageInbox(String username, Mail mail) throws MessagingException {
273         String url = "mailbox://" + username + "@localhost/";
274         poster.post(url, mail.getMessage());
275     }
