376     public void addHeader(String name, String value) throws MessagingException {
377         Object newValue;
378         Object existingValue = m_contentHeaders.get(name);
379         if (existingValue == null) {
380             newValue = value;
381         } else if (existingValue instanceof String) {
382             List values = new ArrayList();
383             values.add(existingValue);
384             values.add(value);
385             newValue = values;
386         } else {
387             List values = (List) existingValue;
388             values.add(value);
389             newValue = values;
390         }
391         m_contentHeaders.put(name, newValue);

398     public Enumeration getAllHeaders() throws MessagingException {
399         final Collection results = new ArrayList();
400         final Collection entries = m_contentHeaders.entrySet();
401         for (Iterator it = entries.iterator(); it.hasNext();) {
402             Map.Entry entry = (Map.Entry) it.next();
403             String name = entry.getKey().toString();
404             Object value = entry.getValue();
405             if (value == null) {
406                 // ignore
407             } else if (value instanceof Collection) {
408                 Collection values = (Collection) value;
409                 for (Iterator iterValues = values.iterator(); iterValues
410                         .hasNext();) {
411                     String stringValue = (String) iterValues.next();
412                     results.add(new Header(name, stringValue));
413                 }
414             } else {
415                 results.add(new Header(name, value.toString()));
416             }
417         }
418         return Collections.enumeration(results);
419     }
