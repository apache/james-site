343     public void addHeader(String name, String value) throws MessagingException {
344         Object newValue;
345         Object existingValue = m_contentHeaders.get(name);
346         if (existingValue == null) {
347             newValue = value;
348         }
349         else if (existingValue instanceof String) {
350             List values = new ArrayList();
351             values.add(existingValue);
352             values.add(value);
353             newValue = values;
354         } else {
355             List values = (List) existingValue;
356             values.add(value);
357             newValue = values;
358         }
359         m_contentHeaders.put(name, newValue);
360     }

366     public Enumeration getAllHeaders() throws MessagingException {
367         final Collection results = new ArrayList();
368         final Collection entries = m_contentHeaders.entrySet();
369         for (Iterator it = entries.iterator(); it.hasNext();) {
370             Map.Entry entry = (Map.Entry) it.next();
371             String name = entry.getKey().toString();
372             Object value = entry.getValue();
373             if (value == null) {
374                 // ignore
375             } else if (value instanceof Collection) {
376                 Collection values = (Collection) value;
377                 for (Iterator iterValues = values.iterator();iterValues.hasNext();) {
378                     String stringValue = (String) iterValues.next();
379                     results.add(new Header(name, stringValue));
380                 }
381             } else {
382                 results.add(new Header(name, value.toString()));
383             }
384         }
385         return Collections.enumeration(results);
386     }
