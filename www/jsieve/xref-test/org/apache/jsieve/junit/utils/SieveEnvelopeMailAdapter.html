21  package org.apache.jsieve.junit.utils;
22  
23  import java.util.ArrayList;
24  import java.util.HashMap;
25  import java.util.Iterator;
26  import java.util.List;
27  import java.util.Map;
28  
29  import javax.mail.internet.MimeMessage;
30  
31  import org.apache.jsieve.mail.SieveMailException;
32  import org.apache.jsieve.mail.optional.EnvelopeAccessors;
33  
34  /**
35   * <p>Class SieveEnvelopeMailAdapter extends class SieveMailAdapter, a mock 
36   * implementation of a MailAdapter, to add support for EnvelopeAccessors.<p>
37   * 
38   * <p>As the Envelope Test is an optional Sieve test, MailAdapter support for the
39   * interface is optional too.</p>
40   */
41  public class SieveEnvelopeMailAdapter
42      extends SieveMailAdapter
43      implements EnvelopeAccessors
44  {
45      /**
46       * The FROM address used in the SMTP MAIL command.
47       */
48      private String fieldEnvelopeFrom;
49      
50      /**
51       * The TO address used in the SMTP RCPT command that resulted in this message
52       *  getting delivered to this user.
53       */ 
54      private String fieldEnvelopeTo;
55  
56      /**
57       * Constructor for SieveEnvelopeMailAdapter.
58       * @param message
59       */
60      public SieveEnvelopeMailAdapter(MimeMessage message)
61      {
62          super(message);
63      }
64      
65      /**
66       * Method getEnvelopes.
67       * @return Map
68       */
69      protected Map getEnvelopes()
70      {
71          Map envelopes = new HashMap(2);
72          if (null != getEnvelopeFrom())
73              envelopes.put("From", getEnvelopeFrom());
74          if (null != getEnvelopeTo())
75              envelopes.put("To", getEnvelopeTo());
76          return envelopes;
77      }
78          
79      /**
80       * @see org.apache.jsieve.mail.optional.EnvelopeAccessors#getEnvelope(String)
81       */
82      public List getEnvelope(String name) throws SieveMailException
83      {
84          List values = new ArrayList(1);
85          Object value = getEnvelopes().get(name);
86          if (null != value)
87              values.add(value);
88          return values;
89      }
90  
91      /**
92       * @see org.apache.jsieve.mail.optional.EnvelopeAccessors#getEnvelopeNames()
93       */
94      public List getEnvelopeNames() throws SieveMailException
95      {
96          return new ArrayList(getEnvelopes().keySet());
97      }
98  
99      /**
100      * @see org.apache.jsieve.mail.optional.EnvelopeAccessors#getMatchingEnvelope(String)
101      */
102     public List getMatchingEnvelope(String name) throws SieveMailException
103     {
104         Iterator envelopeNamesIter = getEnvelopeNames().iterator();
105         List matchedEnvelopeValues = new ArrayList(32);
106         while (envelopeNamesIter.hasNext())
107         {
108             String envelopeName = (String) envelopeNamesIter.next();
109             if (envelopeName.trim().equalsIgnoreCase(name))
110                 matchedEnvelopeValues.addAll(getEnvelope(envelopeName));
111         }
112 
113         return matchedEnvelopeValues;
114     }
115 
116     /**
117      * Returns the from.
118      * @return String
119      */
120     public String getEnvelopeFrom()
121     {
122         return fieldEnvelopeFrom;
123     }
124 
125     /**
126      * Returns the recipient.
127      * @return String
128      */
129     public String getEnvelopeTo()
130     {
131         return fieldEnvelopeTo;
132     }
133 
134     /**
135      * Sets the from.
136      * @param from The from to set
137      */
138     public void setEnvelopeFrom(String from)
139     {
140         fieldEnvelopeFrom = from;
141     }
142 
143     /**
144      * Sets the recipient.
145      * @param recipient The recipient to set
146      */
147     public void setEnvelopeTo(String recipient)
148     {
149         fieldEnvelopeTo = recipient;
150     }
