20  package org.apache.jsieve.junit;
21  
22  import java.io.ByteArrayInputStream;
23  import java.io.InputStream;
24  import java.io.StringBufferInputStream;
25  import java.io.StringReader;
26  import java.util.List;
27  
28  import junit.framework.TestCase;
29  
30  import org.apache.jsieve.util.check.ScriptChecker;
31  
32  public class MultipleToTest extends TestCase {
33      
34      private static final String SOLO_TO_EMAIL = 
35          "Date: Sun, 1 Apr 2007 1100:00:00 +0100 (BST)\r\f" +
36          "From: roadrunner@acme.example.com\r\f" +
37          "To: coyote@desert.example.org\r\f" +
38          "Subject: Who's The Fool?\r\f" +
39          "\r\f" +
40          "Beep-Beep\r\f"; 
41      
42      private static final String MULTIPLE_TO_EMAIL = 
43          "Date: Sun, 1 Apr 2007 1100:00:00 +0100 (BST)\r\f" +
44          "From: roadrunner@acme.example.com\r\f" +
45          "To: coyote@desert.example.org, bugs@example.org, " +
46          "    elmer@hunters.example.org,\r\f" +
47          "Subject: Who's The Fool?\r\f" +
48          "\r\f" +
49          "Beep-Beep\r\f"; 
50      
51      private static final String FILTER_SCRIPT = "require \"fileinto\";\r\f" +
52              "if address :is :all \"to\" \"coyote@desert.example.org\" {\r\f" +
53              "  fileinto \"coyote\";\r\f}\r\f" +
54              "if address :is :all \"to\" \"bugs@example.org\" {\r\f" +
55              "  fileinto \"bugs\";\r\f}\r\f" +
56              "if address :is :all \"to\" \"roadrunneracme.@example.org\" {\r\f" +
57              "  fileinto \"rr\";\r\f}\r\f" +
58              "if address :is :all \"to\" \"elmer@hunters.example.org\" {\r\f" +
59              "  fileinto \"elmer\";\r\f}\r\f";
60      
61      public void testSingleTo() throws Exception {
62          ScriptChecker checker = new ScriptChecker();
63          ScriptChecker.Results results = checker.check(toStream(SOLO_TO_EMAIL), toStream(FILTER_SCRIPT));
64          if (results.getException() != null) {
65              fail(results.getException().toString());
66          }
67          final List actionsExecuted = results.getActionsExecuted();
68          assertEquals(1, actionsExecuted.size());
69          assertTrue(results.isActionFileInto("coyote", 0));
70      }
71      
72      public void testMultipleTo() throws Exception {
73          ScriptChecker checker = new ScriptChecker();
74          ScriptChecker.Results results = checker.check(toStream(MULTIPLE_TO_EMAIL), toStream(FILTER_SCRIPT));
75          if (results.getException() != null) {
76              fail(results.getException().toString());
77          }
78          final List actionsExecuted = results.getActionsExecuted();
79          assertEquals(3, actionsExecuted.size());
80          assertTrue(results.isActionFileInto("coyote", 0));
81          assertTrue(results.isActionFileInto("bugs", 1));
82          assertTrue(results.isActionFileInto("elmer", 2));
83      }
84      
85      private InputStream toStream(String in) throws Exception {
86          byte[] bytes = in.getBytes("US-ASCII");
87          ByteArrayInputStream result = new ByteArrayInputStream(bytes);
88          return result;
89      }
90  }
