428         }

436     /**
437      * Writes the start of an element.
438      * 
439      * @param elementName the name of the element, not null
440      * @throws InvalidXmlException if the name is not valid for an xml element
441      * @throws OperationNotAllowedException 
442      * if called after the first element has been closed
443      */
444     public void openElement(final CharSequence elementName) throws IOException {
445         if (elementsWritten && elementNames.isEmpty()) {
446             throw new OperationNotAllowedException("Root element already closed. Cannot open new element.");
447         }
448         if (!isValidName(elementName)) {
449             throw new InvalidXmlException("'" + elementName + "' is not a valid element name");
450         }
451         elementsWritten = true;
452         if (inElement) {
453             writer.write('>');
454         }
455         writer.write('<');
456         rawWrite(elementName);
457         inElement = true;
458         elementNames.push(elementName);
459         currentAttributes.clear();
460     }

475         if (elementNames.isEmpty()) {

533     private void writeEscaped(final CharSequence content, boolean isAttributeContent) throws IOException {
534         final int length = content.length();
535         for (int i=0;i<length;i++) {
536             char character = content.charAt(i);
537             if (character == '&') {
538                 writer.write("&amp;");
539             } else if (character == '<') {
540                 writer.write("&lt;");
541             } else if (character == '>') {
542                 writer.write("&gt;");
543             } else if (isAttributeContent && character == '\'') {
544                 writer.write("&apos;");
545             } else if (isAttributeContent && character == '\"') {
546                 writer.write("&quot;");
547             } else if (isOutOfRange(character)) {
548                 writer.write('?');
549             } else {
550                 writer.write(character);
551             }
552         }
553     }

568         if (elementNames.isEmpty()) {
569             if (elementsWritten) {
570                 throw new OperationNotAllowedException("Root element has already been closed.");
571             } else {
572                 throw new OperationNotAllowedException("Close called before an element has been opened.");            
573             }
574         }
575         final CharSequence elementName = (CharSequence) elementNames.pop();
576         if (inElement) {
577             writer.write('/');
578             writer.write('>');

617     private boolean isValidName(final CharSequence sequence) {
618         boolean result = true;
619         final int length = sequence.length();
620         for (int i=0;i<length;i++) {
621             char character = sequence.charAt(i);
622             if (i==0) {
623                 if (!isValidNameStart(character)) {
624                     result = false;
625                     break;
626                 }
627             } else {
628                 if (!isValidNameBody(character)) {
629                     result = false;
630                     break;
631                 }
632             }
633         }
634         return result;
635     }
