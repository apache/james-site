25  /**
26   * The Util class provide some static methods which can be usefull in mailets
27   */
28  public class MailetUtil {
29      
30      /**
31       * <p>This takes the subject string and reduces (normailzes) it.
32       * Multiple "Re:" entries are reduced to one, and capitalized.  The
33       * prefix is always moved/placed at the beginning of the line, and
34       * extra blanks are reduced, so that the output is always of the
35       * form:</p>
36       * <code>
37       * &lt;prefix&gt; + &lt;one-optional-"Re:"*gt; + &lt;remaining subject&gt;
38       * </code>
39       * <p>I have done extensive testing of this routine with a standalone
40       * driver, and am leaving the commented out debug messages so that
41       * when someone decides to enhance this method, it can be yanked it
42       * from this file, embedded it with a test driver, and the comments
43       * enabled.</p>
44       */
45      public static String normalizeSubject(String subj, String prefix) {
46          StringBuffer subject = new StringBuffer(subj);
47          int prefixLength = prefix.length();
48  
49          // If the "prefix" is not at the beginning the subject line, remove it
50          int index = subject.indexOf(prefix);
51          if (index != 0) {
52  
53              if (index > 0) {
54                  subject.delete(index, index + prefixLength);
55              }
56              subject.insert(0, prefix); // insert prefix at the front
57          }
58  
59          // Replace Re: with RE:
60          String match = "Re:";
61          index = subject.indexOf(match, prefixLength);
62  
63          while(index > -1) {
64              subject.replace(index, index + match.length(), "RE:");
65              index = subject.indexOf(match, prefixLength);
66          }
67  
68          // Reduce them to one at the beginning
69          match ="RE:";
70          int indexRE = subject.indexOf(match, prefixLength) + match.length();
71          index = subject.indexOf(match, indexRE);
72          while(index > 0) {    
73              subject.delete(index, index + match.length());
74              index = subject.indexOf(match, indexRE);
75          }
76  
77          // Reduce blanks
78          match = "  ";
79          index = subject.indexOf(match, prefixLength);
80          while(index > -1) {
81              subject.replace(index, index + match.length(), " ");
82              index = subject.indexOf(match, prefixLength);
83          }
84          return subject.toString();
85      }
