27  /**
28   * A representation of an email address.
29   * <p>This class encapsulates functionalities to access to different
30   * parts of an email address without dealing with its parsing.</p>
31   *
32   * <p>A MailAddress is an address specified in the MAIL FROM and
33   * RCPT TO commands in SMTP sessions.  These are either passed by
34   * an external server to the mailet-compliant SMTP server, or they
35   * are created programmatically by the mailet-compliant server to
36   * send to another (external) SMTP server.  Mailets and matchers
37   * use the MailAddress for the purpose of evaluating the sender
38   * and recipient(s) of a message.</p>
39   *
40   * <p>MailAddress parses an email address as defined in RFC 821
41   * (SMTP) p. 30 and 31 where addresses are defined in BNF convention.
42   * As the mailet API does not support the aged "SMTP-relayed mail"
43   * addressing protocol, this leaves all addresses to be a <mailbox>,
44   * as per the spec. 
45   *
46   * <p>This class is a good way to validate email addresses as there are
47   * some valid addresses which would fail with a simpler approach
48   * to parsing address.  It also removes parsing burden from
49   * mailets and matchers that might not realize the flexibility of an
50   * SMTP address.  For instance, "serge@home"@lokitech.com is a valid
51   * SMTP address (the quoted text serge@home is the local-part and
52   * lokitech.com is the domain).  This means all current parsing to date
53   * is incorrect as we just find the first @ and use that to separate
54   * local-part from domain.</p>
55   *
56   * <p>This parses an address as per the BNF specification for <mailbox>
57   * from RFC 821 on page 30 and 31, section 4.1.2. COMMAND SYNTAX.
58   * http://www.freesoft.org/CIE/RFC/821/15.htm</p>
59   *
60   * @version 1.0
61   */
62  public class MailAddress implements java.io.Serializable {

77      /**
78       * strip source routing, according to RFC-2821 it is an allowed approach to handle mails
79       * contaning RFC-821 source-route information
80       */
81      private void stripSourceRoute(String address) {
82          if (pos < address.length()) {
83              if(address.charAt(pos)=='@') { 
84                  int i = address.indexOf(':');
85                  if(i != -1) {
86                      pos = i+1;
87                  }
88              }
89          }
90      }

92      /**
93       * <p>Construct a MailAddress parsing the provided <code>String</code> object.</p>
94       *
95       * <p>The <code>personal</code> variable is left empty.</p>
96       *
97       * @param   address the email address compliant to the RFC2822 3.4.1. Addr-spec specification
98       * @throws  AddressException    if the parse failed
99       */
100     public MailAddress(String address) throws AddressException {
101         address = address.trim();
102 
103         // Test if mail address has source routing information (RFC-821) and get rid of it!!
104         //must be called first!! (or at least prior to updating pos)
105         stripSourceRoute(address);
106 
107         StringBuffer localPartSB = new StringBuffer();
108         StringBuffer domainSB = new StringBuffer();
109         //Begin parsing
110         //<mailbox> ::= <local-part> "@" <domain>
111 
112         try {
113             //parse local-part
114             //<local-part> ::= <dot-string> | <quoted-string>
115             if (address.charAt(pos) == '\"') {
116                 localPartSB.append(parseQuotedLocalPart(address));
117                 if (localPartSB.toString().length() == 2) {
118                     throw new AddressException("No quoted local-part (user account) found at position " + (pos + 2) + " in '" + address + "'",address,pos+2);
119                 }
120             } else {
121                 localPartSB.append(parseUnquotedLocalPart(address));
122                 if (localPartSB.toString().length() == 0) {
123                     throw new AddressException("No local-part (user account) found at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
124                 }
125             }
126 
127             //find @
128             if (pos >= address.length() || address.charAt(pos) != '@') {
129                 throw new AddressException("Did not find @ between local-part and domain at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
130             }
131             pos++;
132 
133             //parse domain
134             //<domain> ::=  <element> | <element> "." <domain>
135             //<element> ::= <name> | "#" <number> | "[" <dotnum> "]"
136             while (true) {
137                 if (address.charAt(pos) == '#') {
138                     domainSB.append(parseNumber(address));
139                 } else if (address.charAt(pos) == '[') {
140                     domainSB.append(parseDomainLiteral(address));
141                 } else {
142                     domainSB.append(parseDomain(address));
143                 }
144                 if (pos >= address.length()) {
145                     break;
146                 }
147                 if (address.charAt(pos) == '.') {
148                     domainSB.append('.');
149                     pos++;
150                     continue;
151                 }
152                 break;
153             }
154 
155             if (domainSB.toString().length() == 0) {
156                 throw new AddressException("No domain found at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
157             }
158         } catch (IndexOutOfBoundsException ioobe) {
159             throw new AddressException("Out of data at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
160         }
161 
162         localPart = localPartSB.toString();
163         domain = domainSB.toString();
164     }

289     private String parseQuotedLocalPart(String address) throws AddressException {
290         StringBuffer resultSB = new StringBuffer();
291         resultSB.append('\"');
292         pos++;
293         //<quoted-string> ::=  """ <qtext> """
294         //<qtext> ::=  "\" <x> | "\" <x> <qtext> | <q> | <q> <qtext>
295         while (true) {
296             if (address.charAt(pos) == '\"') {
297                 resultSB.append('\"');
298                 //end of quoted string... move forward
299                 pos++;
300                 break;
301             }
302             if (address.charAt(pos) == '\\') {
303                 resultSB.append('\\');
304                 pos++;
305                 //<x> ::= any one of the 128 ASCII characters (no exceptions)
306                 char x = address.charAt(pos);
307                 if (x < 0 || x > 127) {
308                     throw new AddressException("Invalid \\ syntaxed character at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
309                 }
310                 resultSB.append(x);
311                 pos++;
312             } else {
313                 //<q> ::= any one of the 128 ASCII characters except <CR>,
314                 //<LF>, quote ("), or backslash (\)
315                 char q = address.charAt(pos);
316                 if (q <= 0 || q == '\n' || q == '\r' || q == '\"' || q == '\\') {
317                     throw new AddressException("Unquoted local-part (user account) must be one of the 128 ASCI characters exception <CR>, <LF>, quote (\"), or backslash (\\) at position " + (pos + 1) + " in '" + address + "'");
318                 }
319                 resultSB.append(q);
320                 pos++;
321             }
322         }
323         return resultSB.toString();
324     }

326     private String parseUnquotedLocalPart(String address) throws AddressException {
327         StringBuffer resultSB = new StringBuffer();
328         //<dot-string> ::= <string> | <string> "." <dot-string>
329         boolean lastCharDot = false;
330         while (true) {
331             //<string> ::= <char> | <char> <string>
332             //<char> ::= <c> | "\" <x>
333             if (address.charAt(pos) == '\\') {
334                 resultSB.append('\\');
335                 pos++;
336                 //<x> ::= any one of the 128 ASCII characters (no exceptions)
337                 char x = address.charAt(pos);
338                 if (x < 0 || x > 127) {
339                     throw new AddressException("Invalid \\ syntaxed character at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
340                 }
341                 resultSB.append(x);
342                 pos++;
343                 lastCharDot = false;
344             } else if (address.charAt(pos) == '.') {
345                 resultSB.append('.');
346                 pos++;
347                 lastCharDot = true;
348             } else if (address.charAt(pos) == '@') {
349                 //End of local-part
350                 break;
351             } else {
352                 //<c> ::= any one of the 128 ASCII characters, but not any
353                 //    <special> or <SP>
354                 //<special> ::= "<" | ">" | "(" | ")" | "[" | "]" | "\" | "."
355                 //    | "," | ";" | ":" | "@"  """ | the control
356                 //    characters (ASCII codes 0 through 31 inclusive and
357                 //    127)
358                 //<SP> ::= the space character (ASCII code 32)
359                 char c = address.charAt(pos);
360                 if (c <= 31 || c >= 127 || c == ' ') {
361                     throw new AddressException("Invalid character in local-part (user account) at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
362                 }
363                 for (int i = 0; i < SPECIAL.length; i++) {
364                     if (c == SPECIAL[i]) {
365                         throw new AddressException("Invalid character in local-part (user account) at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
366                     }
367                 }
368                 resultSB.append(c);
369                 pos++;
370                 lastCharDot = false;
371             }
372         }
373         if (lastCharDot) {
374             throw new AddressException("local-part (user account) ended with a \".\", which is invalid in address '" + address + "'",address,pos);
375         }
376         return resultSB.toString();
377     }

379     private String parseNumber(String address) throws AddressException {
380         //<number> ::= <d> | <d> <number>
381 
382         StringBuffer resultSB = new StringBuffer();
383         //We keep the position from the class level pos field
384         while (true) {
385             if (pos >= address.length()) {
386                 break;
387             }
388             //<d> ::= any one of the ten digits 0 through 9
389             char d = address.charAt(pos);
390             if (d == '.') {
391                 break;
392             }
393             if (d < '0' || d > '9') {
394                 throw new AddressException("In domain, did not find a number in # address at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
395             }
396             resultSB.append(d);
397             pos++;
398         }
399         return resultSB.toString();
400     }

402     private String parseDomainLiteral(String address) throws AddressException {
403         //throw away all irrelevant '\' they're not necessary for escaping of '.' or digits, and are illegal as part of the domain-literal
404         while(address.indexOf("\\")>-1){
405              address= address.substring(0,address.indexOf("\\")) + address.substring(address.indexOf("\\")+1);
406         }
407         StringBuffer resultSB = new StringBuffer();
408         //we were passed the string with pos pointing the the [ char.
409         // take the first char ([), put it in the result buffer and increment pos
410         resultSB.append(address.charAt(pos));
411         pos++;
412 
413         //<dotnum> ::= <snum> "." <snum> "." <snum> "." <snum>
414         for (int octet = 0; octet < 4; octet++) {
415             //<snum> ::= one, two, or three digits representing a decimal
416             //                      integer value in the range 0 through 255
417             //<d> ::= any one of the ten digits 0 through 9
418             StringBuffer snumSB = new StringBuffer();
419             for (int digits = 0; digits < 3; digits++) {
420                 char d = address.charAt(pos);
421                 if (d == '.') {
422                     break;
423                 }
424                 if (d == ']') {
425                     break;
426                 }
427                 if (d < '0' || d > '9') {
428                     throw new AddressException("Invalid number at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
429                 }
430                 snumSB.append(d);
431                 pos++;
432             }
433             if (snumSB.toString().length() == 0) {
434                 throw new AddressException("Number not found at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
435             }
436             try {
437                 int snum = Integer.parseInt(snumSB.toString());
438                 if (snum > 255) {
439                     throw new AddressException("Invalid number at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
440                 }
441             } catch (NumberFormatException nfe) {
442                 throw new AddressException("Invalid number at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
443             }
444             resultSB.append(snumSB.toString());
445             if (address.charAt(pos) == ']') {
446                 if (octet < 3) {
447                     throw new AddressException("End of number reached too quickly at " + (pos + 1) + " in '" + address + "'",address,pos+1);
448                 } 
449                 break;
450             }
451             if (address.charAt(pos) == '.') {
452                 resultSB.append('.');
453                 pos++;
454             }
455         }
456         if (address.charAt(pos) != ']') {
457             throw new AddressException("Did not find closing bracket \"]\" in domain at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
458         }
459         resultSB.append(']');
460         pos++;
461         return resultSB.toString();
462     }

464     private String parseDomain(String address) throws AddressException {
465         StringBuffer resultSB = new StringBuffer();
466         //<name> ::= <a> <ldh-str> <let-dig>
467         //<ldh-str> ::= <let-dig-hyp> | <let-dig-hyp> <ldh-str>
468         //<let-dig> ::= <a> | <d>
469         //<let-dig-hyp> ::= <a> | <d> | "-"
470         //<a> ::= any one of the 52 alphabetic characters A through Z
471         //  in upper case and a through z in lower case
472         //<d> ::= any one of the ten digits 0 through 9
473 
474         // basically, this is a series of letters, digits, and hyphens,
475         // but it can't start with a digit or hypthen
476         // and can't end with a hyphen
477 
478         // in practice though, we should relax this as domain names can start
479         // with digits as well as letters.  So only check that doesn't start
480         // or end with hyphen.
481         while (true) {
482             if (pos >= address.length()) {
483                 break;
484             }
485             char ch = address.charAt(pos);
486             if ((ch >= '0' && ch <= '9') ||
487                 (ch >= 'a' && ch <= 'z') ||
488                 (ch >= 'A' && ch <= 'Z') ||
489                 (ch == '-')) {
490                 resultSB.append(ch);
491                 pos++;
492                 continue;
493             }
494             if (ch == '.') {
495                 break;
496             }
497             throw new AddressException("Invalid character at " + pos + " in '" + address + "'",address,pos);
498         }
499         String result = resultSB.toString();
500         if (result.startsWith("-") || result.endsWith("-")) {
501             throw new AddressException("Domain name cannot begin or end with a hyphen \"-\" at position " + (pos + 1) + " in '" + address + "'",address,pos+1);
502         }
503         return result;
504     }
