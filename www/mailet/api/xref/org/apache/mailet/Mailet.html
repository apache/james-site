21  package org.apache.mailet;
22  
23  /**
24   * Draft of a Mailet inteface. The <code>service</code> perform all needed work
25   * on the Mail object. Whatever remains at the end of the service is considered
26   * to need futher processing and will go to the next Mailet if there is one
27   * configured or will go to the error processor if not.
28   * Setting a Mail state (setState(String)) to Mail.GHOST or cleaning its recipient
29   * list has the same meaning that s no more processing is needed.
30   * Instead of creating new messages, the mailet can put a message with new recipients
31   * at the top of the mail queue, or insert them immediately after it's execution
32   * through the API are provided by the MailetContext interface.
33   * <p>
34   * This interface defines methods to initialize a mailet, to service messages, and to
35   * remove a mailet from the server. These are known as life-cycle methods and are called
36   * in the following sequence:
37   * <ol>
38   * <li>The mailet is constructed, then initialized with the init method. </li>
39   * <li>Any messages for the service method are handled.</li>
40   * <li>The mailet is taken out of service, then destroyed with the destroy method,
41   *      then garbage collected and finalized.</li>
42   * </ol>
43   * In addition to the life-cycle methods, this interface provides the getMailetConfig
44   * method, which the mailet can use to get any startup information, and the
45   * getMailetInfo method, which allows the mailet to return basic information about itself,
46   * such as author, version, and copyright.
47   *
48   * @version 1.0.0, 24/04/1999
49   */
50  public interface Mailet {
51  
52      /**
53       * Called by the mailet container to indicate to a mailet that the
54       * mailet is being taken out of service. This method is only called once
55       * all threads within the mailet's service method have exited or after a
56       * timeout period has passed. After the mailet container calls this method,
57       * it will not call the service method again on this mailet.
58       * <p>
59       * This method gives the mailet an opportunity to clean up any resources that
60       * are being held (for example, memory, file handles, threads) and make sure
61       * that any persistent state is synchronized with the mailet's current state in memory.
62       */
63      void destroy();
64  
65      /**
66       * Returns information about the mailet, such as author, version, and
67       * copyright.
68       * <p>
69       * The string that this method returns should be plain text and not markup
70       * of any kind (such as HTML, XML, etc.).
71       *
72       * @return a String containing servlet information
73       */
74      String getMailetInfo();
75  
76      /**
77       * Returns a MailetConfig object, which contains initialization and
78       * startup parameters for this mailet.
79       * <p>
80       * Implementations of this interface are responsible for storing the MailetConfig
81       * object so that this method can return it. The GenericMailet class, which implements
82       * this interface, already does this.
83       *
84       * @return the MailetConfig object that initializes this mailet
85       */
86      MailetConfig getMailetConfig();
87  
88      /**
89       * Called by the mailet container to indicate to a mailet that the
90       * mailet is being placed into service.
91       * <p>
92       * The mailet container calls the init method exactly once after
93       * instantiating the mailet. The init method must complete successfully
94       * before the mailet can receive any requests.
95       *
96       * @param config - a MailetConfig object containing the mailet's configuration
97       *          and initialization parameters
98       * @throws javax.mail.MessagingException - if an exception has occurred that interferes with
99       *          the mailet's normal operation
100      */
101     void init(MailetConfig config) throws javax.mail.MessagingException;
102 
103     /**
104      * Called by the mailet container to allow the mailet to process to
105      * a message.
106      * <p>
107      * This method is only called after the mailet's init() method has completed
108      * successfully.
109      * <p>
110      * Mailets typically run inside multithreaded mailet containers that can handle
111      * multiple requests concurrently. Developers must be aware to synchronize access
112      * to any shared resources such as files, network connections, as well as the
113      * mailet's class and instance variables. More information on multithreaded
114      * programming in Java is available in <a href="http://java.sun.com/Series/Tutorial/java/threads/multithreaded.html">the
115      * Java tutorial on multi-threaded programming</a>.
116      *
117      * @param mail - the Mail object that contains the message and routing information
118      * @throws javax.mail.MessagingException - if a message or address parsing exception occurs or
119      *      an exception that interferes with the mailet's normal operation
120      */
121     void service(Mail mail) throws javax.mail.MessagingException;
122 }
