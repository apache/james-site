19  /**
20   * <p>
21   * This mailet takes an attachment stored in an attribute and attach it back to
22   * the message
23   * </p>
24   * <p>
25   * This may be used to place back attachment stripped by StripAttachment and
26   * stored in the attribute
27   * <code>org.apache.james.transport.mailets.StripAttachment.saved</code>
28   * </p>
29   * <p>
30   * 
31   * <pre>
32   *   &lt;mailet match=&quot;All&quot; class=&quot;RecoverAttachment&quot; &gt;
33   *     &lt;attribute&gt;my.attribute.name&lt;/attribute&gt;
34   *   &lt;/mailet &gt;
35   * </pre>
36   * 
37   * </p>
38   */
39  public class RecoverAttachment extends GenericMailet {
40  
41      public static final String ATTRIBUTE_PARAMETER_NAME = "attribute";
42  
43      private String attributeName = null;
44  
45      /**
46       * Checks if the mandatory parameters are present
47       * 
48       * @throws MailetException
49       */
50      public void init() throws MailetException {
51          attributeName = getInitParameter(ATTRIBUTE_PARAMETER_NAME);
52  
53          if (attributeName == null) {
54              throw new MailetException(ATTRIBUTE_PARAMETER_NAME
55                      + " is a mandatory parameter");
56          }
57  
58          log("RecoverAttachment is initialised with attribute [" + attributeName
59                  + "]");
60      }
61  
62      /**
63       * Service the mail: check for the attribute and attach the attachment to
64       * the mail.
65       * 
66       * @param mail
67       *            The mail to service
68       * @throws MailetException
69       *             Thrown when an error situation is encountered.
70       */
71      public void service(Mail mail) throws MailetException {
72          Map attachments = (Map) mail.getAttribute(attributeName);
73          if (attachments != null) {
74  
75              MimeMessage message = null;
76              try {
77                  message = mail.getMessage();
78              } catch (MessagingException e) {
79                  throw new MailetException(
80                          "Could not retrieve message from Mail object", e);
81              }
82  
83              Iterator i = attachments.values().iterator();
84              try {
85                  while (i.hasNext()) {
86                      byte[] bytes = (byte[]) i.next();
87                      InputStream is = new BufferedInputStream(
88                              new ByteArrayInputStream(bytes));
89                      MimeBodyPart p = new MimeBodyPart(is);
90                      if (!(message.isMimeType("multipart/*") && (message
91                              .getContent() instanceof MimeMultipart))) {
92                          Object content = message.getContent();
93                          String contentType = message.getContentType();
94                          MimeMultipart mimeMultipart = new MimeMultipart();
95                          message.setContent(mimeMultipart);
96                          // This saveChanges is required when the MimeMessage has
97                          // been created from
98                          // an InputStream, otherwise it is not saved correctly.
99                          message.saveChanges();
100                         mimeMultipart.setParent(message);
101                         MimeBodyPart bodyPart = new MimeBodyPart();
102                         mimeMultipart.addBodyPart(bodyPart);
103                         bodyPart.setContent(content, contentType);
104                     }
105                     ((MimeMultipart) message.getContent()).addBodyPart(p);
106                 }
107                 message.saveChanges();
108             } catch (MessagingException e) {
109                 log("MessagingException in recoverAttachment", e);
110             } catch (IOException e) {
111                 log("IOException in recoverAttachment", e);
112             }
113         }
114     }
