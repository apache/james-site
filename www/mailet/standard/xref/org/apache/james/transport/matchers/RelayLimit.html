22  package org.apache.james.transport.matchers;
23  
24  import org.apache.mailet.base.RFC2822Headers;
25  import org.apache.mailet.base.GenericMatcher;
26  import org.apache.mailet.Mail;
27  
28  import javax.mail.Header;
29  import javax.mail.MessagingException;
30  import javax.mail.internet.MimeMessage;
31  import java.util.Collection;
32  import java.util.Enumeration;
33  
34  /**
35   * Matches mail which has been relayed more than a given number of times.
36   * @version 1.0.0, 1/5/2000
37   */
38  
39  public class RelayLimit extends GenericMatcher {
40      int limit = 30;
41  
42      public void init() throws MessagingException {
43          try {
44              limit = Integer.parseInt(getCondition());
45          } catch (NumberFormatException e) {
46              throw new MessagingException("No valid integer: " + getCondition());
47          }
48      }
49  
50      public Collection match(Mail mail) throws javax.mail.MessagingException {
51          MimeMessage mm = mail.getMessage();
52          int count = 0;
53          for (Enumeration e = mm.getAllHeaders(); e.hasMoreElements();) {
54              Header hdr = (Header)e.nextElement();
55              if (hdr.getName().equals(RFC2822Headers.RECEIVED)) {
56                  count++;
57              }
58          }
59          if (count >= limit) {
60              return mail.getRecipients();
61          } else {
62              return null;
63          }
64      }
65  }
