106     /**
107      * Checks if the mandatory parameters are present, creates the directory to
108      * save the files ni (if not present).
109      * 
110      * @throws MailetException
111      */
112     public void init() throws MailetException {
113         patternString = getInitParameter(PATTERN_PARAMETER_NAME);
114         notpatternString = getInitParameter(NOTPATTERN_PARAMETER_NAME);
115         if (patternString == null && notpatternString == null) {
116             throw new MailetException("No value for " + PATTERN_PARAMETER_NAME
117                     + " parameter was provided.");
118         }
119 
120         directoryName = getInitParameter(DIRECTORY_PARAMETER_NAME);
121         attributeName = getInitParameter(ATTRIBUTE_PARAMETER_NAME);
122 
123         removeAttachments = getInitParameter(REMOVE_ATTACHMENT_PARAMETER_NAME,
124                 REMOVE_NONE).toLowerCase();
125         if (!REMOVE_MATCHED.equals(removeAttachments)
126                 && !REMOVE_ALL.equals(removeAttachments)) {
127             removeAttachments = REMOVE_NONE;
128         }
129 
130         try {
131             // if (patternString != null) regExPattern = new
132             // Perl5Compiler().compile(patternString);
133             if (patternString != null)
134                 regExPattern = Pattern.compile(patternString);
135         } catch (Exception e) {
136             throw new MailetException("Could not compile regex ["
137                     + patternString + "].");
138         }
139         try {
140             // if (notpatternString != null) notregExPattern = new
141             // Perl5Compiler().compile(notpatternString);
142             if (notpatternString != null)
143                 notregExPattern = Pattern.compile(notpatternString);
144         } catch (Exception e) {
145             throw new MailetException("Could not compile regex ["
146                     + notpatternString + "].");
147         }
148 
149         if (directoryName != null) {
150             try {
151                 File saveDirectory = null;
152                 saveDirectory = new File(directoryName);
153                 if (!saveDirectory.exists()) {
154                     saveDirectory.mkdirs();
155                 }
156             } catch (Exception e) {
157                 throw new MailetException("Could not create directory ["
158                         + directoryName + "].", e);
159             }
160         }
161 
162         decodeFilename = getBooleanParameter(
163                 getInitParameter(DECODE_FILENAME_PARAMETER_NAME),
164                 decodeFilename);
165         if (getInitParameter(REPLACE_FILENAME_PATTERN_PARAMETER_NAME) != null) {
166             List[] pl = ReplaceContent
167                     .getPatternsFromString(getInitParameter(REPLACE_FILENAME_PATTERN_PARAMETER_NAME));
168             replaceFilenamePatterns = (Pattern[]) pl[0].toArray(new Pattern[0]);
169             replaceFilenameSubstitutions = (String[]) pl[1]
170                     .toArray(new String[0]);
171             replaceFilenameFlags = (Integer[]) pl[2].toArray(new Integer[0]);
172         }
173 
174         String toLog = "StripAttachment is initialised with regex pattern ["
175                 + patternString + " / " + notpatternString + "]";
176         if (directoryName != null) {
177             toLog += " and will save to directory [" + directoryName + "]";
178         }
179         if (attributeName != null) {
180             toLog += " and will store attachments to attribute ["
181                     + attributeName + "]";
182         }
183         log(toLog);
184     }

186     /**
187      * Service the mail: scan it for attchemnts matching the pattern, store the
188      * content of a matchin attachment in the given directory.
189      * 
190      * @param mail
191      *            The mail to service
192      * @throws MailetException
193      *             Thrown when an error situation is encountered.
194      */
195     public void service(Mail mail) throws MailetException {
196         MimeMessage message = null;
197         try {
198             message = mail.getMessage();
199         } catch (MessagingException e) {
200             throw new MailetException(
201                     "Could not retrieve message from Mail object", e);
202         }
203         // All MIME messages with an attachment are multipart, so we do nothing
204         // if it is not mutlipart
205         try {
206             if (message.isMimeType("multipart/*")) {
207                 analyseMultipartPartMessage(message, mail);
208             }
209         } catch (MessagingException e) {
210             throw new MailetException(
211                     "Could not retrieve contenttype of message.", e);
212         } catch (Exception e) {
213             throw new MailetException("Could not analyse message.", e);
214         }
215     }

226     /**
227      * Checks every part in this part (if it is a Multipart) for having a
228      * filename that matches the pattern. If the name matches, the content of
229      * the part is stored (using its name) in te given diretcory.
230      * 
231      * Note: this method is recursive.
232      * 
233      * @param part
234      *            The part to analyse.
235      * @param mail
236      * @return
237      * @throws Exception
238      */
239     private boolean analyseMultipartPartMessage(Part part, Mail mail)
240             throws Exception {
241         if (part.isMimeType("multipart/*")) {
242             try {
243                 Multipart multipart = (Multipart) part.getContent();
244                 boolean atLeastOneRemoved = false;
245                 int numParts = multipart.getCount();
246                 for (int i = 0; i < numParts; i++) {
247                     Part p = multipart.getBodyPart(i);
248                     if (p.isMimeType("multipart/*")) {
249                         atLeastOneRemoved |= analyseMultipartPartMessage(p,
250                                 mail);
251                     } else {
252                         boolean removed = checkMessageRemoved(p, mail);
253                         if (removed) {
254                             multipart.removeBodyPart(i);
255                             atLeastOneRemoved = true;
256                             i--;
257                             numParts--;
258                         }
259                     }
260                 }
261                 if (atLeastOneRemoved) {
262                     part.setContent(multipart);
263                     if (part instanceof Message) {
264                         ((Message) part).saveChanges();
265                     }
266                 }
267                 return atLeastOneRemoved;
268             } catch (Exception e) {
269                 log("Could not analyse part.", e);
270             }
271         }
272         return false;
273     }

275     private boolean checkMessageRemoved(Part part, Mail mail)
276             throws MessagingException, Exception {
277         String fileName = null;
278         fileName = part.getFileName();
279 
280         // filename or name of part can be null, so we have to be careful
281         boolean ret = false;
282 
283         if (fileName != null) {
284             if (decodeFilename)
285                 fileName = MimeUtility.decodeText(fileName);
286 
287             if (replaceFilenamePatterns != null)
288                 fileName = ReplaceContent.applyPatterns(
289                         replaceFilenamePatterns, replaceFilenameSubstitutions,
290                         replaceFilenameFlags, fileName, 0, this);
291 
292             if (fileNameMatches(fileName)) {
293                 if (directoryName != null) {
294                     String filename = saveAttachmentToFile(part, fileName);
295                     if (filename != null) {
296                         Collection c = (Collection) mail
297                                 .getAttribute(SAVED_ATTACHMENTS_ATTRIBUTE_KEY);
298                         if (c == null) {
299                             c = new ArrayList();
300                             mail.setAttribute(SAVED_ATTACHMENTS_ATTRIBUTE_KEY,
301                                     (ArrayList) c);
302                         }
303                         c.add(filename);
304                     }
305                 }
306                 if (attributeName != null) {
307                     Map m = (Map) mail.getAttribute(attributeName);
308                     if (m == null) {
309                         m = new LinkedHashMap();
310                         mail.setAttribute(attributeName, (LinkedHashMap) m);
311                     }
312                     ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
313                     OutputStream os = new BufferedOutputStream(
314                             byteArrayOutputStream);
315                     part.writeTo(os);
316                     m.put(fileName, byteArrayOutputStream.toByteArray());
317                 }
318                 if (removeAttachments.equals(REMOVE_MATCHED)) {
319                     ret = true;
320                 }
321             }
322             if (!ret) {
323                 ret = removeAttachments.equals(REMOVE_ALL);
324             }
325             if (ret) {
326                 Collection c = (Collection) mail
327                         .getAttribute(REMOVED_ATTACHMENTS_ATTRIBUTE_KEY);
328                 if (c == null) {
329                     c = new ArrayList();
330                     mail.setAttribute(REMOVED_ATTACHMENTS_ATTRIBUTE_KEY,
331                             (ArrayList) c);
332                 }
333                 c.add(fileName);
334             }
335         }
336         return ret;
337     }

339     /**
340      * Checks if the given name matches the pattern.
341      * 
342      * @param name
343      *            The name to check for a match.
344      * @return True if a match is found, false otherwise.
345      */
346     private boolean fileNameMatches(String name) {
347         boolean result = true;
348         if (regExPattern != null)
349             result = regExPattern.matcher(name).matches();
350         if (result && notregExPattern != null)
351             result = !notregExPattern.matcher(name).matches();
352 
353         String log = "attachment " + name + " ";
354         if (!result)
355             log += "does not match";
356         else
357             log += "matches";
358         log(log);
359         return result;
360     }

362     /**
363      * Saves the content of the part to a file in the given directoy, using the
364      * name of the part. If a file with that name already exists, it will
365      * 
366      * @param part
367      *            The MIME part to save.
368      * @return
369      * @throws Exception
370      */
371     private String saveAttachmentToFile(Part part, String fileName)
372             throws Exception {
373         BufferedOutputStream os = null;
374         InputStream is = null;
375         File f = null;
376         try {
377             if (fileName == null)
378                 fileName = part.getFileName();
379             int pos = -1;
380             if (fileName != null) {
381                 pos = fileName.lastIndexOf(".");
382             }
383             String prefix = pos > 0 ? (fileName.substring(0, pos)) : fileName;
384             String suffix = pos > 0 ? (fileName.substring(pos)) : ".bin";
385             while (prefix.length() < 3)
386                 prefix += "_";
387             if (suffix.length() == 0)
388                 suffix = ".bin";
389             f = File.createTempFile(prefix, suffix, new File(directoryName));
390             log("saving content of " + f.getName() + "...");
391             os = new BufferedOutputStream(new FileOutputStream(f));
392             is = part.getInputStream();
393             if (!(is instanceof BufferedInputStream)) {
394                 is = new BufferedInputStream(is);
395             }
396             int c;
397             while ((c = is.read()) != -1) {
398                 os.write(c);
399             }
400 
401             return f.getName();
402         } catch (Exception e) {
403             log("Error while saving contents of ["
404                     + (f != null ? f.getName() : (part != null ? part
405                             .getFileName() : "NULL")) + "].", e);
406             throw e;
407         } finally {
408             is.close();
409             os.close();
410         }
411     }
