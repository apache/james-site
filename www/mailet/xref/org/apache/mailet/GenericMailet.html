30  /***
31   * GenericMailet makes writing mailets easier. It provides simple
32   * versions of the lifecycle methods init and destroy and of the methods
33   * in the MailetConfig interface. GenericMailet also implements the log
34   * method, declared in the MailetContext interface.
35   * <p>
36   * To write a generic mailet, you need only override the abstract service
37   * method.
38   *
39   * @version 1.0.0, 24/04/1999
40   */
41  public abstract class GenericMailet implements Mailet, MailetConfig {
42      private MailetConfig config = null;
43  
44      /***
45       * Called by the mailer container to indicate to a mailet that the
46       * mailet is being taken out of service.
47       */
48      public void destroy() {
49          //Do nothing
50      }
51  
52      /***
53       * Returns a String containing the value of the named initialization
54       * parameter, or null if the parameter does not exist.
55       * <p>
56       * This method is supplied for convenience. It gets the value of the
57       * named parameter from the mailet's MailetConfig object.
58       *
59       * @param name - a String specifying the name of the initialization parameter
60       * @return a String containing the value of the initalization parameter
61       */
62      public String getInitParameter(String name) {
63          return config.getInitParameter(name);
64      }
65  
66      /***
67       * Returns a String containing the value of the named initialization
68       * parameter, or defValue if the parameter does not exist.
69       * <p>
70       * This method is supplied for convenience. It gets the value of the
71       * named parameter from the mailet's MailetConfig object.
72       *
73       * @param name - a String specifying the name of the initialization parameter
74       * @param defValue - a String specifying the default value when the parameter
75       *                    is not present
76       * @return a String containing the value of the initalization parameter
77       */
78      public String getInitParameter(String name, String defValue) {
79          String res = config.getInitParameter(name);
80          if (res == null) {
81              return defValue;
82          } else {
83              return res;
84          }
85      }

87      /***
88       * Returns the names of the mailet's initialization parameters as an
89       * Iterator of String objects, or an empty Iterator if the mailet has no

91       * <p>
92       * This method is supplied for convenience. It gets the parameter names from
93       * the mailet's MailetConfig object.
94       *
95       * @return an Iterator of String objects containing the names of
96       *          the mailet's initialization parameters
97       */
98      public Iterator getInitParameterNames() {
99          return config.getInitParameterNames();
100     }

209     /***
210      * Utility method: Checks if there are unallowed init parameters specified in the 
211      * configuration file against the String[] allowedInitParameters.
212      * @param allowedArray array of strings containing the allowed parameter names
213      * @throws MessagingException if an unknown parameter name is found
214      */
215     protected final void checkInitParameters(String[] allowedArray) throws MessagingException {
216         // if null then no check is requested
217         if (allowedArray == null) {
218             return;
219         }
220         
221         Collection allowed = new HashSet();
222         Collection bad = new ArrayList();
223         
224         for (int i = 0; i < allowedArray.length; i++) {
225             allowed.add(allowedArray[i]);
226         }
227         
228         Iterator iterator = getInitParameterNames();
229         while (iterator.hasNext()) {
230             String parameter = (String) iterator.next();
231             if (!allowed.contains(parameter)) {
232                 bad.add(parameter);
233             }
234         }
235         
236         if (bad.size() > 0) {
237             throw new MessagingException("Unexpected init parameters found: "
238                     + arrayToString(bad.toArray()));
239         }
240     }
241     
242     /***
243      * Utility method for obtaining a string representation of an array of Objects.
244      */
245     protected final String arrayToString(Object[] array) {
246         if (array == null) {
247             return "null";
248         }
249         StringBuffer sb = new StringBuffer(1024);
250         sb.append("[");
251         for (int i = 0; i < array.length; i++) {
252             if (i > 0) {
253                 sb.append(",");
254             }
255             sb.append(array[i]);
256         }
257         sb.append("]");
258         return sb.toString();
259     }
