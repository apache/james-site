21  package org.apache.mailet;
22  import javax.mail.MessagingException;
23  import javax.mail.internet.MimeMessage;
24  import java.io.Serializable;
25  import java.util.Collection;
26  import java.util.Date;
27  import java.util.Iterator;
28  
29  /***
30   * Wrap a MimeMessage with routing information (from SMTP) such
31   * as SMTP specified recipients, sender, and ip address and hostname
32   * of sending server.  It also contains its state which represents
33   * which processor in the mailet container it is currently running.
34   * Special processor names are "root" and "error".
35   *
36   * @version CVS $Revision: 534853 $ $Date: 2007-05-03 13:39:16 +0000 (gio, 03 mag 2007) $
37   */
38  public interface Mail extends Serializable, Cloneable {
39      String GHOST = "ghost";
40      String DEFAULT = "root";
41      String ERROR = "error";
42      String TRANSPORT = "transport";
43      /***
44       * Returns the message name of this message
45       * 
46       * @return the message name
47       * @since Mailet API v2.3
48       */
49      String getName();
50      /***
51       * Set the message name of this message
52       * 
53       * @param newName new name
54       * @since Mailet API v2.3
55       */
56      void setName(String newName);
57      /***
58       * Returns the MimeMessage stored in this message
59       *
60       * @return the MimeMessage that this Mail object wraps
61       * @throws MessagingException - an error occured while loading this object
62       */
63      MimeMessage getMessage() throws MessagingException;
64      /***
65       * Returns a Collection of MailAddress objects that are recipients of this message
66       *
67       * @return a Collection of MailAddress objects that are recipients of this message
68       */
69      Collection getRecipients();
70      /***
71       * Method setRecipients.
72       * @param recipients a Collection of MailAddress Objects representing the recipients of this message
73       * @since Mailet API v3.0-unstable
74       */
75      void setRecipients(Collection recipients);
76      /***
77       * The sender of the message, as specified by the MAIL FROM header, or internally defined
78       *
79       * @return a MailAddress of the sender of this message
80       */
81      MailAddress getSender();
82      /***
83       * The current state of the message, such as GHOST, ERROR, or DEFAULT
84       *
85       * @return the state of this message
86       */
87      String getState();
88      /***
89       * The remote hostname of the server that connected to send this message
90       *
91       * @return a String of the hostname of the server that connected to send this message
92       */
93      String getRemoteHost();
94      /***
95       * The remote ip address of the server that connected to send this message
96       *
97       * @return a String of the ip address of the server that connected to send this message
98       */
99      String getRemoteAddr();
100     /***
101      * The error message, if any, associated with this message.  Not sure why this is needed.
102      *
103      * @return a String of a descriptive error message
104      */
105     String getErrorMessage();
106     /***
107      * Sets the error message associated with this message.  Not sure why this is needed.
108      *
109      * @param msg - a descriptive error message
110      */
111     void setErrorMessage(String msg);
112     /***
113      * Sets the MimeMessage associated with this message via the object.
114      *
115      * @param message - the new MimeMessage that this Mail object will wrap
116      */
117     void setMessage(MimeMessage message);
118     /***
119      * Sets the state of this message.
120      *
121      * @param state - the new state of this message
122      */
123     void setState(String state);
124     /***
125      * Returns the Mail session attribute with the given name, or null
126      * if there is no attribute by that name.
127      * An attribute allows a mailet to give this Mail instance additional information
128      * not already provided by this interface.<p>
129      * A list of currently set attributes can be retrieved using getAttributeNames.
130      * <p>
131      * The attribute is returned as a java.lang.Object or some subclass. Attribute
132      * names should follow the same convention as package names. The Mailet API
133      * specification reserves names matching <I>org.apache.james.*</I>
134      * and <I>org.apache.mailet.*</I>.
135      *
136      * @param name - a String specifying the name of the attribute
137      * @return an Object containing the value of the attribute, or null if no attribute
138      *      exists matching the given name
139      * @since Mailet API v2.1
140      */
141     Serializable getAttribute(String name);
142     /***
143      * Returns an Iterator containing the attribute names currently available within
144      * this Mail instance.  Use the getAttribute(java.lang.String) method with an
145      * attribute name to get the value of an attribute.
146      *
147      * @return an Iterator of attribute names
148      * @since Mailet API v2.1
149      */
150     Iterator getAttributeNames();
151     /***
152      * @return true if this Mail instance has any attributes set.
153      * @since Mailet API v2.1
154      **/
155     boolean hasAttributes();
156     /***
157      * Removes the attribute with the given name from this Mail instance. After
158      * removal, subsequent calls to getAttribute(java.lang.String) to retrieve
159      * the attribute's value will return null.
160      *
161      * @param name - a String specifying the name of the attribute to be removed
162      * @return previous attribute value associated with specified name, or null
163      * if there was no mapping for name (null can also mean that null
164      * was bound to the name)
165      * @since Mailet API v2.1
166      */
167     Serializable removeAttribute(String name);
168     /***
169      * Removes all the attributes associated with this Mail instance.  
170      * @since Mailet API v2.1
171      **/
172     void removeAllAttributes();
173     /***
174      * Binds an object to a given attribute name in this Mail instance. If the name
175      * specified is already used for an attribute, this method will remove the old
176      * attribute and bind the name to the new attribute.
177      * As instances of Mail is Serializable, it is necessary that the attributes being
178      * Serializable as well
179      * <p>
180      * Attribute names should follow the same convention as package names.
181      * The Mailet API specification reserves names matching <I>org.apache.james.*</I>
182      * and <I>org.apache.mailet.*</I>.
183      *
184      * @param name - a String specifying the name of the attribute
185      * @param object - a Serializable Object representing the attribute to be bound
186      * @return the object previously bound to the name, null if the name was
187      * not bound (null can also mean that null was bound to the name)
188      * @since Mailet API v2.1
189      */
190     Serializable setAttribute(String name, Serializable object);
191     /***
192      * @return message size
193      * @since Mailet API v2.3
194      */
195     long getMessageSize() throws MessagingException;
196     /***
197      * @return the last update date
198      * @since Mailet API v2.3
199      */
200     Date getLastUpdated();
201     /***
202      * @param lastUpdated the new last updated date
203      * @since Mailet API v2.3
204      */
205     void setLastUpdated(Date lastUpdated);
206 }
